56 Contact NameServer::unregisterName(const std::string& name)
57 {
58  Contact prev = queryName(name);
59  if (prev.isValid()) {
60  if (prev.getPort() != -1) {
61  NameRecord& rec = getNameRecord(prev.getRegName());
62  if (rec.isReusablePort()) {
63  HostRecord& host = getHostRecord(prev.getHost());
64  host.release(prev.getPort());
65  }
66  if (rec.isReusableIp()) {
67  if (rec.getAddress().getCarrier() == "mcast") {
68  mcastRecord.releaseAddress(rec.getAddress().getHost().c_str());
69  }
70  }
71  rec.clear();
72  tmpNames.release(name);
73 
74  Bottle event;
75  event.addVocab(Vocab::encode("del"));
76  event.addString(name.c_str());
77  onEvent(event);
78  }
79  }
80 
81  return queryName(name);
82 }

85 Contact NameServer::registerName(const std::string& name,
86  const Contact& address,
87  const std::string& remote)
88 {
89  bool reusablePort = false;
90  bool reusableIp = false;
91 
92  //YARP_DEBUG(Logger::get(), "in registerName...");
93 
94  if (name != "...") {
95  unregisterName(name);
96  }
97 
98  Contact suggestion = address;
99 
100  if (!suggestion.isValid()) {
101  suggestion = Contact(name, "...", "...", 0);
102  }
103 
104  std::string portName = name;
105  if (portName == "...") {
106  portName = tmpNames.get();
107  }
108 
109  std::string carrier = suggestion.getCarrier();
110  if (carrier == "...") {
111  carrier = "tcp";
112  }
113 
114  std::string machine = suggestion.getHost();
115  int overridePort = 0;
116  if (machine == "...") {
117  if (carrier != "mcast") {
118  if (remote == "...") {
119  YARP_ERROR(Logger::get(), "remote machine name was not found! can only guess it is local...");
120  machine = "127.0.0.1";
121  } else {
122  machine = remote;
123  }
124  } else {
125  machine = mcastRecord.get();
126  overridePort = mcastRecord.lastPortNumber();
127  reusableIp = true;
128  }
129  }
130 
131  int port = suggestion.getPort();
132  if (port == 0) {
133  if (overridePort != 0) {
134  port = overridePort;
135  } else {
136  port = getHostRecord(machine).get();
137  reusablePort = true;
138  }
139  }
140 
141  suggestion = Contact(portName, carrier, machine, port);
142 
143  YARP_DEBUG(Logger::get(), std::string("Registering ") + suggestion.toURI() + " for " + suggestion.getRegName());
144 
145  NameRecord& nameRecord = getNameRecord(suggestion.getRegName());
146  nameRecord.setAddress(suggestion, reusablePort, reusableIp);
147 
148  Bottle event;
149  event.addVocab(Vocab::encode("add"));
150  event.addString(suggestion.getRegName().c_str());
151  onEvent(event);
152 
153  return nameRecord.getAddress();
154 }

157 Contact NameServer::queryName(const std::string& name)
158 {
159  std::string base = name;
160  std::string pat;
161  if (name.find("/net=") == 0) {
162  size_t patStart = 5;
163  size_t patEnd = name.find('/', patStart);
164  if (patEnd >= patStart && patEnd != std::string::npos) {
165  pat = name.substr(patStart, patEnd - patStart);
166  base = name.substr(patEnd);
167  YARP_DEBUG(Logger::get(), std::string("Special query form ") + name + " (" + pat + "/" + base + ")");
168  }
169  }
170 
171  NameRecord* rec = getNameRecord(base, false);
172  if (rec != nullptr) {
173  if (!pat.empty()) {
174  std::string ip = rec->matchProp("ips", pat);
175  if (!ip.empty()) {
176  SplitString sip(ip.c_str());
177  Contact c = rec->getAddress();
178  c.setHost(sip.get(0));
179  return c;
180  }
181  }
182  return rec->getAddress();
183  }
184  return Contact();
185 }

333 std::string NameServer::cmdRoute(int argc, char* argv[])
334 {
335  // ignore source
336  argc--;
337  argv++;
338 
339  if (argc < 2) {
340  return terminate("need at least two arguments: the source port and the target port\n(followed by an optional list of carriers in decreasing order of desirability)");
341  }
342  std::string src = STR(argv[0]);
343  std::string dest = STR(argv[1]);
344 
345  argc -= 2;
346  argv += 2;
347 
348  const char* altArgv[] = {
349  "local",
350  "shmem",
351  "mcast",
352  "udp",
353  "tcp",
354  "text"};
355  int altArgc = 6;
356 
357  if (argc == 0) {
358  argc = altArgc;
359  argv = (char**)altArgv;
360  }
361 
362 
363  NameRecord& srcRec = getNameRecord(src);
364  NameRecord& destRec = getNameRecord(dest);
365  std::string pref;
366 
367  for (int i = 0; i < argc; i++) {
368  std::string carrier = argv[i];
369  if (srcRec.checkProp("offers", carrier) && destRec.checkProp("accepts", carrier)) {
370  bool ok = true;
371  if (carrier == "local" || carrier == "shmem") {
372  if (srcRec.getProp("ips") == destRec.getProp("ips")) {
373  if (carrier == "local") {
374  if (srcRec.getProp("process") != destRec.getProp("process")) {
375  ok = false;
376  }
377  }
378  } else {
379  ok = false;
380  }
381  }
382  if (ok) {
383  pref = carrier;
384  break;
385  }
386  }
387  }
388  if (!pref.empty()) {
389  pref = pref + ":/" + dest;
390  } else {
391  pref = dest;
392  }
393 
394  std::string result = "port ";
395  result += src + " route " + dest + " = " + pref + "\n";
396  return terminate(result);
397 }

711  if (!result.empty()) {

716  //YARP_DEBUG(Logger::get(), std::string("name server request -- ") + txt);

750  bool read(ConnectionReader& reader) override
751  {
752  YTRACE("NameServer::read start");
753  std::string ref = "NAME_SERVER ";
754  bool ok = true;
755  std::string msg = "?";
756  bool haveMessage = false;
757  if (ok) {
758  if (reader.isTextMode()) {
759  msg = reader.expectText();
760  } else {
761  // migrate to binary mode support, eventually optimize
762  Bottle b;
763  b.read(reader);
764  msg = b.toString();
765  }
766  haveMessage = (!msg.empty());
767  msg = ref + msg;
768  }
769  if (reader.isActive() && haveMessage) {
770  YARP_DEBUG(Logger::get(), std::string("name server got message ") + msg);
