58 boolean fill_net_input_buffer (j_decompress_ptr cinfo)
59 {
60  // The whole JPEG data is expected to reside in the supplied memory
61  // buffer, so any request for more data beyond the given buffer size
62  // is treated as an error.
63  auto* mybuffer = (JOCTET *) cinfo->client_data;
64  fprintf(stderr, "JPEG data unusually large\n");
65  // Insert a fake EOI marker
66  mybuffer[0] = (JOCTET) 0xFF;
67  mybuffer[1] = (JOCTET) JPEG_EOI;
68  cinfo->src->next_input_byte = mybuffer;
69  cinfo->src->bytes_in_buffer = 2;
70  return TRUE;
71 }

79 void skip_net_input_data (j_decompress_ptr cinfo, long num_bytes)
80 {
81  auto src = (net_src_ptr) cinfo->src;
82 
83  if (num_bytes > 0) {
84  while (num_bytes > (long) src->bytes_in_buffer) {
85  num_bytes -= (long) src->bytes_in_buffer;
86  (void) (*src->fill_input_buffer) (cinfo);
87  }
88  src->next_input_byte += (size_t) num_bytes;
89  src->bytes_in_buffer -= (size_t) num_bytes;

141  bool decompress(const Bytes& cimg, FlexImage& img) {
142  bool debug = false;
143 
144  if (!active) {
145  init();
146  active = true;
147  }
148  cinfo.client_data = &error_buffer;
149  cinfo.err = jpeg_std_error(&jerr.pub);
150  jerr.pub.error_exit = net_error_exit;
151 
152  if (setjmp(jerr.setjmp_buffer)) {
153  jpeg_finish_decompress(&cinfo);
154  return false;
155  }
156 
157  jpeg_net_src(&cinfo,(char*)cimg.get(),cimg.length());
158  jpeg_save_markers(&cinfo, JPEG_COM, 0xFFFF);
159  jpeg_read_header(&cinfo, TRUE);
160  jpeg_calc_output_dimensions(&cinfo);
161 
162  if(cinfo.jpeg_color_space == JCS_GRAYSCALE) {
163  img.setPixelCode(VOCAB_PIXEL_MONO);
164  }
165  else
166  {
167  img.setPixelCode(VOCAB_PIXEL_RGB);
168  }
169 
170  if (debug) printf("Got image %dx%d\n", cinfo.output_width, cinfo.output_height);
171  img.resize(cinfo.output_width,cinfo.output_height);
172  jpeg_start_decompress(&cinfo);
173  //int row_stride = cinfo.output_width * cinfo.output_components;
174 
175  int at = 0;
176  while (cinfo.output_scanline < cinfo.output_height) {
177  JSAMPLE *lines[1];
178  lines[0] = (JSAMPLE*)(img.getPixelAddress(0,at));
179  jpeg_read_scanlines(&cinfo, lines, 1);
180  at++;
181  }
182  if(readEnvelopeCallback && cinfo.marker_list && cinfo.marker_list->data_length > 0) {
183  Bytes envelope(reinterpret_cast<char*>(cinfo.marker_list->data), cinfo.marker_list->data_length);
184  readEnvelopeCallback(readEnvelopeCallbackData, envelope);
185  }
186  if (debug) printf("Read image!\n");
187  jpeg_finish_decompress(&cinfo);
188  return true;
189  }

192  jpeg_destroy_decompress(&cinfo);
193  }

198  active = false;
