65 bool ShmemCarrier::becomeShmemVersionHybridStream(yarp::os::ConnectionState& proto, bool sender)
66 {
67  auto* stream = new ShmemHybridStream();
68  yAssert(stream != nullptr);
69  yarp::os::Contact base;
70 
71  bool ok = true;
72 
73  if (!sender) {
74  ACE_INET_Addr anywhere((u_short)0, (ACE_UINT32)INADDR_ANY);
75  base = yarp::os::Contact(anywhere.get_host_addr(), anywhere.get_port_number());
76  bool ok = stream->open(base, sender) == 0;
77  if (ok) {
78  int myPort = stream->getLocalAddress().getPort();
79  writeYarpInt(myPort, proto);
80  stream->accept();
81  proto.takeStreams(nullptr);
82  proto.takeStreams(stream);
83  }
84  } else {
85  int altPort = readYarpInt(proto);
86  std::string myName = proto.getStreams().getLocalAddress().getHost();
87  proto.takeStreams(nullptr);
88  base = yarp::os::Contact(myName, altPort);
89  ok = stream->open(base, sender) == 0;
90  if (ok) {
91  proto.takeStreams(stream);
92  }
93  }
94 
95  if (!ok) {
96  delete stream;
97  stream = nullptr;
98  return false;
99  }
100 
101  return true;
102 }
