70 Contact AllocatorOnTriples::completePortNumber(const Contact& c) {
71  if (c.getPort()!=-1 && c.getPort()!=0) {
72  return c;
73  }
74 
75  // unlike standard yarp name server, port number allocation
76  // is global across the network, rather than per machine.
77 
78  // we also try to keep port numbers stable for port names,
79  // when possible.
80 
81  string npref;
82  int pref = -1;
83  string nstring;
84  int number = -1;
85  Triple t;
86  t.setNsNameValue("alloc","regid","*");
87  TripleContext context;
88  context.setRid(db->find(t, nullptr));
89  if (context.rid>=0) {
90  t.setNsNameValue("prefer","*",c.getName().c_str());
91  list<Triple> match = db->query(t,&context);
92  if (match.size()>0) {
93  npref = match.begin()->name;
94  pref = atoi(npref.c_str());
95  t.setNsNameValue("alloc",npref.c_str(),"in_use");
96  match = db->query(t,&context);
97  if (match.size()==0) {
98  nstring = npref;
99  number = pref;
100  }
101  }
102  }
103 
104  if (nstring=="") {
105  if (regid==-1) {
106  Triple t;
107  t.setNsNameValue("alloc","regid","*");
108  list<Triple> lst = db->query(t, nullptr);
109  if (lst.size()>0) {
110  regid = atoi(lst.begin()->value.c_str());
111  }
112  if (regid==-1) {
113  regid = config.minPortNumber-1;
114  }
115  }
116  if (regid>=config.maxPortNumber && config.maxPortNumber!=0) {
117  if (nstring == "") {
118  t.setNsNameValue("alloc","*","free");
119  list<Triple> match = db->query(t,&context);
120  if (match.size()>0) {
121  nstring = match.begin()->name;
122  number = atoi(nstring.c_str());
123  }
124  }
125  if (nstring=="") {
126  fprintf(stderr,"Ran out of port numbers\n");
127  fprintf(stderr,"* Make sure ports/programs get closed properly.\n");
128  fprintf(stderr,"* If programs terminate without closing ports, run \"yarp clean\" from time to time..\n");
129  std::exit(1);
130  }
131  } else {
132  regid++;
133  Triple t;
134  char buf[256];
135  sprintf(buf,"%d",regid);
136  t.setNsNameValue("alloc","regid",buf);
137  db->update(t, nullptr);
138  t.setNsNameValue("alloc","regid","*");
139  context.setRid(db->find(t, nullptr));
140  nstring = buf;
141  number = regid;
142  }
143  }
144  t.setNsNameValue("alloc",nstring.c_str(),"in_use");
145  db->update(t,&context);
146  t.setNsNameValue("prefer",nstring.c_str(),c.getName().c_str());
147  db->update(t,&context);
148 
149  Contact contact = c;
150  contact.setPort(number);
151  return contact;
152 }

155 Contact AllocatorOnTriples::completeHost(const yarp::os::Contact& c) {
156  // deal with allocating multicast ips
157 
158  if (c.getCarrier()!="mcast") {
159  return c;
160  }
161  if (c.getHost()!="...") {
162  return c;
163  }
164 
165  string name;
166  Triple t;
167  t.setNsNameValue("alloc","mcastCursor","*");
168  TripleContext context;
169  context.setRid(db->find(t, nullptr));
170  if (context.rid>=0) {
171  t.setNsNameValue("alloc","*","free");
172  list<Triple> match = db->query(t,&context);
173  if (match.size()>0) {
174  name = match.begin()->name;
175  }
176  }
177 
178  if (name=="") {
179  if (mcastCursor==-1) {
180  t.setNsNameValue("alloc","mcastCursor","*");
181  list<Triple> lst = db->query(t, nullptr);
182  if (lst.size()>0) {
183  mcastCursor = atoi(lst.begin()->value.c_str());
184  }
185  if (mcastCursor==-1) {
186  mcastCursor = 1;
187  }
188  }
189  mcastCursor++;
190  char buf[256];
191  sprintf(buf,"%d",mcastCursor);
192  t.setNsNameValue("alloc","mcastCursor",buf);
193  db->update(t, nullptr);
194  t.setNsNameValue("alloc","mcastCursor","*");
195  context.setRid(db->find(t, nullptr));
196 
197  int v1 = mcastCursor%255;
198  int v2 = mcastCursor/255;
199  if (v2>=255) {
200  fprintf(stderr,"Ran out of mcast addresses\n");
201  std::exit(1);
202  }
203  sprintf(buf,"224.1.%d.%d", v2+1,v1+1);
204  name = buf;
205  }
206 
207  t.setNsNameValue("alloc",name.c_str(),"in_use");
208  db->update(t,&context);
209 
210  Contact contact = c;
211  contact.setHost(name);
212  return contact;
213 }

216 bool AllocatorOnTriples::freePortResources(const yarp::os::Contact& c) {
217  string portName = c.getName();
218  int portNumber = c.getPort();
219  string hostName = c.getHost();
220 
221  // free up automatic name for port, if one was allocated
222  Triple t;
223  t.setNsNameValue("alloc","tmpid","*");
224  TripleContext context;
225  context.setRid(db->find(t, nullptr));
226  t.setNsNameValue("alloc",portName.c_str(),"in_use");
227  if (db->find(t,&context)>=0) {
228  t.setNsNameValue("alloc",portName.c_str(),"free");
229  db->update(t,&context);
230  }
231 
232  t.setNsNameValue("alloc","regid","*");
233  context.setRid(db->find(t, nullptr));
234  char buf[256];
235  sprintf(buf,"%d",portNumber);
236  t.setNsNameValue("alloc",buf,"in_use");
237  if (db->find(t,&context)>=0) {
238  t.setNsNameValue("alloc",buf,"free");
239  db->update(t,&context);
240  }
241 
242  t.setNsNameValue("alloc","mcastCursor","*");
243  context.setRid(db->find(t, nullptr));
244  t.setNsNameValue("alloc",hostName.c_str(),"in_use");
245  if (db->find(t,&context)>=0) {
246  t.setNsNameValue("alloc",hostName.c_str(),"free");
247  db->update(t,&context);
248  }
249 
250  return true;
251 }
