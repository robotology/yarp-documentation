44 Sound& Sound::operator += (const Sound& alt)
45 {
46  if (alt.m_channels!= m_channels)
47  {
48  yError("unable to concatenate sounds with different number of channels!");
49  return *this;
50  }
51  if (alt.m_frequency!= m_frequency)
52  {
53  yError("unable to concatenate sounds with different sample rate!");
54  return *this;
55  }
56 
57  Sound orig= *this;
58  this->resize(this->m_samples+alt.m_samples,m_channels);
59 
60  unsigned char* p1 = orig.getRawData();
61  unsigned char* p2 = alt.getRawData();
62  unsigned char* pout = this->getRawData();
63 
64  for (size_t ch=0; ch<m_channels; ch++)
65  {
66  size_t out1 = ch* this->getBytesPerSample() * this->m_samples;
67  size_t out2 = ch* this->getBytesPerSample() * this->m_samples + this->getBytesPerSample() * orig.m_samples;
68 
69  size_t ori1 = ch * orig.getBytesPerSample() * orig.m_samples;
70  size_t s1 = orig.getBytesPerSample() * orig.m_samples;
71 
72  size_t alt1 = ch * orig.getBytesPerSample() * alt.m_samples;
73  unsigned int s2 = alt.getBytesPerSample() * alt.m_samples;
74 
75  memcpy((void *) &pout[out1], (void *) (p1+ori1), s1);
76  memcpy((void *) &pout[out2], (void *) (p2+alt1), s2);
77  }
78 
79  this->synchronize();
80  return *this;
81 }

272 Sound Sound::extractChannelAsSound(size_t channel_id) const
273 {
274  Sound news(this->m_bytesPerSample);
275  news.setFrequency(this->m_frequency);
276  news.resize(this->m_samples, 1);
277 
278  unsigned char* p_src = this->getRawData();
279  unsigned char* p_dst = news.getRawData();
280  size_t j = 0;
281  //pointer to the first element of the row of the matrix we need to copy
282  size_t first_sample = 0 + (this->m_samples * this->m_bytesPerSample)*channel_id;
283  //pointer to the last element of the row of the matrix we need to copy
284  size_t last_sample = first_sample + (this->m_samples * this->m_bytesPerSample);
285  for (auto i = first_sample; i < last_sample; i++)
286  {
287  p_dst[j++] = p_src[i];
288  }
289  return news;
