98 bool BatteryWrapper::read(yarp::os::ConnectionReader& connection)
99 {
100  yarp::os::Bottle in;
101  yarp::os::Bottle out;
102  bool ok = in.read(connection);
103  if (!ok) return false;
104 
105  // parse in, prepare out
106  int code = in.get(0).asVocab();
107  bool ret = false;
108  if (code == VOCAB_IBATTERY)
109  {
110  int cmd = in.get(1).asVocab();
111  if (cmd == VOCAB_BATTERY_INFO)
112  {
113  if (m_ibattery_p)
114  {
115  std::string info;
116  m_ibattery_p->getBatteryInfo(info);
117  out.addVocab(VOCAB_IS);
118  out.addVocab(cmd);
119  out.addString(info);
120  ret = true;
121  }
122  }
123  else
124  {
125  yError("Invalid vocab received in BatteryWrapper");
126  }
127  }
128  else
129  {
130  yError("Invalid vocab received in BatteryWrapper");
131  }
132 
133  if (!ret)
134  {
135  out.clear();
136  out.addVocab(VOCAB_FAILED);
137  }
138 
139  yarp::os::ConnectionWriter *returnToSender = connection.getWriter();
140  if (returnToSender != nullptr)
141  {
142  out.write(*returnToSender);
143  }
144  return true;
145 }

152 bool BatteryWrapper::open(yarp::os::Searchable &config)
153 {
154  Property params;
155  params.fromString(config.toString());
156 
157  if (!config.check("period"))
158  {
159  m_period = 1.0;
160  yWarning() << "BatteryWrapper: missing 'period' parameter. Assuming default value 1.0 s\n";
161  }
162  else
163  {
164  m_period = config.find("period").asFloat32();
165  }
166  yInfo() << "BatteryWrapper using period: " << m_period << "s";
167 
168  if (!config.check("quitPortName"))
169  {
170  m_quitPortName = config.find("quitPortName").asString();
171  }
172 
173  if (!config.check("name"))
174  {
175  yError() << "BatteryWrapper: missing 'name' parameter. Check you configuration file; it must be like:";
176  yError() << "--name: prefix of the ports opened by the device, e.g. /robotName/battery1";
177  yError() << "/data:o and /rpc:i are automatically appended by the wrapper at the end";
178  return false;
179  }
180  else
181  {
182  m_streamingPortName = config.find("name").asString() + "/data:o";
183  m_rpcPortName = config.find("name").asString() + "/rpc:i";
184  }
185 
186  m_enable_shutdown = config.check("enable_shutdown", Value(0), "enable/disable the automatic shutdown").asBool();
187  m_enable_log = config.check("enable_log", Value(0), "enable/disable log to file").asBool();
188 
189  if(!initialize_YARP(config))
190  {
191  yError() << m_sensorId << "Error initializing YARP ports";
192  return false;
193  }
194 
195  if (m_enable_log)
196  {
197  yInfo("writing to log file batteryLog.txt");
198  m_logFile = fopen("batteryLog.txt", "w");
199  }
200 
201  if (config.check("subdevice"))
202  {
203  PolyDriverList driverlist;
204  Property p;
205  p.fromString(config.toString());
206  p.unput("device");
207  p.unput("subdevice");
208  p.put("device", config.find("subdevice").asString());
209  p.setMonitor(config.getMonitor(), "subdevice"); // pass on any monitoring
210 
211  if (!m_driver.open(p) || !m_driver.isValid())
212  {
213  yError() << "BatteryWrapper: failed to open subdevice.. check params";
214  return false;
215  }
216 
217  driverlist.push(&m_driver, "1");
218  if (!attachAll(driverlist))
219  {
220  yError() << "BatteryWrapper: failed to open subdevice.. check params";
221  return false;
222  }
223  m_ownDevices = true;
224  }
225 
226  return true;
227 }

249 void BatteryWrapper::run()
250 {
251  if (m_ibattery_p!=nullptr)
252  {
253  m_log_buffer[0] = 0;
254  double battery_charge = 0;
255  double battery_voltage = 0;
256  double battery_current = 0;
257  double battery_temperature = 0;
258  IBattery::Battery_status status;
259 
260  bool ret = true;
261  ret &= m_ibattery_p->getBatteryCharge(battery_charge);
262  ret &= m_ibattery_p->getBatteryVoltage(battery_voltage);
263  ret &= m_ibattery_p->getBatteryCurrent(battery_current);
264  ret &= m_ibattery_p->getBatteryTemperature(battery_temperature);
265  ret &= m_ibattery_p->getBatteryStatus(status);
266 
267  if (ret)
268  {
269  m_lastStateStamp.update();
270  yarp::os::Bottle& b = m_streamingPort.prepare();
271  b.clear();
272  b.addFloat64(battery_voltage); //0
273  b.addFloat64(battery_current); //1
274  b.addFloat64(battery_charge); //2
275  b.addFloat64(battery_temperature); //3
276  b.addInt32(status); //4
277  m_streamingPort.setEnvelope(m_lastStateStamp);
278  m_streamingPort.write();
279 
280  // if the battery is not charging, checks its status of charge
281  if (battery_current>0.4) check_battery_status(battery_charge);
282 
283  // save data to file
284  if (m_enable_log)
285  {
286  time_t rawtime;
287  struct tm * timeinfo;
288  time(&rawtime);
289  timeinfo = localtime(&rawtime);
290  char* battery_timestamp = asctime(timeinfo);
291  snprintf(m_log_buffer, 1024, "battery status: %+6.1fA % 6.1fV charge:% 6.1f%% time: %s", battery_current, battery_voltage, battery_charge, battery_timestamp);
292  fprintf(m_logFile, "%s", m_log_buffer);
293  }
294  }
295  else
296  {
297  yError("BatteryWrapper: %s: Sensor returned error", m_sensorId.c_str());
298  }
299  }
300 }

369 void BatteryWrapper::check_battery_status(double battery_charge)
370 {
371  static bool notify_15 = true;
372  static bool notify_12 = true;
373  static bool notify_10 = true;
374  static bool notify_0 = true;
375 
376  if (battery_charge > 20)
377  {
378  notify_15 = true;
379  notify_12 = true;
380  notify_10 = true;
381  notify_0 = true;
382  }
383 
384  if (battery_charge < 5)
385  {
386  if (notify_0)
387  {
388  if (m_enable_shutdown)
389  {
390  emergency_shutdown("CRITICAL WARNING: battery charge below critical level 5%. The robot will be stopped and the system will shutdown in 2mins.");
391  if (m_quitPortName != "") { stop_robot(m_quitPortName); }
392  notify_0 = false;
393  }
394  else
395  {
396  notify_message("CRITICAL WARNING: battery charge reached critical level 5%, but the emergency shutodown is currently disabled!");
397  notify_0 = false;
398  }
399  }
400  }
401  else if (battery_charge < 10)
402  {
403  if (notify_10) { notify_message("WARNING: battery charge below 10%"); notify_10 = false; }
404  }
405  else if (battery_charge < 12)
406  {
407  if (notify_12) { notify_message("WARNING: battery charge below 12%"); notify_12 = false; }
408  }
409  else if (battery_charge < 15)
410  {
411  if (notify_15) { notify_message("WARNING: battery charge below 15%"); notify_15 = false; }
412  }
413 }
