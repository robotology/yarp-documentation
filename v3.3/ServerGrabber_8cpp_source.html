468 bool ServerGrabber::respond(const yarp::os::Bottle& cmd,
469  yarp::os::Bottle& response, bool left, bool both=false) {
470  int code = cmd.get(0).asVocab();
471  Bottle response2;
472  switch (code)
473  {
474  case VOCAB_FRAMEGRABBER_IMAGE:
475  {
476  switch (cmd.get(1).asVocab())
477  {
478  case VOCAB_GET:
479  {
480  switch (cmd.get(2).asVocab())
481  {
482  case VOCAB_CROP:
483  {
484  response.clear();
485  // If the device driver support it, use the device implementation, because it may be more efficient.
486  // If not, acquire the whole image and crop it here before sending it.
487 
488  Bottle *list = cmd.get(4).asList();
489  int nPoints = list->size() /2; // divided by 2 because each pixel is identified by 2 numbers (u,v)
490 
491  yarp::sig::VectorOf<std::pair<int, int> > vertices;
492  vertices.resize(nPoints);
493 
494  for(int i=0; i<nPoints; i++)
495  {
496  vertices[i].first = list->get(i*2).asInt32();
497  vertices[i].second = list->get(i*2 +1).asInt32();
498  }
499 
500  ImageOf< PixelRgb > cropped;
501 
502  // Choose the interface and eventual offset depending on case.
503 
504  /* HW/SW configurations here: (1a, 1b, 2a, 2b), for each one the user can request a crop on left or right image
505  * 1) single HW camera as a source
506  * 1a) split false: a single image to handle
507  * 1b) split true : 2 images, I have to handle left or right image. If user request a crop in the right side,
508  * of the image, then add an offset
509  *
510  * 2) two HW sources
511  * 2a) split true: choose appropriate image source based on left/right request
512  * 2b) split false: choose appropriate image source based on crop position. Crop request have to belong to a
513  * single frame, either left or right. Example: 2 cameras with 320x240 pixels each placed
514  * one after the other, generates a single stitched image of 640x240.
515  * Anyway a crop request like (200,100)(400,200) shall be rejected, even if it could be
516  * considered as a part of the image resulting from the stitch.
517  * Right now the decision is took based on the first point of vector 'vertices', since all
518  * points are expected to belong to the same frame (left/right)
519  *
520  */
521 
522  // Default values here are valid for cases 1a and `left` side of 2a
523  IFrameGrabberImage *imageInterface = fgImage;
524  int u_offset = 0;
525 
526  if(param.twoCameras == false) // a single HW source of images
527  {
528  imageInterface = fgImage;
529  if(left == false) // if left is false, implicitly split is true
530  u_offset = imageInterface->width()/2; // 1b
531 
532  }
533  else
534  {
535  if(param.split) // 2a, right image
536  {
537  if(left == false)
538  {
539  imageInterface = fgImage2;
540  u_offset = 0;
541  }
542  }
543  else
544  {
545  if(vertices[0].first >= fgImage->width()) // 2b, right image
546  {
547  imageInterface = fgImage2;
548  u_offset = -fgImage->width();
549  }
550  }
551 
552  }
553 
554 
555  if(imageInterface != nullptr)
556  {
557  if(imageInterface->getImageCrop((cropType_id_t) cmd.get(3).asVocab(), vertices, cropped) )
558  {
559  // use the device output
560  }
561  else
562  {
563  // In case the device has not yet implemented this feature, do it here (less efficient)
564  if(cmd.get(3).asVocab() == YARP_CROP_RECT)
565  {
566  if(nPoints != 2)
567  {
568  response.addString("GetImageCrop failed: RECT mode requires 2 vertices.");
569  yError() << "GetImageCrop failed: RECT mode requires 2 vertices, got " << nPoints;
570  return false;
571  }
572  ImageOf< PixelRgb > full;
573  imageInterface->getImage(full);
574 
575  cropped.resize(vertices[1].first - vertices[0].first +1, vertices[1].second - vertices[0].second +1); // +1 to be inclusive
576  cropped.zero();
577  for(int u_in=vertices[0].first + u_offset, u_out=0; u_in<=vertices[1].first + u_offset; u_in++, u_out++)
578  {
579  for(int v_in=vertices[0].second, v_out=0; v_in <= vertices[1].second; v_in++, v_out++)
580  {
581  cropped.pixel(u_out, v_out).r = full.pixel(u_in, v_in).r;
582  cropped.pixel(u_out, v_out).g = full.pixel(u_in, v_in).g;
583  cropped.pixel(u_out, v_out).b = full.pixel(u_in, v_in).b;
584  }
585  }
586  }
587  else if(cmd.get(3).asVocab() == YARP_CROP_LIST)
588  {
589  response.addString("List type not yet implemented");
590  }
591  else
592  {
593  response.addString("Crop type unknown");
594  }
595  }
596  }
597 
598  response.addVocab(VOCAB_CROP);
599  response.addVocab(VOCAB_IS);
600  response.addInt32(cropped.width()); // Actual width of image in pixels, to check everything is ok
601  response.addInt32(cropped.height()); // Actual height of image in pixels, to check everything is ok
602 
603  response.add(Value(cropped.getRawImage(), cropped.getRawImageSize()));
604  return true;
605  } break;
606  } break;
607 
608  } break;
609 
610  case VOCAB_SET: // Nothing to do here yet
611  default:
612  {
613  yError() << "FrameGrabberImage interface received an unknown command " << cmd.toString();
614  } break;
615 
616  }
617 
618  } break;
619 
620  // first check if requests are coming from new iFrameGrabberControl2 interface and process them
621  case VOCAB_FRAMEGRABBER_CONTROL:
622  {
623  if(param.twoCameras)
624  {
625  bool ret;
626  if(both){
627  ret=ifgCtrl_Parser.respond(cmd, response);
628  ret&=ifgCtrl2_Parser.respond(cmd, response2);
629  if(!ret || (response!=response2))
630  {
631  response.clear();
632  response.addVocab(VOCAB_FAILED);
633  ret=false;
634  yWarning()<<"ServerGrabber: response different among cameras or failed";
635  }
636  }
637  else
638  {
639  if(left)
640  {
641  ret=ifgCtrl_Parser.respond(cmd, response);
642  }
643  else
644  {
645  ret=ifgCtrl2_Parser.respond(cmd, response);
646  }
