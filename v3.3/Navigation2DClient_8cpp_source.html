533 bool Navigation2DClient::locations_are_similar(Map2DLocation loc1, Map2DLocation loc2, double linear_tolerance, double angular_tolerance)
534 {
535  if (linear_tolerance < 0) return false;
536  if (angular_tolerance < 0) return false;
537  yAssert(linear_tolerance >= 0);
538  yAssert(angular_tolerance >= 0);
539 
540  if (loc1.map_id != loc2.map_id)
541  {
542  return false;
543  }
544  if (sqrt(pow((loc1.x - loc2.x),2) + pow((loc1.y - loc2.y),2)) > linear_tolerance)
545  {
546  return false;
547  }
548 
549  if (angular_tolerance != std::numeric_limits<double>::infinity())
550  {
551  //In the following blocks, I'm giving two possibile solution to the problem of
552  //determining if the difference of two angles is below a certain threshold.
553  //The problem is tricky, because it must take in account the critical points 0,180,360,-180, -360 etc.
554  //Both the formulas lead to the same result, however I'm not sure I they have the same performances.
555  //Please do not remove the unused block, since it may be a useful reference for the future.
556 #if 1
557  //check in the range 0-360
558  double diff = loc2.theta - loc1.theta + 180.0;
559  diff = fmod(diff, 360.0) - 180.0;
560  diff = (diff < -180.0) ? (diff + 360.0) : (diff);
561  if (fabs(diff) > angular_tolerance)
562 #else
563  //check in the range 0-180
564  double angle1 = normalize_angle(loc1.theta);
565  double angle2 = normalize_angle(loc2.theta);
566  double diff = angle1 - angle2;
567  diff += (diff > 180) ? -360 : (diff < -180) ? 360 : 0;
568  if (fabs(diff) > angular_tolerance)
569 #endif
570  {
571  return false;
572  }
573  }
574  return true;
