87 bool NetworkClock::Private::read(ConnectionReader& reader)
88 {
89  Bottle bot;
90  bool ok = bot.read(reader);
91 
92  if (closing) {
93  _time = -1;
94  return false;
95  }
96 
97  if (!ok && !closing) {
98  YARP_ERROR(Logger::get(), "Error reading clock port");
99  return false;
100  }
101 
102  timeMutex.lock();
103  sec = bot.get(0).asInt32();
104  nsec = bot.get(1).asInt32();
105  _time = sec + (nsec * 1e-9);
106  initted = true;
107  timeMutex.unlock();
108 
109  listMutex.lock();
110  auto waiter_it = waiters->begin();
111  while (waiter_it != waiters->end()) {
112  if (waiter_it->first - _time < 1E-12) {
113  Semaphore* waiterSemaphore = waiter_it->second;
114  waiter_it = waiters->erase(waiter_it);
115  if (waiterSemaphore != nullptr) {
116  waiterSemaphore->post();
117  }
118  } else {
119  ++waiter_it;
120  }
121  }
122  listMutex.unlock();
123  return true;
124 }

189 void NetworkClock::delay(double seconds)
190 {
191  if (seconds <= 1E-12) {
192  return;
193  }
194 
195  mPriv->listMutex.lock();
196  if (mPriv->closing) {
197  // We are shutting down. The time signal is no longer available.
198  // Make a short delay and return.
199  mPriv->listMutex.unlock();
200  SystemClock::delaySystem(seconds);
201  return;
202  }
203 
204  std::pair<double, Semaphore*> waiter(now() + seconds, new Semaphore(0));
205  mPriv->waiters->push_back(waiter);
206  mPriv->listMutex.unlock();
207 
208  waiter.second->wait();
209  delete waiter.second;
210  waiter.second = nullptr;
211 }
