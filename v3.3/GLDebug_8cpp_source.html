24 void yarp::dev::checkGlError(const char* file, int line, const char* func)
25 {
26  GLenum error = glGetError();
27  if (error != GL_NO_ERROR) {
28  yarp::os::LogStream ls = yarp::os::Log(file, line, func).error();
29  switch (error) {
30  case GL_INVALID_ENUM:
31  ls << "OpenGL Error GL_INVALID_ENUM: GLenum argument out of range";
32  break;
33  case GL_INVALID_VALUE:
34  ls << "OpenGL Error GL_INVALID_VALUE: Numeric argument out of range";
35  break;
36  case GL_INVALID_OPERATION:
37  ls << "OpenGL Error GL_INVALID_OPERATION: Operation illegal in current state";
38  break;
39  case GL_STACK_OVERFLOW:
40  ls << "OpenGL Error GL_STACK_OVERFLOW: Command would cause a stack overflow";
41  break;
42  case GL_OUT_OF_MEMORY:
43  ls << "OpenGL Error GL_OUT_OF_MEMORY: Not enough memory left to execute command";
44  break;
45  default:
46  ls << "OpenGL Error " << error;
47  break;
48  }
49  }
50  if (!(error == 0)) { yarp::os::Log(file, line, func).fatal("OpenGL Error at %s:%d (%s)", file, line, func); }
51 }
