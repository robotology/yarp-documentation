155 
156  int l=0;
157  const double *tmp=data();
158  for(l=0;l<header.listLen;l++)
159  connection.appendFloat64(tmp[l]);

161  // if someone is foolish enough to connect in text mode,
162  // let them see something readable.
163  connection.convertTextMode();
164 
165  return true;

169 std::string Matrix::toString(int precision, int width, const char* endRowStr) const {
170 
171  // If the width is less than 1, use tabs, else use width number of spaces.
172  std::string spacer((width<0) ? "\t" : " ");
173 
174  // Buffering.
175  std::string ret = "";
176  char buffer[350];
177  const double* src = (*this).data();
178 
179  // Iterate through copying the contents from the matrix, into a string.
180  // Avoid unnecessary string resizes by only adding spacers at the beginning.
181  size_t r, c;
182  for (r = 0; r < nrows; r++) {
183  if (r) { ret += endRowStr; }
184  for (c = 0; c < ncols; c++) {
185  if (c) { ret += spacer; }
186  sprintf(buffer, "% *.*lf", width, precision, *src); src++;
187  ret += buffer;
188  }
189  }
190 
191  return ret;
192 }
