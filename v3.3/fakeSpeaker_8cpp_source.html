92 void fakeSpeaker::run()

100  //playing implies emptying all the buffer

103  size_t siz_byt = m_outputBuffer->size().getBytes();
104  size_t buffer_size = siz_sam * siz_chn;
105  for (size_t i = 0; i<buffer_size; i++)

107  audio_sample_16t s = m_outputBuffer->read();
