49 bool RpLidar2::open(yarp::os::Searchable& config)
50 {
51  string serial;
52  int baudrate;
53  u_result result;
54 
55  m_device_status = DEVICE_OK_STANBY;
56  m_min_distance = 0.1; //m
57  m_max_distance = 2.5; //m
58  m_pwm_val = 0;
59  bool br = config.check("GENERAL");
60  if (br != false)
61  {
62  yarp::os::Searchable& general_config = config.findGroup("GENERAL");
63  m_clip_max_enable = general_config.check("clip_max");
64  m_clip_min_enable = general_config.check("clip_min");
65  if (m_clip_max_enable) { m_max_distance = general_config.find("clip_max").asFloat64(); }
66  if (m_clip_min_enable) { m_min_distance = general_config.find("clip_min").asFloat64(); }
67  if (general_config.check("max_angle") == false) { yError() << "Missing max_angle param in GENERAL group"; return false; }
68  if (general_config.check("min_angle") == false) { yError() << "Missing min_angle param in GENERAL group"; return false; }
69  if (general_config.check("resolution") == false) { yError() << "Missing resolution param in GENERAL group"; return false; }
70  if (general_config.check("serial_port") == false) { yError() << "Missing serial_port param in GENERAL group"; return false; }
71  if (general_config.check("serial_baudrate") == false) { yError() << "Missing serial_baudrate param in GENERAL group"; return false; }
72  if (general_config.check("sample_buffer_life") == false) { yError() << "Missing sample_buffer_life param in GENERAL group"; return false; }
73 
74  baudrate = general_config.find("serial_baudrate").asInt32();
75  m_serialPort = general_config.find("serial_port").asString();
76  m_max_angle = general_config.find("max_angle").asFloat64();
77  m_min_angle = general_config.find("min_angle").asFloat64();
78  m_resolution = general_config.find("resolution").asFloat64();
79  m_buffer_life = general_config.find("sample_buffer_life").asInt32();
80  m_do_not_clip_infinity_enable = (general_config.find("allow_infinity").asInt32()!=0);
81  if (general_config.check("motor_pwm"))
82  {
83  m_pwm_val = general_config.find("motor_pwm").asInt32();
84  }
85  if (general_config.check("thread_period"))
86  {
87  double thread_period = general_config.find("thread_period").asInt32() / 1000.0;
88  this->setPeriod(thread_period);
89  }
90  }
91  else
92  {
93  yError() << "Missing GENERAL section";
94  return false;
95  }
96 
97  bool bs = config.check("SKIP");
98  if (bs != false)
99  {
100  yarp::os::Searchable& skip_config = config.findGroup("SKIP");
101 
102  Bottle mins = skip_config.findGroup("min");
103  Bottle maxs = skip_config.findGroup("max");
104  size_t s_mins = mins.size();
105  size_t s_maxs = mins.size();
106 
107  if (s_mins == s_maxs && s_maxs > 1 )
108  {
109  for (size_t s = 1; s < s_maxs; s++)
110  {
111  Range_t range;
112  range.max = maxs.get(s).asFloat64();
113  range.min = mins.get(s).asFloat64();
114  if (range.max >= 0 && range.max <= 360 &&
115  range.min >= 0 && range.min <= 360 &&
116  range.max > range.min)
117  {
118  m_range_skip_vector.push_back(range);
119  }
120  else
121  {
122  yError() << "Invalid range in SKIP section";
123  return false;
124  }
125  }
126  }
127 
128  }
129 
130  if (m_max_angle <= m_min_angle) { yError() << "max_angle should be > min_angle"; return false; }
131  double fov = (m_max_angle - m_min_angle);
132  if (fov >360) { yError() << "max_angle - min_angle <= 360"; return false; }
133  m_sensorsNum = (int)(fov/m_resolution);
134  m_laser_data.resize(m_sensorsNum, 0.0);
135 
136  m_drv = RPlidarDriver::CreateDriver(rp::standalone::rplidar::DRIVER_TYPE_SERIALPORT);
137  if (!m_drv)
138  {
139  yError() << "Create Driver fail, exit\n";
140  return false;
141  }
142 
143  if (IS_FAIL(m_drv->connect(m_serialPort.c_str(), (_u32)baudrate)))
144  {
145  yError() << "Error, cannot bind to the specified serial port:", m_serialPort.c_str();
146  RPlidarDriver::DisposeDriver(m_drv);
147  return false;
148  }
149 
150  m_info = deviceinfo();
151  yInfo("max_dist %f, min_dist %f", m_max_distance, m_min_distance);
152 
153  bool m_inExpressMode=false;
154  result = m_drv->checkExpressScanSupported(m_inExpressMode);
155  if (result == RESULT_OK && m_inExpressMode==true)
156  {
157  yInfo() << "Express scan mode is supported";
158  }
159  else
160  {
161  yWarning() << "Device does not supports express scan mode";
162  }
163 
164  result = m_drv->startMotor();
165  if (result != RESULT_OK)
166  {
167  handleError(result);
168  return false;
169  }
170  yInfo() << "Motor started successfully";
171 
172  if (m_pwm_val!=0)
173  {
174  if (m_pwm_val>0 && m_pwm_val<1023)
175  {
176  result = m_drv->setMotorPWM(m_pwm_val);
177  if (result != RESULT_OK)
178  {
179  handleError(result);
180  return false;
181  }
182  yInfo() << "Motor pwm set to "<< m_pwm_val;
183  }
184  else
185  {
186  yError() << "Invalid motor pwm request " << m_pwm_val <<". It should be a value between 0 and 1023.";
187  return false;
188  }
189  }
190  else
191  {
192  yInfo() << "Motor pwm set to default value (600?)";
193  }
194 
195  bool forceScan =false;
196  result = m_drv->startScan(forceScan,m_inExpressMode);
197 
198  if (result != RESULT_OK)
199  {
200  handleError(result);
201  return false;
202  }
203  yInfo() << "Scan started successfully";
204 
205  yInfo() << "Device info:" << m_info;
206  yInfo("max_angle %f, min_angle %f", m_max_angle, m_min_angle);
207  yInfo("resolution %f", m_resolution);
208  yInfo("sensors %d", m_sensorsNum);
209  PeriodicThread::start();
210  return true;
211 }

291 bool RpLidar2::getLaserMeasurement(std::vector<LaserMeasurementData> &data)
292 {
293  std::lock_guard<std::mutex> guard(m_mutex);
294  size_t size = m_laser_data.size();
295  data.resize(size);
296 
297  if (m_max_angle < m_min_angle) { yError() << "getLaserMeasurement failed"; return false; }
298 
299  double laser_angle_of_view = m_max_angle - m_min_angle;
300 
301  for (size_t i = 0; i < size; i++)
302  {
303  double angle = (i / double(size)*laser_angle_of_view + m_min_angle)* DEG2RAD;
304  data[i].set_polar(m_laser_data[i], angle);
305  }
306  m_device_status = yarp::dev::IRangefinder2D::DEVICE_OK_IN_USE;
307 
308  return true;

328 void RpLidar2::run()
329 {
330  u_result op_result;
331  rplidar_response_measurement_node_t nodes[2048];
332  size_t count = _countof(nodes);
333  static int life = 0;
334  op_result = m_drv->grabScanData(nodes, count);
335  if (op_result != RESULT_OK)
336  {
337  yError() << m_serialPort << ": grabbing scan data failed";
338  handleError(op_result);
339  return;
340  }
341 
342  float frequency=0;
343  bool is4kmode=false;
344  op_result = m_drv->getFrequency(m_inExpressMode, count, frequency, is4kmode);
345  if (op_result != RESULT_OK)
346  {
347  yError() << "getFrequency failed";
348  }
349 
350  m_drv->ascendScanData(nodes, count);
351 
352  if (op_result != RESULT_OK)
353  {
354  yError() << "ascending scan data failed\n";
355  handleError(op_result);
356  return;
357  }
358 
359  if (m_buffer_life && life%m_buffer_life == 0)
360  {
361  for (size_t i=0 ;i<m_laser_data.size(); i++)
362  {
363  //m_laser_data[i]=0; //0 is a terribly unsafe value and should be avoided.
364  m_laser_data[i]=std::numeric_limits<double>::infinity();
365  }
366  }
367 
368  for (size_t i = 0; i < count; ++i)
369  {
370 
371  double distance = nodes[i].distance_q2 / 4.0f / 1000.0; //m
372  double angle = (float)((nodes[i].angle_q6_checkbit >> RPLIDAR_RESP_MEASUREMENT_ANGLE_SHIFT)/64.0f); //deg
373  double quality = nodes[i].sync_quality >> RPLIDAR_RESP_MEASUREMENT_QUALITY_SHIFT;
374  angle = (360 - angle);
375 
376  if (angle >= 360)
377  {
378  angle -= 360;
379  }
380 
381  if (quality == 0 || distance == 0)
382  {
383  distance = std::numeric_limits<double>::infinity();
384  }
385 
386  if (angle > 360)
387  {
388  yWarning() << "Invalid angle";
389  }
390 
391  if (m_clip_min_enable)
392  {
393  if (distance < m_min_distance)
394  {
395  //laser angular measurements not read by the device are now set to infinity and not to zero
396  distance = std::numeric_limits<double>::infinity();
397  }
398  }
399  if (m_clip_max_enable)
400  {
401  if (distance > m_max_distance)
402  {
403  if (!m_do_not_clip_infinity_enable && distance <= std::numeric_limits<double>::infinity())
404  {
405  distance = m_max_distance;
406  }
407  }
408  }
409 
410  for (auto& i : m_range_skip_vector)
411  {
412  if (angle > i.min && angle < i.max)
413  {
414  distance = std::numeric_limits<double>::infinity();
415  }
416  }
417 
418  int elem = (int)(angle / m_resolution);
419 
420  if (elem >= 0 && elem < (int)m_laser_data.size())
421  {
422  m_laser_data[elem] = distance;
423  }
424  else
425  {
426  yDebug() << "RpLidar::run() invalid angle: elem" << elem << ">" << "laser_data.size()" << m_laser_data.size();
427  }
428  }
429 
430  life++;
431  return;
432 }
