40 bool yarp::robottestingframework::TestCase::setup(int argc, char** argv)
41 {
42  // check yarp network
43  mPriv->yarp.setVerbosity(-1);
44  ROBOTTESTINGFRAMEWORK_ASSERT_ERROR_IF_FALSE(mPriv->yarp.checkNetwork(),
45  "YARP network does not seem to be available, is the yarp server accessible?");
46 
47  // loading environment properties by parsing the string value
48  // from getEnvironment().
49  std::string strEnv = getEnvironment();
50  yarp::os::Property envprop;
51  envprop.fromArguments(strEnv.c_str());
52  bool useSuiteContext = envprop.check("context");
53 
54  // load the config file and update the environment if available
55  // E.g., "--from mytest.ini"
56  yarp::os::ResourceFinder rf;
57  rf.setVerbose(false);
58  if(useSuiteContext) {
59  rf.setDefaultContext(envprop.find("context").asString().c_str());
60  } else {
61  rf.setDefaultContext("RobotTesting");
62  }
63  rf.configure(argc, argv);
64  yarp::os::Property property;
65 
66  if(rf.check("from")) {
67 
68  std::string cfgname = rf.find("from").asString();
69  ROBOTTESTINGFRAMEWORK_ASSERT_ERROR_IF_FALSE(cfgname.size(),
70  "Empty value was set for the '--from' property");
71 
72  // loading configuration file indicated by --from
73  std::string cfgfile = rf.findFileByName(cfgname.c_str());
74 
75  bool useTestContext = rf.check("context");
76 
77  // if the config file cannot be found from default context or
78  // there is not any context, use the robotname environment as context
79  if(!useSuiteContext && !useTestContext && !cfgfile.size() && envprop.check("robotname")) {
80  rf.setDefaultContext(envprop.find("robotname").asString().c_str());
81  cfgfile = rf.findFileByName(cfgname.c_str());
82  }
83  ROBOTTESTINGFRAMEWORK_ASSERT_ERROR_IF_FALSE(cfgfile.size(),
84  ::robottestingframework::Asserter::format("Cannot find configuration file %s", cfgfile.c_str()));
85  ROBOTTESTINGFRAMEWORK_TEST_REPORT(::robottestingframework::Asserter::format("Loading configuration from %s", cfgfile.c_str()));
86  // update the properties with environment
87  property.fromConfigFile(cfgfile.c_str(), envprop);
88  } else {
89  property.fromString(rf.toString().c_str());
90  }
91 
92  return setup(property);
93 }
