16 static bool checkParam(const Bottle& input, RGBDSensorParamParser::RGBDParam& param, bool& found)
17 {
18  bool ret = false;
19  Bottle bt=input.findGroup(param.name).tail(); // the first element is the name of the parameter
20 
21  if (!bt.isNull())
22  {
23  Bottle* b;
24  if (param.size>1 && bt.size()==1)
25  {
26  b = bt.get(0).asList();
27  }
28  else
29  b = &bt;
30  if (b->isNull())
31  {
32  yError()<<"RGBDSensorParamParser: check"<<param.name<<"in config file";
33  return false;
34  }
35  if (b->size() != param.size)
36  {
37  yError() << "RGBDSensorParamParser: parameter" << param.name << "size should be" << param.size;
38  return false;
39  }
40  param.val.resize(param.size);
41  for (size_t i=0;i<b->size();i++)
42  {
43  ret = true;
44  param.val[i] = b->get(i);
45  found = true;
46  }
47  }
48  else
49  {
50  ret = true;
51  found = false;
52  }
53  return ret;
54 }

56 static bool checkParam(const Bottle& settings, const Bottle& description, RGBDSensorParamParser::RGBDParam &param)
57 {
58  bool ret1, ret2, ret3;
59 
60  ret3 = true;
61  ret1 = checkParam(settings, param, param.isSetting); // look for settings
62  ret2 = checkParam(description, param, param.isDescription);// look for HW_DESCRIPTION
63 
64  if ( (param.isSetting) && (param.isDescription) )
65  {
66  yError() << "Setting " << param.name << " can either be a 'SETTING' or 'HW_DESCRIPTION', not both. Fix the config file. \
67  Look for documentation online.";
68  ret3 = false;
69  }
70  return (ret1 && ret2 && ret3);
71 }

74 static bool parseIntrinsic(const Searchable& config, const string& groupName, yarp::sig::IntrinsicParams &params)
75 {
76 
77  pair<string, double*> realparam;
78  vector<pair<string, double*> > realParams;
79  size_t i;
80 
81  if (!config.check(groupName))
82  {
83  if (params.isOptional)
84  {
85  return true;
86  }
87  else
88  {
89  return false;
90  }
91  }
92  Bottle& intrinsic = config.findGroup(groupName);
93 
94  realparam.first = "physFocalLength"; realparam.second = &params.physFocalLength; realParams.push_back(realparam);
95  realparam.first = "focalLengthX"; realparam.second = &params.focalLengthX; realParams.push_back(realparam);
96  realparam.first = "focalLengthY"; realparam.second = &params.focalLengthY; realParams.push_back(realparam);
97  realparam.first = "principalPointX"; realparam.second = &params.principalPointX; realParams.push_back(realparam);
98  realparam.first = "principalPointY"; realparam.second = &params.principalPointY; realParams.push_back(realparam);
99 
100  for(i = 0; i < realParams.size(); i++)
101  {
102  if (!intrinsic.check(realParams[i].first))
103  {
104  yError() << "RGBDSensorParamParser: missing" << realParams[i].first << "param in" << groupName << "group in the configuration file";
105  return false;
106  }
107 
108  *(realParams[i].second) = intrinsic.find(realParams[i].first).asFloat64();
109  }
110 
111  if (!intrinsic.check("distortionModel"))
112  {
113  yError() << "missing distortionModel param in configuration";
114  return false;
115  }
116 
117  if (!config.check(intrinsic.find("distortionModel").asString()))
118  {
119  yError() << "missing" << intrinsic.find("distortionModel").asString() << "group in configuration file";
120  return false;
121  }
122 
123  Bottle& distortion = config.findGroup(intrinsic.find("distortionModel").asString());
124 
125  if (!distortion.check("name"))
126  {
127  yError() << "RGBDSensorParamParser: missing name param in" << config.find("distortionModel").asString() << "group in configuration file";
128  return false;
129  }
130  if (distortion.find("name").asString() != "plumb_bob")
131  {
132  yError() << "RGBDSensorParamParser: only plumb_bob distortion model is supported at the moment";
133  return false;
134  }
135 
136  realParams.clear();
137  realparam.first = "k1"; realparam.second = &params.distortionModel.k1; realParams.push_back(realparam);
138  realparam.first = "k2"; realparam.second = &params.distortionModel.k2; realParams.push_back(realparam);
139  realparam.first = "t1"; realparam.second = &params.distortionModel.t1; realParams.push_back(realparam);
140  realparam.first = "t2"; realparam.second = &params.distortionModel.t2; realParams.push_back(realparam);
141  realparam.first = "k3"; realparam.second = &params.distortionModel.k3; realParams.push_back(realparam);
142 
143  for(i = 0; i < realParams.size(); i++)
144  {
145  if (!distortion.check(realParams[i].first))
146  {
147  yError() << "RGBDSensorParamParser: missing" << realParams[i].first << "param in" << intrinsic.find("distortionModel").asString() << "group in the configuration file";
148  return false;
149  }
150  *(realParams[i].second) = distortion.find(realParams[i].first).asFloat64();
151  }
152 
153  return true;
154 }

156 bool RGBDSensorParamParser::parseParam(const Searchable &config, std::vector<RGBDParam*>& params)
157 {
158  bool ret = true;
159 
160  if (!config.check("SETTINGS"))
161  {
162  yError() << "RGBDSensorParamParser: missing SETTINGS section on the configuration file";
163  return false;
164  }
165 
166  Bottle& settings = config.findGroup("SETTINGS");
167 
168  if (!config.check("HW_DESCRIPTION"))
169  {
170  yError() << "RGBDSensorParamParser: missing HW_DESCRIPTION section on the configuration file";
171  return false;
172  }
173 
174  Bottle& description = config.findGroup("HW_DESCRIPTION");
175 
176 
177  for (auto& v: params)
178  {
179  if (!checkParam(settings, description, *v) ) {ret = false;}
180  }
181 
182  if (!ret)
183  {
184  yError() << "RGBDSensorParamParser: driver input file not correct, please fix it!";
185  return false;
186  }
187 
188  if (!parseIntrinsic(config, "RGB_INTRINSIC_PARAMETERS", rgbIntrinsic))
189  {
190  yError() << "RGBDSensorParamParser: incomplete or missing RGB_INTRINSIC_PARAMETERS section on the configuration file";
191  return false;
192  }
193 
194 
195  if (!parseIntrinsic(config, "DEPTH_INTRINSIC_PARAMETERS", depthIntrinsic))
196  {
197  yError() << "RGBDSensorParamParser: incomplete or missing DEPTH_INTRINSIC_PARAMETERS section on the configuration file";
198  return false;
199  }
200 
201  if (!config.check("EXTRINSIC_PARAMETERS"))
202  {
203  if (!isOptionalExtrinsic)
204  {
205  yError() << "RGBDSensorParamParser: missing EXTRINSIC_PARAMETERS section on the configuration file";
206  return false;
207  }
208  else
209  {
210  return true;
211  }
212  }
213  else
214  {
215 
216  Bottle& extrinsic = config.findGroup("EXTRINSIC_PARAMETERS");
217 
218  if (!extrinsic.check("transformation"))
219  {
220  yError() << "RGBDSensorParamParser: missing transformation parameter under EXTRINSIC_PARAMETERS group in configuration file";
221  return false;
222  }
223 
224  Bottle transformation = extrinsic.findGroup("transformation").tail();
225  Bottle* tf;
226 
227  if (transformation.size()==1)
228  {
229  tf = transformation.get(0).asList();
230  }
231  else
232  {
233  tf=&transformation;
234  }
235  if (!(tf->size() == 4*4))
236  {
237  yError() << "RGBDSensorParamParser: the size of the transformation matrix is wrong";
238  return false;
239  }
240 
241  for(int i = 0; i < 4; i++)
242  {
243  for(int j = 0; j < 4; j++)
244  {
245  int k = i*4+j;
246  Value& v = tf->get(k);
247  if (!v.isFloat64())
248  {
249  yError() << "wrong data format on transformation matrix (position" << k << ")";
250  return false;
251  }
252  transformationMatrix[i][j] = v.asFloat64();
253  }
254  }
255  }
256 
257  return ret;
258 }
