52 bool VectorBase::read(yarp::os::ConnectionReader& connection) {
53  // auto-convert text mode interaction
54  connection.convertTextMode();
55  VectorPortContentHeader header;
56  bool ok = connection.expectBlock((char*)&header, sizeof(header));
57  if (!ok) return false;
58  if (header.listLen > 0 &&
59  header.listTag == (BOTTLE_TAG_LIST | getBottleTag()))
60  {
61  if ((size_t)getListSize() != (size_t)(header.listLen))
62  resize(header.listLen);
63  char* ptr = getMemoryBlock();
64  yAssert(ptr != nullptr);
65  int elemSize=getElementSize();
66  ok = connection.expectBlock(ptr, elemSize*header.listLen);
67  if (!ok) return false;
68  } else {
69  return false;
70  }
71 
72  return !connection.isError();
73 }

84  int elemSize=getElementSize();
