14 PointCloud<DataXYZ> utils::depthToPC(const yarp::sig::ImageOf<PixelFloat> &depth,
15  const yarp::sig::IntrinsicParams &intrinsic)
16 {
17  yAssert(depth.width() != 0);
18  yAssert(depth.height() != 0);
19  size_t w = depth.width();
20  size_t h = depth.height();
21  PointCloud<DataXYZ> pointCloud;
22  pointCloud.resize(w, h);
23 
24  for (size_t u = 0; u < w; ++u) {
25  for (size_t v = 0; v < h; ++v) {
26  // De-projection equation (pinhole model):
27  // x = (u - ppx)/ fx * z
28  // y = (v - ppy)/ fy * z
29  // z = z
30  pointCloud(u,v).x = (u - intrinsic.principalPointX)/intrinsic.focalLengthX*depth.pixel(u,v);
31  pointCloud(u,v).y = (v - intrinsic.principalPointY)/intrinsic.focalLengthY*depth.pixel(u,v);
32  pointCloud(u,v).z = depth.pixel(u,v);
33  }
34  }
35  return pointCloud;
36 }
