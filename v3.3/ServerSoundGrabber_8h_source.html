195 #ifdef PRINT_DEBUG_MESSAGES

203 #endif

206  yDebug();
207 #endif

216  bool read(yarp::os::ConnectionReader& connection) override
217  {
218  yarp::os::Bottle command;
219  yarp::os::Bottle reply;
220  bool ok = command.read(connection);
221  if (!ok) return false;
222  reply.clear();
223 
224  if (command.get(0).asString()=="start")
225  {
226  mic->startRecording();
227  reply.addVocab(VOCAB_OK);
228  }
229  else if (command.get(0).asString() == "stop")
230  {
231  mic->stopRecording();
232  reply.addVocab(VOCAB_OK);
233  }
234  else if (command.get(0).asString() == "help")
235  {
236  reply.addVocab(yarp::os::Vocab::encode("many"));
237  reply.addString("start");
238  reply.addString("stop");
239  }
240  else
241  {
242  yError() << "Invalid command";
243  reply.addVocab(VOCAB_ERR);
244  }
245 
246  yarp::os::ConnectionWriter *returnToSender = connection.getWriter();
247  if (returnToSender != nullptr)
248  {
249  reply.write(*returnToSender);
250  }
251  return true;
252  }
