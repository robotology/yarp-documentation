86 bool navigation2DServer::open(Searchable& config)
87 {
88  Property params;
89  params.fromString(config.toString().c_str());
90  yDebug() << "navigation2DServer configuration: \n" << config.toString().c_str();
91 
92  if (!config.check("period"))
93  {
94  yInfo() << "navigation2DServer: missing 'period' parameter. Using default value: 0.010";
95  m_period = 0.010;
96  }
97  else
98  {
99  m_period = config.find("period").asFloat64();
100  }
101 
102  string local_name = "/navigationServer";
103  if (!config.check("name"))
104  {
105  yInfo() << "navigation2DServer: missing 'name' parameter. Using default value: /navigationServer";
106  }
107  else
108  {
109  }
110  m_rpcPortName = local_name + "/rpc";
111  m_streamingPortName = local_name + "/streaming:o";
112 
113  if (config.check("subdevice"))
114  {
115  Property p;
116  PolyDriverList driverlist;
117  p.fromString(config.toString(), false);
118  p.put("device", config.find("subdevice").asString());
119 
120  if (!pNav.open(p) || !pNav.isValid())
121  {
122  yError() << "navigation2DServer: failed to open subdevice.. check params";
123  return false;
124  }
125 
126  driverlist.push(&pNav, "1");
127  if (!attachAll(driverlist))
128  {
129  yError() << "navigation2DServer: failed to open subdevice.. check params";
130  return false;
131  }
132  }
133  m_stats_time_last = yarp::os::Time::now();
134 
135  if (!initialize_YARP(config))
136  {
137  yError() << "navigation2DServer: Error initializing YARP ports";
138  return false;
139  }
140 
141  return true;
142 }

188 bool navigation2DServer::parse_respond_vocab(const yarp::os::Bottle& command, yarp::os::Bottle& reply)
189 {
190  if (command.get(0).isVocab() == false)
191  {
192  yError() << "General error in navigation2DServer::parse_respond_vocab()";
193  return false;
194  }
195 
196  if (command.get(0).asVocab() != VOCAB_INAVIGATION ||
197  command.get(1).isVocab() == false)
198  {
199  yError() << "Invalid vocab received";
200  reply.addVocab(VOCAB_ERR);
201  return true;
202  }
203 
204  int request = command.get(1).asVocab();
205  if (request == VOCAB_NAV_GOTOABS)
206  {
207  yarp::dev::Map2DLocation loc;
208  loc.map_id = command.get(2).asString();
209  loc.x = command.get(3).asFloat64();
210  loc.y = command.get(4).asFloat64();
211  loc.theta = command.get(5).asFloat64();
212  bool ret = iNav_target->gotoTargetByAbsoluteLocation(loc);
213  if (ret)
214  {
215  reply.addVocab(VOCAB_OK);
216  }
217  else
218  {
219  yError() << "gotoTargetByAbsoluteLocation() failed";
220  reply.addVocab(VOCAB_ERR);
221  }
222  }
223  else if (request == VOCAB_NAV_RECOMPUTE_PATH)
224  {
225  bool ret = iNav_ctrl->recomputeCurrentNavigationPath();
226  if (ret)
227  {
228  reply.addVocab(VOCAB_OK);
229  }
230  else
231  {
232  yError() << "recomputeCurrentNavigationPath() failed";
233  reply.addVocab(VOCAB_ERR);
234  }
235  reply.addVocab(VOCAB_OK);
236  }
237  else if (request == VOCAB_NAV_GOTOREL)
238  {
239  if (command.size() == 5)
240  {
241  double x = command.get(2).asFloat64();
242  double y = command.get(3).asFloat64();
243  double theta = command.get(4).asFloat64();
244  bool ret = iNav_target->gotoTargetByRelativeLocation(x, y, theta);
245  if (ret)
246  {
247  reply.addVocab(VOCAB_OK);
248  }
249  else
250  {
251  yError() << "gotoTargetByRelativeLocation() failed";
252  reply.addVocab(VOCAB_ERR);
253  }
254  }
255  else if (command.size() == 4)
256  {
257  double x = command.get(2).asFloat64();
258  double y = command.get(3).asFloat64();
259  bool ret = iNav_target->gotoTargetByRelativeLocation(x, y);
260  if (ret)
261  {
262  reply.addVocab(VOCAB_OK);
263  }
264  else
265  {
266  yError() << "gotoTargetByRelativeLocation() failed";
267  reply.addVocab(VOCAB_ERR);
268  }
269  }
270  else
271  {
272  yError() << "Invalid number of params";
273  reply.addVocab(VOCAB_ERR);
274  }
275  }
276  else if (request == VOCAB_NAV_VELOCITY_CMD)
277  {
278  double x_vel = command.get(2).asFloat64();
279  double y_vel = command.get(3).asFloat64();
280  double t_vel = command.get(4).asFloat64();
281  double timeout = command.get(5).asFloat64();
282  bool ret = iNav_target->applyVelocityCommand(x_vel,y_vel,t_vel,timeout);
283  if (ret)
284  {
285  reply.addVocab(VOCAB_OK);
286  reply.addInt32(VOCAB_OK);
287  }
288  else
289  {
290  yError() << "applyVelocityCommand() failed";
291  reply.addVocab(VOCAB_ERR);
292  }
293  }
294  else if (request == VOCAB_NAV_GET_NAVIGATION_STATUS)
295  {
296  yarp::dev::NavigationStatusEnum nav_status = yarp::dev::navigation_status_error;
297  bool ret = iNav_ctrl->getNavigationStatus(nav_status);
298  if (ret)
299  {
300  reply.addVocab(VOCAB_OK);
301  reply.addInt32(nav_status);
302  }
303  else
304  {
305  yError() << "getNavigationStatus() failed";
306  reply.addVocab(VOCAB_ERR);
307  }
308  }
309  else if (request == VOCAB_NAV_STOP)
310  {
311  bool ret = iNav_ctrl->stopNavigation();
312  if (ret)
313  {
314  reply.addVocab(VOCAB_OK);
315  }
316  else
317  {
318  yError() << "stopNavigation() failed";
319  reply.addVocab(VOCAB_ERR);
320  }
321  }
322  else if (request == VOCAB_NAV_SUSPEND)
323  {
324  double time = -1;
325  if (command.size() > 1)
326  {
327  time = command.get(1).asFloat64();
328  bool ret = iNav_ctrl->suspendNavigation(time);
329  if (ret)
330  {
331  reply.addVocab(VOCAB_OK);
332  }
333  else
334  {
335  yError() << "suspendNavigation() failed";
336  reply.addVocab(VOCAB_ERR);
337  }
338  }
339  else
340  {
341  bool ret = iNav_ctrl->suspendNavigation();
342  if (ret)
343  {
344  reply.addVocab(VOCAB_OK);
345  }
346  else
347  {
348  yError() << "suspendNavigation() failed";
349  reply.addVocab(VOCAB_ERR);
350  }
351  }
352  }
353  else if (request == VOCAB_NAV_RESUME)
354  {
355  bool ret = iNav_ctrl->resumeNavigation();
356  if (ret)
357  {
358  reply.addVocab(VOCAB_OK);
359  }
360  else
361  {
362  yError() << "resumeNavigation failed()";
363  reply.addVocab(VOCAB_ERR);
364  }
365  }
366  else if (request == VOCAB_NAV_GET_NAVIGATION_WAYPOINTS)
367  {
368  std::vector<yarp::dev::Map2DLocation> locs;
369  bool ret = iNav_ctrl->getAllNavigationWaypoints(locs);
370  if (ret)
371  {
372  reply.addVocab(VOCAB_OK);
373  Bottle& waypoints = reply.addList();
374  for (size_t i = 0; i < locs.size(); i++)
375  {
376  Bottle& the_waypoint = waypoints.addList();
377  the_waypoint.addString(locs[i].map_id);
378  the_waypoint.addFloat64(locs[i].x);
379  the_waypoint.addFloat64(locs[i].y);
380  the_waypoint.addFloat64(locs[i].theta);
381  }
382  }
383  else
384  {
385  yError() << "getAllNavigationWaypoints() failed";
386  reply.addVocab(VOCAB_ERR);
387  }
388  }
389  else if (request == VOCAB_NAV_GET_CURRENT_WAYPOINT)
390  {
391  yarp::dev::Map2DLocation loc;
392  bool ret = iNav_ctrl->getCurrentNavigationWaypoint(loc);
393  if (ret)
394  {
395  reply.addVocab(VOCAB_OK);
396  reply.addString(loc.map_id);
397  reply.addFloat64(loc.x);
398  reply.addFloat64(loc.y);
399  reply.addFloat64(loc.theta);
400  }
401  else
402  {
403  yError() << "getCurrentNavigationWaypoint() failed";
404  reply.addVocab(VOCAB_ERR);
405  }
406  }
407  else if (request == VOCAB_NAV_GET_NAV_MAP)
408  {
409  yarp::dev::MapGrid2D map;
410  if (iNav_ctrl->getCurrentNavigationMap((yarp::dev::NavigationMapTypeEnum)(command.get(2).asInt32()), map))
411  {
412  reply.addVocab(VOCAB_OK);
413  yarp::os::Bottle& mapbot = reply.addList();
414  Property::copyPortable(map, mapbot);
415  }
416  else
417  {
418  yError() << "getCurrentNavigationMap() failed";
419  reply.addVocab(VOCAB_ERR);
420  }
421  }
422  else if (request == VOCAB_NAV_GET_ABS_TARGET)
423  {
424  yarp::dev::Map2DLocation loc;
425  bool ret;
426  ret = iNav_target->getAbsoluteLocationOfCurrentTarget(loc);
427  if (ret)
428  {
429  reply.addVocab(VOCAB_OK);
430  reply.addString(loc.map_id);
431  reply.addFloat64(loc.x);
432  reply.addFloat64(loc.y);
433  reply.addFloat64(loc.theta);
434  }
435  else
436  {
437  yError() << "getAbsoluteLocationOfCurrentTarget() failed";
438  reply.addVocab(VOCAB_ERR);
439  }
440  }
441  else if (request == VOCAB_NAV_GET_REL_TARGET)
442  {
443  yarp::dev::Map2DLocation loc;
444  bool ret;
445  ret = iNav_target->getRelativeLocationOfCurrentTarget(loc.x, loc.y, loc.theta);
446  if (ret)
447  {
448  reply.addVocab(VOCAB_OK);
449  reply.addFloat64(loc.x);
450  reply.addFloat64(loc.y);
451  reply.addFloat64(loc.theta);
452  }
453  else
454  {
455  yError() << "getRelativeLocationOfCurrentTarget() failed";
456  reply.addVocab(VOCAB_ERR);
457  }
458  }
459  else
460  {
461  yError() << "Invalid vocab received";
462  reply.addVocab(VOCAB_ERR);
463  }
464 
465  return true;
466 }

468 bool navigation2DServer::read(yarp::os::ConnectionReader& connection)
469 {
470  yarp::os::Bottle command;
471  yarp::os::Bottle reply;
472  bool ok = command.read(connection);
473  if (!ok) return false;
474  reply.clear();
475 
476  //^^^^^^^^^^^^^^^^^ STRING SECTION
477  if (command.get(0).isString())
478  {
479  parse_respond_string(command, reply);
480  }
481  //^^^^^^^^^^^^^^^^^ VOCAB SECTION
482  else if (command.get(0).isVocab())
483  {
484  parse_respond_vocab(command, reply);
485  }
486  else
487  {
488  yError() << "Invalid command type";
489  reply.addVocab(VOCAB_ERR);
490  }
491 
492  yarp::os::ConnectionWriter *returnToSender = connection.getWriter();
493  if (returnToSender != nullptr)
494  {
495  reply.write(*returnToSender);
496  }
497 
498  return true;
499 }
