72 void FakeBattery::run()
73 {
74  double timeNow=yarp::os::Time::now();
75  static int counter = 0;
76  mutex.wait();
77 
78  if (timeNow-timeStamp>2.0)
79  {
80  timeStamp = timeNow;
81  if (counter >= 4)
82  {
83  battery_voltage = 50.0;
84  battery_current = 51.0;
85  battery_charge = 52.0;
86  battery_temperature = 53.0;
87  counter = 0;
88  }
89  else
90  {
91  battery_voltage += 10;
92  battery_current += 10;
93  battery_charge += 10;
94  battery_temperature += 10;
95  counter++;
96  }
97  }
