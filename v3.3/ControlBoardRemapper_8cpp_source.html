105 bool ControlBoardRemapper::parseNetworks(const Property& prop)
106 {
107  Bottle *nets=prop.find("networks").asList();
108  if(nets==nullptr)
109  {
110  yError() <<"ControlBoardRemapper: Error parsing parameters: \"networks\" should be followed by a list\n";
111  return false;
112  }
113 
114  if (!prop.check("joints", "number of joints of the part"))
115  {
116  yError() <<"ControlBoardRemapper: joints options not found when reading networks option";
117  return false;
118  }
119 
120  this->setNrOfControlledAxes((size_t)prop.find("joints").asInt32());
121 
122  int nsubdevices=nets->size();
123  remappedControlBoards.lut.resize(controlledJoints);
124  remappedControlBoards.subdevices.resize(nsubdevices);
125 
126  // configure the devices
127  for(size_t k=0;k<nets->size();k++)
128  {
129  Bottle parameters;
130  int wBase;
131  int wTop;
132  int base;
133  int top;
134 
135  parameters=prop.findGroup(nets->get(k).asString());
136 
137  if(parameters.size()==2)
138  {
139  Bottle *bot=parameters.get(1).asList();
140  Bottle tmpBot;
141  if(bot==nullptr)
142  {
143  // probably data are not passed in the correct way, try to read them as a string.
144  std::string bString(parameters.get(1).asString());
145  tmpBot.fromString(bString);
146 
147  if(tmpBot.size() != 4)
148  {
149  yError() << "Error: check network parameters in part description"
150  << "--> I was expecting "<<nets->get(k).asString().c_str() << " followed by a list of four integers in parenthesis"
151  << "Got: "<< parameters.toString().c_str() << "\n";
152  return false;
153  }
154  else
155  {
156  bot = &tmpBot;
157  }
158  }
159 
160  // If I came here, bot is correct
161  wBase=bot->get(0).asInt32();
162  wTop=bot->get(1).asInt32();
163  base=bot->get(2).asInt32();
164  top=bot->get(3).asInt32();
165  }
166  else if (parameters.size()==5)
167  {
168  // yError<<"Parameter networks use deprecated syntax\n";
169  wBase=parameters.get(1).asInt32();
170  wTop=parameters.get(2).asInt32();
171  base=parameters.get(3).asInt32();
172  top=parameters.get(4).asInt32();
173  }
174  else
175  {
176  yError() <<"Error: check network parameters in part description"
177  <<"--> I was expecting "<<nets->get(k).asString().c_str() << " followed by a list of four integers in parenthesis"
178  <<"Got: "<< parameters.toString().c_str() << "\n";
179  return false;
180  }
181 
182  RemappedSubControlBoard *tmpDevice=remappedControlBoards.getSubControlBoard((size_t)k);
183  tmpDevice->setVerbose(_verb);
184 
185  if( (wTop-wBase) != (top-base) )
186  {
187  yError() <<"Error: check network parameters in network "<<nets->get(k).asString().c_str() <<
188  "I was expecting a well form quadruple of numbers, got instead: "<< parameters.toString().c_str();
189  }
190 
191  tmpDevice->id = nets->get(k).asString();
192 
193  for(int j=wBase;j<=wTop;j++)
194  {
195  int off = j-wBase;
196  remappedControlBoards.lut[j].subControlBoardIndex=k;
197  remappedControlBoards.lut[j].axisIndexInSubControlBoard=base+off;
198  }
199  }
200 
201  return true;
202 }

282 bool ControlBoardRemapper::attachAllUsingAxesNames(const PolyDriverList& polylist)
283 {
284  std::map<std::string, axisLocation> axesLocationMap;
285 
286  for(int p=0;p<polylist.size();p++)
287  {
288  // If there is a device with a specific device key, use it
289  // as a calibrator, otherwise rely on the subcontrolboards
290  // as usual
291  std::string deviceKey=polylist[p]->key;
292  if(deviceKey == "Calibrator" || deviceKey == "calibrator")
293  {
294  // Set the IRemoteCalibrator interface, the wrapper must point to the calibrator rdevice
295  yarp::dev::IRemoteCalibrator *calibrator;
296  polylist[p]->poly->view(calibrator);
297 
298  IRemoteCalibrator::setCalibratorDevice(calibrator);
299  continue;
300  }
301 
302  // find if one of the desired axis is in this device
303  yarp::dev::IAxisInfo *iaxinfos = nullptr;
304  yarp::dev::IEncoders *iencs = nullptr;
305  polylist[p]->poly->view(iaxinfos);
306  polylist[p]->poly->view(iencs);
307 
308  if( !iencs ||
309  !iaxinfos )
310  {
311  yError() <<"ControlBoardRemapper: subdevice " << deviceKey << " does not implemented the required IAxisInfo or IEncoders interfaces";
312  return false;
313  }
314 
315  int nrOfSubdeviceAxes;
316  bool ok = iencs->getAxes(&nrOfSubdeviceAxes);
317 
318  if( !ok )
319  {
320  yError() <<"ControlBoardRemapper: subdevice " << deviceKey << " does not implemented the required getAxes method";
321  return false;
322  }
323 
324  for(int axInSubDevice =0; axInSubDevice < nrOfSubdeviceAxes; axInSubDevice++)
325  {
326  std::string axNameYARP;
327  ok = iaxinfos->getAxisName(axInSubDevice,axNameYARP);
328 
329  std::string axName = axNameYARP;
330 
331  if( !ok )
332  {
333  yError() <<"ControlBoardRemapper: subdevice " << deviceKey << " does not implemented the required getAxisName method";
334  return false;
335  }
336 
337  auto it = axesLocationMap.find(axName);
338  if( it != axesLocationMap.end() )
339  {
340  yError() <<"ControlBoardRemapper: multiple axes have the same name " << axName
341  <<" on on device " << polylist[p]->key << " with index " << axInSubDevice
342  <<" and another on device " << it->second.subDeviceKey << " with index " << it->second.indexInSubDevice;
343  return false;
344  }
345 
346  axisLocation newLocation;
347  newLocation.subDeviceKey = polylist[p]->key;
348  newLocation.indexInSubDevice = axInSubDevice;
349  newLocation.indexOfSubDeviceInPolyDriverList = p;
350  axesLocationMap[axName] = newLocation;
351  }
352  }
353 
354  // We store the key of all the devices that we actually use in the remapped control device
355  std::vector<std::string> subControlBoardsKeys;
356  std::map<std::string, size_t> subControlBoardKey2IndexInPolyDriverList;
357  std::map<std::string, size_t> subControlBoardKey2IndexInRemappedControlBoards;
358 
359 
360  // Once we build the axis map, we build the mapping between the remapped axes and
361  // the couple subControlBoard, axis in subControlBoard
362  for(const auto& axesName : axesNames)
363  {
364  auto it = axesLocationMap.find(axesName);
365  if( it == axesLocationMap.end() )
366  {
367  yError() <<"ControlBoardRemapper: axis " << axesName
368  <<" specified in axesNames was not found in the axes of the controlboards "
369  <<"passed to attachAll, attachAll failed. ";
370  return false;
371  }
372 
373  axisLocation loc = it->second;
374  std::string key = loc.subDeviceKey;
375 
376  if(std::find(subControlBoardsKeys.begin(), subControlBoardsKeys.end(), key) == subControlBoardsKeys.end())
377  {
378  /* subControlBoardsKeys does not contain key */
379  subControlBoardKey2IndexInRemappedControlBoards[key] = subControlBoardsKeys.size();
380  subControlBoardsKeys.push_back(key);
381  subControlBoardKey2IndexInPolyDriverList[key] = loc.indexOfSubDeviceInPolyDriverList;
382  }
383  }
384 
385  assert(controlledJoints == (int) axesNames.size());
386 
387  // We have now the number of controlboards to attach to
388  size_t nrOfSubControlBoards = subControlBoardsKeys.size();
389  remappedControlBoards.lut.resize(controlledJoints);
390  remappedControlBoards.subdevices.resize(nrOfSubControlBoards);
391 
392  // Open the controlboards
393  for(size_t ctrlBrd=0; ctrlBrd < nrOfSubControlBoards; ctrlBrd++)
394  {
395  size_t p = subControlBoardKey2IndexInPolyDriverList[subControlBoardsKeys[ctrlBrd]];
396  RemappedSubControlBoard *tmpDevice = remappedControlBoards.getSubControlBoard(ctrlBrd);
397  tmpDevice->setVerbose(_verb);
398  tmpDevice->id = subControlBoardsKeys[ctrlBrd];
399  bool ok = tmpDevice->attach(polylist[p]->poly,subControlBoardsKeys[ctrlBrd]);
400 
401  if( !ok )
402  {
403  return false;
404  }
405  }
406 
407 
408  for(size_t l=0; l < axesNames.size(); l++)
409  {
410  axisLocation loc = axesLocationMap[axesNames[l]];
411  remappedControlBoards.lut[l].subControlBoardIndex = subControlBoardKey2IndexInRemappedControlBoards[loc.subDeviceKey];
412  remappedControlBoards.lut[l].axisIndexInSubControlBoard = (size_t)loc.indexInSubDevice;
413  }
414 
415  return true;
416 }

420 {
421  for(int p=0;p<polylist.size();p++)
422  {
423  // look if we have to attach to a calibrator

426  {

431  IRemoteCalibrator::setCalibratorDevice(calibrator);

433  }
434 
435  // find appropriate entry in list of subdevices and attach
436  unsigned int k=0;
437  for(k=0; k<remappedControlBoards.getNrOfSubControlBoards(); k++)
438  {
439  if (remappedControlBoards.subdevices[k].id==subDeviceKey)
440  {
441  if (!remappedControlBoards.subdevices[k].attach(polylist[p]->poly, subDeviceKey))
442  {
443  yError("ControlBoardRemapper: attach to subdevice %s failed\n", polylist[p]->key.c_str());
444  return false;
445  }
446  }
447  }
448  }
449 
450  bool ok = updateAxesName();
451 
452  if( !ok )
453  {
454  yWarning() << "ControlBoardRemapper: unable to update axesNames";
455  }
456 
457  return true;
458 }
