57 std::string InputStream::readLine(const char terminal, bool* success)
58 {
59  std::string buf;
60  bool done = false;
61  int esc = 0;
62  if (success != nullptr) {
63  *success = true;
64  }
65  while (!done) {
66  int v = read();
67  if (v < 0) {
68  if (success != nullptr) {
69  *success = false;
70  }
71  return {};
72  }
73  char ch = (char)v;
74  if (v == '\\') {
75  esc++;
76  }
77  if (v != 0 && v != '\r' && v != '\n') {
78  if (v != '\\' || esc >= 2) {
79  while (esc != 0) {
80  buf += '\\';
81  esc--;
82  }
83  }
84  if (v != '\\') {
85  buf += ch;
86  }
87  }
88  if (ch == terminal) {
89  if (esc == 0) {
90  done = true;
91  } else {
92  esc = 0;
93  }
94  }
95  }
96  return buf;
97 }

99 yarp::conf::ssize_t InputStream::readFull(Bytes& b)
100 {
101  yarp::conf::ssize_t off = 0;
102  yarp::conf::ssize_t fullLen = b.length();
103  yarp::conf::ssize_t remLen = fullLen;
104  yarp::conf::ssize_t result = 1;
105  while (result > 0 && remLen > 0) {
106  result = read(b, off, remLen);
107  if (result > 0) {
108  remLen -= result;
109  off += result;
110  }
111  }
112  return (result <= 0) ? -1 : fullLen;
113 }
