56 void Map2DServer::parse_vocab_command(yarp::os::Bottle& in, yarp::os::Bottle& out)
57 {
58  int code = in.get(0).asVocab();
59 // bool ret = false;
60  if (code == VOCAB_IMAP)
61  {
62  int cmd = in.get(1).asVocab();
63  if (cmd == VOCAB_IMAP_SET_MAP)
64  {
65  MapGrid2D the_map;
66  Value& b = in.get(2);
67  if (Property::copyPortable(b, the_map))
68  {
69  string map_name = the_map.getMapName();
70  auto it = m_maps_storage.find(map_name);
71  if (it == m_maps_storage.end())
72  {
73  //add a new map
74  m_maps_storage[map_name] = the_map;
75  out.clear();
76  out.addVocab(VOCAB_IMAP_OK);
77  }
78  else
79  {
80  //the map already exists
81  m_maps_storage[map_name] = the_map;
82  out.clear();
83  out.addVocab(VOCAB_IMAP_OK);
84  }
85  }
86  else
87  {
88  out.clear();
89  out.addVocab(VOCAB_IMAP_ERROR);
90  yError() << "Error in copyPortable";
91  }
92  }
93  else if (cmd == VOCAB_IMAP_GET_MAP)
94  {
95  string name = in.get(2).asString();
96  auto it = m_maps_storage.find(name);
97  if (it != m_maps_storage.end())
98  {
99  out.clear();
100  out.addVocab(VOCAB_IMAP_OK);
101  yarp::os::Bottle& mapbot = out.addList();
102  Property::copyPortable(it->second, mapbot);
103  }
104  else
105  {
106  out.clear();
107  out.addVocab(VOCAB_IMAP_ERROR);
108  yError() << "Map" << name << "not found";
109  }
110  }
111  else if (cmd == VOCAB_IMAP_GET_NAMES)
112  {
113  out.clear();
114  out.addVocab(VOCAB_IMAP_OK);
115 
116  for (auto& it : m_maps_storage)
117  {
118  out.addString(it.first);
119  }
120  }
121  else if (cmd == VOCAB_IMAP_REMOVE)
122  {
123  string name = in.get(2).asString();
124  size_t rem = m_maps_storage.erase(name);
125  if (rem == 0)
126  {
127  yError() << "Map not found";
128  out.clear();
129  out.addVocab(VOCAB_IMAP_ERROR);
130  }
131  else
132  {
133  out.clear();
134  out.addVocab(VOCAB_IMAP_OK);
135  }
136  }
137  else if (cmd == VOCAB_IMAP_CLEAR)
138  {
139  m_maps_storage.clear();
140  out.clear();
141  out.addVocab(VOCAB_IMAP_OK);
142  }
143  else if (cmd == VOCAB_IMAP_SAVE_COLLECTION)
144  {
145  string mapfile = in.get(2).asString();
146  if (saveMaps(mapfile))
147  {
148  out.clear();
149  out.addVocab(VOCAB_IMAP_OK);
150  }
151  else
152  {
153  yError("Map2DServer: Unable to save collection");
154  out.clear();
155  out.addVocab(VOCAB_IMAP_ERROR);
156  }
157  }
158  else if (cmd == VOCAB_IMAP_LOAD_COLLECTION)
159  {
160  string mapfile = in.get(2).asString();
161  if (loadMaps(mapfile))
162  {
163  out.clear();
164  out.addVocab(VOCAB_IMAP_OK);
165  }
166  else
167  {
168  yError("Map2DServer: Unable to load collection");
169  out.clear();
170  out.addVocab(VOCAB_IMAP_ERROR);
171  }
172  }
173  else
174  {
175  yError("Invalid vocab received in Map2DServer");
176  out.clear();
177  out.addVocab(VOCAB_IMAP_ERROR);
178  }
179  }
180  else if (code == VOCAB_INAVIGATION)
181  {
182  int cmd = in.get(1).asVocab();
183  if (cmd == VOCAB_NAV_GET_LIST_X && in.get(2).asVocab() == VOCAB_NAV_LOCATION)
184  {
185  std::string info;
186 
187  out.addVocab(VOCAB_OK);
188  Bottle& l = out.addList();
189 
190  std::map<std::string, Map2DLocation>::iterator it;
191  for (it = m_locations_storage.begin(); it != m_locations_storage.end(); ++it)
192  {
193  l.addString(it->first);
194  }
195  yInfo() << "The following locations are currently stored in the server: "<<l.toString();
196 // ret = true;
197  }
198  else if (cmd == VOCAB_NAV_GET_LIST_X && in.get(2).asVocab() == VOCAB_NAV_AREA)
199  {
200  std::string info;
201 
202  out.addVocab(VOCAB_OK);
203  Bottle& l = out.addList();
204 
205  std::map<std::string, Map2DArea>::iterator it;
206  for (it = m_areas_storage.begin(); it != m_areas_storage.end(); ++it)
207  {
208  l.addString(it->first);
209  }
210  yInfo() << "The following areas are currently stored in the server: "<<l.toString();
211 // ret = true;
212  }
213  else if (cmd == VOCAB_NAV_CLEAR_X && in.get(2).asVocab() == VOCAB_NAV_LOCATION)
214  {
215  m_locations_storage.clear();
216  yInfo() << "All locations deleted ";
217  out.addVocab(VOCAB_OK);
218 // ret = true;
219  }
220  else if (cmd == VOCAB_NAV_CLEAR_X && in.get(2).asVocab() == VOCAB_NAV_AREA)
221  {
222  m_areas_storage.clear();
223  yInfo() << "All areas deleted ";
224  out.addVocab(VOCAB_OK);
225  // ret = true;
226  }
227  else if (cmd == VOCAB_NAV_DELETE_X && in.get(2).asVocab() == VOCAB_NAV_LOCATION)
228  {
229  std::string name = in.get(3).asString();
230 
231  std::map<std::string, Map2DLocation>::iterator it;
232  it = m_locations_storage.find(name);
233  if (it != m_locations_storage.end())
234  {
235  yInfo() << "Deleted location " << name;
236  m_locations_storage.erase(it);
237  out.addVocab(VOCAB_OK);
238  }
239  else
240  {
241  yError("User requested an invalid location name");
242  out.addVocab(VOCAB_ERR);
243  }
244 
245 // ret = true;
246  }
247  else if (cmd == VOCAB_NAV_RENAME_X && in.get(2).asVocab() == VOCAB_NAV_LOCATION)
248  {
249  std::string orig_name = in.get(3).asString();
250  std::string new_name = in.get(4).asString();
251 
252  std::map<std::string, Map2DLocation>::iterator orig_it;
253  orig_it = m_locations_storage.find(orig_name);
254  std::map<std::string, Map2DLocation>::iterator new_it;
255  new_it = m_locations_storage.find(new_name);
256 
257  if (orig_it != m_locations_storage.end() &&
258  new_it == m_locations_storage.end())
259  {
260  yInfo() << "location: " << orig_name << " renamed to: " << new_name;
261  auto loc = orig_it->second;
262  m_locations_storage.erase(orig_it);
263  m_locations_storage.insert(std::pair<std::string, Map2DLocation>(new_name, loc));
264  out.addVocab(VOCAB_OK);
265  }
266  else
267  {
268  yError("User requested an invalid rename operation");
269  out.addVocab(VOCAB_ERR);
270  }
271  // ret = true;
272  }
273  else if (cmd == VOCAB_NAV_RENAME_X && in.get(2).asVocab() == VOCAB_NAV_AREA)
274  {
275  std::string orig_name = in.get(3).asString();
276  std::string new_name = in.get(4).asString();
277 
278  std::map<std::string, Map2DArea>::iterator orig_it;
279  orig_it = m_areas_storage.find(orig_name);
280  std::map<std::string, Map2DArea>::iterator new_it;
281  new_it = m_areas_storage.find(new_name);
282 
283  if (orig_it != m_areas_storage.end() &&
284  new_it == m_areas_storage.end())
285  {
286  yInfo() << "area: " << orig_name << " renamed to: " << new_name;
287  auto area = orig_it->second;
288  m_areas_storage.erase(orig_it);
289  m_areas_storage.insert(std::pair<std::string, Map2DArea>(new_name,area));
290  out.addVocab(VOCAB_OK);
291  }
292  else
293  {
294  yError("User requested an invalid rename operation");
295  out.addVocab(VOCAB_ERR);
296  }
297 // ret = true;
298  }
299  else if (cmd == VOCAB_NAV_DELETE_X && in.get(2).asVocab() == VOCAB_NAV_AREA)
300  {
301  std::string name = in.get(3).asString();
302 
303  std::map<std::string, Map2DArea>::iterator it;
304  it = m_areas_storage.find(name);
305  if (it != m_areas_storage.end())
306  {
307  yInfo() << "Deleted area " << name;
308  m_areas_storage.erase(it);
309  out.addVocab(VOCAB_OK);
310  }
311  else
312  {
313  yError("User requested an invalid area name");
314  out.addVocab(VOCAB_ERR);
315  }
316 
317  // ret = true;
318  }
319  else if (cmd == VOCAB_NAV_GET_X && in.get(2).asVocab() == VOCAB_NAV_LOCATION)
320  {
321  std::string name = in.get(3).asString();
322 
323  std::map<std::string, Map2DLocation>::iterator it;
324  it = m_locations_storage.find(name);
325  if (it != m_locations_storage.end())
326  {
327  out.addVocab(VOCAB_OK);
328  Map2DLocation loc = it->second;
329  yInfo() << "Retrieved location " << name << "at " << loc.toString();
330  out.addString(loc.map_id);
331  out.addFloat64(loc.x);
332  out.addFloat64(loc.y);
333  out.addFloat64(loc.theta);
334  }
335  else
336  {
337  out.addVocab(VOCAB_ERR);
338  yError("User requested an invalid location name");
339  }
340 
341 // ret = true;
342  }
343  else if (cmd == VOCAB_NAV_GET_X && in.get(2).asVocab() == VOCAB_NAV_AREA)
344  {
345  std::string area_name = in.get(3).asString();
346 
347  std::map<std::string, Map2DArea>::iterator it;
348  it = m_areas_storage.find(area_name);
349  if (it != m_areas_storage.end())
350  {
351  Map2DArea area = it->second;
352  yarp::os::Bottle mapbot;
353  Map2DArea maptemp = area;
354  if (Property::copyPortable(maptemp, mapbot) == false)
355  {
356  yError() << "Map2DServer::VOCAB_NAV_GET_X VOCAB_NAV_AREA failed copyPortable()";
357  out.addVocab(VOCAB_ERR);
358  }
359  else
360  {
361  yInfo() << "Retrieved area " << area_name << "at " << area.toString();
362  out.addVocab(VOCAB_OK);
363 
364  yarp::os::Bottle& mapbot = out.addList();
365  Property::copyPortable(maptemp, mapbot);
366  }
367  }
368  else
369  {
370  out.addVocab(VOCAB_ERR);
371  yError("User requested an invalid area name");
372  }
373 
374 // ret = true;
375  }
376  else if (cmd == VOCAB_NAV_STORE_X && in.get(2).asVocab() == VOCAB_NAV_LOCATION)
377  {
378  Map2DLocation location;
379  std::string name = in.get(3).asString();
380 
381  location.map_id = in.get(4).asString();
382  location.x = in.get(5).asFloat64();
383  location.y = in.get(6).asFloat64();
384  location.theta = in.get(7).asFloat64();
385 
386  m_locations_storage.insert(std::pair<std::string, Map2DLocation>(name, location));
387  yInfo() << "Added location " << name << "at " << location.toString();
388  out.addVocab(VOCAB_OK);
389 // ret = true;
390  }
391  else if (cmd == VOCAB_NAV_STORE_X && in.get(2).asVocab() == VOCAB_NAV_AREA)
392  {
393  Map2DArea area;
394  std::string area_name = in.get(3).asString();
395 
396  Value& b = in.get(4);
397  if (Property::copyPortable(b, area))
398  {
399  m_areas_storage.insert(std::pair<std::string, Map2DArea>(area_name, area));
400  yInfo() << "Added area " << area_name << "at " << area.toString();
401  out.addVocab(VOCAB_OK);
402  }
403  else
404  {
405  yError() << "Map2DServer::VOCAB_NAV_STORE_X VOCAB_NAV_AREA failed copyPortable()";
406  out.addVocab(VOCAB_ERR);
407  }
408 
409 
410 // ret = true;
411  }
412  else
413  {
414  yError() << "Invalid vocab received in LocationsServer:" << in.toString();
415  out.clear();
416  out.addVocab(VOCAB_ERR);
417  }
418  }
419  else
420  {
421  yError() << "Invalid vocab received in Map2DServer" << in.toString();
422  out.clear();
423  out.addVocab(VOCAB_IMAP_ERROR);
424  }

427 void Map2DServer::parse_string_command(yarp::os::Bottle& in, yarp::os::Bottle& out)
428 {
429  if (in.get(0).asString() == "save_locations&areas" && in.get(1).isString())

431  if(save_locations_and_areas(in.get(1).asString()))
432  {
433  out.addString(in.get(1).asString() + " successfully saved");
434  }
435  else
436  {
437  out.addString("save_locations&areas failed");
438  }
439  }
440  else if (in.get(0).asString() == "load_locations&areas" && in.get(1).isString())
441  {
442  if(load_locations_and_areas(in.get(1).asString()))
443  {
444  out.addString(in.get(1).asString() + " successfully loaded");
445  }
446  else
447  {
448  out.addString("load_locations&areas failed");
449  }
450  }
451  else if(in.get(0).asString() == "list_locations")
452  {
453  std::map<std::string, Map2DLocation>::iterator it;
454  for (it = m_locations_storage.begin(); it != m_locations_storage.end(); ++it)
455  {
456  out.addString(it->first);
457  }
458  }
459  else if (in.get(0).asString() == "list_areas")
460  {
461  std::map<std::string, Map2DArea>::iterator it;
462  for (it = m_areas_storage.begin(); it != m_areas_storage.end(); ++it)
463  {
464  out.addString(it->first);
465  }
466  }
467  else if (in.get(0).asString() == "save_maps" && in.get(1).isString())
468  {
469  if(saveMaps(in.get(1).asString()))
470  {
471  out.addString(in.get(1).asString() + " successfully saved");
472  }
473  else
474  {
475  out.addString("save_maps failed");
476  }
477  }
478  else if (in.get(0).asString() == "load_maps" && in.get(1).isString())
479  {
480  if(loadMaps(in.get(1).asString()))
481  {
482  out.addString(in.get(1).asString() + " successfully loaded");
483  }
484  else
485  {
486  out.addString("load_maps failed");
487  }
488  }
489  else if (in.get(0).asString() == "save_map" && in.get(1).isString() && in.get(2).isString())
490  {
491  std::string map_name = in.get(1).asString();
492  std::string map_file = in.get(2).asString() + ".map";
493  auto p = m_maps_storage.find(map_name);

495  {
496  out.addString("save_map failed: map " + map_name + " not found");
497  }
498  else

501  if (b)
502  {
503  out.addString(map_file + " successfully saved");
504  }
505  else
506  {
507  out.addString("save_map failed: unable to save " + map_name + " to "+ map_file);
508  }
509  }
510  }
511  else if (in.get(0).asString() == "load_map" && in.get(1).isString())
512  {
513  yarp::dev::MapGrid2D map;

515  if(r)
516  {
517  string map_name= map.getMapName();
518  auto p = m_maps_storage.find(map_name);
519  if (p == m_maps_storage.end())

521  m_maps_storage[map_name] = map;
522  out.addString(in.get(1).asString() + " successfully loaded.");
523  }
524  else
525  {
526  out.addString(in.get(1).asString() + " already exists, skipping.");
527  }
528  }
529  else
530  {
531  out.addString("load_map failed. Unable to load " + in.get(1).asString());
532  }
533  }
534  else if(in.get(0).asString() == "list_maps")
535  {
536  std::map<std::string, MapGrid2D>::iterator it;
537  for (it = m_maps_storage.begin(); it != m_maps_storage.end(); ++it)
538  {
539  out.addString(it->first);
540  }
541  }
542  else if(in.get(0).asString() == "clear_all_locations")
543  {
544  m_locations_storage.clear();
545  out.addString("all locations cleared");
546  }
547  else if (in.get(0).asString() == "clear_all_areas")
548  {
549  m_areas_storage.clear();
550  out.addString("all areas cleared");
551  }
552  else if(in.get(0).asString() == "clear_all_maps")
553  {
554  m_maps_storage.clear();
555  out.addString("all maps cleared");
556  }
557  else if(in.get(0).asString() == "help")
558  {
559  out.addVocab(Vocab::encode("many"));
560  out.addString("'save_locations&areas <full path filename>' to save locations and areas on a file");
561  out.addString("'load_locations&areas <full path filename>' to load locations and areas from a file");
562  out.addString("'list_locations' to view a list of all stored locations");
563  out.addString("'list_areas' to view a list of all stored areas");
564  out.addString("'clear_all_locations' to clear all stored locations");
565  out.addString("'clear_all_areas' to clear all stored areas");
566  out.addString("'save_maps <full path>' to save a map collection to a folder");
567  out.addString("'load_maps <full path>' to load a map collection from a folder");
568  out.addString("'save_map <map_name> <full path>' to save a single map");
569  out.addString("'load_map <full path>' to load a single map");
570  out.addString("'list_maps' to view a list of all stored maps");
571  out.addString("'clear_all_maps' to clear all stored maps");
572  }
573  else
574  {
575  out.addString("request not understood, call 'help' to see a list of available commands");
576  }

581 bool Map2DServer::read(yarp::os::ConnectionReader& connection)
582 {
583  std::lock_guard<std::mutex> lock(m_mutex);
584  yarp::os::Bottle in;
585  yarp::os::Bottle out;
586  bool ok = in.read(connection);
587  if (!ok) return false;
588 
589  //parse string command
590  if(in.get(0).isString())
591  {
592  parse_string_command(in, out);
593  }
594  // parse vocab command
595  else if(in.get(0).isVocab())
596  {
597  parse_vocab_command(in, out);
598  }
599 
600  yarp::os::ConnectionWriter *returnToSender = connection.getWriter();
601  if (returnToSender != nullptr)
602  {
603  out.write(*returnToSender);
604  }
605  else
606  {
607  yError() << "Map2DServer: invalid return to sender";
608  }
609  return true;
610 }

612 bool Map2DServer::saveMaps(std::string mapsfile)
613 {
614  if (m_maps_storage.size() == 0)
615  {
616  yError() << "map storage is empty";
617  return false;
618  }
619  std::ofstream file;
620  file.open(mapsfile.c_str());
621  if (!file.is_open())
622  {
623  yError() << "sorry unable to open" << mapsfile;
624  return false;
625  }
626  bool ret = true;
627  for (auto& it : m_maps_storage)
628  {
629  string map_filename = it.first + ".map";
630  file << "mapfile: ";
631  file << map_filename;
632  file << endl;
633  ret &= it.second.saveToFile(map_filename);
634  }
635  file.close();
636  return ret;
637 }

639 bool Map2DServer::loadMaps(std::string mapsfile)
640 {
641  bool ret = true;
642  std::ifstream file;
643  file.open(mapsfile.c_str());
644  if (!file.is_open())
645  {
646  yError() << "Map2DServer::loadMaps() Unable to open:" << mapsfile;
647  return false;
648  }
649  while (!file.eof())
650  {
651  string dummy;
652  string buffer;
653  std::getline(file, buffer);
654  std::istringstream iss(buffer);
655  iss >> dummy;
656  if (dummy == "") break;

1018  double ypos = b.get(ai+1).asFloat64();
1019  area.points.push_back(yarp::math::Vec2D<double>(xpos,ypos));
1020  }
