52 void RPCMessagesParser::handleImpedanceMsg(const yarp::os::Bottle& cmd,
53  yarp::os::Bottle& response, bool *rec, bool *ok)
54 {
55  if (ControlBoardWrapper_p->verbose())
56  yDebug("Handling IImpedance message\n");
57  if (!rpc_IImpedance)
58  {
59  yError("controlBoardWrapper: I do not have a valid interface");
60  *ok=false;
61  return;
62  }
63 
64  int code = cmd.get(0).asVocab();
65  *ok=false;
66  switch(code)
67  {
68  case VOCAB_SET:
69  {
70  if (ControlBoardWrapper_p->verbose())
71  yDebug("handleImpedanceMsg::VOCAB_SET command\n");
72  switch (cmd.get(2).asVocab())
73  {
74  case VOCAB_IMP_PARAM:
75  {
76  Bottle *b = cmd.get(4).asList();
77  if (b!=nullptr)
78  {
79  double stiff = b->get(0).asFloat64();
80  double damp = b->get(1).asFloat64();
81  *ok = rpc_IImpedance->setImpedance(cmd.get(3).asInt32(),stiff,damp);
82  *rec=true;
83  }
84  }
85  break;
86  case VOCAB_IMP_OFFSET:
87  {
88  Bottle *b = cmd.get(4).asList();
89  if (b!=nullptr)
90  {
91  double offs = b->get(0).asFloat64();
92  *ok = rpc_IImpedance->setImpedanceOffset(cmd.get(3).asInt32(),offs);
93  *rec=true;
94  }
95  }
96  break;
97  }
98  }
99  break;
100  case VOCAB_GET:
101  {
102  double stiff = 0;
103  double damp = 0;
104  double offs = 0;
105  if (ControlBoardWrapper_p->verbose())
106  yDebug("handleImpedanceMsg::VOCAB_GET command\n");
107 
108  response.addVocab(VOCAB_IS);
109  response.add(cmd.get(1));
110  switch (cmd.get(2).asVocab())
111  {
112  case VOCAB_IMP_PARAM:
113  {
114  *ok = rpc_IImpedance->getImpedance(cmd.get(3).asInt32(),&stiff, &damp);
115  Bottle& b = response.addList();
116  b.addFloat64(stiff);
117  b.addFloat64(damp);
118  *rec=true;
119  }
120  break;
121  case VOCAB_IMP_OFFSET:
122  {
123  *ok = rpc_IImpedance->getImpedanceOffset(cmd.get(3).asInt32(),&offs);
124  Bottle& b = response.addList();
125  b.addFloat64(offs);
126  *rec=true;
127  }
128  break;
129  case VOCAB_LIMITS:
130  {
131  double min_stiff = 0;
132  double max_stiff = 0;
133  double min_damp = 0;
134  double max_damp = 0;
135  *ok = rpc_IImpedance->getCurrentImpedanceLimit(cmd.get(3).asInt32(),&min_stiff, &max_stiff, &min_damp, &max_damp);
136  Bottle& b = response.addList();
137  b.addFloat64(min_stiff);
138  b.addFloat64(max_stiff);
139  b.addFloat64(min_damp);
140  b.addFloat64(max_damp);
141  *rec=true;
142  }
143  break;
144  }
145  }
146  lastRpcStamp.update();
147  appendTimeStamp(response, lastRpcStamp);
148  break; // case VOCAB_GET
149  default:
150  {
151  *rec=false;
152  }
153  break;
154  }

194  *rec = false;
195  }
196  }
197  break;
198 
199  case VOCAB_CM_CONTROL_MODE_GROUP:
200  {
201  int n_joints = cmd.get(3).asInt32();
202  Bottle& jList = *(cmd.get(4).asList());
203  Bottle& modeList= *(cmd.get(5).asList());
204 
205  int *js = new int [n_joints];
206  int *modes = new int [n_joints];
207 
208  for(int i=0; i<n_joints; i++)
209  {
210  js[i] = jList.get(i).asInt32();

212 
213  for(int i=0; i<n_joints; i++)
214  {
215  modes[i] = modeList.get(i).asVocab();

217  if(rpc_iCtrlMode)
218  *ok = rpc_iCtrlMode->setControlModes(n_joints, js, modes);

220  {
221  *rec = false;

223  }

228 
229  case VOCAB_CM_CONTROL_MODES:
230  {
231  yarp::os::Bottle *modeList;
232  modeList = cmd.get(3).asList();
233 
234  if(modeList->size() != (size_t) controlledJoints)
235  {
236  yError("received an invalid setControlMode message. Size of vector doesnÂ´t match the number of controlled joints\n");
237  *ok = false;

240  int *modes = new int [controlledJoints];

293  break;
294

301  break;
302 
303  case VOCAB_CM_PWM:
304  if (rpc_iCtrlMode)
305  *ok = rpc_iCtrlMode->setControlMode(axis, VOCAB_CM_PWM);
306  else
307  {
308  *rec = false;
309  *ok = false;
310  }
311  break;

331  break;

335  *ok = rpc_iCtrlMode->setControlMode(axis, VOCAB_CM_FORCE_IDLE);
336  else
337  {
338  *rec = false;
339  *ok = false;

341  break;
342 
343  default:
344  // if (ControlBoardWrapper_p->verbose())
345  yError("SET unknown controlMode : %s \n", cmd.toString().c_str());
346  *ok = false;

348  break;
349  }
350  }
351  break; // close default case
352  }
353  }
354  break; // close SET case
355 
356  case VOCAB_GET:
357  {
358  if (ControlBoardWrapper_p->verbose())
359  yDebug("GET command\n");
360 
361  int method = cmd.get(2).asVocab();
362 
363  switch(method)
364  {
365  case VOCAB_CM_CONTROL_MODES:
366  {
367  if (ControlBoardWrapper_p->verbose())
368  yDebug("getControlModes\n");
369  int *p = new int[controlledJoints];
370  for (int i = 0; i < controlledJoints; ++i) {
371  p[i] = -1;
372  }
373  if(rpc_iCtrlMode)
374  *ok = rpc_iCtrlMode->getControlModes(p);

385  *rec=true;

391  if (ControlBoardWrapper_p->verbose())
392  yDebug("getControlMode\n");
393 
394  int p=-1;
395  int axis = cmd.get(3).asInt32();
396  if(rpc_iCtrlMode)
397  *ok = rpc_iCtrlMode->getControlMode(axis, &p);

399  response.addVocab(VOCAB_IS);
400  response.addInt32(axis);
401  response.addVocab(p);
402 
403  //yError("Returning %d\n", p);
404  *rec=true;
405  }
406  break;
407 
408  case VOCAB_CM_CONTROL_MODE_GROUP:
409  {
410  if (ControlBoardWrapper_p->verbose())
411  yDebug("getControlMode group\n");
412 
413  int n_joints = cmd.get(3).asInt32();

422  }

426  {
427  *rec = false;
428  *ok = false;
429  }
430 
431  response.addVocab(VOCAB_IS);
432  response.addVocab(VOCAB_CM_CONTROL_MODE_GROUP);
433  Bottle& b = response.addList();
434  for(int i=0; i<n_joints; i++)
435  {
436  b.addVocab(modes[i]);
437  }
438 
439  delete[] js;
440  delete[] modes;
441 
442  //yDebugf("Returning %d\n", p);
443  *rec=true;
444  }
445  break;
446 
447  default:
448  yError("received a GET ICONTROLMODE command not understood");
449  break;
450  }
451  }
452 
453  lastRpcStamp.update();
454  appendTimeStamp(response, lastRpcStamp);
455  break; // case VOCAB_GET
456

458  {
459  *rec=false;
460  }
461  break;
462  }

466 void RPCMessagesParser::handleTorqueMsg(const yarp::os::Bottle& cmd,
467  yarp::os::Bottle& response, bool *rec, bool *ok)
468 {
469  if (ControlBoardWrapper_p->verbose())
470  yDebug("Handling ITorqueControl message\n");
471 
472  if (!rpc_ITorque)
473  {
474  yError("Error, I do not have a valid ITorque interface");
475  *ok=false;
476  return;
477  }
478 
479  int code = cmd.get(0).asVocab();
480  switch (code)
481  {
482  case VOCAB_SET:
483  {
484  *rec = true;
485  if (ControlBoardWrapper_p->verbose())
486  yDebug("set command received\n");
487 
488  switch(cmd.get(2).asVocab())
489  {
490  case VOCAB_REF:
491  {
492  *ok = rpc_ITorque->setRefTorque(cmd.get(3).asInt32(), cmd.get(4).asFloat64());
493  }
494  break;
495 
496  case VOCAB_MOTOR_PARAMS:
497  {
498  yarp::dev::MotorTorqueParameters params;
499  int joint = cmd.get(3).asInt32();
500  Bottle *b = cmd.get(4).asList();
501 
502  if (b==nullptr)
503  break;
504 
505  if (b->size() != 4)
506  {
507  yError("received a SET VOCAB_MOTOR_PARAMS command not understood, size!=4");
508  break;
509  }
510 
511  params.bemf = b->get(0).asFloat64();
512  params.bemf_scale = b->get(1).asFloat64();
513  params.ktau = b->get(2).asFloat64();
514  params.ktau_scale = b->get(3).asFloat64();
515 
516  *ok = rpc_ITorque->setMotorTorqueParams(joint, params);
517  }
518  break;
519 
520  case VOCAB_REFS:
521  {
522  Bottle *b = cmd.get(3).asList();
523  if (b==nullptr)
524  break;
525 
526  int i;
527  const int njs = b->size();
528  if (njs==controlledJoints)
529  {
530  auto* p = new double[njs]; // LATER: optimize to avoid allocation.
531  for (i = 0; i < njs; i++)
532  p[i] = b->get(i).asFloat64();
533  *ok = rpc_ITorque->setRefTorques (p);
534  delete[] p;
535  }
536  }
537  break;
538 
539  case VOCAB_TORQUE_MODE:
540  {
541  if(rpc_iCtrlMode)
542  {
543  int *modes = new int[controlledJoints];
544  for(int i=0; i<controlledJoints; i++)
545  modes[i] = VOCAB_CM_TORQUE;
546  *ok = rpc_iCtrlMode->setControlModes(modes);
547  delete [] modes;
548  }
549  else
550  {
551  *ok = false;
552  }
553  }
554  break;
555 
556  }
557  }
558  break;
559 
560  case VOCAB_GET:
561  {
562  *rec = true;
563  if (ControlBoardWrapper_p->verbose())
564  yDebug("get command received\n");
565  double dtmp = 0.0;
566  double dtmp2 = 0.0;
567  response.addVocab(VOCAB_IS);
568  response.add(cmd.get(1));
569 
570  switch(cmd.get(2).asVocab())
571  {
572  case VOCAB_AXES:
573  {
574  int tmp;
575  *ok = rpc_ITorque->getAxes(&tmp);
576  response.addInt32(tmp);
577  }
578  break;
579 
580  case VOCAB_TRQ:
581  {
582  *ok = rpc_ITorque->getTorque(cmd.get(3).asInt32(), &dtmp);
583  response.addFloat64(dtmp);
584  }
585  break;
586 
587  case VOCAB_MOTOR_PARAMS:
588  {
589  yarp::dev::MotorTorqueParameters params;
590  int joint = cmd.get(3).asInt32();
591 
592  // get the data
593  *ok = rpc_ITorque->getMotorTorqueParams(joint, &params);
594 
595  // convert it back to yarp message
596  Bottle& b = response.addList();
597 
598  b.addFloat64(params.bemf);
599  b.addFloat64(params.bemf_scale);
600  b.addFloat64(params.ktau);
601  b.addFloat64(params.ktau_scale);
602  }
603  break;
604  case VOCAB_RANGE:
605  {
606  *ok = rpc_ITorque->getTorqueRange(cmd.get(3).asInt32(), &dtmp, &dtmp2);
607  response.addFloat64(dtmp);
608  response.addFloat64(dtmp2);
609  }
610  break;
611 
612  case VOCAB_TRQS:
613  {
614  int i=0;
615  auto* p = new double[controlledJoints];
616  *ok = rpc_ITorque->getTorques(p);
617  Bottle& b = response.addList();
618  for (i = 0; i < controlledJoints; i++)
619  b.addFloat64(p[i]);
620  delete[] p;
621  }
622  break;
623 
624  case VOCAB_RANGES:
625  {
626  auto* p1 = new double[controlledJoints];
627  auto* p2 = new double[controlledJoints];
628  *ok = rpc_ITorque->getTorqueRanges(p1,p2);
629  Bottle& b1 = response.addList();
630  int i;
631  for (i = 0; i < controlledJoints; i++)
632  b1.addFloat64(p1[i]);
633  Bottle& b2 = response.addList();
634  for (i = 0; i < controlledJoints; i++)
635  b2.addFloat64(p2[i]);
636  delete[] p1;
637  delete[] p2;
638  }
639  break;
640 
641  case VOCAB_REFERENCE:
642  {
643  *ok = rpc_ITorque->getRefTorque(cmd.get(3).asInt32(), &dtmp);
644  response.addFloat64(dtmp);
645  }
646  break;
647 
648  case VOCAB_REFERENCES:
649  {
650  auto* p = new double[controlledJoints];
651  *ok = rpc_ITorque->getRefTorques(p);
652  Bottle& b = response.addList();
653  int i;
654  for (i = 0; i < controlledJoints; i++)
655  b.addFloat64(p[i]);
656  delete[] p;
657  }
658  break;
659  }
660  }
661  lastRpcStamp.update();
662  appendTimeStamp(response, lastRpcStamp);
663  break; // case VOCAB_GET
664  }
