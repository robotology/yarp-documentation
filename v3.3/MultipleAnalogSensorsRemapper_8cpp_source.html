88 // Return an empty list if the key is not found, and an error (false) if the key was found but it is not a list of strings
89 bool getVectorOfStringFromListInConfig(const std::string& key, const yarp::os::Searchable& config, std::vector<std::string> & vectorOfStrings)
90 {
91  yarp::os::Property prop;
92  prop.fromString(config.toString());
93  bool keyExists = prop.check(key);
94 
95  yarp::os::Bottle *propList=prop.find(key).asList();
96  if (!propList && keyExists)
97  {
98  yError() <<"MultipleAnalogSensorsRemapper : Error parsing parameters: if present " << key << " should be followed by a list of strings.\n";
99  return false;
100  }
101 
102  if (!propList && !keyExists)
103  {
104  vectorOfStrings.resize(0);
105  return true;
106  }
107 
108  vectorOfStrings.resize(propList->size());
109  for (size_t ax=0; ax < propList->size(); ax++)
110  {
111  vectorOfStrings[ax] = propList->get(ax).asString();
112  }
113 
114  return true;
115 }

117 bool MultipleAnalogSensorsRemapper::parseOptions(const Property& prop)
118 {
119  bool ok = true;
120 
121  m_remappedSensors.resize(MAS_NrOfSensorTypes);
122 
123  for (size_t i = 0; i < MAS_NrOfSensorTypes; i++)
124  {
125  auto sensType = static_cast<MAS_SensorType>(i);
126  std::string optionName = MAS_getTagFromEnum(sensType) +"Names";
127  ok = getVectorOfStringFromListInConfig(optionName , prop, m_remappedSensors[i]);
128  if (!ok)
129  {
130  yError() << "MultipleAnalogSensorsRemapper: " << optionName << " should be followed by a list of string.";
131  return false;
132  }
133  }
134 
135  return ok;
136 }

143 template<typename Interface>
144 bool MultipleAnalogSensorsRemapper::genericAttachAll(const MAS_SensorType sensorType,
145  std::vector<Interface *>& subDeviceVec,
146  const PolyDriverList &polylist,
147  bool (Interface::*getNameMethodPtr)(size_t, std::string&) const,
148  size_t (Interface::*getNrOfSensorsMethodPtr)() const)
149 {
150  std::map<std::string, SensorInSubDevice> sensorLocationMap;
151 
152  subDeviceVec.resize(polylist.size());
153 
154  for(int p=0; p<polylist.size(); p++)
155  {
156  // If this fails it is ok, this just means that this devices does not expose this kind of sensors
157  polylist[p]->poly->view(subDeviceVec[p]);
158 
159  if (subDeviceVec[p])
160  {
161  size_t nrOfSensorsInSubDevice = MAS_CALL_MEMBER_FN(subDeviceVec[p], getNrOfSensorsMethodPtr)();
162  for (size_t s=0; s < nrOfSensorsInSubDevice; s++)
163  {
164  std::string name;
165  bool ok = MAS_CALL_MEMBER_FN(subDeviceVec[p], getNameMethodPtr)(s,name);
166  if (!ok)
167  {
168  yError() << "MultipleAnalogSensorsRemapper: Failure in getting a name in the device " << polylist[p]->key;
169  return false;
170  }
171 
172  // If the name is already in the map, raise an error
173  if (sensorLocationMap.find(name) != sensorLocationMap.end())
174  {
175  SensorInSubDevice deviceWithSameName = sensorLocationMap.find(name)->second;
176  yError() << "MultipleAnalogSensorsRemapper: sensor ambiguity: sensor with name " << name
177  << " present on both device " << polylist[p]->key << " " << polylist[deviceWithSameName.subDevice]->key;
178  return false;
179  }
180 
181  sensorLocationMap[name] = SensorInSubDevice(p, s);
182  }
183  }
184  }
185 
186  // Fill the indices map given the name of all the subdevices
187  std::vector<SensorInSubDevice>& sensIndicesMap = m_indicesMap[static_cast<size_t>(sensorType)];
188  sensIndicesMap.resize(m_remappedSensors[sensorType].size());
189  for(size_t i=0; i < m_remappedSensors[sensorType].size(); i++)
190  {
191  std::string name = m_remappedSensors[sensorType][i];
192  if (sensorLocationMap.find(name) == sensorLocationMap.end())
193  {
194  yError() << "MultipleAnalogSensorsRemapper: impossible to find sensor name " << name << ", exiting.";
195  return false;
196  }
197 
198  sensIndicesMap[i] = sensorLocationMap.find(name)->second;
199  }
200 
201  return true;
202 }

206 bool MultipleAnalogSensorsRemapper::attachAll(const PolyDriverList &polylist)
207 {
208  bool ok = true;
209  m_indicesMap.resize(MAS_NrOfSensorTypes);
210  ok = ok && genericAttachAll(ThreeAxisGyroscopes, m_iThreeAxisGyroscopes, polylist,
211  &IThreeAxisGyroscopes::getThreeAxisGyroscopeName, &IThreeAxisGyroscopes::getNrOfThreeAxisGyroscopes);
212  ok = ok && genericAttachAll(ThreeAxisLinearAccelerometers, m_iThreeAxisLinearAccelerometers, polylist,
213  &IThreeAxisLinearAccelerometers::getThreeAxisLinearAccelerometerName, &IThreeAxisLinearAccelerometers::getNrOfThreeAxisLinearAccelerometers);
214  ok = ok && genericAttachAll(ThreeAxisMagnetometers, m_iThreeAxisMagnetometers, polylist,
215  &IThreeAxisMagnetometers::getThreeAxisMagnetometerName, &IThreeAxisMagnetometers::getNrOfThreeAxisMagnetometers);
216  ok = ok && genericAttachAll(OrientationSensors, m_iOrientationSensors, polylist,
217  &IOrientationSensors::getOrientationSensorName, &IOrientationSensors::getNrOfOrientationSensors);
218  ok = ok && genericAttachAll(TemperatureSensors, m_iTemperatureSensors, polylist,
219  &ITemperatureSensors::getTemperatureSensorName, &ITemperatureSensors::getNrOfTemperatureSensors);
220  ok = ok && genericAttachAll(SixAxisForceTorqueSensors, m_iSixAxisForceTorqueSensors, polylist,
221  &ISixAxisForceTorqueSensors::getSixAxisForceTorqueSensorName, &ISixAxisForceTorqueSensors::getNrOfSixAxisForceTorqueSensors);
222  ok = ok && genericAttachAll(ContactLoadCellArrays, m_iContactLoadCellArrays, polylist,
223  &IContactLoadCellArrays::getContactLoadCellArrayName, &IContactLoadCellArrays::getNrOfContactLoadCellArrays);
224  ok = ok && genericAttachAll(EncoderArrays, m_iEncoderArrays, polylist,
225  &IEncoderArrays::getEncoderArrayName, &IEncoderArrays::getNrOfEncoderArrays);
226  ok = ok && genericAttachAll(SkinPatches, m_iSkinPatches, polylist,
227  &ISkinPatches::getSkinPatchName, &ISkinPatches::getNrOfSkinPatches);
228 
229  return ok;
230 }

249 MAS_status MultipleAnalogSensorsRemapper::genericGetStatus(const MAS_SensorType sensorType,
250  size_t& sens_index,
251  const std::vector<Interface *>& subDeviceVec,
252  MAS_status (Interface::*methodPtr)(size_t) const) const
253 {
254  size_t nrOfAvailableSensors = m_indicesMap[sensorType].size();
255  if (sens_index >= nrOfAvailableSensors)
256  {
257  if (m_verbose)
258  {
259  yError("MultipleAnalogSensorsRemapper::genericGetStatus sens_index %zu out of range of available sensors (%zu).", sens_index, nrOfAvailableSensors);
260  }
261  return MAS_ERROR;
262  }
263 
264  SensorInSubDevice subDeviceSensor = m_indicesMap[sensorType][sens_index];
265  return MAS_CALL_MEMBER_FN(subDeviceVec[subDeviceSensor.subDevice], methodPtr)(subDeviceSensor.indexInSubDevice);
266 }
