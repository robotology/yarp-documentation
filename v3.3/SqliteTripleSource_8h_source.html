46  std::string condition(Triple& t, TripleContext *context) {
47  int rid = (context != nullptr) ? context->rid : -1;
48  std::string cond = "";
49  if (rid==-1) {
50  cond = "rid IS NULL";
51  } else {
52  cond = "rid = " + expressContext(context);
53  }
54  if (t.hasNs) {
55  if (t.ns!="*") {
56  char *query = nullptr;
57  query = sqlite3_mprintf(" AND ns = %Q",t.getNs());
58  cond = cond + query;
59  sqlite3_free(query);
60  }
61  } else {
62  cond += " AND ns IS NULL";
63  }
64  if (t.hasName) {
65  if (t.name!="*") {
66  char *query = nullptr;
67  query = sqlite3_mprintf(" AND name = %Q",t.getName());
68  cond = cond + query;
69  sqlite3_free(query);
70  }
71  } else {
72  cond += " AND name IS NULL";
73  }
74  if (t.hasValue) {
75  if (t.value!="*") {
76  char *query = nullptr;
77  query = sqlite3_mprintf(" AND value = %Q",t.getValue());
78  cond = cond + query;
79  sqlite3_free(query);
80  }
81  } else {
82  cond += " AND value IS NULL";
83  }
84  return cond;
85  }

141  std::list<Triple> query(Triple& ti, TripleContext *context) override {
142  std::list<Triple> q;
143  sqlite3_stmt *statement = nullptr;
144  char *query = nullptr;
145  query = sqlite3_mprintf("SELECT id, ns, name, value FROM tags WHERE %s",condition(ti,context).c_str());
146  if (verbose) {
147  printf("Query: %s\n", query);
148  }
149  int result = sqlite3_prepare_v2(db, query, -1, &statement, nullptr);
150  if (result!=SQLITE_OK) {
151  printf("Error in query\n");
152  }
153  while (result == SQLITE_OK && sqlite3_step(statement) == SQLITE_ROW) {
154  //int id = sqlite3_column_int(statement,0);
155  char *ns = (char *)sqlite3_column_text(statement,1);
156  char *name = (char *)sqlite3_column_text(statement,2);
157  char *value = (char *)sqlite3_column_text(statement,3);
158  Triple t;
159  if (ns != nullptr) {
160  t.ns = ns;
161  t.hasNs = true;
162  }
163  if (name != nullptr) {
164  t.name = name;
165  t.hasName = true;
166  }
167  if (value != nullptr) {
168  t.value = value;
169  t.hasValue = true;
170  }
171  q.push_back(t);
172  }
173  sqlite3_finalize(statement);
174  sqlite3_free(query);
175  return q;
176  }

210  void update(Triple& t, TripleContext *context) override {
211  char *msg = nullptr;
212  char *query = nullptr;

214  Triple t2(t);
215  t2.value = "*";

218  condition(t2,context).c_str());
219  } else {
220  query = sqlite3_mprintf("UPDATE tags SET value = %Q WHERE id = %Q",
221  t.getValue(),
222  expressContext(context).c_str());
223  }
224  if (verbose) {
225  printf("Query: %s\n", query);
226  }
227  int result = sqlite3_exec(db, query, nullptr, nullptr, &msg);
228  if (result!=SQLITE_OK) {
229  if (msg != nullptr) {
230  fprintf(stderr,"Error: %s\n", msg);
231  sqlite3_free(msg);
232  }
233  }

235  if (ct==0 && (t.hasName||t.hasNs)) {
236  insert(t,context);
