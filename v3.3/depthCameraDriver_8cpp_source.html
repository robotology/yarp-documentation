98 void streamFrameListener::onNewFrame(openni::VideoStream& stream)
99 {
100  std::lock_guard<std::mutex> guard(mutex);
101  stream.readFrame(&frameRef);
102 
103  if (!frameRef.isValid() || !frameRef.getData())
104  {
105  yInfo() << "frame lost";
106  return;
107  }
108 
109  int pixC;
110 
111  pixF = stream.getVideoMode().getPixelFormat();
112  pixC = depthCameraDriver::pixFormatToCode(pixF);
113  w = frameRef.getWidth();
114  h = frameRef.getHeight();
115  dataSize = frameRef.getDataSize();
116 
117  if (isReady == false)
118  {
119  isReady = true;
120  }
121 
122  if (pixC == VOCAB_PIXEL_INVALID)
123  {
124  yError() << "depthCameraDriver: Pixel Format not recognized";
125  return;
126  }
127 
128  image.setPixelCode(pixC);
129  image.resize(w, h);
130 
131  if (image.getRawImageSize() != (size_t) frameRef.getDataSize())
132  {
133  yError() << "depthCameraDriver:device and local copy data size doesn't match";
134  return;
135  }
136 
137  memcpy((void*)image.getRawImage(), (void*)frameRef.getData(), frameRef.getDataSize());
138  stamp.update();
139  return;
140 }

169 bool depthCameraDriver::initializeOpeNIDevice()
170 {
171  Status rc;
172 
173  rc = OpenNI::initialize();
174  if (rc != STATUS_OK)
175  {
176  yError() << "depthCameraDriver: Initialize failed," << OpenNI::getExtendedError();
177  return false;
178  }
179 
180  rc = m_device.open(ANY_DEVICE);
181  if (rc != STATUS_OK)
182  {
183  yError() << "depthCameraDriver: Couldn't open device," << OpenNI::getExtendedError();
184  return false;
185  }
186 
187  if (m_device.getSensorInfo(SENSOR_COLOR) != nullptr)
188  {
189  rc = m_imageStream.create(m_device, SENSOR_COLOR);
190  if (rc != STATUS_OK)
191  {
192  yError() << "depthCameraDriver: Couldn't create color stream," << OpenNI::getExtendedError();
193  return false;
194  }
195  }
196 
197  rc = m_imageStream.start();
198  if (rc != STATUS_OK)
199  {
200  yError() << "depthCameraDriver: Couldn't start the color stream," << OpenNI::getExtendedError();
201  return false;
202  }
203 
204  if (m_device.getSensorInfo(SENSOR_DEPTH) != nullptr)
205  {
206  rc = m_depthStream.create(m_device, SENSOR_DEPTH);
207  if (rc != STATUS_OK)
208  {
209  yError() << "depthCameraDriver: Couldn't create depth stream," << OpenNI::getExtendedError();
210  return false;
211  }
212  }
213 
214  if (m_depthRegistration)
215  {
216  if (m_device.isImageRegistrationModeSupported(IMAGE_REGISTRATION_DEPTH_TO_COLOR))
217  {
218  if (m_device.setImageRegistrationMode(IMAGE_REGISTRATION_DEPTH_TO_COLOR) == STATUS_OK)
219  {
220  yInfo() << "DepthCameraDriver:Depth successfully registered on rgb sensor";
221  }
222  else
223  {
224  yWarning() << "DepthCameraDriver: Depth registration failed.. sending unregistered images";
225  }
226  }
227  else
228  {
229  yWarning() << "DepthCameraDriver: depth image registration not supported by this device";
230  }
231  }
232 
233  rc = m_depthStream.start();
234  if (rc != STATUS_OK)
235  {
236  yError() << "depthCameraDriver: Couldn't start the depth stream," << OpenNI::getExtendedError();
237  return false;
238  }
239 
240  m_imageStream.addNewFrameListener(m_imageFrame);
241  m_depthStream.addNewFrameListener(m_depthFrame);
242 
243  return true;
244 }

253 bool depthCameraDriver::setParams()
254 {
255  bool ret;
256  ret = true;
257 
258  // Do all required settings
259 
260  //ACCURACY
261  if (params_map[accuracy].isSetting && ret)
262  {
263  if (!params_map[accuracy].val[0].isFloat64() )
264  settingErrorMsg("Param " + params_map[accuracy].name + " is not a double as it should be.", ret);
265 
266  if (! setDepthAccuracy(params_map[accuracy].val[0].asFloat64() ) )
267  settingErrorMsg("Setting param " + params_map[accuracy].name + " failed... quitting.", ret);
268  }
269 
270  //CLIP_PLANES
271  if (params_map[clipPlanes].isSetting && ret)
272  {
273  if (!params_map[clipPlanes].val[0].isFloat64() )
274  settingErrorMsg("Param " + params_map[clipPlanes].name + " is not a double as it should be.", ret);
275 
276  if (!params_map[clipPlanes].val[1].isFloat64() )
277  settingErrorMsg("Param " + params_map[clipPlanes].name + " is not a double as it should be.", ret);
278 
279  if (! setDepthClipPlanes(params_map[clipPlanes].val[0].asFloat64(), params_map[clipPlanes].val[1].asFloat64() ) )
280  settingErrorMsg("Setting param " + params_map[clipPlanes].name + " failed... quitting.", ret);
281  }
282 
283  //DEPTH_FOV
284  if (params_map[depth_Fov].isSetting && ret)
285  {
286  Value p1, p2;
287  p1 = params_map[depth_Fov].val[0];
288  p2 = params_map[depth_Fov].val[1];
289 
290  if (!p1.isFloat64() || !p2.isFloat64() )
291  settingErrorMsg("Param " + params_map[depth_Fov].name + " is not a double as it should be.", ret);
292 
293  if (! setDepthFOV(p1.asFloat64(), p2.asFloat64() ) )
294  settingErrorMsg("Setting param " + params_map[depth_Fov].name + " failed... quitting.", ret);
295  }
296 
297 
298 
299  //RGB_FOV
300  if (params_map[rgb_Fov].isSetting && ret)
301  {
302  Value p1, p2;
303  p1 = params_map[rgb_Fov].val[0];
304  p2 = params_map[rgb_Fov].val[1];
305 
306  if (!p1.isFloat64() || !p2.isFloat64() )
307  {
308  settingErrorMsg("Param " + params_map[rgb_Fov].name + " is not a double as it should be.", ret);
309  }
310 
311  if (! setRgbFOV(p1.asFloat64(), p2.asFloat64() ) )
312  {
313  settingErrorMsg("Setting param " + params_map[rgb_Fov].name + " failed... quitting.", ret);
314  }
315  }
316 
317  //DEPTH_RES
318  if (params_map[depthRes].isSetting && ret)
319  {
320  Value p1, p2;
321  p1 = params_map[depthRes].val[0];
322  p2 = params_map[depthRes].val[1];
323 
324  if (!p1.isInt32() || !p2.isInt32() )
325  {
326  settingErrorMsg("Param " + params_map[depthRes].name + " is not a int as it should be.", ret);
327  }
328 
329  if (! setDepthResolution(p1.asInt32(), p2.asInt32()))
330  {
331  settingErrorMsg("Setting param " + params_map[depthRes].name + " failed... quitting.", ret);
332  }
333  }
334 
335  //RGB_RES
336  if (params_map[rgbRes].isSetting && ret)
337  {
338  Value p1, p2;
339  p1 = params_map[rgbRes].val[0];
340  p2 = params_map[rgbRes].val[1];
341 
342  if (!p1.isInt32() || !p2.isInt32() )
343  {
344  settingErrorMsg("Param " + params_map[rgbRes].name + " is not a int as it should be.", ret);
345  }
346 
347  if (! setRgbResolution(p1.asInt32(), p2.asInt32()))
348  {
349  settingErrorMsg("Setting param " + params_map[rgbRes].name + " failed... quitting.", ret);
350  }
351  }
352 
353  // rgb MIRRORING
354  if (params_map[rgbMirroring].isSetting && ret)
355  {
356  //the device usually fail to set the mirror properties at the start.
357  //so we will try to set it for 5 times with a little delay before returning false
358  bool mirrorOk;
359  Value& v = params_map[rgbMirroring].val[0];
360  mirrorOk = false;
361 
362  if (!v.isBool())
363  {
364  settingErrorMsg("Param " + params_map[rgbMirroring].name + " is not a bool as it should be.", ret);
365  return false;
366  }
367 
368  for (int t = 0; t < 5; t++)
369  {
370  yInfo() << "depthCamera: trying to set rgb mirroring parameter for the" << t+1 << "time/s";
371  yarp::os::SystemClock::delaySystem(0.5);
372  if (setRgbMirroring(v.asBool()))
373  {
374  yInfo() << "depthCamera: rgb mirroring parameter set successfully";
375  mirrorOk = true;
376  break;
377  }
378  }
379  if (!mirrorOk)
380  {
381  settingErrorMsg("Setting param " + params_map[rgbMirroring].name + " failed... quitting.", ret);
382  }
383  }
384  // depth MIRRORING
385  if (params_map[depthMirroring].isSetting && ret)
386  {
387  //the device usually fail to set the mirror properties at the start.
388  //so we will try to set it for 5 times with a little delay before returning false
389  bool mirrorOk;
390  Value& v = params_map[depthMirroring].val[0];
391  mirrorOk = false;
392 
393  if (!v.isBool() )
394  {
395  settingErrorMsg("Param " + params_map[depthMirroring].name + " is not a bool as it should be.", ret);
396  }
397 
398  for (int t = 0; t < 5; t++)
399  {
400  yInfo() << "depthCamera: trying to set depth mirroring parameter for the" << t+1 << "time/s";
401  yarp::os::SystemClock::delaySystem(0.5);
402  if (setDepthMirroring(v.asBool()))
403  {
404  yInfo() << "depthCamera: depth mirroring parameter set successfully";
405  mirrorOk = true;
406  break;
407  }
408  }
409  if (!mirrorOk)
410  {
411  settingErrorMsg("Setting param " + params_map[depthMirroring].name + " failed... quitting.", ret);
412  }
413  }
414  return ret;
415 }

418 bool depthCameraDriver::open(Searchable& config)
419 {
420  std::vector<RGBDSensorParamParser::RGBDParam*> params;
421  for (auto& p:params_map)
422  {
423  params.push_back(&(p.second));
424  }
425 
426  if (!m_paramParser->parseParam(config, params))
427  {
428  yError()<<"depthCameraDriver: failed to parse the parameters";
429  return false;
430  }
431 
432  //"registered" is a hidden parameter for debugging pourpose
433  m_depthRegistration = !(config.check("registered") && config.find("registered").isBool() && config.find("registered").asBool() == false);
434 
435  if (!initializeOpeNIDevice())
436  {
437  return false;
438  }
439 
440  // setting Parameters
441  if (!setParams())
442  {
443  return false;
444  }
445 
446  return true;
447 }

564 bool depthCameraDriver::setDepthAccuracy(double _accuracy)
565 {
566  if (params_map[accuracy].isDescription)
567  {
568  return false;
569  }
570  bool a1, a2;
571  a1 = fabs(_accuracy - 0.001) < 0.00001;
572  a2 = fabs(_accuracy - 0.0001) < 0.00001;
573  if (!a1 && !a2)
574  {
575  yError() << "depthCameraDriver: supporting accuracy of 1mm (0.001) or 100um (0.0001) only at the moment";
576  return false;
577  }
578 
579  PixelFormat pf;
580  VideoMode vm;
581  bool ret;
582 
583  vm = m_imageStream.getVideoMode();
584  pf = fabs(_accuracy - 0.001) < 0.00001 ? PIXEL_FORMAT_DEPTH_1_MM : PIXEL_FORMAT_DEPTH_100_UM;
585 
586  vm.setPixelFormat(pf);
587  m_depthStream.stop();
588  ret = m_depthStream.setVideoMode(vm) == STATUS_OK;
589  RETURN_FALSE_STATUS_NOT_OK(m_depthStream.start());
590 
591  if (!ret)
592  {
593  yError() << OpenNI::getExtendedError();
594  }
595  return ret;
596 }
