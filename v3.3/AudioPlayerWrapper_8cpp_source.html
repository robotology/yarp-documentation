104 bool AudioPlayerWrapper::read(yarp::os::ConnectionReader& connection)
105 {
106  yarp::os::Bottle command;
107  yarp::os::Bottle reply;
108  bool ok = command.read(connection);
109  if (!ok) return false;
110  reply.clear();
111 
112  if (command.get(0).asString() == "start")
113  {
114  m_irender->startPlayback();
115  reply.addVocab(VOCAB_OK);
116  }
117  else if (command.get(0).asString() == "stop")
118  {
119  m_irender->stopPlayback();
120  reply.addVocab(VOCAB_OK);
121  }
122  else if (command.get(0).asString() == "clear")
123  {
124  m_irender->resetPlaybackAudioBuffer();
125  reply.addVocab(VOCAB_OK);
126  }
127  else if (command.get(0).asString() == "help")
128  {
129  reply.addVocab(yarp::os::Vocab::encode("many"));
130  reply.addString("start");
131  reply.addString("stop");
132  reply.addString("clear");
133  }
134  else
135  {
136  yError() << "Invalid command";
137  reply.addVocab(VOCAB_ERR);
138  }
139 
140  yarp::os::ConnectionWriter *returnToSender = connection.getWriter();
141  if (returnToSender != nullptr)
142  {
143  reply.write(*returnToSender);
144  }
145  return true;
146 }

153 bool AudioPlayerWrapper::open(yarp::os::Searchable &config)
154 {
155  Property params;
156  params.fromString(config.toString());
157 
158  if (config.check("debug"))
159  {
160  m_debug_enabled = true;
161  }
162 
163  if (config.check("period"))
164  {
165  m_period = config.find("period").asFloat64();
166  }
167 
168  string name = "/audioPlayerWrapper";
169  if (config.check("name"))
170  {
171  name = config.find("name").asString();
172  }
173  m_audioInPortName = name + "/audio:i";
174  m_rpcPortName = name + "/rpc:i";
175  m_statusPortName = name + "/status:o";
176 
177  if(!initialize_YARP(config) )
178  {
179  yError() << "AudioPlayerWrapper: Error initializing YARP ports";
180  return false;
181  }
182 
183  if (config.check("playback_network_buffer_lenght"))
184  {
185  m_buffer_delay = config.find("playback_network_buffer_lenght").asFloat64();
186  }
187  yInfo() << "Using a 'playback_network_buffer_lenght' of" << m_buffer_delay << "s";
188  yInfo() << "Increase this value to robustify the real-time audio stream (it will increase latency too)";
189 
190  if(config.check("subdevice"))
191  {
192  Property p;
193  PolyDriverList driverlist;
194  p.fromString(config.toString(), false);
195  p.put("device", config.find("subdevice").asString());
196 
197  if(!m_driver.open(p) || !m_driver.isValid())
198  {
199  yError() << "audioPlayerWrapper: failed to open subdevice.. check params";
200  return false;
201  }
202 
203  driverlist.push(&m_driver, "1");
204  if(!attachAll(driverlist))
205  {
206  yError() << "audioPlayerWrapper: failed to open subdevice.. check params";
207  return false;
208  }
209  m_isDeviceOwned = true;
210  }
211 
212 
213  if (m_irender == nullptr)
214  {
215  yError("m_irender is null\n");
216  return false;
217  }
218 
219  bool b=m_irender->getPlaybackAudioBufferMaxSize(m_max_buffer_size);
220  if (!b)
221  {
222  yError("getPlaybackAudioBufferMaxSize failed\n");
223  return false;
224  }
225 
226  return true;
227 }

260 void AudioPlayerWrapper::run()
261 {
262  double current_time = yarp::os::Time::now();
263 
264  Sound* s = m_audioInPort.read(false);
265  if (s != nullptr)
266  {
267  scheduled_sound_type ss;
268 #if 1
269  //This is simple, but we don't know how big the sound is...
270  ss.scheduled_time = current_time + m_buffer_delay;
271 #elif 0
272  //This is ok, but it doesn't work if the sounds have different durations...
273  ss.scheduled_time = current_time + 5.0 * s.getDuration();
274 #else
275  ss.scheduled_time = current_time + m_buffer_delay > 5.0 * s.getDuration() ? (m_buffer_delay) : (5.0 * s.getDuration());
276 #endif
277  ss.sound_data = *s;
278  m_sound_buffer.push(ss);
279  }
280 
281  if (!m_sound_buffer.empty() && current_time > m_sound_buffer.front().scheduled_time)
282  {
283  m_irender->renderSound(m_sound_buffer.front().sound_data);
284  m_sound_buffer.pop();
285  }
286 
287  m_irender->getPlaybackAudioBufferCurrentSize(m_current_buffer_size);
288  if (m_debug_enabled)
289  {
290  static double printer_wdt = yarp::os::Time::now();
291  if (yarp::os::Time::now() - printer_wdt > 1.0)
292  {
293  yDebug() << m_current_buffer_size.getSamples() << "/" << m_max_buffer_size.getSamples() << "samples";
294  printer_wdt = yarp::os::Time::now();
295  }
296  }
297 }
