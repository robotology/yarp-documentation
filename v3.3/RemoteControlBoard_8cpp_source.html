117 bool RemoteControlBoard::checkProtocolVersion(bool ignore)
118 {
119  bool error=false;
120  // verify protocol
121  Bottle cmd, reply;
122  cmd.addVocab(VOCAB_GET);
123  cmd.addVocab(VOCAB_PROTOCOL_VERSION);
124  rpc_p.write(cmd, reply);
125 
126  // check size and format of messages, expected [prot] int int int [ok]
127  if (reply.size()!=5)
128  error=true;
129 
130  if (reply.get(0).asVocab()!=VOCAB_PROTOCOL_VERSION)
131  error=true;
132 
133  if (!error)
134  {
135  protocolVersion.major=reply.get(1).asInt32();
136  protocolVersion.minor=reply.get(2).asInt32();
137  protocolVersion.tweak=reply.get(3).asInt32();
138 
139  //verify protocol
140  if (protocolVersion.major!=PROTOCOL_VERSION_MAJOR)
141  error=true;
142 
143  if (protocolVersion.minor!=PROTOCOL_VERSION_MINOR)
144  error=true;
145  }
146 
147  if (!error)
148  return true;
149 
150  // protocol did not match
151  yError("expecting protocol %d %d %d, but the device we are connecting to has protocol version %d %d %d\n",
152  PROTOCOL_VERSION_MAJOR, PROTOCOL_VERSION_MINOR, PROTOCOL_VERSION_TWEAK,
153  protocolVersion.major, protocolVersion.minor, protocolVersion.tweak);
154 
155  bool ret;
156  if (ignore)
157  {
158  yWarning(" ignoring error but please update YARP or the remotecontrolboard implementation\n");
159  ret = true;
160  }
161  else
162  {
163  yError(" please update YARP or the remotecontrolboard implementation\n");
164  ret = false;
165  }
166 
167  return ret;
168 }

170 bool RemoteControlBoard::open(Searchable& config)
171 {
172  remote = config.find("remote").asString();
173  local = config.find("local").asString();
174 
175  if (config.check("timeout"))
176  {
177  extendedIntputStatePort.setTimeout(config.find("timeout").asFloat64());
178  }
179  // check the Qos perefernces if available (local and remote)
180  yarp::os::QosStyle localQos;
181  if (config.check("local_qos")) {
182  Bottle& qos = config.findGroup("local_qos");
183  if(qos.check("thread_priority"))
184  localQos.setThreadPriority(qos.find("thread_priority").asInt32());
185  if(qos.check("thread_policy"))
186  localQos.setThreadPolicy(qos.find("thread_policy").asInt32());
187  if(qos.check("packet_priority"))
188  localQos.setPacketPriority(qos.find("packet_priority").asString());
189  }
190 
191  yarp::os::QosStyle remoteQos;
192  if (config.check("remote_qos")) {
193  Bottle& qos = config.findGroup("remote_qos");
194  if(qos.check("thread_priority"))
195  remoteQos.setThreadPriority(qos.find("thread_priority").asInt32());
196  if(qos.check("thread_policy"))
197  remoteQos.setThreadPolicy(qos.find("thread_policy").asInt32());
198  if(qos.check("packet_priority"))
199  remoteQos.setPacketPriority(qos.find("packet_priority").asString());
200  }
201 
202  bool writeStrict_isFound = config.check("writeStrict");
203  if(writeStrict_isFound)
204  {
205  Value &gotStrictVal = config.find("writeStrict");
206  if(gotStrictVal.asString() == "on")
207  {
208  writeStrict_singleJoint = true;
209  writeStrict_moreJoints = true;
210  yInfo("RemoteControlBoard is ENABLING the writeStrict option for all commands\n");
211  }
212  else if(gotStrictVal.asString() == "off")
213  {
214  writeStrict_singleJoint = false;
215  writeStrict_moreJoints = false;
216  yInfo("RemoteControlBoard is DISABLING the writeStrict opition for all commands\n");
217  }
218  else
219  yError("Found writeStrict opition with wrong value. Accepted options are 'on' or 'off'\n");
220  }
221 
222  if (local=="") {
223  yError("Problem connecting to remote controlboard, 'local' port prefix not given\n");
224  return false;
225  }
226 
227  if (remote=="") {
228  yError("Problem connecting to remote controlboard, 'remote' port name not given\n");
229  return false;
230  }
231 
232  std::string carrier =
233  config.check("carrier",
234  Value("udp"),
235  "default carrier for streaming robot state").asString();
236 
237  bool portProblem = false;
238  if (local != "") {
239  std::string s1 = local;
240  s1 += "/rpc:o";
241  if (!rpc_p.open(s1)) { portProblem = true; }
242  s1 = local;
243  s1 += "/command:o";
244  if (!command_p.open(s1)) { portProblem = true; }
245  s1 = local;
246  s1 += "/stateExt:i";
247  if (!extendedIntputStatePort.open(s1)) { portProblem = true; }
248  if (!portProblem)
249  {
250  extendedIntputStatePort.useCallback();
251  }
252  }
253 
254  bool connectionProblem = false;
255  if (remote != "" && !portProblem)
256  {
257  std::string s1 = remote;
258  s1 += "/rpc:i";
259  std::string s2 = local;
260  s2 += "/rpc:o";
261  bool ok = false;
262  // RPC port needs to be tcp, therefore no carrier option is added here
263  // ok=Network::connect(s2.c_str(), s1.c_str()); //This doesn't take into consideration possible YARP_PORT_PREFIX on local ports
264  // ok=Network::connect(rpc_p.getName(), s1.c_str()); //This should work also with YARP_PORT_PREFIX because getting back the name of the port will return the modified name
265  ok=rpc_p.addOutput(s1); //This works because we are manipulating only remote side and let yarp handle the local side
266  if (!ok) {
267  yError("Problem connecting to %s, is the remote device available?\n", s1.c_str());
268  connectionProblem = true;
269  }
270 
271  s1 = remote;
272  s1 += "/command:i";
273  s2 = local;
274  s2 += "/command:o";
275  //ok = Network::connect(s2.c_str(), s1.c_str(), carrier);
276  // ok=Network::connect(command_p.getName(), s1.c_str(), carrier); //doesn't take into consideration possible YARP_PORT_PREFIX on local ports
277  ok = command_p.addOutput(s1, carrier);
278  if (!ok) {
279  yError("Problem connecting to %s, is the remote device available?\n", s1.c_str());
280  connectionProblem = true;
281  }
282  // set the QoS preferences for the 'command' port
283  if (config.check("local_qos") || config.check("remote_qos"))
284  NetworkBase::setConnectionQos(command_p.getName(), s1, localQos, remoteQos, false);
285 
286  s1 = remote;
287  s1 += "/stateExt:o";
288  s2 = local;
289  s2 += "/stateExt:i";
290  // not checking return value for now since it is wip (different machines can have different compilation flags
291  ok = Network::connect(s1, extendedIntputStatePort.getName(), carrier);
292  if (ok)
293  {
294  // set the QoS preferences for the 'state' port
295  if (config.check("local_qos") || config.check("remote_qos"))
296  NetworkBase::setConnectionQos(s1, extendedIntputStatePort.getName(), remoteQos, localQos, false);
297  }
298  else
299  {
300  yError("Problem connecting to %s, is the remote device available?\n", s1.c_str());
301  connectionProblem = true;
302  }
303  }
304 
305  if (connectionProblem||portProblem) {
306 
307  rpc_p.close();
308  command_p.close();
309  extendedIntputStatePort.close();
310  return false;
311  }
312 
313  state_buffer.setStrict(false);
314  command_buffer.attach(command_p);
315 
316  if (!checkProtocolVersion(config.check("ignoreProtocolCheck")))
317  {
318  yError() << "checkProtocolVersion failed";
319  command_buffer.detach();
320  rpc_p.close();
321  command_p.close();
322  extendedIntputStatePort.close();
323  return false;
324  }
325 
326  if (!isLive()) {
327  if (remote!="") {
328  yError("Problems with obtaining the number of controlled axes\n");
329  command_buffer.detach();
330  rpc_p.close();
331  command_p.close();
332  extendedIntputStatePort.close();
333  return false;
334  }
335  }
336 
337  if (config.check("diagnostic"))
338  {
339  diagnosticThread = new DiagnosticThread(DIAGNOSTIC_THREAD_PERIOD);
340  diagnosticThread->setOwner(&extendedIntputStatePort);
341  diagnosticThread->start();
342  }
343  else
344  diagnosticThread=nullptr;
345 
346  // allocate memory for helper struct
347  // single joint
348  last_singleJoint.jointPosition.resize(1);
349  last_singleJoint.jointVelocity.resize(1);
350  last_singleJoint.jointAcceleration.resize(1);
351  last_singleJoint.motorPosition.resize(1);
352  last_singleJoint.motorVelocity.resize(1);
353  last_singleJoint.motorAcceleration.resize(1);
354  last_singleJoint.torque.resize(1);
355  last_singleJoint.pwmDutycycle.resize(1);
356  last_singleJoint.current.resize(1);
357  last_singleJoint.controlMode.resize(1);
358  last_singleJoint.interactionMode.resize(1);
359 
360  // whole part (safe here because we already got the nj
361  last_wholePart.jointPosition.resize(nj);
362  last_wholePart.jointVelocity.resize(nj);
363  last_wholePart.jointAcceleration.resize(nj);
364  last_wholePart.motorPosition.resize(nj);
365  last_wholePart.motorVelocity.resize(nj);
366  last_wholePart.motorAcceleration.resize(nj);
367  last_wholePart.torque.resize(nj);
368  last_wholePart.current.resize(nj);
369  last_wholePart.pwmDutycycle.resize(nj);
370  last_wholePart.controlMode.resize(nj);
371  last_wholePart.interactionMode.resize(nj);
372  return true;
373 }
