149 void TestFrameGrabber::timing() {
150  double now = yarp::os::Time::now();
151 
152  if (now-prev>1000) {
153  first = now;
154  prev = now;
155  }
156  double dt = period-(now-prev);
157 
158  if (dt>0) {
159  yarp::os::Time::delay(dt);
160  }
161 
162  // this is the controlled instant when we consider the
163  // image as going out
164  prev += period;
165 }

273 void TestFrameGrabber::printTime(unsigned char* pixbuf, int pixbuf_w, int pixbuf_h, int x, int y, char* s, int size)
274 {
275  int pixelsize = 5;
276  for (int i = 0; i<size; i++)
277  {
278  char* num_p = nullptr;
279  switch (s[i])
280  {
281  case '0': num_p = num[0].data; break;
282  case '1': num_p = num[1].data; break;
283  case '2': num_p = num[2].data; break;
284  case '3': num_p = num[3].data; break;
285  case '4': num_p = num[4].data; break;
286  case '5': num_p = num[5].data; break;
287  case '6': num_p = num[6].data; break;
288  case '7': num_p = num[7].data; break;
289  case '8': num_p = num[8].data; break;
290  case '9': num_p = num[9].data; break;
291  case ' ': num_p = num[10].data; break;
292  case '.': num_p = num[11].data; break;
293  }
294 
295  for (int yi = 0; yi<5; yi++)
296  for (int xi = 0; xi<3; xi++)
297  {
298  int ii = yi * 3 + xi;
299  if (num_p[ii] == '*')
300  {
301  for (int r = yi * pixelsize; r<yi*pixelsize + pixelsize; r++)
302  {
303  int off = i * (pixelsize + 20);
304  for (int c = xi * pixelsize + off; c<xi*pixelsize + pixelsize + off; c++)
305  {
306  if (c >= pixbuf_h ||
307  r >= pixbuf_w)
308  {
309  //avoid drawing out of the image memory
310  return;
311  }
312  unsigned char *pixel = pixbuf;
313  size_t offset = c * 3 + r * (pixbuf_w * 3);
314  pixel = pixel + offset;
315  pixel[0] = 0;
316  pixel[1] = 0;
317  pixel[2] = 255;
318  }
319  }
320  }
321  }
322  }

325 void TestFrameGrabber::createTestImage(yarp::sig::ImageOf<yarp::sig::PixelRgb>&
326  image) {
327  // to test IPreciselyTimed, make timestamps be mysteriously NNN.NNN42
328  double t = Time::now();
329  t -= ((t*1000)-(int)t)/1000;
330  t+= 0.00042;
331  stamp.update(t);
332  if (background.width()>0) {
333  image.copy(background);
334  } else {
335  image.resize(w,h);
336  image.zero();
337  }
338  switch (mode) {
339  case VOCAB_TIMETEXT:
340  {
341  char txtbuf[50];
342  double time = yarp::os::Time::now() - start_time;
343  snprintf(txtbuf, 50, "%.3f", time);
344  int len = strlen(txtbuf);
345  if (len < 20)
346  {
347  printTime((unsigned char*)image.getRawImage(), image.width(), image.height(), 0, 0, txtbuf, len);
348  }
349  }
350  break;
351  case VOCAB_BALL:
352  {
353  addCircle(image,PixelRgb(0,255,0),bx,by,15);
354  addCircle(image,PixelRgb(0,255,255),bx,by,8);
355  if (ct%5!=0) {
356  rnd *= 65537;
357  rnd += 17;
358  bx += (rnd%5)-2;
359  rnd *= 65537;
360  rnd += 17;
361  by += (rnd%5)-2;
362  } else {
363  int dx = w/2 - bx;
364  int dy = h/2 - by;
365  if (dx>0) { bx++; }
366  if (dx<0) { bx--; }
367  if (dy>0) { by++; }
368  if (dy<0) { by--; }
369  }
370  }
371  break;
372  case VOCAB_GRID:
373  {
374  size_t ww = image.width();
375  size_t hh = image.height();
376  if (ww>1&&hh>1) {
377  for (size_t x=0; x<ww; x++) {
378  for (size_t y=0; y<hh; y++) {
379  double xx = ((double)x)/(ww-1);
380  double yy = ((double)y)/(hh-1);
381  int r = int(0.5+255*xx);
382  int g = int(0.5+255*yy);
383  bool act = (y==ct);
384  image.pixel(x,y) = PixelRgb(r,g,act*255);
385  }
386  }
387  }
388  }
389  break;
390  case VOCAB_GRID_MULTISIZE:
391  {
392  static int count = 0;
393  count++;
394  if (count== 100)
395  {
396  yDebug() << "size 100, 100";
397  image.resize(100,100);
398  }
399  else if (count == 200)
400  {
401  yDebug() << "size 200, 100";
402  image.resize(200, 100);
403  }
404  else if (count == 300)
405  {
406  yDebug() << "size 300, 50";
407  image.resize(300, 50);
408  count = 0;
409  }
410 
411  size_t ww = w = image.width();
412  size_t hh = h = image.height();
413  if (ww>1 && hh>1) {
414  for (size_t x = 0; x<ww; x++) {
415  for (size_t y = 0; y<hh; y++) {
416  double xx = ((double)x) / (ww - 1);
417  double yy = ((double)y) / (hh - 1);
418  int r = int(0.5 + 255 * xx);
419  int g = int(0.5 + 255 * yy);
420  bool act = (y == ct);
421  image.pixel(x, y) = PixelRgb(r, g, act * 255);
422  }
423  }
424  }
425  }
426  break;
427  case VOCAB_LINE:
428  default:
429  {
430  for (size_t i=0; i<image.width(); i++) {
431  image.pixel(i,ct).r = 255;
432  }
433  char ttxt[50];
434  sprintf(ttxt, "%021.10f", t);

511 bool TestFrameGrabber::makeSimpleBayer(
512  ImageOf<PixelRgb>& img,
513  ImageOf<PixelMono>& bayer) {
514 
515  bayer.resize(img.width(), img.height());

517  const size_t w = img.width();
518  const size_t h = img.height();
519 
520  size_t i, j;
521  for (i = 0; i < h; i++) {
522  auto* row = (PixelRgb *)img.getRow(i);
523  auto* rd = (PixelMono *)bayer.getRow(i);
524 
525  for (j = 0; j < w; j++) {
526 
527  if ((i%2) == 0) {
528  switch (j%4) {
529  case 0:
530  case 2:
531  *rd++ = row->g;
532  row++;
533  break;
534 
535  case 1:
536  case 3:
537  *rd++ = row->r;
538  row++;
539  break;
540  }
541  }
542 
543  if ((i%2) == 1) {
544  switch (j%4) {
545  case 1:
546  case 3:
547  *rd++ = row->g;
548  row++;
549  break;
550 
551  case 0:
552  case 2:
553  *rd++ = row->b;
554  row++;
555  break;
556  }
557  }
