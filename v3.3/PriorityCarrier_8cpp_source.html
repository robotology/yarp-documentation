123 double PriorityCarrier::getActualStimulation(double t)
124 {
125  //
126  // +P| ---___
127  // | | -__
128  // | | -_
129  // 0|------------------> (Active state)
130  // Ta Tc
131  //
132  // Ta Tr
133  // 0|-------------_----> (Resting state && P<0)
134  // | | __-
135  // | | ___-
136  // -P| ---
137  //
138  //
139  // P(t) = Pi * (1-exp((t-Tc-Ta)/Tc*5) + exp(-5))
140  // t:time, Pi: temporal Priority level
141  // Tc: reset time, Ta: arrival time
142  //
143  // we do not consider ports which has not seen any message
144  // from them yet.
145  //if(timeArrival == 0)
146  // return 0;
147 
148  double dt = t - timeArrival;
149  // Temporal priority is inverted if this is a neuron model and the temporal
150  // stimulation has already reached to STIMUL_THRESHOLD and waited for Tc.
151  if((timeResting > 0)
152  && (dt >= fabs(timeConstant))
153  && (temporalStimulation >= STIMUL_THRESHOLD))
154  temporalStimulation = -temporalStimulation;
155 
156  double actualStimulation;
157  if(!isResting(temporalStimulation)) // behavior is in stimulation state
158  {
159  // After a gap bigger than Tc, the
160  // priority is set to zero to avoid redundant calculation.
161  if(dt > fabs(timeConstant))
162  actualStimulation = 0;
163  else
164  actualStimulation = temporalStimulation *
165  (1.0 - exp((dt-timeConstant)/timeConstant*5.0) + exp(-5.0));
166  }
167  else // behavior is in resting state
168  {
169  // it is in waiting state for Tc
170  if(temporalStimulation > 0)
171  actualStimulation = temporalStimulation;
172  else
173  {
174  dt -= fabs(timeConstant);
175  // After a gap bigger than Tr, the
176  // priority is set to zero to avoid redundant calculation.
177  if(dt > fabs(timeResting))
178  actualStimulation = 0;
179  else
180  actualStimulation = temporalStimulation *
181  (1.0 - exp((dt-timeResting)/timeResting*5.0) + exp(-5.0));
182  }
183  }
184 
185  if(actualStimulation <= 0)
186  isActive = false;
187 
188  return actualStimulation;
189 }

191 double PriorityCarrier::getActualInput(double t)
192 {
193  // calculating E(t) = Sum(e.I(t)) + b
194  if(!isActive)
195  return 0.0;
196 
197  double E = 0;
198  for (auto& it : group->peerSet)
199  {
200  PriorityCarrier *peer = it.first;
201  if(peer != this)
202  {
203  for(size_t i=0; i<peer->excitation.size(); i++)
204  {
205  Value v = peer->excitation.get(i);
206  if(v.isList() && (v.asList()->size()>=2))
207  {
208  Bottle* b = v.asList();
209  // an exitatory to this priority carrier
210  if(sourceName == b->get(0).asString())
211  E += peer->getActualInput(t) * (b->get(1).asFloat64()/10.0);
212  }
213  }
214 
215  }
216  }
217  E += bias;
218  double I = E * getActualStimulation(t);
219  return ((I<0) ? 0 : I); //I'(t)
220 }

227 bool PriorityGroup::recalculate(double t)
228 {
229  //TODO: find the correct way to get the size of peerSet
230  int nConnections = 0;
231  for(auto it=peerSet.begin(); it!=peerSet.end(); it++)
232  nConnections++;
233 
234  // calculate matrices X, B, InvA and Y
235  X.resize(nConnections, 1);
236  B.resize(nConnections, 1);
237  Y.resize(nConnections, 1);
238  InvA.resize(nConnections, nConnections);
239  InvA.eye();
240 
241  int row = 0;
242  for(auto& rowItr : peerSet)
243  {
244  PriorityCarrier* peer = rowItr.first;
245  // call 'getActualStimulation' to update 'isActive'
246  peer->getActualStimulation(t);
247  double xi = (peer->isActive) ? STIMUL_THRESHOLD : 0.0;
248  B(row,0) = peer->bias * xi;
249  X(row,0) = xi;
250 
251  int col = 0;
252  for(auto& it : peerSet)
253  {
254  PriorityCarrier *peerCol = it.first;
255  for(size_t i=0; i<peerCol->excitation.size(); i++)
256  {
257  Value v = peerCol->excitation.get(i);
258  if(v.isList() && (v.asList()->size()>=2))
259  {
260  Bottle* b = v.asList();
261  // an exitatory link to this connection
262  if(peer->sourceName == b->get(0).asString())
263  InvA(row,col) = -(b->get(1).asFloat64()/10.0)*xi;
264  }
265  }
266  col++;
267  }
268  row++;
269  }
270 
271  //fprintf(stdout, "A:\n %s\n", InvA.toString(1).c_str());
272 
273  // calclulating the determinant
274  double determinant = yarp::math::det(InvA);
275  if(determinant == 0)
276  {
277  yError("Inconsistent regulation! non-invertible weight matrix");
278  return false;
279  }
280 
281  // inverting the weight matrix
282  InvA = yarp::math::luinv(InvA);
283  Y = InvA * B;
284 
285  //fprintf(stdout, "X:\n %s\n", X.toString(1).c_str());
286  //fprintf(stdout, "B:\n %s\n", B.toString(1).c_str());
287  //fprintf(stdout, "Y:\n %s\n", Y.toString(1).c_str());
288 
289  return true;
290 }

292 // Decide whether data should be accepted, for real.
293 bool PriorityGroup::acceptIncomingData(yarp::os::ConnectionReader& reader,
294  PriorityCarrier *source) {
295 
296  bool accept;
297  // updates message's arrival time
298  double tNow = yarp::os::Time::now();
299  source->stimulate(tNow);
300 
301  if(!recalculate(tNow))
302  return false;
303 
304  int row = 0;
305  PriorityCarrier *maxPeer = nullptr;
306  double maxStimuli = 0.0;
307  for(auto& it : peerSet)
308  {
309  PriorityCarrier *peer = it.first;
310  double output = Y(row,0) * X(row,0);
311  peer->yi = output; // only for debug purpose
312 
313  if(!peer->isVirtual)
314  {
315  if(output > maxStimuli)
316  {
317  maxStimuli = output;
318  maxPeer = peer;
319  }
320  }
321  row++;
322  }
323  accept = (maxPeer == source);
324 
325  // a virtual message will never be delivered. It will be only
326  // used for the coordination
327  if(source->isVirtual)
328  accept = false;
329 
330  return accept;
331 }
