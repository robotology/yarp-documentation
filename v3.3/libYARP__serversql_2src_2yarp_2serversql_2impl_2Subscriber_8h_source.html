90  virtual bool apply(yarp::os::Bottle& cmd,
91  yarp::os::Bottle& reply,
92  yarp::os::Bottle& event,
93  const yarp::os::Contact& remote) override {
94  std::string tag = cmd.get(0).asString();
95  bool ok = false;
96  if (tag=="subscribe"||tag=="unsubscribe"||tag=="announce"||
97  tag=="topic"||tag=="untopic"||tag=="type") {
98  if (!silent) printf("-> %s\n", cmd.toString().c_str());
99  }
100  if (tag=="subscribe") {
101  std::string src = cmd.get(1).asString();
102  std::string dest = cmd.get(2).asString();
103  std::string mode = cmd.get(3).asString();
104  if (dest!="") {
105  ok = addSubscription(src,
106  dest,
107  mode);
108  reply.clear();
109  reply.addVocab(ok?yarp::os::createVocab('o','k'):yarp::os::createVocab('f','a','i','l'));
110  return ok;
111  } else {
112  // list subscriptions
113  listSubscriptions(src,reply);
114  }
115  return true;
116  }
117  if (tag=="unsubscribe") {
118  ok = removeSubscription(cmd.get(1).asString().c_str(),
119  cmd.get(2).asString().c_str());
120  reply.clear();
121  reply.addVocab(ok?yarp::os::createVocab('o','k'):yarp::os::createVocab('f','a','i','l'));
122  return ok;
123  }
124  if (tag=="announce") {
125  if (cmd.get(2).isInt32()) {
126  welcome(cmd.get(1).asString().c_str(),cmd.get(2).asInt32()?1:0);
127  } else {
128  welcome(cmd.get(1).asString().c_str(),true);
129  }
130  reply.clear();
131  reply.addVocab(yarp::os::createVocab('o','k'));
132  return true;
133  }
134  if (tag=="topic") {
135  if (cmd.size()>=2) {
136  bool result = setTopic(cmd.get(1).asString().c_str(),
137  cmd.get(2).asString().c_str(),true);
138  reply.clear();
139  reply.addVocab(replyCode(result));
140  return true;
141  } else {
142  reply.clear();
143  listTopics(reply);
144  return true;
145  }
146  }
147  if (tag=="type") {
148  if (cmd.size()==4) {
149  bool result = setType(cmd.get(1).asString().c_str(),
150  cmd.get(2).asString().c_str(),
151  cmd.get(3).asString().c_str());
152  reply.clear();
153  reply.addVocab(replyCode(result));
154  return true;
155  } else if (cmd.size()==3) {
156  std::string result =
157  getType(cmd.get(1).asString().c_str(),
158  cmd.get(2).asString().c_str());
159  reply.clear();
160  if (result=="") {
161  reply.addVocab(replyCode(false));
162  } else {
163  reply.addString(cmd.get(0).asString());
164  reply.addString(cmd.get(1).asString());
165  reply.addString(cmd.get(2).asString());
166  reply.addString(result);
167  }
168  return true;
169  } else {
170  reply.clear();
171  reply.addVocab(replyCode(false));
172  return true;
173  }
174  }
175  if (tag=="untopic") {
176  bool result = setTopic(cmd.get(1).asString().c_str(),"",false);
177  reply.clear();
178  reply.addVocab(replyCode(result));
179  return true;
180  }
181  return ok;
182  }
