148return_transcribe ISpeechTranscriptionMsgsd::transcribe(const yarp::sig::Sound& sound)
149{
150 std::lock_guard <std::mutex> lg(m_mutex);
151 return_transcribe ret;
152 if (m_isptr)
153 {
154 std::string transcription;
155 double score;
156 ReturnValue b = m_isptr->transcribe(sound, transcription, score);
157 ret.ret = b;
158 ret.transcription = transcription;
159 ret.score = score;
160
161 if (m_output_port && m_output_port->getOutputCount()>0)
162 {
163 yarp::os::Bottle bot;
164 bot.addString(transcription);
165 bot.addFloat64(score);
166 m_output_port->write(bot);
167 }
168
169 return ret;
170 }
171 yCError(SPEECHTR_NWS, "ISpeechTranscription interface was not set");
172 ret.ret = ReturnValue::return_code::return_value_error_generic;
173 return ret;
174}
