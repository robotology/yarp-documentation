16bool ChatBot_nwc_yarp::open(yarp::os::Searchable &config)
17{
18 if (!parseParams(config)) { return false; }
19
20 if (!m_thriftClientPort.open(m_local))
21 {
22 yCError(CHATBOT_NWC_YARP) << "Cannot open rpc port, check network";
23
24 return false;
25 }
26
27 bool ok = false;
28
29 ok = yarp::os::Network::connect(m_local, m_remote);
30
31 if (!ok)
32 {
33 yCError(CHATBOT_NWC_YARP) << "open() error could not connect to" << m_remote;
34 return false;
35 }
36
37 if (!m_thriftClient.yarp().attachAsClient(m_thriftClientPort))
38 {
39 yCError(CHATBOT_NWC_YARP) << "Cannot attach the m_thriftClientPort port as client";
40 return false;
41 }
42
43 yCDebug(CHATBOT_NWC_YARP) << "Opening of nwc successful";
44 return true;
45}
