67void ILLMRPCd::m_stream_conversation()
68{
69 std::vector<yarp::dev::LLM_Message> conversation;
70 if (!m_iLlm->getConversation(conversation)) {
71 yCError(LLMSERVER, "Unable to retrieve the conversation");
72 return;
73 }
74
75 auto& bot = m_streaming_port.prepare();
76 bot.clear();
77 auto& list = bot.addList();
78 for (const auto& message : conversation) {
79 auto& message_bot = list.addList();
80 message_bot.addString(message.type);
81 message_bot.addString(message.content);
82 auto& params_bot = message_bot.addList();
83 for(const auto& params: message.parameters)
84 {
85 params_bot.addString(params);
86 }
87
88 auto& args_bot = message_bot.addList();

90 {
91 args_bot.addString(args);
92 }
93 }
