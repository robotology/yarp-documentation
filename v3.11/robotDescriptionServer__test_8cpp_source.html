50 DeviceDescription dev2; dev2.device_name = "/icubTest/left_leg"; dev2.device_type = "controlBoard_nws_yarp";
51 DeviceDescription dev3; dev3.device_name = "/icubTest/test"; dev3.device_type = "testDevice";
52 idesc->registerDevice(dev1);

60 CHECK(std::find(list1.begin(), list1.end(), dev1) != list1.end());
61 CHECK(std::find(list1.begin(), list1.end(), dev2) != list1.end());
62 CHECK(std::find(list1.begin(), list1.end(), dev3) != list1.end());
63 // IRobotDescription::getAllDevices() successfully tested
64
65 idesc->getAllDevicesByType("controlBoard_nws_yarp", list2);
66 CHECK(list2.size() == 2);

70 // IRobotDescription::getControlBoardWrapperDevices() successfully tested
71
72 // Test unregister device
73 idesc->unregisterDevice("/icubTest/test");

75 CHECK(list2.size() == 0); // IRobotDescription::unregisterDevice() successfully tested
