30 {
31  std::lock_guard<std::mutex> lock (m_rpc_mutex);
32  yarp::os::Bottle in;
33  yarp::os::Bottle out;
34  bool ok = in.read(connection);
35  if (!ok) {
36  return false;

38

56  out.write(*returnToSender);
