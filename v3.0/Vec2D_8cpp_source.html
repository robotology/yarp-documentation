35 template<>
36 bool Vec2D<double>::read(yarp::os::ConnectionReader& connection)
37 {
38  // auto-convert text mode interaction
39  connection.convertTextMode();
40  Vec2DPortContentHeader header;
41  bool ok = connection.expectBlock((char*)&header, sizeof(header));
42  if (!ok) return false;
43 
44  if (header.listLen == 2 && header.listTag == (BOTTLE_TAG_LIST | BOTTLE_TAG_FLOAT64))
45  {
46  this->x = connection.expectFloat64();
47  this->y = connection.expectFloat64();
48  }
49  else
50  {
51  return false;
52  }
53 
54  return !connection.isError();
55 }

158 {
