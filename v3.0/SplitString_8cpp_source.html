55 void SplitString::apply(const char *command, char splitter)
56 {
57  int at = 0;
58  int sub_at = 0;
59  unsigned int i;
60  for (i=0; i<strlen(command)+1; i++) {
61  if (at<MAX_ARG_CT) {
62  // yarpserver can deal with quoting.
63  char ch = command[i];
64  if (ch=='\"') {
65  ch = ' ';
66  }
67  if (ch>=32||ch=='\0'||ch=='\n') {
68  if (ch==splitter||ch=='\n') {
69  ch = '\0';
70  }
71  if (sub_at<MAX_ARG_LEN) {
72  buf[at][sub_at] = ch;
73  sub_at++;
74  }
75  }
76  if (ch == '\0') {
77  if (sub_at>1) {
78  at++;
79  }
80  sub_at = 0;
81  }
82  }
83  }
84  for (i=0; i<MAX_ARG_CT; i++) {
85  argv[i] = buf[i];

87  }
