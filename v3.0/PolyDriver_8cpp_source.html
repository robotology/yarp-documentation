30  virtual void report(const SearchReport& report, const char *context) override {
31  std::string ctx = context;
32  std::string key = report.key.c_str();
33  std::string prefix = "";
34 
35  prefix = ctx;
36  prefix += ".";
37 
38  key = prefix + key;
39  if (key.substr(0,1)==".") {
40  key = key.substr(1,key.length());
41  }
42 
43  if (!present.check(key.c_str())) {
44  present.put(key.c_str(),"present");
45  order.addString(key.c_str());
46  }
47 
48  if (report.isFound) {
49  actual.put(key.c_str(),report.value);
50  return;
51  }
52 
53  if (report.isComment==true) {
54  comment.put(key.c_str(),report.value);
55  return;
56  }
57 
58  if (report.isDefault==true) {
59  fallback.put(key.c_str(),report.value);
60  return;
61  }
62  }

106 bool PolyDriver::open(yarp::os::Searchable& config) {
107  if (isValid()) {
108  // already open - should close first
109  return false;
110  }
111  if (system_resource==nullptr) {
112  system_resource = new YarpDevMonitor;
113  }
114  yAssert(system_resource!=nullptr);
115  bool removeMonitorAfterwards = false;
116  if (config.getMonitor()==nullptr) {
117  config.setMonitor(&HELPER(system_resource));
118  removeMonitorAfterwards = true;
119  }
120 
121  //dd = Drivers::factory().open(config);
122  coreOpen(config);
123  HELPER(system_resource).info.fromString(config.toString());
124  if (removeMonitorAfterwards) {
125  config.setMonitor(nullptr);
126  }
127  return isValid();
128 }

212 bool PolyDriver::coreOpen(yarp::os::Searchable& prop) {
213  yarp::os::Searchable *config = &prop;
214  Property p;
215  std::string str = prop.toString();
216  Value *part;
217  if (prop.check("device",part)) {
218  str = part->toString().c_str();
219  }
220 
221  DeviceDriver *driver = nullptr;
222 
223  DriverCreator *creator = Drivers::factory().find(str.c_str());
224  if (creator!=nullptr) {
225  Value *val;
226  if (config->check("wrapped",val)&&(creator->getWrapper()!="")) {
227  std::string wrapper = creator->getWrapper();
228  DriverCreator *wrapCreator =
229  Drivers::factory().find(wrapper.c_str());
230  if (wrapCreator!=nullptr) {
231  p.fromString(config->toString());
232  p.unput("wrapped");
233  config = &p;
234  if (wrapCreator!=creator) {
235  p.put("subdevice",str.c_str());
236  p.put("device",wrapper.c_str());
237  p.setMonitor(prop.getMonitor(),
238  wrapper.c_str()); // pass on any monitoring
239  driver = wrapCreator->create();
240  creator = wrapCreator;
241  } else {
242  // already wrapped
243  driver = creator->create();
244  }
245  }
246  } else {
247  driver = creator->create();
248  }
249  } else {
250  // FIXME do not use yarpdev here
251  printf("yarpdev: ***ERROR*** could not find device <%s>\n", str.c_str());
252  return false;
253  }
254 
255  if (driver!=nullptr) {
256  PolyDriver *manager = creator->owner();
257  if (manager!=nullptr) {
258  link(*manager);
259  return true;
260  }
261 
262  //printf("yarpdev: parameters are %s\n", config->toString().c_str());
263  bool ok = driver->open(*config);
264  if (!ok) {
265  printf("yarpdev: ***ERROR*** driver <%s> was found but could not open\n", config->find("device").toString().c_str());
266  //YARP_DEBUG(Logger::get(),String("Discarding ") + str);
267  delete driver;
268  //YARP_DEBUG(Logger::get(),String("Discarded ") + str);
269  driver = nullptr;
270  } else {
271  yarp::dev::DeprecatedDeviceDriver *ddd = nullptr;
272  driver->view(ddd);
273  if(ddd) {
274  if(config->check("allow-deprecated-devices")) {
275  yWarning("Device \"%s\" is deprecated. Opening since the \"allow-deprecated-devices\" option was passed in the configuration.", str.c_str());
276  } else {
277  yError("Device \"%s\" is deprecated. Pass the \"allow-deprecated-devices\" option in the configuration if you want to open it anyway.", str.c_str());
278  driver->close();
279  delete driver;
280  return false;
281  }
282  }
283  std::string name = creator->getName();
284  std::string wrapper = creator->getWrapper();
285  std::string code = creator->getCode();
286  yInfo("created %s <%s>. See C++ class %s for documentation.",
287  ((name==wrapper)?"wrapper":"device"),
288  name.c_str(),
289  code.c_str());
290  }
291  dd = driver;
292  return true;
293  }
294 
295  return false;
296 }
