127 static int enactConnection(const Contact& src,
128  const Contact& dest,
129  const ContactStyle& style,
130  int mode,
131  bool reversed)
132 {
133  ContactStyle rpc;
134  rpc.admin = true;
135  rpc.quiet = style.quiet;
136  rpc.timeout = style.timeout;
137 
138  CARRIER_DEBUG("enactConnection: SRC %s DST %s using carrier %s, MODE=%d, rev=%d\n",
139  src.getName().c_str(), dest.getName().c_str(), style.carrier.c_str(), mode, reversed);
140 
141  if (style.persistent) {
142  bool ok = false;
143  // we don't talk to the ports, we talk to the nameserver
144  NameSpace& ns = getNameSpace();
145  if (mode==YARP_ENACT_CONNECT) {
146  ok = ns.connectPortToPortPersistently(src, dest, style);
147  } else if (mode==YARP_ENACT_DISCONNECT) {
148  ok = ns.disconnectPortToPortPersistently(src, dest, style);
149  } else {
150  fprintf(stderr, "Failure: cannot check subscriptions yet\n");
151  return 1;
152  }
153  if (!ok) {
154  return 1;
155  }
156  if (!style.quiet) {
157  fprintf(stderr, "Success: port-to-port persistent connection added.\n");
158  }
159  return 0;
160  }
161 
162  Bottle cmd, reply;
163  cmd.addVocab(Vocab::encode("list"));
164  cmd.addVocab(Vocab::encode(reversed?"in":"out"));
165  cmd.addString(dest.getName().c_str());
166  YARP_SPRINTF2(Logger::get(), debug, "asking %s: %s",
167  src.toString().c_str(), cmd.toString().c_str());
168  bool ok = NetworkBase::write(src, cmd, reply, rpc);
169  if (!ok) {
170  noteDud(src);
171  return 1;
172  }
173  if (reply.check("carrier")) {
174  std::string carrier = reply.find("carrier").asString();
175  if (!style.quiet) {
176  printf("Connection found between %s and %s using carrier %s\n",
177  src.getName().c_str(), dest.getName().c_str(), carrier.c_str());
178  }
179  if (mode==YARP_ENACT_EXISTS) {
180  return (carrier == style.carrier) ? 0 : 1;
181  }
182 
183  // This is either a connect or a disconnect command, but the current
184  // connection is connectionless, the other side will not know that we
185  // are closing the connection and therefore will continue sending data.
186  // Therefore we send an explicit disconnect here.
187  bool currentIsConnectionLess = false;
188  bool currentIsPush = true;
189  if (reply.check("push")) {
190  currentIsPush = reply.find("push").asBool();
191  }
192  if (reply.check("connectionless")) {
193  currentIsConnectionLess = reply.find("connectionless").asBool();
194  }
195  if (currentIsConnectionLess && ((reversed && currentIsPush) || (!reversed && !currentIsPush))) {
196  enactConnection(dest, src, style, YARP_ENACT_DISCONNECT, !reversed);
197  }
198  }
199  if (mode==YARP_ENACT_EXISTS) {
200  return 1;
201  }
202 
203  int act = (mode==YARP_ENACT_DISCONNECT)?VOCAB3('d', 'e', 'l'):VOCAB3('a', 'd', 'd');
204 
205  // Let's ask the destination to connect/disconnect to the source.
206  // We assume the YARP carrier will reverse the connection if
207  // appropriate when connecting.
208  cmd.clear();
209  reply.clear();
210  cmd.addVocab(act);
211  Contact c = dest;
212  if (style.carrier!="") {
213  c.setCarrier(style.carrier);
214  }
215  if (mode!=YARP_ENACT_DISCONNECT) {
216  cmd.addString(c.toString());
217  } else {
218  cmd.addString(c.getName());
219  }
220 
221  Contact c2 = src;
222  if (c2.getPort()<=0) {
223  c2 = NetworkBase::queryName(c2.getName());
224  }
225 
226  YARP_SPRINTF2(Logger::get(), debug, "** asking %s: %s",
227  src.toString().c_str(), cmd.toString().c_str());
228  ok = NetworkBase::write(c2, cmd, reply, rpc);
229  if (!ok) {
230  noteDud(src);
231  return 1;
232  }
233  std::string msg = "";
234  if (reply.get(0).isInt32()) {
235  ok = (reply.get(0).asInt32()==0);
236  msg = reply.get(1).asString();
237  } else {
238  // older protocol
239  msg = reply.get(0).asString();
240  ok = msg[0]=='A'||msg[0]=='R';
241  }
242  if (mode==YARP_ENACT_DISCONNECT && !ok) {
243  msg = "no such connection\n";
244  }
245  if (mode==YARP_ENACT_CONNECT && !ok) {
246  noteDud(dest);
247  }
248  if (!style.quiet) {
249  if (style.verboseOnSuccess||!ok) {
250  fprintf(stderr, "%s %s",
251  ok?"Success:":"Failure:",
252  msg.c_str());
253  }
254  }
255  return ok ? 0 : 1;
256 }

300 /*
301 
302  Connect two ports, bearing in mind that one of them may not be
303  a regular YARP port.
304 
305  Normally, YARP sends a request to the source port asking it to
306  connect to the destination port. But the source port may not
307  be capable of initiating connections, in which case we can
308  request the destination port to connect to the source (this
309  is appropriate for carriers that can reverse the initiative).
310 
311  The source or destination could also be topic ports, which are
312  entirely virtual. In that case, we just need to tell the name
313  server, and it will take care of the details.
314 
315 */

317 static int metaConnect(const std::string& src,
318  const std::string& dest,
319  ContactStyle style,
320  int mode) {
321  YARP_SPRINTF3(Logger::get(), debug,
322  "working on connection %s to %s (%s)",
323  src.c_str(),
324  dest.c_str(),
325  (mode==YARP_ENACT_CONNECT)?"connect":((mode==YARP_ENACT_DISCONNECT)?"disconnect":"check")
326  );
327  // check if source name and destination name contain spaces
328  if(dest.find(" ") != std::string::npos || src.find(" ") != std::string::npos)
329  {
330  fprintf(stderr, "Failure: no way to make connection %s->%s,\n", src.c_str(), dest.c_str());
331  return 1;
332  }
333 
334  CARRIER_DEBUG("METACONNECT: src=%s dest=%s style=%s\n", src.c_str(), dest.c_str(), style.carrier.c_str());
335  // get the expressed contacts, without name server input
336  Contact dynamicSrc = Contact::fromString(src);
337  Contact dynamicDest = Contact::fromString(dest);
338 
339  CARRIER_DEBUG("DYNAMIC_SRC: name=%s, carrier=%s\n", dynamicSrc.getName().c_str(), dynamicSrc.getCarrier().c_str());
340  CARRIER_DEBUG("DYNAMIC_DST: name=%s, carrier=%s\n", dynamicDest.getName().c_str(), dynamicDest.getCarrier().c_str());
341 
342  if(!NetworkBase::isValidPortName(dynamicSrc.getName()))
343  {
344  fprintf(stderr, "Failure: no way to make connection, invalid source '%s'\n", dynamicSrc.getName().c_str());
345  return 1;
346  }
347  if(!NetworkBase::isValidPortName(dynamicDest.getName()))
348  {
349  fprintf(stderr, "Failure: no way to make connection, invalid destination '%s'\n", dynamicDest.getName().c_str());
350  return 1;
351  }
352 
353  bool topical = style.persistent;
354  if (dynamicSrc.getCarrier()=="topic" ||
355  dynamicDest.getCarrier()=="topic") {
356  topical = true;
357  }
358 
359  bool topicalNeedsLookup = !getNameSpace().connectionHasNameOfEndpoints();
360 
361  // fetch completed contacts from name server, if needed
362  Contact staticSrc;
363  Contact staticDest;
364  if (needsLookup(dynamicSrc)&&(topicalNeedsLookup||!topical)) {
365  staticSrc = NetworkBase::queryName(dynamicSrc.getName());
366  if (!staticSrc.isValid()) {
367  if (!style.persistent) {
368  if (!style.quiet) {
369  fprintf(stderr, "Failure: could not find source port %s\n",
370  src.c_str());
371  }
372  return 1;
373  } else {
374  staticSrc = dynamicSrc;
375  }
376  }
377  } else {
378  staticSrc = dynamicSrc;
379  }
380  if (staticSrc.getCarrier()=="") {
381  staticSrc.setCarrier("tcp");
382  }
383  if (staticDest.getCarrier()=="") {
384  staticDest.setCarrier("tcp");
385  }
386 
387  if (needsLookup(dynamicDest)&&(topicalNeedsLookup||!topical)) {
388  staticDest = NetworkBase::queryName(dynamicDest.getName());
389  if (!staticDest.isValid()) {
390  if (!style.persistent) {
391  if (!style.quiet) {
392  fprintf(stderr, "Failure: could not find destination port %s\n",
393  dest.c_str());
394  }
395  return 1;
396  } else {
397  staticDest = dynamicDest;
398  }
399  }
400  } else {
401  staticDest = dynamicDest;
402  }
403 
404  CARRIER_DEBUG("STATIC_SRC: name=%s, carrier=%s\n", staticSrc.getName().c_str(), staticSrc.getCarrier().c_str());
405  CARRIER_DEBUG("STATIC_DST: name=%s, carrier=%s\n", staticDest.getName().c_str(), staticDest.getCarrier().c_str());
406 
407  //DynamicSrc and DynamicDst are the contacts created by connect command
408  //while staticSrc and staticDest are contacts created by querying th server
409 
410  if (staticSrc.getCarrier()=="xmlrpc" &&
411  (staticDest.getCarrier()=="xmlrpc"||(staticDest.getCarrier().find("rossrv")==0))&&
412  mode==YARP_ENACT_CONNECT) {
413  // Unconnectable in general
414  // Let's assume the first part is a YARP port, and use "tcp" instead
415  staticSrc.setCarrier("tcp");
416  staticDest.setCarrier("tcp");
417  }
418 
419  std::string carrierConstraint = "";
420 
421  // see if we can do business with the source port
422  bool srcIsCompetent = false;
423  bool srcIsTopic = false;
424  if (staticSrc.getCarrier()!="topic") {
425  if (!topical) {
426  Carrier *srcCarrier = nullptr;
427  CARRIER_DEBUG("staticSrc.getCarrier= %s ", staticSrc.getCarrier().c_str());
428  if (staticSrc.getCarrier()!="") {
429  srcCarrier = Carriers::chooseCarrier(staticSrc.getCarrier().c_str());
430  }
431  if (srcCarrier!=nullptr) {
432  CARRIER_DEBUG("srcCarrier is NOT null; its name is %s; ", srcCarrier->getName().c_str());
433  std::string srcBootstrap = srcCarrier->getBootstrapCarrierName();
434  if (srcBootstrap!="") {
435 
436  CARRIER_DEBUG(" it is competent(bootstrapname is %s), while its name is %s )\n\n", srcBootstrap.c_str(), srcCarrier->getName().c_str());
437  srcIsCompetent = true;
438  } else {
439  //if the srcCarrier is not competent, (that is it can't perform the starting yarp handshaking)
440  //set the carrier constraint equal to the carrier with which the posrt had been registered.
441  carrierConstraint = staticSrc.getCarrier();
442  CARRIER_DEBUG(" it is NOT competent. its constraint is %s\n\n", carrierConstraint.c_str());
443  }
444  delete srcCarrier;
445  srcCarrier = nullptr;
446  }
447  }
448  } else {
449  srcIsTopic = true;
450  }
451 
452  // see if we can do business with the destination port
453  bool destIsCompetent = false;
454  bool destIsTopic = false;
455  if (staticDest.getCarrier()!="topic") {
456  if (!topical) {
457  Carrier *destCarrier = nullptr;
458  CARRIER_DEBUG("staticDest.getCarrier= %s ", staticDest.getCarrier().c_str());
459  if (staticDest.getCarrier()!="") {
460  destCarrier = Carriers::chooseCarrier(staticDest.getCarrier().c_str());
461  }
462  if (destCarrier!=nullptr) {
463  CARRIER_DEBUG("destCarrier is NOT null; its name is %s; ", destCarrier->getName().c_str());
464  std::string destBootstrap = destCarrier->getBootstrapCarrierName();
465  if (destBootstrap!="") {
466  CARRIER_DEBUG(" it is competent(bootstrapname is %s), while its name is %s )\n\n\n\n", destBootstrap.c_str(), destCarrier->getName().c_str() );
467  destIsCompetent = true;
468  } else {
469  //if the destCarrier is not competent, (that is it can't perform the starting yarp handshaking)
470  //set the carrier constraint equal to the carrier with which the posrt had been registered.
471  carrierConstraint = staticDest.getCarrier();
472  CARRIER_DEBUG(" it is NOT competent. its constraint is %s\n\n", carrierConstraint.c_str());
473  }
474  delete destCarrier;
475  destCarrier = nullptr;
476  }
477  }
478  } else {
479  destIsTopic = true;
480  }
481 
482  if (srcIsTopic||destIsTopic) {
483  Bottle cmd, reply;
484  NameSpace& ns = getNameSpace();
485 
486  bool ok = false;
487  if (srcIsTopic) {
488  if (mode==YARP_ENACT_CONNECT) {
489  ok = ns.connectTopicToPort(staticSrc, staticDest, style);
490  } else if (mode==YARP_ENACT_DISCONNECT) {
491  ok = ns.disconnectTopicFromPort(staticSrc, staticDest, style);
492  } else {
493  fprintf(stderr, "Failure: cannot check subscriptions yet\n");
494  return 1;
495  }
496  } else {
497  if (mode==YARP_ENACT_CONNECT) {
498  ok = ns.connectPortToTopic(staticSrc, staticDest, style);
499  } else if (mode==YARP_ENACT_DISCONNECT) {
500  ok = ns.disconnectPortFromTopic(staticSrc, staticDest, style);
501  } else {
502  fprintf(stderr, "Failure: cannot check subscriptions yet\n");
503  return 1;
504  }
505  }
506  if (!ok) {
507  return 1;
508  }
509  if (!style.quiet) {
510  if (style.verboseOnSuccess) {
511  fprintf(stderr, "Success: connection to topic %s.\n", mode==YARP_ENACT_CONNECT ? "added" : "removed");
512  }
513  }
514  return 0;
515  }
516 
517  CARRIER_DEBUG("---------\n");
518  CARRIER_DEBUG("dynamicSrc.getCarrier() = %s\n ", dynamicSrc.getCarrier().c_str());
519  CARRIER_DEBUG("dynamicDest.getCarrier() = %s\n ", dynamicDest.getCarrier().c_str());
520  CARRIER_DEBUG("staticSrc.getCarrier() = %s\n ", staticSrc.getCarrier().c_str());
521  CARRIER_DEBUG("staticDest.getCarrier() = %s\n ", staticDest.getCarrier().c_str());
522  CARRIER_DEBUG("carrierConstraint is %s\n ", carrierConstraint.c_str());
523  CARRIER_DEBUG("---------\n");
524 
525  CARRIER_DEBUG("style.carrier (1) is %s\n ", style.carrier.c_str());
526 
527 
528  if (dynamicSrc.getCarrier()!="") { //if in connect command the user specified the carrier of src port
529  style.carrier = dynamicSrc.getCarrier();
530  CARRIER_DEBUG("style.carrier is %s ==> in connect command the user specified the carrier of src port\n ", style.carrier.c_str());
531  }
532 
533  if (dynamicDest.getCarrier()!="") { //if in connect command the user specified the carrier of dest port or the carrier of the connection
534  style.carrier = dynamicDest.getCarrier();
535  CARRIER_DEBUG("style.carrier is %s ==> in connect command the user specified the carrier of dest port or the carrier of the connection\n ", style.carrier.c_str());
536  }
537 
538  CARRIER_DEBUG("at the end style style.carrier is %s\n ", style.carrier.c_str());
539 
540  //here we'll check if the style carrier and the constraint carrier are equal.
541  //note that in both string may contain params of carrier, so we need to comapare only the name of carrier.
542  if(style.carrier!="" && carrierConstraint!="") {
543  //get only carrier name of style.
544  std::string style_carrier_name = extractCarrierNameOnly(style.carrier);
545 
546  //get only carrier name of carrierConstraint.
547  std::string carrier_constraint_name = extractCarrierNameOnly(carrierConstraint);
548 
549  if (style_carrier_name!=carrier_constraint_name) {
550  fprintf(stderr, "Failure: conflict between %s and %s\n",
551  style_carrier_name.c_str(),
552  carrier_constraint_name.c_str());
553  return 1;
554  }
555  CARRIER_DEBUG("style_carrier_name=%s and carrier_constraint_name=%s are equals!\n", style_carrier_name.c_str(), carrier_constraint_name.c_str());
556 
557  }
558  //we are going to choose the carrier of this connection, and we collect parameters specified by user
559  //in order to pass them to the carrier, so it can configure itself.
560  if (carrierConstraint!="") {
561  style.carrier = carrierConstraint;
562  //if I'm here means that sorce or dest is not competent.
563  //so we need to get parameters of carrier given in connect command.
564  CARRIER_DEBUG("if I'm here means that sorce or dest is not competent\n");
565  std::string c = dynamicSrc.getCarrier();
566  if(extractCarrierNameOnly(c) == extractCarrierNameOnly(style.carrier))
567  style.carrier+=collectParams(dynamicSrc);
568  c = dynamicDest.getCarrier();
569  if(extractCarrierNameOnly(c) == extractCarrierNameOnly(style.carrier))
570  style.carrier+=collectParams(dynamicDest);
571  }
572  if (style.carrier=="") {
573  style.carrier = staticDest.getCarrier();
574  //if I'm here means that both src and dest are copentent and the user didn't specified a carrier in the connect command
575  CARRIER_DEBUG("if I'm here means that both src and dest are copentent and the user didn't specified a carrier in the connect command\n");
576  std::string c = dynamicSrc.getCarrier();
577  if(extractCarrierNameOnly(c) == extractCarrierNameOnly(style.carrier))
578  style.carrier+=collectParams(staticSrc);
579  }
580 
581  if (style.carrier=="") {
582  style.carrier = staticSrc.getCarrier();
583  CARRIER_DEBUG("the chosen style carrier is static src\n ");
584  }
585 
586  //now stylecarrier contains the carrier chosen for this connection
587 
588  CARRIER_DEBUG("style_carrier with params =%s\n", style.carrier.c_str());
589 
590  bool connectionIsPush = false;
591  bool connectionIsPull = false;
592  Carrier *connectionCarrier = nullptr;
593  if (style.carrier!="topic") {
594  connectionCarrier = Carriers::chooseCarrier(style.carrier.c_str());
595  if (connectionCarrier!=nullptr) {
596  connectionIsPush = connectionCarrier->isPush();
597  connectionIsPull = !connectionIsPush;
598  }
599  }
600 
601  int result = -1;
602  if ((srcIsCompetent&&connectionIsPush)||topical) {
603  // Classic case.
604  Contact c = Contact::fromString(dest);
605  if (connectionCarrier!=nullptr) delete connectionCarrier;
606  return enactConnection(staticSrc, c, style, mode, false);
607  }
608  if (destIsCompetent&&connectionIsPull) {
609  Contact c = Contact::fromString(src);
610  if (connectionCarrier!=nullptr) delete connectionCarrier;
611  return enactConnection(staticDest, c, style, mode, true);
612  }
613 
614  if (connectionCarrier!=nullptr) {
615  if (!connectionIsPull) {
616  Contact c = Contact::fromString(dest);
617  result = connectionCarrier->connect(staticSrc, c, style, mode, false);
618  } else {
619  Contact c = Contact::fromString(src);
620  result = connectionCarrier->connect(staticDest, c, style, mode, true);
621  }
622  }
623  if (connectionCarrier!=nullptr) {
624  delete connectionCarrier;
625  connectionCarrier = nullptr;
626  }
627  if (result!=-1) {
628  if (!style.quiet) {
629  if (result==0) {
630  if (style.verboseOnSuccess) {
631  printf("Success: added connection using custom carrier method\n");
632  }
633  } else {
634  printf("Failure: custom carrier method did not work\n");
635  }
636  }
637  return result;
638  }
639 
640  if (mode!=YARP_ENACT_DISCONNECT) {
641  fprintf(stderr, "Failure: no way to make connection %s->%s\n", src.c_str(), dest.c_str());
642  }
643
