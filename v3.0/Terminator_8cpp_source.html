23 bool Terminator::terminateByName(const char *name) {
24  if (name == nullptr)
25  return false;
26 
27  std::string s(name);
28 
29  if (s.find("/quit")==std::string::npos) {
30  // name doesn't include /quit
31  // old mechanism won't work, let's try new
32  PortCommand pc('\0', "i");
33  NetworkBase::sendMessage(s, pc, true);
34  return true;
35  }
36 
37  if (name[0] != '/') {
38  s.clear();
39  s += "/";
40  s += name;
41  }
42 
43  Bottle cmd("quit"), reply;
44  Contact c = NetworkBase::queryName(s.c_str());
45  if (!c.isValid()) {
46  fprintf(stderr, "Terminator port not found\n");
47  return false;
48  }

50  style.quiet = false;
51  style.carrier = "text_ack";
52  style.expectReply = true;
53  NetworkBase::write(c, cmd, reply, style);

56 }
