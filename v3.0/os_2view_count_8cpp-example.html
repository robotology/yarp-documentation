#include <yarp/os/all.h>
#include <stdio.h>
using namespace yarp::os;
int main(int argc, char *argv[]) {
if (argc!=2) return 1;
Network yarp;
BufferedPort<Bottle> in;
in.open(argv[1]);
int count = 1;
while (count>0) {
Bottle *msg = in.read();
count = msg->get(1).asInt32();
printf("at %d\n", count);
}
return 0;
}
