46 void TextureStatic::createTexture()
47 {
48  ovrTextureSwapChainDesc desc = {};
49  desc.Type = ovrTexture_2D;
50  desc.Format = OVR_FORMAT_R8G8B8A8_UNORM_SRGB;
51  desc.ArraySize = 1;
52  desc.Width = width;
53  desc.Height = height;
54  desc.MipLevels = 1;
55  desc.SampleCount = 1;
56  desc.StaticImage = ovrTrue;
57  desc.MiscFlags = ovrTextureMisc_None;
58  desc.BindFlags = ovrTextureBind_None;
59 
60  if (ovr_CreateTextureSwapChainGL(session, &desc, &textureSwapChain) != ovrSuccess) {
61  yFatal() << "Failed to create texture swap chain";
62  }
63 
64  ovr_GetTextureSwapChainBufferGL(session, textureSwapChain, -1, &chainTexId);
65  checkGlErrorMacro;
66 
67  glBindTexture(GL_TEXTURE_2D, chainTexId);
68  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER, GL_LINEAR);
69  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MAG_FILTER, GL_LINEAR);
70  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_S, GL_REPEAT);
71  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_T, GL_REPEAT);
72  glTexSubImage2D(GL_TEXTURE_2D, 0, 0, 0, width, height, GL_RGBA, GL_UNSIGNED_BYTE, ptr);
73  checkGlErrorMacro;
74 
75  glBindTexture(GL_TEXTURE_2D, 0);
76 
77  ovr_CommitTextureSwapChain(session, textureSwapChain);
78  checkGlErrorMacro;
79 }
