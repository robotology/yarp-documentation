129  bool isResting(double priority) {

135  void stimulate(double t) {
136 
137  temporalStimulation = getActualStimulation(t);
138  // if it is in active state, stimulates.
139  if(!isResting(temporalStimulation))
140  {
141  temporalStimulation += stimulation;
142  // respecting priority ceiling
143  if(temporalStimulation >= STIMUL_THRESHOLD)
144  {
145  temporalStimulation = STIMUL_THRESHOLD;
146  isActive = true;
147  }
148  // updating arrival time
149  timeArrival = t;
150  }
151  }
