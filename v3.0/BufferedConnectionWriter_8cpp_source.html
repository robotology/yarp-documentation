98 bool BufferedConnectionWriter::addPool(const yarp::os::Bytes& data)
99 {
100  if (pool != nullptr) {
101  if (data.length() + poolIndex > pool->length()) {
102  pool = nullptr;
103  }
104  }
105  if (pool == nullptr && data.length() < poolLength) {
106  bool add = false;
107  if (*target_used < target->size()) {
108  yarp::os::ManagedBytes*& bytes = (*target)[*target_used];
109  if (bytes->length() < poolLength) {
110  delete bytes;
111  bytes = new yarp::os::ManagedBytes(poolLength);
112  }
113  pool = bytes;
114  if (pool == nullptr) {
115  return false;
116  }
117  } else {
118  pool = new yarp::os::ManagedBytes(poolLength);
119  if (pool == nullptr) {
120  return false;
121  }
122  add = true;
123  }
124  (*target_used)++;
125  poolCount++;
126  poolIndex = 0;
127  if (poolLength < 65536) {
128  poolLength *= 2;
129  }
130  pool->setUsed(0);
131  if (add)
132  target->push_back(pool);
133  }
134  if (pool != nullptr) {
135  memcpy(pool->get() + poolIndex, data.get(), data.length());
136  poolIndex += data.length();
137  pool->setUsed(poolIndex);
138  return true;
139  }
140  return false;
141 }

153 void BufferedConnectionWriter::push(const Bytes& data, bool copy)
154 {
155  if (copy) {
156  if (addPool(data))
157  return;
158  }
159  yarp::os::ManagedBytes* buf = nullptr;
160  if (*target_used < target->size()) {
161  yarp::os::ManagedBytes*& bytes = (*target)[*target_used];
162  if (bytes->isOwner() != copy || bytes->length() < data.length()) {
163  delete bytes;
164  bytes = new yarp::os::ManagedBytes(data, false);
165  if (copy) {
166  bytes->copy();
167  }
168  (*target_used)++;
169  return;
170  }
171  buf = bytes;
172  bytes->setUsed(data.length());
173  }
174  if (buf == nullptr) {
175  buf = new yarp::os::ManagedBytes(data, false);
176  if (copy)
177  buf->copy();
178  target->push_back(buf);
179  } else {
180  if (copy) {
181  buf->copy();
182  memmove(buf->get(), data.get(), data.length());
183  } else {
184  *buf = ManagedBytes(data, false);
185  }
186  }
187  (*target_used)++;
188 }

503 bool BufferedConnectionWriter::applyConvertTextMode()
504 {
505  if (convertTextModePending) {
506  convertTextModePending = false;
507 
508  Bottle b;
509  StringOutputStream sos;
510  for (size_t i = 0; i < lst_used; i++) {
511  yarp::os::ManagedBytes& m = *(lst[i]);
512  sos.write(m.usedBytes());
513  }
514  const std::string& str = sos.str();
515  b.fromBinary(str.c_str(), (int)str.length());
516  std::string replacement = b.toString() + "\n";
517  for (size_t i = 0; i < lst.size(); i++) {
518  delete lst[i];
519  }
520  lst_used = 0;
521  target = &lst;
522  lst.clear();
523  stopPool();
524  Bytes data((char*)replacement.c_str(), replacement.length());
525  appendBlockCopy(data);
526  }
527  return true;
528 }
