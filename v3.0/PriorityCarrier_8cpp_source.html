129 double PriorityCarrier::getActualStimulation(double t)
130 {
131  //
132  // +P| ---___
133  // | | -__
134  // | | -_
135  // 0|------------------> (Active state)
136  // Ta Tc
137  //
138  // Ta Tr
139  // 0|-------------_----> (Resting state && P<0)
140  // | | __-
141  // | | ___-
142  // -P| ---
143  //
144  //
145  // P(t) = Pi * (1-exp((t-Tc-Ta)/Tc*5) + exp(-5))
146  // t:time, Pi: temporal Priority level
147  // Tc: reset time, Ta: arrival time
148  //
149  // we do not consider ports which has not seen any message
150  // from them yet.
151  //if(timeArrival == 0)
152  // return 0;
153 
154  double dt = t - timeArrival;
155  // Temporal priority is inverted if this is a neuron model and the temporal
156  // stimulation has already reached to STIMUL_THRESHOLD and waited for Tc.
157  if((timeResting > 0)
158  && (dt >= fabs(timeConstant))
159  && (temporalStimulation >= STIMUL_THRESHOLD))
160  temporalStimulation = -temporalStimulation;
161 
162  double actualStimulation;
163  if(!isResting(temporalStimulation)) // behavior is in stimulation state
164  {
165  // After a gap bigger than Tc, the
166  // priority is set to zero to avoid redundant calculation.
167  if(dt > fabs(timeConstant))
168  actualStimulation = 0;
169  else
170  actualStimulation = temporalStimulation *
171  (1.0 - exp((dt-timeConstant)/timeConstant*5.0) + exp(-5.0));
172  }
173  else // behavior is in resting state
174  {
175  // it is in waiting state for Tc
176  if(temporalStimulation > 0)
177  actualStimulation = temporalStimulation;
178  else
179  {
180  dt -= fabs(timeConstant);
181  // After a gap bigger than Tr, the
182  // priority is set to zero to avoid redundant calculation.
183  if(dt > fabs(timeResting))
184  actualStimulation = 0;
185  else
186  actualStimulation = temporalStimulation *
187  (1.0 - exp((dt-timeResting)/timeResting*5.0) + exp(-5.0));
188  }
189  }
190 
191  if(actualStimulation <= 0)
192  isActive = false;
193 
194  return actualStimulation;
195 }

197 double PriorityCarrier::getActualInput(double t)
198 {
199  // calculating E(t) = Sum(e.I(t)) + b
200  if(!isActive)
201  return 0.0;
202 
203  double E = 0;
204  for (PriorityGroup::iterator it = group->peerSet.begin(); it!=group->peerSet.end(); it++)
205  {
206  PriorityCarrier *peer = it->first;
207  if(peer != this)
208  {
209  for(size_t i=0; i<peer->excitation.size(); i++)
210  {
211  Value v = peer->excitation.get(i);
212  if(v.isList() && (v.asList()->size()>=2))
213  {
214  Bottle* b = v.asList();
215  // an exitatory to this priority carrier
216  if(sourceName == b->get(0).asString().c_str())
217  E += peer->getActualInput(t) * (b->get(1).asFloat64()/10.0);
218  }
219  }
220 
221  }
222  }
223  E += baias;
224  double I = E * getActualStimulation(t);
225  return ((I<0) ? 0 : I); //I'(t)
226 }

233 bool PriorityGroup::recalculate(double t)
234 {
235 #ifdef WITH_YARPMATH
236  //TODO: find the correct way to get the size of peerSet
237  int nConnections = 0;
238  for(PriorityGroup::iterator it=peerSet.begin(); it!=peerSet.end(); it++)
239  nConnections++;
240 
241  // calculate matrices X, B, InvA and Y
242  X.resize(nConnections, 1);
243  B.resize(nConnections, 1);
244  Y.resize(nConnections, 1);
245  InvA.resize(nConnections, nConnections);
246  InvA.eye();
247 
248  int row = 0;
249  for(PriorityGroup::iterator rowItr=peerSet.begin(); rowItr!=peerSet.end(); rowItr++)
250  {
251  PriorityCarrier *peer = rowItr->first;
252  // call 'getActualStimulation' to update 'isActive'
253  peer->getActualStimulation(t);
254  double xi = (peer->isActive) ? STIMUL_THRESHOLD : 0.0;
255  B(row,0) = peer->baias * xi;
256  X(row,0) = xi;
257 
258  int col = 0;
259  for(PriorityGroup::iterator colItr=peerSet.begin(); colItr!=peerSet.end(); colItr++)
260  {
261  PriorityCarrier *peerCol = colItr->first;
262  for(size_t i=0; i<peerCol->excitation.size(); i++)
263  {
264  Value v = peerCol->excitation.get(i);
265  if(v.isList() && (v.asList()->size()>=2))
266  {
267  Bottle* b = v.asList();
268  // an exitatory link to this connection
269  if(peer->sourceName == b->get(0).asString().c_str())
270  InvA(row,col) = -(b->get(1).asFloat64()/10.0)*xi;
271  }
272  }
273  col++;
274  }
275  row++;
276  }
277 
278  //fprintf(stdout, "A:\n %s\n", InvA.toString(1).c_str());
279 
280  // calclulating the determinant
281  double determinant = yarp::math::det(InvA);
282  if(determinant == 0)
283  {
284  yError("Inconsistent regulation! non-invertible weight matrix");
285  return false;
286  }
287 
288  // inverting the weight matrix
289  InvA = yarp::math::luinv(InvA);
290  Y = InvA * B;
291 
292  //fprintf(stdout, "X:\n %s\n", X.toString(1).c_str());
293  //fprintf(stdout, "B:\n %s\n", B.toString(1).c_str());
294  //fprintf(stdout, "Y:\n %s\n", Y.toString(1).c_str());
295 
296  return true;
297 #else
298  return false;
299 #endif

303 // Decide whether data should be accepted, for real.
304 bool PriorityGroup::acceptIncomingData(yarp::os::ConnectionReader& reader,
305  PriorityCarrier *source) {
306 
307  bool accept;
308  // updates message's arrival time
309  double tNow = yarp::os::Time::now();
310  source->stimulate(tNow);
311 
312 #ifdef WITH_YARPMATH
313  if(!recalculate(tNow))
314  return false;
315 
316  int row = 0;
317  PriorityCarrier *maxPeer = nullptr;
318  double maxStimuli = 0.0;
319  for(PriorityGroup::iterator it=peerSet.begin(); it!=peerSet.end(); it++)
320  {
321  PriorityCarrier *peer = it->first;
322  double output = Y(row,0) * X(row,0);
323  peer->yi = output; // only for debug purpose
324 
325  if(!peer->isVirtual)
326  {
327  if(output > maxStimuli)
328  {
329  maxStimuli = output;
330  maxPeer = peer;
331  }
332  }
333  row++;
334  }
335  accept = (maxPeer == source);
336 
337 #else
338  // first checks whether actual input signal is positive or not
339  double actualInput = source->getActualInput(tNow);
340  accept = (actualInput > 0);
341  if(accept)
342  {
343  for (PriorityGroup::iterator it = peerSet.begin(); it!=peerSet.end(); it++)
344  {
345  PriorityCarrier *peer = it->first;
346  if(peer != source)
347  {
348  if(actualInput < peer->getActualInput(tNow))
349  {
350  accept = false;
351  break;
352  }
353  }
354  }
355  }
356 #endif
357 
358  // a virtual message will never be delivered. It will be only
359  // used for the coordination
360  if(source->isVirtual)
361  accept = false;
362 
363  return accept;
364 }
