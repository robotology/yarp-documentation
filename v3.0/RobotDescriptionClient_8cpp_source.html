76 bool yarp::dev::RobotDescriptionClient::getAllDevicesByType(const std::string &type, std::vector<DeviceDescription>& dev_list)
77 {
78  yarp::os::Bottle b;
79  yarp::os::Bottle resp;
80  dev_list.clear();
81 
82  b.addVocab(VOCAB_IROBOT_DESCRIPTION);
83  b.addVocab(VOCAB_IROBOT_GET);
84  b.addVocab(VOCAB_IROBOT_BY_TYPE);
85  b.addString(type);
86  bool ret = m_rpc_port.write(b, resp);
87  if (ret)
88  {
89  if (resp.get(0).asVocab() != VOCAB_OK)
90  {
91  yError() << "RobotDescriptionClient::getAllDevices() received error from server";
92  return false;
93  }
94  else
95  {
96  Bottle *b = resp.get(1).asList();
97  for (size_t i = 0; i < b->size(); i += 2)
98  {
99  DeviceDescription desc;
100  desc.device_name = b->get(i).asString();
101  desc.device_type = b->get(i + 1).asString();
102  dev_list.push_back(desc);
103  }
104  return true;
105  }
106  }
107  else
108  {
109  yError() << "RobotDescriptionClient: error on writing on rpc port";
110  return false;
111  }
112  return true;
113 }
