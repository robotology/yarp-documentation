107 bool BootstrapServer::configFileBootstrap(yarp::os::Contact& contact,
108  bool configFileRequired,
109  bool mayEditConfigFile) {
110  Contact suggest = contact;
111 
112  // see what address is lying around
113  Contact prev;
114  NameConfig conf;
115  if (conf.fromFile()) {
116  prev = conf.getAddress();
117  } else if (configFileRequired) {
118  fprintf(stderr,"Could not read configuration file %s\n",
119  conf.getConfigFileName().c_str());
120  return false;
121  }
122 
123  // merge
124  if (prev.isValid()) {
125  if (suggest.getHost() == "...") {
126  suggest.setHost(prev.getHost());
127  }
128  if (suggest.getCarrier() == "...") {
129  suggest.setCarrier(prev.getCarrier());
130  }
131  if (suggest.getPort() == 0) {
132  suggest.setPort(prev.getPort());
133  }
134  }
135 
136  if (suggest.getRegName() == "...") {
137  suggest.setName(conf.getNamespace());
138  }
139 
140  // still something not set?
141  if (suggest.getPort() == 0) {
142  suggest.setPort(10000);
143  }
144  if (suggest.getHost() == "...") {
145  // should get my IP
146  suggest.setHost(conf.getHostName());
147  }
148 
149  if (!configFileRequired) {
150  // finally, should make sure IP is local, and if not, correct it
151  if (!conf.isLocalName(suggest.getHost())) {
152  fprintf(stderr,"Overriding non-local address for name server\n");
153  suggest.setHost(conf.getHostName());
154  } else {
155  // Let's just check we're not a loopback
156  std::string betterHost = conf.getHostName(false,suggest.getHost());
157  if (betterHost!=suggest.getHost()) {
158  fprintf(stderr,"Overriding loopback address for name server\n");
159  suggest.setHost(betterHost);
160  }
161  }
162  }
163 
164  bool changed = false;
165  if (prev.isValid()) {
166  changed = (prev.getHost() != suggest.getHost()) ||
167  (prev.getPort() != suggest.getPort()) ||
168  (conf.getMode() != "yarp" && conf.getMode() != "");
169  }
170  if (changed && !mayEditConfigFile) {
171  fprintf(stderr,"PROBLEM: need to change settings in %s\n",
172  conf.getConfigFileName().c_str());
173  fprintf(stderr," Current settings: host %s port %d family %s\n",
174  prev.getHost().c_str(), prev.getPort(),
175  (conf.getMode()=="")?"yarp":conf.getMode().c_str());
176  fprintf(stderr," Desired settings: host %s port %d family %s\n",
177  suggest.getHost().c_str(), suggest.getPort(), "yarp");
178  fprintf(stderr,"Please specify '--write' if it is ok to overwrite current settings, or\n");
179  fprintf(stderr,"Please specify '--read' to use the current settings, or\n");
180  fprintf(stderr,"delete %s\n", conf.getConfigFileName().c_str());
181  return false;
182  }
183  bool shouldSave = changed || !prev.isValid();
184 
185  if (shouldSave) {
186  // and save
187  conf.setAddress(suggest);
188  if (!conf.toFile()) {
189  fprintf(stderr,"Could not save configuration file %s\n",
190  conf.getConfigFileName().c_str());
191  }
192  }
193 
194  contact = suggest;
195  return true;
196 }
