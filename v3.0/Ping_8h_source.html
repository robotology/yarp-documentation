48  void add(const Stat& alt) {
49  tot += alt.tot;

51  ct += alt.ct;
52  }

54  double mean() {
55  compute();
56  return mu;
57  }

59  double deviation() {
60  compute();

62  }

65  return ct;
66  }
67 
68  operator double() {
69  return mean();
70  }
71 
72 private:
73  void compute() {
74  if (ct!=at) {
75  // ct must be > 0
76  mu = tot/ct;
77  sigma = tot2/ct - mu*mu;
78  if (sigma<0) sigma = 0; // round-off error
79  sigma = sqrt(sigma);
80  at = ct;
81  }
82  }
83 
84  int ct, at;
85  double tot, tot2;
86  double mu, sigma;
