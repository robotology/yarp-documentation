592 template <typename T>
593 inline std::int32_t subCoder(T& content)
594 {
595  std::int32_t c = -1;
596  bool ok = false;
597  for (unsigned int i = 0; i < content.size(); ++i) {
598  std::int32_t sc = content.get(i).getCode();
599  if (c == -1) {
600  c = sc;
601  ok = true;
602  }
603  if (sc != c) {
604  ok = false;
605  }
606  }
607  // just optimize primitive types
608  if ((c & GROUP_MASK) != 0) {
609  ok = false;
610  }
611  c = ok ? c : 0;
612  content.specialize(c);
613  return c;
614 }
