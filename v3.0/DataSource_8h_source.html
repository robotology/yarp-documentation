81  virtual void run() override {
82 
84  double now=yarp::os::Time::now();
85  double deltaT=0.0;
86 
87  if (counter==0)
88  {
89  lastSpoke=now;
90  timePrevious=now;
91  cumulativeT=0.0;
92  }
93  else
94  {
95  deltaT=now-timePrevious;
96  cumulativeT+=deltaT;
97  if (deltaT>maxT)
98  maxT=deltaT;
99  if (deltaT<minT)
100  minT=deltaT;
101  timePrevious=now;
102  }
103 
104  counter++;
105 
106  // print report
107  if (now-lastSpoke>REPORT_TIME)
108  {
109  yInfo("Read [%d] frames in %d[s], average period %.2lf[ms], min %.2lf[ms], max %.2lf[ms]\n",
110  counter,
111  REPORT_TIME,
112  (cumulativeT/counter)*1000,
113  minT*1000, maxT*1000);
114  cumulativeT=0;
115  counter=0;
116  minT=1e10;
117  maxT=0.0;
118  lastSpoke=now;
119  }
121 
122  T& datum = writer.get();
123 
124  dater.getDatum(datum);
125  if (addStamp) {
126  if (pPrecTime)
127  {
128  stamp=pPrecTime->getLastInputStamp();
129  }
130  else
131  {
132  stamp.update();
133  }
134  port.setEnvelope(stamp);
135  }
136  writer.write(!canDrop);
137  }
