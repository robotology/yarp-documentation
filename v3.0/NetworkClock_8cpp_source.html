116 void NetworkClock::delay(double seconds) {
117  if (seconds<=1E-12) {
118  return;
119  }
120 
121  listMutex.lock();
122  if (closing) {
123  // We are shutting down. The time signal is no longer available.
124  // Make a short delay and return.
125  listMutex.unlock();
126  SystemClock::delaySystem(seconds);
127  return;
128  }
129 
130  Waiters* waiters = static_cast<Waiters*>(pwaiters);
131 
132  std::pair<double, Semaphore*> waiter(now() + seconds, new Semaphore(0));
133  waiters->insert(waiters->end(), waiter);
134  listMutex.unlock();
135 
136  waiter.second->wait();
137  delete waiter.second;
138  waiter.second = nullptr;
139 }

145 bool NetworkClock::read(ConnectionReader& reader) {
146  Bottle bot;
147  bool ok = bot.read(reader);
148 
149  if(closing)
150  {
151  _time = -1;
152  return false;
153  }
154 
155  if (!ok && !closing)
156  {
157  YARP_ERROR(Logger::get(), "Error reading clock port");
158  return false;
159  }
160 
161  timeMutex.lock();
162  sec = bot.get(0).asInt32();
163  nsec = bot.get(1).asInt32();
164  _time = sec + (nsec*1e-9);
165  initted = true;
166  timeMutex.unlock();
167 
168  listMutex.lock();
169  Waiters* waiters = static_cast<Waiters*>(pwaiters);
170  Waiters::iterator waiter_i;
171 
172  waiter_i = waiters->begin();
173  while (waiter_i != waiters->end())
174  {
175  if (waiter_i->first - _time < 1E-12 )
176  {
177  Semaphore *waiterSemaphore = waiter_i->second;
178  waiter_i = waiters->erase(waiter_i);
179  if (waiterSemaphore)
180  waiterSemaphore->post();
181  }
182  else
183  ++waiter_i;
184  }
185  listMutex.unlock();
186  return true;
