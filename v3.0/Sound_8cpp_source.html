40 Sound& Sound::operator += (const Sound& alt) {
41  if (alt.channels!= channels)
42  {
43  printf ("unable to concatenate sounds with different number of channels!");
44  return *this;
45  }
46  if (alt.frequency!= frequency)
47  {
48  printf ("unable to concatenate sounds with different sample rate!");
49  return *this;
50  }
51 
52  Sound orig= *this;
53  this->resize(this->samples+alt.samples,channels);
54 
55  unsigned char* p1 = orig.getRawData();
56  unsigned char* p2 = alt.getRawData();
57  unsigned char* pout = this->getRawData();
58 
59  for (size_t ch=0; ch<channels; ch++)
60  {
61  size_t out1 = ch* this->getBytesPerSample() * this->samples;
62  size_t out2 = ch* this->getBytesPerSample() * this->samples + this->getBytesPerSample() * orig.samples;
63 
64  size_t ori1 = ch * orig.getBytesPerSample() * orig.samples;
65  size_t s1 = orig.getBytesPerSample() * orig.samples;
66 
67  size_t alt1 = ch * orig.getBytesPerSample() * alt.samples;
68  unsigned int s2 = alt.getBytesPerSample() * alt.samples;
69 
70  memcpy((void *) &pout[out1], (void *) (p1+ori1), s1);

73

75  return *this;
76 }
