92  virtual bool connectTopic(const std::string& dir,
93  bool srcIsTopic,
94  bool destIsTopic,
95  const Contact& src,
96  const Contact& dest,
97  ContactStyle style)
98  {
99  YARP_UNUSED(srcIsTopic);
100  Contact dynamicSrc = src;
101  Contact dynamicDest = dest;
102  Bottle cmd, reply;
103  cmd.addString(dir.c_str());
104  if (style.carrier!="") {
105  if (!destIsTopic) {
106  dynamicDest.setCarrier(style.carrier);
107  } else {
108  dynamicSrc.setCarrier(style.carrier);
109  }
110  }
111  cmd.addString(dynamicSrc.toString().c_str());
112  cmd.addString(dynamicDest.toString().c_str());
113  if (style.persistent) {
114  switch (style.persistenceType) {
115  case ContactStyle::END_WITH_FROM_PORT:
116  cmd.addString("from");
117  break;
118  case ContactStyle::END_WITH_TO_PORT:
119  cmd.addString("to");
120  break;
121  default:
122  break;
123  }
124  }
125  bool ok = false;
126  if (!NetworkBase::getQueryBypass()) {
127  ok = NetworkBase::write(getNameServerContact(),
128  cmd,
129  reply);
130  } else {
131  ContactStyle style;
132  ok = NetworkBase::writeToNameServer(cmd,
133  reply,
134  style);
135  }
136  bool fail = (reply.get(0).toString()=="fail")||!ok;
137  if (fail) {
138  if (!style.quiet) {
139  fprintf(stderr, "Failure: name server did not accept connection to topic.\n");
140  }
141  }
142  return !fail;
143  }
