41 AuthHMAC::AuthHMAC() :
42  authentication_enabled(false)
43 {
44  memset(&context, 0, sizeof(HMAC_CONTEXT));
45  static int auth_warning_shown = false;
46  if (auth_warning_shown) {
47  // If the warning was already shown, we have nothing to do.
48  // return as soon as possible
49  return;
50  }
51  std::string key;
52  ResourceFinder& rf = ResourceFinder::getResourceFinderSingleton();
53  std::string fname;
54  Network::lock();
55  ResourceFinderOptions opt;
56  opt.messageFilter = ResourceFinderOptions::ShowNone;
57  fname = rf.findFile("auth.conf", opt);
58  Network::unlock();
59 
60 
61  if (fname.empty()) {
62  //yInfo("Cannot find auth.conf file. Authentication disabled.\n");
63  auth_warning_shown = true;
64  return;
65  }
66 
67  Property config;
68  config.fromConfigFile(fname.c_str());
69  Bottle group = config.findGroup("AUTH");
70 
71  if (group.isNull()) {
72  yWarning("No \"AUTH\" group found in auth.conf file. Authentication disabled.\n");
73  auth_warning_shown = true;
74  return;
75  }
76 
77  key = group.find("key").asString();
78  if (!(key.length() > 0)) {
79  yWarning("No \"key\" found in \"AUTH\" group in auth.conf file. Authentication disabled.\n");
80  auth_warning_shown = true;
81  return;
82  }
83 
84  size_t key_len = key.length();
85  unsigned char * tmp = new unsigned char[key_len];
86  strcpy((char*) tmp, key.c_str());
87  HMAC_INIT(&context, tmp, (unsigned int)key_len);
88  delete[] tmp;
89  srand((unsigned)time(nullptr));
90 
91  if (!authentication_enabled) {
92  yInfo("Authentication enabled.\n");
93  authentication_enabled = true;
94  }
95 }

98 bool AuthHMAC::authSource(InputStream *streamIn, OutputStream *streamOut)
99 {
100 
101  if (!authentication_enabled) {
102  return true;
103  }
104 
105  /* ---------------
106  * 3-way auth
107  * Port A
108  * ---------------
109  */
110 
111  unsigned char nonce1[NONCE_LEN];
112  unsigned char nonce2[NONCE_LEN];
113  unsigned char nonce3[NONCE_LEN];
114 
115  unsigned char mac[DIGEST_SIZE];
116  unsigned char mac_check[DIGEST_SIZE];
117 
118  /* ---------------
119  * Send first msg: A->B
120  */
121  fill_nonce(nonce1);
122  HMAC_REINIT(&context);
123  HMAC_UPDATE(&context, nonce1, NONCE_LEN);
124  HMAC_FINAL(&context, mac, DIGEST_SIZE);
125  if (! send_hmac(streamOut, nonce1, mac)) {
126  return false;
127  }
128 
129  /* ---------------
130  * Receive and check second msg: B->A
131  */
132  if (! receive_hmac(streamIn, nonce2, mac)) {
133  return false;
134  }
135 
136  HMAC_REINIT(&context);
137  HMAC_UPDATE(&context, nonce1, NONCE_LEN);
138  HMAC_UPDATE(&context, nonce2, NONCE_LEN);
139  HMAC_FINAL(&context, mac_check, DIGEST_SIZE);
140  if (! check_hmac(mac, mac_check)) {
141  return false;
142  }
143  /* Authentication of B successful */
144 
145 
146  /* ---------------
147  * Send third msg: A->B
148  */
149  fill_nonce(nonce3);
150  HMAC_REINIT(&context);
151  HMAC_UPDATE(&context, nonce1, NONCE_LEN);
152  HMAC_UPDATE(&context, nonce2, NONCE_LEN);
153  HMAC_UPDATE(&context, nonce3, NONCE_LEN);
154  HMAC_FINAL(&context, mac, DIGEST_SIZE);
155  if (! send_hmac(streamOut, nonce3, mac)) {
156  return false;
157  }
158 
159  return true;
