34 bool yarp::dev::Navigation2DClient::open(yarp::os::Searchable &config)
35 {
36  m_local_name.clear();
37  m_navigation_server_name.clear();
38  m_map_locations_server_name.clear();
39  m_localization_server_name.clear();
40 
41  m_local_name = config.find("local").asString().c_str();
42  m_navigation_server_name = config.find("navigation_server").asString().c_str();
43  m_map_locations_server_name = config.find("map_locations_server").asString().c_str();
44  m_localization_server_name = config.find("localization_server").asString().c_str();
45 
46  if (m_local_name == "")
47  {
48  yError("Navigation2DClient::open() error you have to provide a valid 'local' param");
49  return false;
50  }
51 
52  if (m_navigation_server_name == "")
53  {
54  yError("Navigation2DClient::open() error you have to provide a valid 'navigation_server' param");
55  return false;
56  }
57 
58  if (m_map_locations_server_name == "")
59  {
60  yError("Navigation2DClient::open() error you have to provide valid 'map_locations_server' param");
61  return false;
62  }
63 
64  if (m_localization_server_name == "")
65  {
66  yError("Navigation2DClient::open() error you have to provide valid 'localization_server' param");
67  return false;
68  }
69 
70  if (config.check("period"))
71  {
72  m_period = config.find("period").asInt32();
73  }
74  else
75  {
76  m_period = 10;
77  yWarning("Navigation2DClient: using default period of %d ms" , m_period);
78  }
79 
80  std::string
81  local_rpc_1,
82  local_rpc_2,
83  local_rpc_3,
84  remote_rpc_1,
85  remote_rpc_2,
86  remote_rpc_3,
87  remote_streaming_name,
88  local_streaming_name;
89 
90  local_rpc_1 = m_local_name + "/navigation/rpc";
91  local_rpc_2 = m_local_name + "/locations/rpc";
92  local_rpc_3 = m_local_name + "/localization/rpc";
93  remote_rpc_1 = m_navigation_server_name + "/rpc";
94  remote_rpc_2 = m_map_locations_server_name + "/rpc";
95  remote_rpc_3 = m_localization_server_name + "/rpc";
96  remote_streaming_name = m_localization_server_name + "/stream:o";
97  local_streaming_name = m_local_name + "/stream:i";
98 
99  if (!m_rpc_port_navigation_server.open(local_rpc_1.c_str()))
100  {
101  yError("Navigation2DClient::open() error could not open rpc port %s, check network", local_rpc_1.c_str());
102  return false;
103  }
104 
105  if (!m_rpc_port_map_locations_server.open(local_rpc_2.c_str()))
106  {
107  yError("Navigation2DClient::open() error could not open rpc port %s, check network", local_rpc_2.c_str());
108  return false;
109  }
110 
111  if (!m_rpc_port_localization_server.open(local_rpc_3.c_str()))
112  {
113  yError("Navigation2DClient::open() error could not open rpc port %s, check network", local_rpc_3.c_str());
114  return false;
115  }
116 
117  /*
118  //currently unused
119  bool ok=Network::connect(remote_streaming_name.c_str(), local_streaming_name.c_str(), "tcp");
120  if (!ok)
121  {
122  yError("Navigation2DClient::open() error could not connect to %s", remote_streaming_name.c_str());
123  return false;
124  }*/
125 
126  bool ok = true;
127 
128  ok = Network::connect(local_rpc_1.c_str(), remote_rpc_1.c_str());
129  if (!ok)
130  {
131  yError("Navigation2DClient::open() error could not connect to %s", remote_rpc_1.c_str());
132  return false;
133  }
134 
135  ok = Network::connect(local_rpc_2.c_str(), remote_rpc_2.c_str());
136  if (!ok)
137  {
138  yError("Navigation2DClient::open() error could not connect to %s", remote_rpc_2.c_str());
139  return false;
140  }
141 
142  ok = Network::connect(local_rpc_3.c_str(), remote_rpc_3.c_str());
143  if (!ok)
144  {
145  yError("Navigation2DClient::open() error could not connect to %s", remote_rpc_3.c_str());
146  return false;
147  }
148 
149  return true;
150 }

160 bool yarp::dev::Navigation2DClient::getNavigationStatus(NavigationStatusEnum& status)
161 {
162  yarp::os::Bottle b;
163  yarp::os::Bottle resp;
164 
165  b.addVocab(VOCAB_INAVIGATION);
166  b.addVocab(VOCAB_NAV_GET_STATUS);
167  bool ret = m_rpc_port_navigation_server.write(b, resp);
168  if (ret)
169  {
170  if (resp.get(0).asVocab() != VOCAB_OK)
171  {
172  yError() << "Navigation2DClient::getNavigationStatus() received error from navigation server";
173  return false;
174  }
175  else
176  {
177  status = (NavigationStatusEnum) resp.get(1).asInt32();
178  return true;
179  }
180  }
181  else
182  {
183  yError() << "Navigation2DClient::getNavigationStatus() error on writing on rpc port";
184  return false;
185  }
186  return true;
