65 void yarp::os::Nodes::Private::clear()
66 {
67  mutex.lock();
68  for (auto& n : nodes_map) {
69  if (n.second.first) {
70  if (!n.second.second) {
71  delete n.second.first;
72  n.second.first = nullptr;
73  }
74  }
75  }
76  nodes_map.clear();
77  mutex.unlock();
78 
79  active = true;
80  active_name = "";
81  if (dummy != nullptr) {
82  delete dummy;
83  dummy = nullptr;
84  }
85 }

87 Node* yarp::os::Nodes::Private::getNode(const std::string& name, bool create)
88 {
89  NestedContact nc(name);
90  if (!nc.isNested()) {
91  return nullptr;
92  }
93  Node* node = nullptr;
94  auto it = nodes_map.find(nc.getNodeName());
95  if (it != nodes_map.end()) {
96  node = it->second.first;
97  } else if (create) {
98  mutex.lock();
99  node = new Node();
100  it = nodes_map.find(nc.getNodeName());
101  if (it == nodes_map.end()) {
102  nodes_map[nc.getNodeName()] = std::make_pair(node, false);
103  node->prepare(nc.getNodeName());
104  } else {
105  // The node was created by some other thread while this
106  // thread was waiting on the lock.
107  delete node;
108  node = it->second.first;
109  }
110  mutex.unlock();
111  }
112  return node;
113 }

168 Contact yarp::os::Nodes::Private::query(const std::string& name, const std::string& category)
169 {
170  if (!active) {
171  return Contact();
172  }
173  Contact result;
174  mutex.lock();
175  for (const auto& n : nodes_map) {
176  result = n.second.first->query(name, category);
