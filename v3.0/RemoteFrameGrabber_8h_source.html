398  virtual bool getImage(yarp::sig::ImageOf<yarp::sig::PixelRgb>& image) override {
399  mutex.lock();
400  if(no_stream == true)
401  {
402  image.zero();
403  mutex.unlock();
404  return false;
405  }
406 
407  if (reader.read(true)!=NULL) {
408  image = *(reader.lastRead());
409  lastHeight = image.height();
410  lastWidth = image.width();
411  mutex.unlock();
412  return true;
413  }
414  mutex.unlock();
415  return false;
416  }

418  virtual bool getImageCrop(cropType_id_t cropType, yarp::sig::VectorOf<std::pair<int, int> > vertices, yarp::sig::ImageOf<yarp::sig::PixelRgb>& image) override
419  {
420  yarp::os::Bottle cmd, response;
421  cmd.addVocab(VOCAB_FRAMEGRABBER_IMAGE);
422  cmd.addVocab(VOCAB_GET);
423  cmd.addVocab(VOCAB_CROP);
424  cmd.addInt32(cropType);
425  yarp::os::Bottle & list = cmd.addList();
426  for(size_t i=0; i<vertices.size(); i++)
427  {
428  list.addInt32(vertices[i].first);
429  list.addInt32(vertices[i].second);
430  }
431  port.write(cmd,response);
432 
433  // Parse the response
434  image.zero();
435  if( (response.get(0).asVocab() != VOCAB_CROP) || (response.size() != 5) || (!response.get(4).isBlob()))
436  {
437  yError() << "getImageCrop: malformed response message. Size is " << response.size();
438  return false;
439  }
440 
441  image.resize(response.get(2).asInt32(), response.get(3).asInt32());
442  unsigned char *pixelOut = image.getRawImage();
443 
444  if(response.get(4).asBlob())
445  memcpy(pixelOut, response.get(4).asBlob(), (size_t) image.getRawImageSize());
446 
447  return true;
448  }

469  virtual bool open(yarp::os::Searchable& config) override {

476  "port name to use locally").asString();
477  std::string carrier =
478  config.check("stream",yarp::os::Value("tcp"),
479  "carrier to use for streaming").asString();
480  port.open(local);
481  if (remote!="") {

483 
484  if(!config.check("no_stream") )
485  {
