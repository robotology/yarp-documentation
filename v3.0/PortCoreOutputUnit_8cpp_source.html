77 void PortCoreOutputUnit::run()
78 {
79  running = true;
80  sending = false;
81 
82  // By default, we don't start up a thread for outputs.
83 
84  if (!threaded) {
85  runSingleThreaded();
86  phase.post();
87  } else {
88  phase.post();
89  Route r = getRoute();
90  Logger log(r.toString().c_str(), Logger::get());
91  while (!closing) {
92  YARP_DEBUG(log, "PortCoreOutputUnit waiting");
93  activate.wait();
94  YARP_DEBUG(log, "PortCoreOutputUnit woken");
95  if (!closing) {
96  if (sending) {
97  YARP_DEBUG(log, "write something in background");
98  sendHelper();
99  YARP_DEBUG(log, "wrote something in background");
100  trackerMutex.lock();
101  if (cachedTracker != nullptr) {
102  void *t = cachedTracker;
103  cachedTracker = nullptr;
104  sending = false;
105  getOwner().notifyCompletion(t);
106  } else {
107  sending = false;
108  }
109  trackerMutex.unlock();
110  }
111  }
112  YARP_DEBUG(log, "wrote something in background");
113  }
114  YARP_DEBUG(log,
115  "PortCoreOutputUnit thread closing");
116  sending = false;
117  }
118 }

157 void PortCoreOutputUnit::closeBasic()
158 {
159  bool waitForOther = false;
160  if (op != nullptr) {
161  op->getConnection().prepareDisconnect();
162  Route route = op->getRoute();
163  if (op->getConnection().isConnectionless()||
164  op->getConnection().isBroadcast()) {
165  YARP_SPRINTF1(Logger::get(),
166  debug,
167  "output for route %s asking other side to close by out-of-band means",
168  route.toString().c_str());
169  NetworkBase::disconnectInput(route.getToName().c_str(),
170  route.getFromName().c_str(), true);
171  } else {
172  if (op->getConnection().canEscape()) {
173  BufferedConnectionWriter buf(op->getConnection().isTextMode(),
174  op->getConnection().isBareMode());
175  PortCommand pc('\0', std::string("q"));
176  pc.write(buf);
177  //printf("Asked for %s to close...\n",
178  // op->getRoute().toString().c_str());
179  waitForOther = op->write(buf);
180  }
181  }
182 
183  std::string msg = std::string("Removing output from ") +
184  route.getFromName() + " to " + route.getToName();
185 
186  if (Name(route.getToName()).isRooted()) {
187  if (Name(route.getFromName()).isRooted()) {
188  YARP_INFO(Logger::get(), msg);
189  }
190  }
191 
192  getOwner().reportUnit(this, false);
193 
194  // Report the disappearing connection
195  PortInfo info;
196  info.message = msg.c_str();
197  info.tag = yarp::os::PortInfo::PORTINFO_CONNECTION;
198  info.incoming = false;
199  info.created = false;
200  info.sourceName = route.getFromName().c_str();
201  info.targetName = route.getToName().c_str();
202  info.portName = info.sourceName;
203  info.carrierName = route.getCarrierName().c_str();
204  getOwner().report(info);
205  }
206 
207 
208  if (op != nullptr) {
209  if (waitForOther) {
210  // quit is only acknowledged in certain conditions
211  if (op->getConnection().isTextMode()&&
212  op->getConnection().supportReply()) {
213  InputStream& is = op->getInputStream();
214  ManagedBytes dummy(1);
215  is.read(dummy.bytes());
216  }
217  }
218  op->close();
219  delete op;
220  op = nullptr;
221  }
222 }

264 bool PortCoreOutputUnit::sendHelper()
265 {
266  bool replied = false;
267  if (op != nullptr) {
268  bool done = false;
269  BufferedConnectionWriter buf(op->getConnection().isTextMode(),
270  op->getConnection().isBareMode());
271  if (cachedReader != nullptr) {
272  buf.setReplyHandler(*cachedReader);
273  }
274 
275  if (op->getSender().modifiesOutgoingData())
276  {
277  if (op->getSender().acceptOutgoingData(*cachedWriter))
278  cachedWriter = &op->getSender().modifyOutgoingData(*cachedWriter);
279  else
280  return (done = true);
281  }
282 
283  if (op->getConnection().isLocal()) {
284  // WARNING Cast away const qualifier.
285  // This may actually cause bugs when using the local carrier
286  // with something that is actually const (i.e. that is using
287  // some parts of memory that cannot be written.
288  yarp::os::PortWriter* pw = const_cast<yarp::os::PortWriter*>(cachedWriter);
289  yarp::os::Portable* p = dynamic_cast<yarp::os::Portable*>(pw);
290  if(p == nullptr) {
291  YARP_ERROR(Logger::get(), "PortCoreOutputUnit: cast failed.");
292  return false;
293  }
294  buf.setReference(p);
295  } else {
296  yAssert(cachedWriter != nullptr);
297  bool ok = cachedWriter->write(buf);
298  if (!ok) {
299  done = true;
300  }
301 
302  bool suppressReply = (buf.getReplyHandler() == nullptr);
303 
304  if (!done) {
305  if (!op->getConnection().canEscape()) {
306  if (cachedEnvelope!="") {
307  op->getConnection().handleEnvelope(cachedEnvelope);
308  }
309  } else {
310  buf.addToHeader();
311 
312  if (cachedEnvelope!="") {
313  if (cachedEnvelope=="__ADMIN") {
314  PortCommand pc('a', "");
315  pc.write(buf);
316  } else {
317  PortCommand pc('\0', std::string(suppressReply ? "D " : "d ") + cachedEnvelope);
318  pc.write(buf);
319  }
320  } else {
321  PortCommand pc(suppressReply ? 'D' : 'd', "");
322  pc.write(buf);
323  }
324  }
325  }
326  }
327 
328  if (!done) {
329  if (op->getConnection().isActive()) {
330  replied = op->write(buf);
331  if (replied && op->getSender().modifiesReply() && cachedReader != nullptr) {
332  cachedReader = &op->getSender().modifyReply(*cachedReader);
333  }
334  }
335  if (!op->isOk()) {
336  done = true;
337  }
338  }
339 
340  if (buf.dropRequested()) {
341  done = true;
342  }
343  if (done) {
344  closeBasic();
345  closing = true;
346  finished = true;
347  setDoomed();
348  }
349  }
350 
351 
352  return replied;
353 }

355 void *PortCoreOutputUnit::send(const yarp::os::PortWriter& writer,
356  yarp::os::PortReader *reader,
357  const yarp::os::PortWriter *callback,
358  void *tracker,
359  const std::string& envelopeString,
360  bool waitAfter,
361  bool waitBefore,
362  bool *gotReply)
363 {
364  bool replied = false;
365 
366  if (op != nullptr) {
367  if (!op->getConnection().isActive()) {
368  return tracker;
369  }
370  }
371 
372  if (!waitBefore || !waitAfter) {
373  if (running == false) {
374  // we must have a thread if we're going to be skipping waits
375  threaded = true;
376  YARP_DEBUG(Logger::get(), "starting a thread for output");
377  start();
378  YARP_DEBUG(Logger::get(), "started a thread for output");
379  }
380  }
381 
382  if ((!waitBefore)&&waitAfter) {
383  YARP_ERROR(Logger::get(), "chosen port wait combination not yet implemented");
384  }
385  if (!sending) {
386  cachedWriter = &writer;
387  cachedReader = reader;
388  cachedCallback = callback;
389  cachedEnvelope = envelopeString;
390 
391  sending = true;
392  if (waitAfter==true) {
393  replied = sendHelper();
394  sending = false;
395  } else {
396  trackerMutex.lock();
397  void *nextTracker = tracker;
398  tracker = cachedTracker;
399  cachedTracker = nextTracker;
400  activate.post();
401  trackerMutex.unlock();
402  }
403  } else {
404  YARP_DEBUG(Logger::get(),
405  "skipping connection tagged as sending something");
406  }
407 
408  if (waitAfter) {
409  if (gotReply != nullptr) {
410  *gotReply = replied;
411  }
412  }
413 
414  // return tracker that we no longer need
415  return tracker;
416 }
