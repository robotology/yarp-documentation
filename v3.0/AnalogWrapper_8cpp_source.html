52 bool AnalogServerHandler::_handleIAnalog(yarp::os::Bottle &cmd, yarp::os::Bottle &reply)
53 {
54  if (is==nullptr)
55  return false;
56 
57  int msgsize=cmd.size();
58 
59  int code=cmd.get(1).asVocab();
60  switch (code)
61  {
62  case VOCAB_CALIBRATE:
63  if (msgsize==2)
64  is->calibrateSensor();
65  else
66  {
67  //read Vector of values and pass to is->calibrate();
68  }
69  return true;
70  break;
71  case VOCAB_CALIBRATE_CHANNEL:
72  if (msgsize==3)
73  {
74  int ch=cmd.get(2).asInt32();
75  is->calibrateChannel(ch);
76  }
77  if (msgsize==4)
78  {
79  int ch=cmd.get(2).asInt32();
80  double v=cmd.get(3).asFloat64();
81  is->calibrateChannel(ch, v);
82  }
83 
84  return true;
85  break;
86  default:
87  return false;
88  }
89 }

91 bool AnalogServerHandler::read(yarp::os::ConnectionReader& connection)
92 {
93  yarp::os::Bottle in;
94  yarp::os::Bottle out;
95  bool ok=in.read(connection);
96  if (!ok) return false;
97 
98  // parse in, prepare out
99  int code = in.get(0).asVocab();
100  bool ret=false;
101  if (code==VOCAB_IANALOG)
102  {
103  ret=_handleIAnalog(in, out);
104  }
105 
106  if (!ret)
107  {
108  out.clear();
109  out.addVocab(VOCAB_FAILED);
110  }
111 
112  yarp::os::ConnectionWriter *returnToSender = connection.getWriter();
113  if (returnToSender!=nullptr) {
114  out.write(*returnToSender);
115  }
116  return true;
117 }

369 bool AnalogWrapper::checkROSParams(Searchable &config)
370 {
371  // check for ROS parameter group
372  if(!config.check("ROS") )
373  {
374  useROS = ROS_disabled;
375  yInfo() << "No ROS group found in config file ... skipping ROS initialization.";
376  return true;
377  }
378 
379  yInfo() << "ROS group was FOUND in config file.";
380 
381  Bottle &rosGroup = config.findGroup("ROS");
382  if(rosGroup.isNull())
383  {
384  yError() << sensorId << "ROS group params is not a valid group or empty";
385  useROS = ROS_config_error;
386  return false;
387  }
388 
389  // check for useROS parameter
390  if(!rosGroup.check("useROS"))
391  {
392  yError() << sensorId << " cannot find useROS parameter, mandatory when using ROS message. \n \
393  Allowed values are true, false, ROS_only";
394  useROS = ROS_config_error;
395  return false;
396  }
397  std::string ros_use_type = rosGroup.find("useROS").asString();
398  if(ros_use_type == "false")
399  {
400  yInfo() << sensorId << "useROS topic if set to 'false'";
401  useROS = ROS_disabled;
402  return true;
403  }
404  else if(ros_use_type == "true")
405  {
406  yInfo() << sensorId << "useROS topic if set to 'true'";
407  useROS = ROS_enabled;
408  }
409  else if(ros_use_type == "only")
410  {
411  yInfo() << sensorId << "useROS topic if set to 'only";
412  useROS = ROS_only;
413  }
414  else
415  {
416  yInfo() << sensorId << "useROS parameter is set to unvalid value ('" << ros_use_type << "'), supported values are 'true', 'false', 'only'";
417  useROS = ROS_config_error;
418  return false;
419  }
420 
421  // check for ROS_nodeName parameter
422  if(!rosGroup.check("ROS_nodeName"))
423  {
424  yError() << sensorId << " cannot find ROS_nodeName parameter, mandatory when using ROS message";
425  useROS = ROS_config_error;
426  return false;
427  }
428  rosNodeName = rosGroup.find("ROS_nodeName").asString(); // TODO: check name is correct
429  yInfo() << sensorId << "rosNodeName is " << rosNodeName;
430 
431  // check for ROS_topicName parameter
432  if(!rosGroup.check("ROS_topicName"))
433  {
434  yError() << sensorId << " cannot find ROS_topicName parameter, mandatory when using ROS message";
435  useROS = ROS_config_error;
436  return false;
437  }
438  rosTopicName = rosGroup.find("ROS_topicName").asString();
439  yInfo() << sensorId << "ROS_topicName is " << rosTopicName;
440 
441  // check for ROS_msgType parameter
442  if (!rosGroup.check("ROS_msgType"))
443  {
444  yError() << sensorId << " cannot find ROS_msgType parameter, mandatory when using ROS message";
445  useROS = ROS_config_error;
446  return false;
447  }
448  rosMsgType = rosGroup.find("ROS_msgType").asString();
449 
450  // check for frame_id parameter
451  if (rosMsgType == "geometry_msgs/WrenchStamped")
452  {
453  yInfo() << sensorId << "ROS_msgType is " << rosMsgType;
454  if (!rosGroup.check("frame_id"))
455  {
456  yError() << sensorId << " cannot find frame_id parameter, mandatory when using ROS message";
457  useROS = ROS_config_error;
458  return false;
459  }
460  frame_id = rosGroup.find("frame_id").asString();
461  yInfo() << sensorId << "frame_id is " << frame_id;
462  }
463  else if (rosMsgType == "sensor_msgs/JointState")
464  {
465  std::string jointName = "";
466  yInfo() << sensorId << "ROS_msgType is " << rosMsgType;
467  bool oldParam = false;
468  bool newParam = false;
469 
470  if(rosGroup.check("joint_names"))
471  {
472  oldParam = true;
473  jointName = "joint_names";
474  yWarning() << sensorId << " using DEPRECATED 'joint_names' parameter. Please use 'jointNames' instead.";
475  }
476 
477  if(rosGroup.check("jointNames"))
478  {
479  newParam = true;
480  jointName = "jointNames";
481  }
482 
483  if(!oldParam && !newParam)
484  {
485  yError() << sensorId << " missing 'jointNames' parameter needed when broadcasting 'sensor_msgs/JointState' message type";
486  useROS = ROS_config_error;
487  return false;
488  }
489  // Throw an error if both new and old are present
490  if(oldParam && newParam)
491  {
492  yError() << sensorId << " found both DEPRECATED 'joint_names' and new 'jointNames' parameters. Please remove the old 'joint_names' from your config file.";
493  useROS = ROS_config_error;
494  return false;
495  }
496 
497  yarp::os::Bottle& jnam = rosGroup.findGroup(jointName);
498  if(jnam.isNull())
499  {
500  yError() << sensorId << "Cannot find 'jointNames' parameters.";
501  return false;
502  }
503 
504  // Cannot check number of channels here because need to wait for the attach function
505  int joint_names_size = jnam.size()-1;
506  for (int i = 0; i < joint_names_size; i++)
507  {
508  ros_joint_names.push_back(jnam.get(i+1).asString());
509  }
510  }
511  else
512  {
513  yError() << sensorId << "ROS_msgType '" << rosMsgType << "' not supported ";
514  return false;
515  }
516 
517  return true;

562  success = true;

579  yError() << sensorId << " something went wrong with ROS configuration, we should never be here!!!";

581  } break;
582  }

586 bool AnalogWrapper::open(yarp::os::Searchable &config)
587 {
588  Property params;
589  params.fromString(config.toString().c_str());
590  yTrace() << "AnalogWrapper params are: " << config.toString();
591 
592  if (!config.check("period"))
593  {
594  yError() << "AnalogWrapper: missing 'period' parameter. Check you configuration file\n";
595  return false;
596  }
597  else
598  {
599  _rate = config.find("period").asInt32();
600  }
601 
602  if (config.check("deviceId"))
603  {
604  yError() << "AnalogWrapper: the parameter 'deviceId' has been removed, please use parameter 'name' instead. \n"
605  << "e.g. In the FT wrapper configuration files of your robot, replace '<param name=""deviceId""> left_arm </param>' \n"
606  << "with '/icub/left_arm/analog:o' ";
607  return false;
608  }
609 
610  if (!config.check("name"))
611  {
612  yError() << "AnalogWrapper: missing 'name' parameter. Check you configuration file; it must be like:\n"
613  " name: full name of the port, like /robotName/deviceId/sensorType:o";
614  return false;
615  }
616  else
617  {
618  streamingPortName = config.find("name").asString().c_str();
619  setId("AnalogServer");
620  }
621 
622  if(!checkROSParams(config) )
623  {
624  yError() << sensorId << "ROS parameter are not correct, check your configuration file";
625  return false;
626  }
627 
628  if(!initialize_YARP(config) )
629  {
630  yError() << sensorId << "Error initializing YARP ports";
631  return false;
632  }
633 
634  if(!initialize_ROS() )
635  {
636  yError() << sensorId << "Error initializing ROS topics";
637  return false;
638  }
639 
640  // check if we need to create subdevice or if they are
641  // passed later on thorugh attachAll()
642  if(config.check("subdevice"))
643  {
644  ownDevices=true;
645  if(! openAndAttachSubDevice(config))
646  {
647  yError("AnalogWrapper: error while opening subdevice\n");
648  return false;
649  }
650  }
651  else
652  {
653  ownDevices=false;
654  if(!openDeferredAttach(config))
655  return false;
656  }
657 
658  return true;
659 }

661 bool AnalogWrapper::initialize_YARP(yarp::os::Searchable &params)
662 {
663  switch(useROS)
664  {
665  case ROS_only:
666  {
667  yInfo() << sensorId << " No YARP initialization required";
668  return true;
669  } break;
670 
671  default:
672  {
673  // Create the list of ports
674  // port names are optional, do not check for correctness.
675  if(!params.check("ports"))
676  {
677  // if there is no "ports" section open only 1 port and use name as is.
678  if (Network::exists(streamingPortName + "/rpc:i") || Network::exists(streamingPortName))
679  {
680  yError() << "AnalogWrapper: unable to open the analog server, address conflict";
681  return false;
682  }
683  createPort((streamingPortName ).c_str(), _rate );
684  // since createPort always return true, check the port is really been opened is done here
685  if(! Network::exists(streamingPortName + "/rpc:i"))
686  return false;
687  }
688  else
689  {
690  Bottle *ports=params.find("ports").asList();
691 
692  Value &deviceChannels = params.find("channels");
693  if (deviceChannels.isNull())
694  {
695  yError("AnalogWrapper: 'channels' parameters was not found in config file.");
696  return false;
697  }
698 
699  int nports=ports->size();
700  int sumOfChannels = 0;
701  std::vector<AnalogPortEntry> tmpPorts;
702  tmpPorts.resize(nports);
703 
704  for(size_t k=0; k<ports->size(); k++)
705  {
706  Bottle parameters=params.findGroup(ports->get(k).asString().c_str());
707 
708  if (parameters.size()!=5)
709  {
710  yError() << "AnalogWrapper: check skin port parameters in part description, I was expecting "
711  << ports->get(k).asString().c_str() << " followed by four integers";
712  yError() << " your param is " << parameters.toString();
713  return false;
714  }
715 
716  if (Network::exists(streamingPortName + "/" + string(ports->get(k).asString().c_str()) + "/rpc:i")
717  || Network::exists(streamingPortName + "/" + string(ports->get(k).asString().c_str())))
718  {
719  yError() << "AnalogWrapper: unable to open the analog server, address conflict";
720  return false;
721  }
722  int wBase=parameters.get(1).asInt32();
723  int wTop=parameters.get(2).asInt32();
724  int base=parameters.get(3).asInt32();
725  int top=parameters.get(4).asInt32();
726 
727  yDebug()<<"--> "<<wBase<<" "<<wTop<<" "<<base<<" "<<top;
728 
729  //check consistenty
730  if(wTop-wBase != top-base){
731  yError() << "AnalogWrapper: numbers of mapped taxels do not match, check "
732  << ports->get(k).asString().c_str() << " port parameters in part description";
733  return false;
734  }
735  int portChannels = top-base+1;
736 
737  tmpPorts[k].length = portChannels;
738  tmpPorts[k].offset = wBase;
739  yDebug() << "opening port " << ports->get(k).asString().c_str();
740  tmpPorts[k].port_name = streamingPortName+ "/" + string(ports->get(k).asString().c_str());
741 
742  sumOfChannels+=portChannels;
743  }
744  createPorts(tmpPorts, _rate);
745 
746  if (sumOfChannels!=deviceChannels.asInt32())
747  {
748  yError() << "AnalogWrapper: Total number of mapped taxels does not correspond to total taxels";
749  return false;
750  }
751  }
752  } break;
753  }
754  return true;
755 }

766 void AnalogWrapper::run()
767 {
768  int first, last, ret;
769 
770  if (analogSensor_p!=nullptr)
771  {
772  ret=analogSensor_p->read(lastDataRead);
773 
774  if (ret==yarp::dev::IAnalogSensor::AS_OK)
775  {
776  if (lastDataRead.size()>0)
777  {
778  if(useROS != ROS_only)
779  {
780  lastStateStamp.update();
781  // send the data on the port(s), splitting them as specified in the config file
782  for(unsigned int i=0; i<analogPorts.size(); i++)
783  {
784  yarp::sig::Vector &pv = analogPorts[i].port.prepare();

786  if(analogPorts[i].length==-1) // read the max length available
787  last = lastDataRead.size()-1;

790

794  <<" Vector size expected to be at least "<<last<<" whereas it is "<< lastDataRead.size();
795  continue;
796  }
797  pv = lastDataRead.subVector(first, last);
798 
799  analogPorts[i].port.setEnvelope(lastStateStamp);
800  analogPorts[i].port.write();
801  }
802  }
803 
804  if (useROS != ROS_disabled && rosMsgType == "geometry_msgs/WrenchStamped")

808  rosData.header.stamp = yarp::os::Time::now();

814

816  rosData.wrench.torque.y = lastDataRead[4];
817  rosData.wrench.torque.z = lastDataRead[5];
818

820  }

822  {
823  yarp::rosmsg::sensor_msgs::JointState rosData;
824  size_t data_size = lastDataRead.size();
825  rosData.name.resize(data_size);
826  rosData.position.resize(data_size);
827  rosData.velocity.resize(data_size);
828  rosData.effort.resize(data_size);
829 
830  if (data_size != ros_joint_names.size())
831  {
832  yDebug() << "Invalid jointNames size:" << data_size << "!=" << ros_joint_names.size();
833  }
834  else
835  {
836  for (size_t i = 0; i< data_size; i++)
837  {

843  rosData.velocity[i] = 0;
844  rosData.effort[i] = 0;
845  }
846  }
847  }
848  rosData.header.seq = rosMsgCounter++;
849  rosData.header.stamp = yarp::os::Time::now();
850  rosPublisherJointPort.write(rosData);
851  }
852  }
853  else
854  {
855  yError("AnalogWrapper: %s: vector size non valid: %lu", sensorId.c_str(), static_cast<unsigned long> (lastDataRead.size()));
856  }
857  }
858  else
859  {
860  switch(ret)
861  {
862  case IAnalogSensor::AS_OVF:
863  yError("AnalogWrapper: %s: Sensor returned overflow error (code %d).", sensorId.c_str(), ret);
864  break;
865  case IAnalogSensor::AS_TIMEOUT:

867  break;
868  case IAnalogSensor::AS_ERROR:
869  default:
870  yError("AnalogWrapper: %s: Sensor returned error with code %d.", sensorId.c_str(), ret);
871  break;
872  }
873  }
874  }
