31 bool MultipleAnalogSensorsServer::open(os::Searchable& config)
32 {
33  if (!config.check("name"))
34  {
35  yError("MultipleAnalogSensorsServer: missing name parameter, exiting.");
36  return false;
37  }
38 
39  if (!config.check("period"))
40  {
41  yError("MultipleAnalogSensorsClient: missing period parameter, exiting.");
42  return false;
43  }
44 
45 
46  if (!config.find("period").isInt32())
47  {
48  yError("MultipleAnalogSensorsClient: period parameter is present but it is not an integer, exiting.");
49  return false;
50  }
51 
52  m_periodInS = config.find("period").asInt32() / 1000.0;
53 
54  if (m_periodInS <= 0)
55  {
56  yError("MultipleAnalogSensorsClient: period parameter is present (%f) but it is not a positive integer, exiting.", m_periodInS);
57  return false;
58  }
59 
60  std::string name = config.find("name").asString().c_str();
61 
62  // Reserve a fair amount of elements
63  // It would be great if yarp::sig::Vector had a reserve method
64  m_buffer.resize(100);
65  m_buffer.resize(0);
66 
67  // TODO(traversaro) Add port name validation when ready,
68  // see https://github.com/robotology/yarp/pull/1508
69  m_RPCPortName = name + "/rpc:o";
70  m_streamingPortName = name + "/measures:o";
71 
72  return true;
73 }

85 bool MultipleAnalogSensorsServer::populateSensorsMetadata(Interface * wrappedDeviceInterface,
86  std::vector<SensorMetadata>& metadataVector, const std::string& tag,
87  size_t (Interface::*getNrOfSensorsMethodPtr)() const,
88  bool (Interface::*getNameMethodPtr)(size_t, std::string&) const,
89  bool (Interface::*getFrameNameMethodPtr)(size_t, std::string&) const)
90 {
91  if (wrappedDeviceInterface)
92  {
93  size_t nrOfSensors = MAS_CALL_MEMBER_FN(wrappedDeviceInterface, getNrOfSensorsMethodPtr)();
94  metadataVector.resize(nrOfSensors);
95  for (size_t i=0; i < nrOfSensors; i++)
96  {
97  std::string sensorName;
98  bool ok = MAS_CALL_MEMBER_FN(wrappedDeviceInterface, getNameMethodPtr)(i, sensorName);
99  if (!ok)
100  {
101  yError() << "MultipleAnalogSensorsServer: Failure in reading name of sensor of type "
102  << tag.c_str() << " at index " << i << " .";
103  return false;
104  }
105  std::string frameName;
106  ok = MAS_CALL_MEMBER_FN(wrappedDeviceInterface, getFrameNameMethodPtr)(i, frameName);
107  if (!ok)
108  {
109  yError() << "MultipleAnalogSensorsServer: Failure in reading frame name of sensor of type "
110  << tag.c_str() << " at index " << i << " .";
111  return false;
112  }
113 
114  metadataVector[i].name = sensorName;
115  metadataVector[i].frameName = frameName;
116  metadataVector[i].additionalMetadata = "";
117  }
118 
119  }
120  else
121  {
122  metadataVector.resize(0);
123  }
124  return true;
125 }

202 bool MultipleAnalogSensorsServer::attachAll(const PolyDriverList& p)
203 {
204  // Attach the device
205  if (p.size() > 1)
206  {
207  yError("MultipleAnalogSensorsServer: this device only supports exposing a "
208  "single MultipleAnalogSensors device on YARP ports, but %d devices have been passed in attachAll.", p.size());
209  yError("MultipleAnalogSensorsServer: please use the multipleanalogsensorsremapper device to combine several device in a new device.");
210  detachAll();
211  return false;
212  }
213 
214  if (p.size() == 0)
215  {
216  yError("MultipleAnalogSensorsServer: no device passed to attachAll, please pass a device to expose on YARP ports.");
217  return false;
218  }
219 
220  PolyDriver* poly = p[0]->poly;
221 
222  if (!poly)
223  {
224  yError("MultipleAnalogSensorsServer: null pointer passed to attachAll.");
225  return false;
226  }
227 
228  // View all the interfaces
229  poly->view(m_iThreeAxisGyroscopes);
230  poly->view(m_iThreeAxisLinearAccelerometers);
231  poly->view(m_iThreeAxisMagnetometers);
232  poly->view(m_iOrientationSensors);
233  poly->view(m_iTemperatureSensors);
234  poly->view(m_iSixAxisForceTorqueSensors);
235  poly->view(m_iContactLoadCellArrays);
236  poly->view(m_iEncoderArrays);
237  poly->view(m_iSkinPatches);
238 
239 
240  // Populate the RPC data to be served on the RPC port
241  bool ok = populateAllSensorsMetadata();
242 
243  if(!ok)
244  {
245  detachAll();
246  return false;
247  }
248 
249  // Attach was successful, open the ports
250  ok = m_streamingPort.open(m_streamingPortName);
251  if (!ok)
252  {
253  yError("MultipleAnalogSensorsServer: failure in opening port named %s.", m_streamingPortName.c_str());
254  detachAll();
255  return false;
256  }
257 
258  ok = this->yarp().attachAsServer(m_rpcPort);
259  if (!ok)
260  {
261  yError("MultipleAnalogSensorsServer: failure in attaching RPC port to thrift RPC interface.");
262  detachAll();
263  return false;
264  }
265 
266  ok = m_rpcPort.open(m_RPCPortName);
267  if (!ok)
268  {
269  yError("MultipleAnalogSensorsServer: failure in opening port named %s.", m_RPCPortName.c_str());
270  detachAll();
271  return false;
272  }
273 
274  // Set rate period
275  ok = this->setPeriod(m_periodInS);
276  ok = ok && this->start();
277 
278  return ok;
279 }

300 template<typename Interface>
301 bool MultipleAnalogSensorsServer::genericStreamData(Interface* wrappedDeviceInterface,
302  const std::vector< SensorMetadata >& metadataVector,
303  std::vector< SensorMeasurement >& streamingDataVector,
304  MAS_status (Interface::*getStatusMethodPtr)(size_t) const,
305  bool (Interface::*getMeasureMethodPtr)(size_t, yarp::sig::Vector&, double&) const)
306 {
307  if (wrappedDeviceInterface)
308  {
309  size_t nrOfSensors = metadataVector.size();
310  streamingDataVector.resize(nrOfSensors);
311  for (size_t i=0; i < nrOfSensors; i++)
312  {
313  yarp::sig::Vector& outputBuffer = streamingDataVector[i].measurement;
314  double& outputTimestamp = streamingDataVector[i].timestamp;
315  // TODO(traversaro): resize the buffer to the correct size
316  MAS_CALL_MEMBER_FN(wrappedDeviceInterface, getMeasureMethodPtr)(i, outputBuffer, outputTimestamp);
317  MAS_status status = MAS_CALL_MEMBER_FN(wrappedDeviceInterface, getStatusMethodPtr)(i);
318  if (status != MAS_OK)
319  {
320  yError("MultipleAnalogSensorsServer: failure in reading data from sensor %s, no data will be sent on the port.",
321  m_sensorMetadata.ThreeAxisGyroscopes[i].name.c_str());
322  return false;
323  }
324  }
325  }
326 
327  return true;
328 }
