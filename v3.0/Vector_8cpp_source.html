44 bool VectorBase::read(yarp::os::ConnectionReader& connection) {
45  // auto-convert text mode interaction
46  connection.convertTextMode();
47  VectorPortContentHeader header;
48  bool ok = connection.expectBlock((char*)&header, sizeof(header));
49  if (!ok) return false;
50  if (header.listLen > 0 &&
51  header.listTag == (BOTTLE_TAG_LIST | getBottleTag()))
52  {
53  if ((size_t)getListSize() != (size_t)(header.listLen))
54  resize(header.listLen);
55  char* ptr = getMemoryBlock();
56  yAssert(ptr != nullptr);
57  int elemSize=getElementSize();
58  ok = connection.expectBlock(ptr, elemSize*header.listLen);
59  if (!ok) return false;
60  } else {
61  return false;
62  }
63 
64  return !connection.isError();
65 }
