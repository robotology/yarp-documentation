48  void finishWrites() {
49  YARP_DEBUG(Logger::get(), "finishing writes");
50  bool done = false;
51  while (!done) {
52  stateSema.wait();
53  if (port) {
54  if (!port->isOpen()) outCt = 0;
55  }
56  done = (outCt == 0);
57  if (!done) {
58  finishing = true;
59  }
60  stateSema.post();
61  if (!done) {
62  completionSema.wait();
63  }
64  }
65  YARP_DEBUG(Logger::get(), "finished writes");
66  }
67 
68  const void* get() {
69  if (callback!=nullptr) {
70  // (Safe to check outside mutex)
71  // oops, there is already a prepared and unwritten
72  // object. best remove it.
73  YARP_DEBUG(Logger::get(), "releasing unused buffer");
74  release();
75  }
76  stateSema.wait();
77  PortCorePacket *packet = packets.getFreePacket();
78  yAssert(packet!=nullptr);
79  if (packet->getContent()==nullptr) {
80  YARP_DEBUG(Logger::get(), "creating a writer buffer");
81  //packet->setContent(owner.create(*this, packet), true);
82  yarp::os::PortWriterWrapper *wrapper =
83  owner.create(*this, packet);
84  //packet->setContent(wrapper, true);
85  packet->setContent(wrapper->getInternal(), false,
86  wrapper, true);
87  }
88  stateSema.post();
89 
90  current = packet->getContent();
91  callback = packet->getCallback();
92  return callback;
93  }
94 
95  bool release() {
96  stateSema.wait();
97  const PortWriter* cback = callback;
98  current = nullptr;
99  callback = nullptr;
100  stateSema.post();
101  if (cback!=nullptr) {
102  stateSema.wait();
103  outCt++;
104  stateSema.post();
105  cback->onCompletion();
106  }
107  return cback!=nullptr;
108  }
109 
110  virtual void onCompletion(void *tracker) override {
111  stateSema.wait();
112  YARP_DEBUG(Logger::get(), "freeing up a writer buffer");
113  packets.freePacket((PortCorePacket*)tracker, false);
114  outCt--;
115  bool sig = finishing;
116  finishing = false;
117  stateSema.post();
118  if (sig) {
119  completionSema.post();
120  }
121  }

135  void write(bool strict) {
136  if (strict) {
137  finishWrites();
138  }
139  stateSema.wait();
140  const PortWriter *active = current;
141  const PortWriter *cback = callback;
142  current = nullptr;
143  callback = nullptr;
144  stateSema.post();
145  if (active!=nullptr && port!=nullptr) {
146  stateSema.wait();
147  outCt++;
148  stateSema.post();
149  port->write(*active, cback);
150  }
151  }
