24 class MultiNameSpaceHelper {
25 public:

42  void clear() {
43  // remove all namespaces and reset flags
44  for (int i=0; i<(int)spaces.size(); i++) {
45  NameSpace *ns = spaces[i];
46  if (ns) {
47  delete ns;
48  ns = nullptr;
49  }
50  }
51  spaces.clear();
52  _localOnly = true;
53  _usesCentralServer = false;
54  _serverAllocatesPortNumbers = false;
55  _connectionHasNameOfEndpoints = true;
56  }
57 
58  void scan() {
59  // reset flags
60  _localOnly = true;
61  _usesCentralServer = false;
62  _serverAllocatesPortNumbers = true;
63  // now scan each namespace
64  for (int i=0; i<(int)spaces.size(); i++) {
65  NameSpace *ns = spaces[i];
66  if (!ns) continue;
67  // if any namespace is nonlocal, combination is nonlocal
68  if (!ns->localOnly()) _localOnly = false;
69  // if any namespace uses a central server, combination also does
70  if (ns->usesCentralServer()) _usesCentralServer = true;
71  // if any namespace doesn't allocate port numbers, combination
72  // cannot be relied on to do so either
73  if (!ns->serverAllocatesPortNumbers()) {
74  _serverAllocatesPortNumbers = false;
75  }
76  // if any namespace lacks informed connections, combination
77  // cannot be relied on to be informed either
78  if (!ns->connectionHasNameOfEndpoints()) {
79  _connectionHasNameOfEndpoints = false;
80  }
81  }
82  }
83 
84  bool setLocalMode(bool flag) {
85  // remove any existing namespaces
86  clear();
87  if (flag) {
88  // add a dummy local namespace
89  NameSpace *ns = new YarpDummyNameSpace;
90  spaces.push_back(ns);
91  }
92  // cache flags
93  scan();
94  return true;
95  }
96 
97  bool activate(bool force = false) {
98  if (force) {
99  // wipe if forced
100  clear();
101  }
102  // return if namespaces already present
103  if (spaces.size()!=0) return true;
104  // read namespace list from config file
105  NameConfig conf;
106  if (!conf.fromFile()) {
107  double now = SystemClock::nowSystem();
108  static double last_shown = now-10;
109  if (now-last_shown>3) {
110  last_shown = now;
111  fprintf(stderr, "warning: YARP name server(s) not configured, ports will be anonymous\n");
112  fprintf(stderr, "warning: check your namespace and settings with 'yarp detect'\n");
113  }
114  return false;
115  }
116  Bottle ns = conf.getNamespaces();
117  // loop through namespaces
118  for (size_t i=0; i<ns.size(); i++) {
119  std::string n = ns.get(i).asString();
120  NameConfig conf2;
121  // read configuration of individual namespace
122  if (!conf2.fromFile(n.c_str())) {
123  fprintf(stderr, "Could not find namespace %s\n",
124  n.c_str());
125  continue;
126  }
127  std::string mode = conf2.getMode();
128  Contact address = conf2.getAddress();
129  address.setName(n);
130  if (mode=="yarp"||mode=="//") {
131  // add a yarp namespace
132  NameSpace *ns = new YarpNameSpace(address);
133  spaces.push_back(ns);
134  } else if (mode=="ros") {
135  // add a ros namespace
136  NameSpace *ns = new RosNameSpace(address);
137  spaces.push_back(ns);
138  } else if (mode=="local") {
139  NameSpace *ns = new YarpDummyNameSpace;
140  spaces.push_back(ns);
141  } else {
142  // shrug
143  YARP_SPRINTF1(Logger::get(), error,
144  "cannot deal with namespace of type %s",
145  mode.c_str());
146  return false;
147  }
148  }
149  // cache flags
150  scan();
151  return true;
152  }
153 
154  Contact getNameServerContact() {
155  activate(); // make sure we've loaded namespace(s)
156  if (spaces.size()>0) {
157  // return first name server
158  return spaces[0]->getNameServerContact();
159  }
160  return Contact();
161  }
162 
163  Contact queryName(const std::string& name) {
164  activate();
165  // try query against each namespace in order
166  for (int i=0; i<(int)spaces.size(); i++) {
167  NameSpace *ns = spaces[i];
168  if (!ns) continue;
169  if (ns->getNameServerName()==name) {
170  // optimization: return cached server address for
171  // port names that match name of namespace
172  return ns->getNameServerContact();
173  }
174  Contact result = ns->queryName(name);
175  // return a result once we get one, skipping any remaining
176  // namespaces
177  if (result.isValid()) return result;
178  }
179  return Contact();
180  }
181 
182  // return one namespace, any namespace (in fact always first)
183  NameSpace *getOne() {
184  activate();
185  if (spaces.size()==0) {
186  return nullptr;
187  }
188  return spaces[0];
189  }
190 
191  // return full list of namespaces
192  SpaceList& getAll() {
193  activate();
194  return spaces;
195  }
196 };

305 Contact MultiNameSpace::registerName(const std::string& name) {
306  SpaceList lst = HELPER(this).getAll();
307  Contact result;
308  // loop through namespaces
309  for (int i=0; i<(int)lst.size(); i++) {
310  Contact iresult;
311  // Register name with namespace. If contact information is
312  // fleshed out while registering, we carry that along for
313  // registration with the next namespace.
314  if (result.getPort()<=0) {
315  iresult = lst[i]->registerName(name);
316  } else {
317  iresult = lst[i]->registerContact(result);
318  }
319  if (i==0 || result.getPort()<=0) {
320  result = iresult;
321  }
322  }
323  return result;
324 }

372 Contact MultiNameSpace::detectNameServer(bool useDetectedServer,
373  bool& scanNeeded,
374  bool& serverUsed) {
375  // This code looks like a placeholder that never got replaced.
376  // It is using a heuristic that namespaces with "/ros" in the
377  // name are ros namespaces. There's no need for guesswork like
378  // that anymore. Also, code duplication. Should spin this
379  // off into a proper plugin mechanism for namespaces.
380  std::string name = NetworkBase::getNameServerName();
381  Contact fake, r;
382  if (name.find("/ros")!=std::string::npos) {
383  RosNameSpace ns(fake);
384  r = ns.detectNameServer(useDetectedServer, scanNeeded, serverUsed);
385  if (r.isValid()&&useDetectedServer&&scanNeeded) {
386  HELPER(this).activate(true);
387  }
388  } else {
389  YarpNameSpace ns(fake);
390  r = ns.detectNameServer(useDetectedServer, scanNeeded, serverUsed);
391  if (r.isValid()&&useDetectedServer&&scanNeeded) {
392  HELPER(this).activate(true);
393  }
394  }
395  return r;
396 }
