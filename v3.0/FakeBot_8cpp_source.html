28 void FakeBot::init() {
29  int m_w = 640;
30  int m_h = 480;
31  int rr = 30;
32 
33  back.resize(m_w,m_h);
34  PixelRgb w(255,255,255);
35  PixelRgb r(128,128,255);
36  PixelRgb b(0,0,255);
37  IMGFOR(back,x,y) {
38  back(x,y) = r;
39  if (y>=m_h*0.75) {
40  back(x,y) = b;
41  }
42  }
43  for (int i=0; i<=m_w; i+=m_w/10) {
44  addCircle(back,b,i,int(m_h*0.77),rr);
45  }
46 
47  std::default_random_engine randengine;
48  std::uniform_real_distribution<double> udist1(-m_w/4,m_w/4);
49  std::uniform_real_distribution<double> udist2(-int(m_h*0.2),rr);
50  for (int j=0; j<40; j++) {
51  int xx = m_w/2 + udist1(randengine);
52  int yy = int(m_h*0.2) - udist2(randengine);
53  addCircle(back,w,xx,yy,rr);
54  }
55 
56  m_x = (back.width()-this->m_w)/2;
57  m_y = (back.height()-this->m_h)/2;
58  m_dx = m_dy = 0;
59 
60  std::normal_distribution<double> ndist(0,255);
61  for (int k=0; k<MAXRND; k++) {
62  rnds[k] = int(ndist(randengine));
63  }
64 
65  fore.resize(64,64);
66  fore.zero();
67  m_tx = back.width()/2;
68  m_ty = back.height()*0.4;
69  m_tdx = 1;
70  m_tdy = 0;
71 }
