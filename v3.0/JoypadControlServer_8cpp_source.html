54 bool JoypadCtrlParser::respond(const yarp::os::Bottle& cmd, yarp::os::Bottle& response)
55 {
56  bool ret;
57 
58  ret = false;
59  if(cmd.get(0).asVocab() != VOCAB_IJOYPADCTRL || !cmd.get(1).isVocab() || !cmd.get(2).isVocab() || !cmd.get(3).isVocab())
60  {
61  response.addVocab(VOCAB_FAILED);
62  return ret;
63  }
64 
65  if(cmd.get(1).asVocab() == VOCAB_GET)
66  {
67  int toGet;
68 
69  toGet = cmd.get(2).asVocab();
70 
71  if(cmd.get(3).asVocab() == VOCAB_COUNT)
72  {
73  if(countGetters.find(toGet) != countGetters.end())
74  {
75  unsigned int count;
76  getcountmethod getter;
77  getter = countGetters[toGet];
78  if((device->*getter)(count))
79  {
80  response.addVocab(VOCAB_OK);
81  response.addInt32(count);
82  ret = true;
83  }
84  }
85  else if (toGet == VOCAB_STICKDOF && cmd.get(4).isInt32())
86  {
87  unsigned int count;
88  if (device->getStickDoF(cmd.get(4).asInt32(), count))
89  {
90  response.addVocab(VOCAB_OK);
91  response.addInt32(count);
92  ret = true;
93  }
94  else
95  {
96  response.addVocab(VOCAB_FAILED);
97  ret = false;
98  }
99  }
100  else
101  {
102  response.addVocab(VOCAB_FAILED);
103  ret = false;
104  }
105  }
106  else if(cmd.get(3).asVocab() == VOCAB_VALUE)
107  {
108  switch (cmd.get(2).asVocab()) {
109  case VOCAB_BUTTON:
110  {
111  float value;
112  if(cmd.get(4).isInt32() && device->getButton(cmd.get(4).asInt32(), value))
113  {
114  response.addVocab(VOCAB_OK);
115  response.addFloat64(value);
116  ret = true;
117  }
118  break;
119  }
120  case VOCAB_AXIS:
121  {
122  double value;
123  if(cmd.get(4).isInt32() && device->getAxis(cmd.get(4).asInt32(), value))
124  {
125  response.addVocab(VOCAB_OK);
126  response.addFloat64(value);
127  ret = true;
128  }
129  break;
130  }
131  case VOCAB_STICK:
132  {
133  if(cmd.get(4).isVocab())
134  {
135  yarp::sig::Vector frame;
136 
137  auto mode = cmd.get(4).asVocab() == VOCAB_CARTESIAN ? yarp::dev::IJoypadController::JypCtrlcoord_CARTESIAN : yarp::dev::IJoypadController::JypCtrlcoord_POLAR;
138  if(cmd.get(5).isInt32() && device->getStick(cmd.get(5).asInt32(), frame, mode))
139  {
140  response.addVocab(VOCAB_OK);
141  for(size_t i = 0; i < frame.size(); ++i)
142  {
143  response.addFloat64(frame[i]);
144  }
145 
146  ret = true;
147  }
148  }
149  break;
150  }
151  case VOCAB_STICKDOF:
152  {
153 
154  unsigned int dofCount;
155 
156  if(cmd.get(5).isInt32() && device->getStickDoF(cmd.get(5).asInt32(), dofCount))
157  {
158  response.addVocab(VOCAB_OK);
159  response.addInt32(dofCount);
160  ret = true;
161  }
162 
163  break;
164  }
165  case VOCAB_TOUCH:
166  {
167  yarp::sig::Vector pos;
168  unsigned int id;
169 
170  id = cmd.get(4).asInt32();
171  if(cmd.get(4).isInt32() && device->getTouch(id, pos))
172  {
173  response.addVocab(VOCAB_OK);
174  for(size_t i = 0; i < pos.size(); ++i)
175  {
176  response.addFloat64(pos[i]);
177  }
178  ret = true;
179  }
180  break;
181  }
182  case VOCAB_TRACKBALL:
183  {
184  yarp::sig::Vector axes;
185  unsigned int id;
186 
187  id = cmd.get(4).asInt32();
188  if(cmd.get(4).isInt32() && device->getTrackball(id, axes))
189  {
190  response.addVocab(VOCAB_OK);
191  for(size_t i = 0; i < axes.size(); ++i)
192  {
193  response.addFloat64(axes[i]);
194  }
195  ret = true;
196  }
197  break;
198  }
199  case VOCAB_HAT:
200  {
201  unsigned char value;
202  if(cmd.get(4).isInt32() && device->getHat(cmd.get(4).asInt32(), value))
203  {
204  response.addVocab(VOCAB_OK);
205  response.addInt32(value);
206  ret = true;
207  }
208  break;
209  }
210  default:
211  break;
212  }
213  }
214  }
215  return ret;

242 bool JoypadControlServer::open(yarp::os::Searchable& params)
243 {
244  if(params.check("help"))
245  {
246  yInfo() << "parameters:\n\n" <<
247  "period - refresh period of the broadcasted values in ms.. default" << DEFAULT_THREAD_PERIOD * 1000<< "\n"
248  "use_separate_ports - set it to 1 to use separate ports (buttons, axes, trackballs, hats) and 0 to stream all in one single port\n" <<
249  "name - Prefix name of the ports opened by the JoypadControlServer, e.g. /robotName/joypad\n" <<
250  "subdevice - name of the subdevice to open\n" <<
251  "profile - print the joypad data for debugging purpose";
252  return false;
253  }
254  std::string rootName;
255  if (!params.check("period", "refresh period of the broadcasted values in ms"))
256  {
257  yInfo() << "JoypadControlServer: using default 'period' parameter of " << DEFAULT_THREAD_PERIOD << "s";
258  }
259  else
260  {
261  m_period = params.find("period").asInt32() / 1000.0;
262  }
263 
264  m_profile = params.check("profile");
265 
266  if(params.check("use_separate_ports"))
267  {
268  m_separatePorts = params.find("use_separate_ports").asBool();
269  if(!m_separatePorts)

271  yError() << "single port mode not supported at the moment";
272  return false;
273  }
274  }
275  else
276  {
277  yError() << "JoypadControlServer: missing use_separate_ports in configuration";
278  return false;
279  }
280  m_coordsMode = yarp::dev::IJoypadController::JypCtrlcoord_CARTESIAN;
281  rootName = params.check("name",Value("/"), "starting '/' if needed.").asString().c_str();
282

284  {
285  yError() << "JoypadControlServer: missing 'name' parameter. Check you configuration file; it must be like:";

287  return false;
288  }
289 
290  rootName = params.find("name").asString().c_str();
291  m_rpcPortName = rootName + "/rpc:i";

299

301  // passed later on thorugh attachAll()
302  if(params.check("subdevice"))
303  {
304  m_isSubdeviceOwned=true;
305  if(!openAndAttachSubDevice(params))
306  {
307  yError("JoypadControlServer: error while opening subdevice\n");
308  return false;

311  else

313  m_isSubdeviceOwned=false;
314  }
315  return true;

318 bool JoypadControlServer::openAndAttachSubDevice(Searchable& prop)
319 {
320  Property p;
321 
322  m_subDeviceOwned = new PolyDriver;
323 
324  p.fromString(prop.toString().c_str());
325  p.setMonitor(prop.getMonitor(), "subdevice"); // pass on any monitoring
326  p.unput("device");
327  p.put("device",prop.find("subdevice").asString()); // subdevice was already checked before
328 
329  // if errors occurred during open, quit here.

338  if(!attach(m_subDeviceOwned))
339  return false;

642 {

645  if (m_portButtons.valid)

650  b.clear();
651  for(size_t i = 0; i < m_portButtons.count; ++i)
652  {
653  float v;
654  if(!m_device->getButton(i, v))
655  {
656  write = false;
