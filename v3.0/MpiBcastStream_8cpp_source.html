25 void MpiBcastStream::execCmd(int cmd) {
26  switch (cmd) {
27  case CMD_JOIN:
28  // Connect:
29  // Let a new port join the broadcast group
30  comm->accept();
31  break;
32  case CMD_DISCONNECT:
33  // Disconnect:
34  // Let a port leave the broadcast group
35  int length;
36  MPI_Bcast(&length, 1, MPI_INT, 0,comm->comm);
37  char* remote = new char[length];
38  MPI_Bcast(remote, length, MPI_CHAR, 0,comm->comm);
39  terminate = !strcmp(remote, name.c_str());
40  #ifdef MPI_DEBUG
41  printf("[MpiBcastStream @ %s] Got disconnect : %s => %d\n", name.c_str(), remote, terminate);
42  #endif
43  delete [] remote;
44  comm->disconnect(terminate);
45  break;
46  }
47 }

54 ssize_t MpiBcastStream::read(Bytes& b) {
55  if (terminate) {
56  return -1;
57  }
58  if (readAvail == 0) {
59  // get new data
60  reset();
61  int size;
62  #ifdef MPI_DEBUG
63  printf("[MpiBcastStream @ %s] Trying to read\n", name.c_str());
64  #endif
65 
66  MPI_Bcast(&size, 1, MPI_INT, 0,comm->comm);
67  #ifdef MPI_DEBUG
68  printf("[MpiBcastStream @ %s] got size %d\n", name.c_str(), size);
69  #endif
70  if (size < 0) {
71  execCmd(size);
72  return 0;
73  }
74  if ((size_t)size == b.length()) {
75  // size of received data matches expected data
76  // do not use buffer, but write directly
77  MPI_Bcast(b.get(), size, MPI_BYTE, 0, comm->comm);
78  return size;
79  }
80  else {
81  // allocate new buffer
82  readBuffer = new char[size];
83  MPI_Bcast(readBuffer, size, MPI_BYTE, 0, comm->comm);
84  //printf("got new msg of size %d\n", size);
85  readAvail = size;
86  readAt = 0;
87  }
88  }
89  if (readAvail>0) {
90  // copy data from buffer to destination object
91  int take = readAvail;
92  if (take>(int)b.length()) {
93  take = b.length();
94  }
95  memcpy(b.get(),readBuffer+readAt,take);
96  //printf("read %d of %d \n", take, readAvail);
97  readAt += take;
98  readAvail -= take;
99  return take;
100  }
101  return 0;
102 }
