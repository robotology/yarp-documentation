226 bool ServerFrameGrabber::respond(const yarp::os::Bottle& cmd,
227  yarp::os::Bottle& response) {
228  int code = cmd.get(0).asVocab();
229 
230  IFrameGrabberControlsDC1394* fgCtrlDC1394=dynamic_cast<IFrameGrabberControlsDC1394*>(fgCtrl);
231 
232  switch (code)
233  {
234  // first check if requests are coming from new iFrameGrabberControl2 interface and process them
235  case VOCAB_FRAMEGRABBER_CONTROL:
236  {
237  return ifgCtrl_Parser.respond(cmd, response); // I don't like all those returns everywhere!!! :-(
238  } break;
239  case VOCAB_RGB_VISUAL_PARAMS:
240  {
241  return rgbParser.respond(cmd,response);
242  } break;
244  // DC1394 COMMANDS
246  case VOCAB_FRAMEGRABBER_CONTROL_DC1394:
247  {
248  if (fgCtrlDC1394)
249  {
250  int codeDC1394 = cmd.get(1).asVocab();
251  switch(codeDC1394)
252  {
253  case VOCAB_DRGETMSK: // VOCAB_DRGETMSK 12
254  response.addInt32(int(fgCtrlDC1394->getVideoModeMaskDC1394()));
255  return true;
256  case VOCAB_DRGETVMD: // VOCAB_DRGETVMD 13
257  response.addInt32(int(fgCtrlDC1394->getVideoModeDC1394()));
258  return true;
259  case VOCAB_DRSETVMD: // VOCAB_DRSETVMD 14
260  response.addInt32(int(fgCtrlDC1394->setVideoModeDC1394(cmd.get(1).asInt32())));
261  return true;
262  case VOCAB_DRGETFPM: // VOCAB_DRGETFPM 15
263  response.addInt32(int(fgCtrlDC1394->getFPSMaskDC1394()));
264  return true;
265  case VOCAB_DRGETFPS: // VOCAB_DRGETFPS 16
266  response.addInt32(int(fgCtrlDC1394->getFPSDC1394()));
267  return true;
268  case VOCAB_DRSETFPS: // VOCAB_DRSETFPS 17
269  response.addInt32(int(fgCtrlDC1394->setFPSDC1394(cmd.get(1).asInt32())));
270  return true;
271 
272  case VOCAB_DRGETISO: // VOCAB_DRGETISO 18
273  response.addInt32(int(fgCtrlDC1394->getISOSpeedDC1394()));
274  return true;
275  case VOCAB_DRSETISO: // VOCAB_DRSETISO 19
276  response.addInt32(int(fgCtrlDC1394->setISOSpeedDC1394(cmd.get(1).asInt32())));
277  return true;
278 
279  case VOCAB_DRGETCCM: // VOCAB_DRGETCCM 20
280  response.addInt32(int(fgCtrlDC1394->getColorCodingMaskDC1394(cmd.get(1).asInt32())));
281  return true;
282  case VOCAB_DRGETCOD: // VOCAB_DRGETCOD 21
283  response.addInt32(int(fgCtrlDC1394->getColorCodingDC1394()));
284  return true;
285  case VOCAB_DRSETCOD: // VOCAB_DRSETCOD 22
286  response.addInt32(int(fgCtrlDC1394->setColorCodingDC1394(cmd.get(1).asInt32())));
287  return true;
288  case VOCAB_DRGETF7M: // VOCAB_DRGETF7M 25
289  {
290  unsigned int xstep,ystep,xdim,ydim,xoffstep,yoffstep;
291  fgCtrlDC1394->getFormat7MaxWindowDC1394(xdim,ydim,xstep,ystep,xoffstep,yoffstep);
292  response.addInt32(xdim);
293  response.addInt32(ydim);
294  response.addInt32(xstep);
295  response.addInt32(ystep);
296  response.addInt32(xoffstep);
297  response.addInt32(yoffstep);
298  }
299  return true;
300  case VOCAB_DRGETWF7: // VOCAB_DRGETWF7 26
301  {
302  unsigned int xdim,ydim;
303  int x0,y0;
304  fgCtrlDC1394->getFormat7WindowDC1394(xdim,ydim,x0,y0);
305  response.addInt32(xdim);
306  response.addInt32(ydim);
307  response.addInt32(x0);
308  response.addInt32(y0);
309  }
310  return true;
311  case VOCAB_DRSETWF7: // VOCAB_DRSETWF7 27
312  response.addInt32(int(fgCtrlDC1394->setFormat7WindowDC1394(cmd.get(1).asInt32(),cmd.get(2).asInt32(),cmd.get(3).asInt32(),cmd.get(4).asInt32())));
313  return true;
314  case VOCAB_DRSETOPM: // VOCAB_DRSETOPM 28
315  response.addInt32(int(fgCtrlDC1394->setOperationModeDC1394(cmd.get(1).asInt32()!=0)));
316  return true;
317  case VOCAB_DRGETOPM: // VOCAB_DRGETOPM 29
318  response.addInt32(fgCtrlDC1394->getOperationModeDC1394());
319  return true;
320 
321  case VOCAB_DRSETTXM: // VOCAB_DRSETTXM 30
322  response.addInt32(int(fgCtrlDC1394->setTransmissionDC1394(cmd.get(1).asInt32()!=0)));
323  return true;
324  case VOCAB_DRGETTXM: // VOCAB_DRGETTXM 31
325  response.addInt32(fgCtrlDC1394->getTransmissionDC1394());
326  return true;
327  /*
328  case VOCAB_DRSETBAY: // VOCAB_DRSETBAY 32
329  response.addInt32(int(fgCtrlDC1394->setBayerDC1394(bool(cmd.get(1).asInt32()))));
330  return true;
331  case VOCAB_DRGETBAY: // VOCAB_DRGETBAY 33
332  response.addInt32(fgCtrlDC1394->getBayerDC1394());
333  return true;
334  */
335  case VOCAB_DRSETBCS: // VOCAB_DRSETBCS 34
336  response.addInt32(int(fgCtrlDC1394->setBroadcastDC1394(cmd.get(1).asInt32()!=0)));
337  return true;
338  case VOCAB_DRSETDEF: // VOCAB_DRSETDEF 35
339  response.addInt32(int(fgCtrlDC1394->setDefaultsDC1394()));
340  return true;
341  case VOCAB_DRSETRST: // VOCAB_DRSETRST 36
342  response.addInt32(int(fgCtrlDC1394->setResetDC1394()));
343  return true;
344  case VOCAB_DRSETPWR: // VOCAB_DRSETPWR 37
345  response.addInt32(int(fgCtrlDC1394->setPowerDC1394(cmd.get(1).asInt32()!=0)));
346  return true;
347  case VOCAB_DRSETCAP: // VOCAB_DRSETCAP 38
348  response.addInt32(int(fgCtrlDC1394->setCaptureDC1394(cmd.get(1).asInt32()!=0)));
349  return true;
350  case VOCAB_DRSETBPP: // VOCAB_DRSETCAP 39
351  response.addInt32(int(fgCtrlDC1394->setBytesPerPacketDC1394(cmd.get(1).asInt32())));
352  return true;
353  case VOCAB_DRGETBPP: // VOCAB_DRGETTXM 40
354  response.addInt32(fgCtrlDC1394->getBytesPerPacketDC1394());
355  return true;
356  default:
357  return DeviceResponder::respond(cmd,response);
358  }
359  }
360  }
361  }
362  yError() << "ServerFrameGrabber: command not recognized" << cmd.toString();
363  return DeviceResponder::respond(cmd,response);
364 }
