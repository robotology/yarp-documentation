30 void Ping::connect() {
31  lastConnect.clear();
32  double start = SystemClock::nowSystem();
33  Contact c = NetworkBase::queryName(target);
34  double afterQuery = SystemClock::nowSystem();
35  if (!c.isValid()) {
36  yError("Port not found during ping");
37  }
38  ContactStyle rpc;
39  rpc.admin = true;
40  rpc.quiet = true;
41  Bottle cmd, reply;
42  cmd.addVocab(Vocab::encode("ver"));
43  bool ok = NetworkBase::write(c, cmd, reply, rpc);
44  if (!ok) {
45  yError("Port did not respond as expected");
46  }
47  double stop = SystemClock::nowSystem();
48  lastConnect.totalTime.add(stop-start);
49  lastConnect.targetTime.add(stop-afterQuery);
50  accumConnect.add(lastConnect);
51 }

73 std::string Ping::renderTime(double t, int space, int decimal) {
74  std::string unit = "";
75  double times = 1;
76  if (space<0) {
77  yError("Negative space");
78  }
79  if (t>=1) {
80  unit = "sec";
81  } else if (t>1e-3) {
82  unit = " ms";
83  times = 1e3;
84  } else if (t>1e-6) {
85  unit = " us";
86  times = 1e6;
87  } else if (t>1e-9) {
88  unit = " ns";
89  times = 1e9;
90  }
91  char buf[512];
92  safe_printf(buf, sizeof(buf), "%.*f%s", decimal, t*times,
93  unit.c_str());
94  return buf;
95 }

98 class PingSampler : public PortReader {
99 public:
100  Mutex mutex;
101  int ct;
102  double lastTime;
103  Stat period;
104 
105  PingSampler() : mutex() { ct = 0; lastTime = 0; }
106 
107  virtual bool read(ConnectionReader& connection) override {
108  double now = SystemClock::nowSystem();
109  Bottle b;
110  bool ok = b.read(connection);
111  if (ok) {
112  mutex.lock();
113  ct++;
114  if (ct>1) {
115  double dt = now - lastTime;
116  period.add(dt);
117  }
118  lastTime = now;
119  printf("Period is %g +/- %g (%d)\n",
120  period.mean(),
121  period.deviation(),
122  ct);
123  mutex.unlock();
124  }
125  return ok;
126  }
127 };
