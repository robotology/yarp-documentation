131 void TestFrameGrabber::timing() {
132  double now = yarp::os::Time::now();
133 
134  if (now-prev>1000) {
135  first = now;
136  prev = now;
137  }
138  double dt = period-(now-prev);
139 
140  if (dt>0) {
141  yarp::os::Time::delay(dt);
142  }
143 
144  // this is the controlled instant when we consider the
145  // image as going out
146  prev += period;
147 }

256 void TestFrameGrabber::createTestImage(yarp::sig::ImageOf<yarp::sig::PixelRgb>&
257  image) {
258  // to test IPreciselyTimed, make timestamps be mysteriously NNN.NNN42
259  double t = Time::now();
260  t -= ((t*1000)-(int)t)/1000;
261  t+= 0.00042;
262  stamp.update(t);
263  if (background.width()>0) {
264  image.copy(background);
265  } else {
266  image.resize(w,h);
267  image.zero();
268  }
269  switch (mode) {
270  case VOCAB_BALL:
271  {
272  addCircle(image,PixelRgb(0,255,0),bx,by,15);
273  addCircle(image,PixelRgb(0,255,255),bx,by,8);
274  if (ct%5!=0) {
275  rnd *= 65537;
276  rnd += 17;
277  bx += (rnd%5)-2;
278  rnd *= 65537;
279  rnd += 17;
280  by += (rnd%5)-2;
281  } else {
282  int dx = w/2 - bx;
283  int dy = h/2 - by;
284  if (dx>0) { bx++; }
285  if (dx<0) { bx--; }
286  if (dy>0) { by++; }
287  if (dy<0) { by--; }
288  }
289  }
290  break;
291  case VOCAB_GRID:
292  {
293  size_t ww = image.width();
294  size_t hh = image.height();
295  if (ww>1&&hh>1) {
296  for (size_t x=0; x<ww; x++) {
297  for (size_t y=0; y<hh; y++) {
298  double xx = ((double)x)/(ww-1);
299  double yy = ((double)y)/(hh-1);
300  int r = int(0.5+255*xx);
301  int g = int(0.5+255*yy);
302  bool act = (y==ct);
303  image.pixel(x,y) = PixelRgb(r,g,act*255);
304  }
305  }
306  }
307  }
308  break;
309  case VOCAB_GRID_MULTISIZE:
310  {
311  static int count = 0;
312  count++;
313  if (count== 100)
314  {
315  yDebug() << "size 100, 100";
316  image.resize(100,100);
317  }
318  else if (count == 200)
319  {
320  yDebug() << "size 200, 100";
321  image.resize(200, 100);
322  }
323  else if (count == 300)
324  {
325  yDebug() << "size 300, 50";
326  image.resize(300, 50);
327  count = 0;
328  }
329 
330  size_t ww = w = image.width();
331  size_t hh = h = image.height();
332  if (ww>1 && hh>1) {
333  for (size_t x = 0; x<ww; x++) {
334  for (size_t y = 0; y<hh; y++) {
335  double xx = ((double)x) / (ww - 1);
336  double yy = ((double)y) / (hh - 1);
337  int r = int(0.5 + 255 * xx);
338  int g = int(0.5 + 255 * yy);
339  bool act = (y == ct);
340  image.pixel(x, y) = PixelRgb(r, g, act * 255);
341  }
342  }
343  }
344  }
345  break;
346  case VOCAB_LINE:
347  default:
348  {
349  for (size_t i=0; i<image.width(); i++) {
350  image.pixel(i,ct).r = 255;
351  }
352  char ttxt[50];
353  sprintf(ttxt, "%021.10f", t);
354  image.pixel(0, 0).r = ttxt[0] - '0';
355  image.pixel(0, 0).g = ttxt[1] - '0';
356  image.pixel(0, 0).b = ttxt[2] - '0';
357 
358  image.pixel(1, 0).r = ttxt[3] - '0';
359  image.pixel(1, 0).g = ttxt[4] - '0';
360  image.pixel(1, 0).b = ttxt[5] - '0';
361 
362  image.pixel(2, 0).r = ttxt[6] - '0';
363  image.pixel(2, 0).g = ttxt[7] - '0';
364  image.pixel(2, 0).b = ttxt[8] - '0';
365 
366  image.pixel(3, 0).r = ttxt[9] - '0';
367  image.pixel(3, 0).g = ttxt[10] - '0';
368  image.pixel(3, 0).b = ttxt[11] - '0';
369 
370  image.pixel(4, 0).r = ttxt[12] - '0';
371  image.pixel(4, 0).g = ttxt[13] - '0';
372  image.pixel(4, 0).b = ttxt[14] - '0';
373 
374  image.pixel(5, 0).r = ttxt[15] - '0';
375  image.pixel(5, 0).g = ttxt[16] - '0';
376  image.pixel(5, 0).b = ttxt[17] - '0';
377 
378  image.pixel(6, 0).r = ttxt[18] - '0';
379  image.pixel(6, 0).g = ttxt[19] - '0';
380  image.pixel(6, 0).b = ttxt[20] - '0';
381  }
382  break;
383  case VOCAB_RAND:
384  {
385  // from Alessandro Scalzo
386 
387  static unsigned char r=128,g=128,b=128;
388 
389  size_t ww = image.width();
390  size_t hh = image.height();
391 
392  if (ww>1&&hh>1) {
393  std::default_random_engine randengine;
394  std::uniform_real_distribution<double> udist(-1.0, 1.0);
395  for (size_t x=0; x<ww; x++) {
396  for (size_t y=0; y<hh; y++) {
397  //r+=(rand()%3)-1;
398  //g+=(rand()%3)-1;
399  //b+=(rand()%3)-1;
400  r += udist(randengine);
401  g += udist(randengine);
402  b += udist(randengine);
403  image.pixel(x,y) = PixelRgb(r,g,b);
404  }
405  }
406  }
407  }
408  break;
409  case VOCAB_NONE:
410  break;
411  }
412  ct++;
413  if (ct>=image.height()) {
414  ct = 0;
415  }
416  if (by>=image.height()) {
417  by = image.height()-1;
418  }
419  if (bx>=image.width()) {
420  bx = image.width()-1;
421  }
422 }

430 bool TestFrameGrabber::makeSimpleBayer(
431  ImageOf<PixelRgb>& img,
432  ImageOf<PixelMono>& bayer) {
433 
434  bayer.resize(img.width(), img.height());
435 
436  const size_t w = img.width();
437  const size_t h = img.height();
438 
439  size_t i, j;
440  for (i = 0; i < h; i++) {
441  PixelRgb *row = (PixelRgb *)img.getRow(i);
442  PixelMono *rd = (PixelMono *)bayer.getRow(i);
443 
444  for (j = 0; j < w; j++) {
445 
446  if ((i%2) == 0) {
447  switch (j%4) {
448  case 0:
449  case 2:
450  *rd++ = row->g;
451  row++;
452  break;
453 
454  case 1:
455  case 3:
456  *rd++ = row->r;
457  row++;
458  break;
459  }
460  }
461 
462  if ((i%2) == 1) {
463  switch (j%4) {
464  case 1:
465  case 3:
466  *rd++ = row->g;
467  row++;
468  break;
469 
470  case 0:
471  case 2:
472  *rd++ = row->b;
473  row++;
474  break;
475  }
476  }
477  }
478  }
479 
480  return true;
