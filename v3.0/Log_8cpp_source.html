85 void yarp::os::impl::LogImpl::print_callback(yarp::os::Log::LogType t,
86  const char *msg,
87  const char *file,
88  const unsigned int line,
89  const char *func)
90 {
91  switch (t) {
92  case yarp::os::Log::TraceType:
93  if (trace_output) {
94  if (verbose_output) {
95  std::cout << "[" << WHITE << "TRACE" << CLEAR << "]" << file << ":" << line << " " << WHITE << func << CLEAR << ": " << msg << std::endl;
96  } else {
97  std::cout << "[" << WHITE << "TRACE" << CLEAR << "]" << func << msg << std::endl;
98  }
99  }
100  break;
101  case yarp::os::Log::DebugType:
102  if (debug_output) {
103  if (verbose_output) {
104  std::cout << "[" << GREEN << "DEBUG" << CLEAR << "]" << file << ":" << line << " " << GREEN << func << CLEAR << ": " << msg << std::endl;
105  } else {
106  std::cout << "[" << GREEN << "DEBUG" << CLEAR << "]" << msg << std::endl;
107  }
108  }
109  break;
110  case yarp::os::Log::InfoType:
111  if (verbose_output) {
112  std::cout << "[" << BLUE << "INFO" << CLEAR << "]" << file << ":" << line << " " << BLUE << func << CLEAR << ": " << msg << std::endl;
113  } else {
114  std::cout << "[" << BLUE << "INFO" << CLEAR << "]" << msg << std::endl;
115  }
116  break;
117  case yarp::os::Log::WarningType:
118  if (verbose_output) {
119  std::cerr << "[" << YELLOW << "WARNING" << CLEAR << "]" << file << ":" << line << " " << YELLOW << func << CLEAR << ": " << msg << std::endl;
120  } else {
121  std::cerr << "[" << YELLOW << "WARNING" << CLEAR << "]" << msg << std::endl;
122  }
123  break;
124  case yarp::os::Log::ErrorType:
125  if (verbose_output) {
126  std::cerr << "[" << RED << "ERROR" << CLEAR << "]" << file << ":" << line << " " << RED << func << CLEAR << ": " << msg << std::endl;
127  } else {
128  std::cerr << "[" << RED << "ERROR" << CLEAR << "]" << msg << std::endl;
129  }
130  break;
131  case yarp::os::Log::FatalType:
132  if (verbose_output) {
133  std::cerr << "["<< WHITE << RED_BG << "FATAL" << CLEAR << "]" << file << ":" << line << " " << WHITE << RED_BG << func << CLEAR << ": " << msg << std::endl;
134  } else {
135  std::cerr << "["<< WHITE << RED_BG << "FATAL" << CLEAR << "]" << msg << std::endl;
136  }
137  break;
138  default:
139  break;
140  }

233 void yarp::os::Log::trace(const char *msg, ...) const
234 {
235  va_list args;
236  va_start(args, msg);
237  if (msg) {
238  char buf[YARP_MAX_LOG_MSG_SIZE];
239  int w = vsnprintf(buf, YARP_MAX_LOG_MSG_SIZE, msg, args);
240  buf[YARP_MAX_LOG_MSG_SIZE-1]=0;
241  if (w>0 && buf[w-1]=='\n') {
242  buf[w-1]=0;
243  }
244  if (print_callback) {
245  print_callback(yarp::os::Log::TraceType, buf, mPriv->file, mPriv->line, mPriv->func);
246  }
247  if (forward_callback) {
248  forward_callback(yarp::os::Log::TraceType, buf, mPriv->file, mPriv->line, mPriv->func);
249  }
250  if (w > YARP_MAX_LOG_MSG_SIZE-1) {
251  yarp::os::Log(mPriv->file, mPriv->line, mPriv->func).warning("Previous trace message was truncated");
252  }
253  }
254  va_end(args);
255 }

420 void yarp_print_trace(FILE *out, const char *file, int line) {
421 #ifdef YARP_HAS_ACE
422  YARP_UNUSED(file);
423  YARP_UNUSED(line);
424  ACE_Stack_Trace st(-1);
425  // TODO demangle symbols using <cxxabi.h> and abi::__cxa_demangle
426  // when available.
427  fprintf(out, "%s", st.c_str());
428 #elif defined(YARP_HAS_EXECINFO_H)
429  const size_t max_depth = 100;
430  size_t stack_depth;
431  void *stack_addrs[max_depth];
432  char **stack_strings;
433  stack_depth = backtrace(stack_addrs, max_depth);
434  stack_strings = backtrace_symbols(stack_addrs, stack_depth);
435  fprintf(out, "Assertion thrown at %s:%d by code called from:\n", file, line);
436  for (size_t i = 1; i < stack_depth; i++) {
437  fprintf(out, " --> %s\n", stack_strings[i]);
438  }
439  free(stack_strings); // malloc()ed by backtrace_symbols
440  fflush(out);
441 #else
442  // Not implemented on this platform
443 #endif
444 }
