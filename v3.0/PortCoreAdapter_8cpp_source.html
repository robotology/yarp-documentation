95 void yarp::os::impl::PortCoreAdapter::finishWriting()
96 {
97  if (isWriting()) {
98  double start = SystemClock::nowSystem();
99  double pause = 0.01;
100  do {
101  SystemClock::delaySystem(pause);
102  pause *= 2;
103  } while (isWriting() && (SystemClock::nowSystem()-start<3));
104  if (isWriting()) {
105  YARP_ERROR(Logger::get(), "Closing port that was sending data (slowly)");
106  }
107  }

119 bool yarp::os::impl::PortCoreAdapter::read(ConnectionReader& reader)
120 {
121  if (permanentReadDelegate!=nullptr) {
122  bool result = permanentReadDelegate->read(reader);
123  return result;
124  }
125 
126  // called by comms code
127  readBlock.wait();
128 
129  if (!reader.isValid()) {
130  // interrupt
131  stateMutex.lock();
132  if (readDelegate!=nullptr) {
133  readResult = readDelegate->read(reader);
134  }
135  stateMutex.unlock();
136  produce.post();
137  readBlock.post();
138  return false;
139  }
140 
141  if (closed) {
142  YARP_DEBUG(Logger::get(), "Port::read shutting down");
143  readBlock.post();
144  return false;
145  }
146 
147  // wait for happy consumer - don't want to miss a packet
148  if (!readBackground) {
149  consume.wait();
150  }
151 
152  stateMutex.lock();
153  readResult = false;
154  if (readDelegate!=nullptr) {
155  readResult = readDelegate->read(reader);
156  } else {
157  // read and ignore
158  YARP_DEBUG(Logger::get(), "data received in Port, no reader for it");
159  Bottle b;
160  b.read(reader);
161  }
162  if (!readBackground) {
163  readDelegate = nullptr;
164  writeDelegate = nullptr;
165  }
166  bool result = readResult;
167  stateMutex.unlock();
168  if (!readBackground) {
169  produce.post();
170  }
171  if (result&&willReply) {
172  consume.wait();
173  if (closed) {
174  YARP_DEBUG(Logger::get(), "Port::read shutting down");
175  readBlock.post();
176  return false;
177  }
178  if (writeDelegate!=nullptr) {
179  stateMutex.lock();
180  ConnectionWriter *writer = reader.getWriter();
181  if (writer!=nullptr) {
182  result = readResult = writeDelegate->write(*writer);
183  }
184  stateMutex.unlock();
185  }
186  if (dropDue) {
187  reader.requestDrop();
188  }
189  produce.post();
190  }
191  readBlock.post();
192  return result;
193 }

195 bool yarp::os::impl::PortCoreAdapter::read(PortReader& reader, bool willReply)
196 {
197  // called by user
198 
199  // user claimed they would reply to last read, but then
200  // decided not to.
201  if (replyDue) {
202  Bottle emptyMessage;
203  reply(emptyMessage, false, false);
204  replyDue = false;
205  dropDue = false;
206  }
207  if (willReply) {
208  replyDue = true;
209  }
210 
211  stateMutex.lock();
212  readActive = true;
213  readDelegate = &reader;
214  checkType(reader);
215  writeDelegate = nullptr;
216  this->willReply = willReply;
217  consume.post(); // happy consumer
218  stateMutex.unlock();
219 
220  produce.wait();
221  stateMutex.lock();
222  if (!readBackground) {
223  readDelegate = nullptr;
224  }
225  bool result = readResult;
226  if (!result) replyDue = false;
227  stateMutex.unlock();
228  return result;
229 }

231 bool yarp::os::impl::PortCoreAdapter::reply(PortWriter& writer, bool drop, bool /*interrupted*/)
232 {
233  // send reply even if interrupt has happened in interim
234  if (!replyDue) return false;
235 
236  replyDue = false;
237  dropDue = drop;
238  writeDelegate = &writer;
239  consume.post();
240  produce.wait();
241  bool result = readResult;
242  return result;
243 }
