91 bool ServerInertial::checkROSParams(yarp::os::Searchable &config)
92 {
93  // check for ROS parameter group
94  if(!config.check("ROS") )
95  {
96  useROS = ROS_disabled;
97  yInfo() << "No ROS group found in config file ... skipping ROS initialization.";
98  return true;
99  }
100 
101  yInfo() << "ROS group was FOUND in config file.";
102 
103  Bottle &rosGroup = config.findGroup("ROS");
104  if(rosGroup.isNull())
105  {
106  yError() << partName << "ROS group params is not a valid group or empty";
107  useROS = ROS_config_error;
108  return false;
109  }
110 
111  // check for useROS parameter
112  if(!rosGroup.check("useROS"))
113  {
114  yError() << partName << " cannot find useROS parameter, mandatory when using ROS message. \n \
115  Allowed values are true, false, ROS_only";
116  useROS = ROS_config_error;
117  return false;
118  }
119  std::string ros_use_type = rosGroup.find("useROS").asString();
120  if(ros_use_type == "false")
121  {
122  yInfo() << partName << "useROS topic if set to 'false'";
123  useROS = ROS_disabled;
124  return true;
125  }
126  else if(ros_use_type == "true")
127  {
128  yInfo() << partName << "useROS topic if set to 'true'";
129  useROS = ROS_enabled;
130  }
131  else if(ros_use_type == "only")
132  {
133  yInfo() << partName << "useROS topic if set to 'only";
134  useROS = ROS_only;
135  }
136  else
137  {
138  yInfo() << partName << "useROS parameter is seet to unvalid value ('" << ros_use_type << "'), supported values are 'true', 'false', 'only'";
139  useROS = ROS_config_error;
140  return false;
141  }
142 
143  // check for ROS_nodeName parameter
144  if(!rosGroup.check("ROS_nodeName"))
145  {
146  yError() << partName << " cannot find ROS_nodeName parameter, mandatory when using ROS message";
147  useROS = ROS_config_error;
148  return false;
149  }
150  rosNodeName = rosGroup.find("ROS_nodeName").asString(); // TODO: check name is correct
151  yInfo() << partName << "rosNodeName is " << rosNodeName;
152 
153  // check for ROS_topicName parameter
154  if(!rosGroup.check("ROS_topicName"))
155  {
156  yError() << partName << " cannot find ROS_topicName parameter, mandatory when using ROS message";
157  useROS = ROS_config_error;
158  return false;
159  }
160  rosTopicName = rosGroup.find("ROS_topicName").asString();
161  yInfo() << partName << "ROS_topicName is " << rosTopicName;
162 
163  // check for frame_id parameter

165  {
166  yError() << partName << " cannot find frame_id parameter, mandatory when using ROS message";
167  useROS = ROS_config_error;
168  return false;
169  }

171  yInfo() << partName << "frame_id is " << frame_id;
172 
173  return true;

176 bool ServerInertial::initialize_ROS()
177 {
178  bool success = false;
179  switch(useROS)
180  {
181  case ROS_enabled:
182  case ROS_only:
183  {
184  rosNode = new yarp::os::Node(rosNodeName); // add a ROS node
185 
186  if(rosNode == nullptr)
187  {
188  yError() << " opening " << rosNodeName << " Node, check your yarp-ROS network configuration\n";
189  success = false;
190  break;
191  }
192 
193  if (!rosPublisherPort.topic(rosTopicName) )
194  {
195  yError() << " opening " << rosTopicName << " Topic, check your yarp-ROS network configuration\n";
196  success = false;
197  break;
198  }
199  success = true;
200  } break;
201 
202  case ROS_disabled:
203  {
204  yInfo() << partName << ": no ROS initialization required";
205  success = true;
206  } break;
207 
208  case ROS_config_error:
209  {
210  yError() << partName << " ROS parameter are not correct, check your configuration file";
211  success = false;
212  } break;
213 
214  default:
215  {
216  yError() << partName << " something went wrong with ROS configuration, we should never be here!!!";
217  success = false;
218  } break;
219  }
220  return success;
221 }

230 bool yarp::dev::ServerInertial::openAndAttachSubDevice(yarp::os::Property& prop)
231 {

235 // yDebug("Subdevice %s\n", subdevice.toString().c_str());

239  yarp::os::Property p;
240  p.setMonitor(prop.getMonitor(), "subdevice"); // pass on any monitoring
241  p.fromString(prop.toString());
242  p.put("device",subdevice.toString());
243  IMU_polydriver->open(p);
244  }
245  else
246  IMU_polydriver->open(subdevice);
247 
248  if (!IMU_polydriver->isValid())
249  {
250  yError("cannot create device <%s>\n", subdevice.toString().c_str());
251  return false;
252  }
253 
254  // if we are here, poly is valid
255  IMU_polydriver->view(IMU); // attach to subdevice
256  if(IMU == nullptr)
257  {
258  yError("Error, subdevice <%s> has no valid IMU interface\n", subdevice.toString().c_str());
259  IMU_polydriver->close();
260  return false;
261  }
262 
263  // iTimed interface
264  IMU_polydriver->view(iTimed); // not mandatory
265  return true;

278 bool yarp::dev::ServerInertial::open(yarp::os::Searchable& config)
279 {
280  Property prop;
281  prop.fromString(config.toString().c_str());
282 
283  p.setReader(*this);
284 
285  period = config.check("period",yarp::os::Value(0.005),"maximum period").asFloat64();
286  strict = config.check("strict",yarp::os::Value(false),"strict write").asBool();
287 
288  //Look for the device name (serial Port). Usually /dev/ttyUSB0
289  // check if we need to create subdevice or if they are
290  // passed later on thorugh attachAll()
291  if(prop.check("subdevice"))
292  {
293  ownDevices=true;
294  if(! openAndAttachSubDevice(prop))
295  {
296  yError("ControlBoardWrapper: error while opening subdevice\n");
297  return false;
298  }
299  }
300  else
301  {
302  ownDevices=false;
303  if(!openDeferredAttach(prop))
304  return false;
305  }
306 
307 
308  checkROSParams(config);
309 
310 
311 
312  //Look for the portname to register (--name option), defaulting to /inertial if missing
313  std::string portName;
314  if(useROS != ROS_only)
315  {
316  if (config.check("name"))
317  portName = config.find("name").asString();
318  else
319  {
320  yInfo() << "Using default values for port name, you can change it by using '--name /myPortName' parameter";
321  portName = "/inertial";
322  }
323 
324  if(!p.open(portName))
325  {
326  yError() << "ServerInertial, cannot open port " << portName;
327  return false;
328  }
329  writer.attach(p);
330  }
331 
332  // call ROS node/topic initialization, if needed
333  if(!initialize_ROS() )
334  {
335  return false;
336  }
337 
338  if( (ownDevices) && (IMU!=nullptr) )
339  {
340  start();
341  }
342 
343  return true;
344 }

398 void yarp::dev::ServerInertial::run()
399 {
400  double before, now;
401  yInfo("Starting server Inertial thread\n");
402  while (!isStopping())
403  {
404  before = yarp::os::Time::now();
405  if (IMU!=nullptr)
406  {
407  Bottle imuData;
408  bool res = getInertial(imuData);
409 
410  // publish data on YARP port if required
411  if(useROS != ROS_only)
412  {
413  yarp::os::Bottle& bot = writer.get();
414  bot = imuData;
415  if (res)
416  {
417  static yarp::os::Stamp ts;
418  if (iTimed)
419  ts=iTimed->getLastInputStamp();
420  else
421  ts.update();
422 
423 
424  curr_timestamp_counter = ts.getCount();
425 
426  if (curr_timestamp_counter!=prev_timestamp_counter)
427  {
428  if (!spoke)
429  {
430  yDebug("Writing an Inertial measurement.\n");
431  spoke = true;
432  }
433  p.setEnvelope(ts);
434  writer.write(strict);
435  }
436  else
437  {
438  trap++;
439  }
440 
441  prev_timestamp_counter = curr_timestamp_counter;
442  }
443  }
444 
445  // publish ROS topic if required
446  if(useROS != ROS_disabled)
447  {
448  double euler_xyz[3], quaternion[4];
449 
450  euler_xyz[0] = imuData.get(0).asFloat64();
451  euler_xyz[1] = imuData.get(1).asFloat64();
452  euler_xyz[2] = imuData.get(2).asFloat64();
453 
454  convertEulerAngleYXZdegrees_to_quaternion(euler_xyz, quaternion);
455 
456  yarp::rosmsg::sensor_msgs::Imu &rosData = rosPublisherPort.prepare();
457 
458  rosData.header.seq = rosMsgCounter++;
459  rosData.header.stamp = yarp::os::Time::now();
460  rosData.header.frame_id = frame_id;
461 
462  rosData.orientation.x = quaternion[0];
463  rosData.orientation.y = quaternion[1];
464  rosData.orientation.z = quaternion[2];
465  rosData.orientation.w = quaternion[3];
466  rosData.orientation_covariance = covariance;
467 
468  rosData.linear_acceleration.x = imuData.get(3).asFloat64(); // [m/s^2]
469  rosData.linear_acceleration.y = imuData.get(4).asFloat64(); // [m/s^2]
470  rosData.linear_acceleration.z = imuData.get(5).asFloat64(); // [m/s^2]
471  rosData.linear_acceleration_covariance = covariance;
472 
473  rosData.angular_velocity.x = imuData.get(6).asFloat64(); // to be converted into rad/s (?) - verify with users
474  rosData.angular_velocity.y = imuData.get(7).asFloat64(); // to be converted into rad/s (?) - verify with users
475  rosData.angular_velocity.z = imuData.get(8).asFloat64(); // to be converted into rad/s (?) - verify with users
476  rosData.angular_velocity_covariance = covariance;
477 
478  rosPublisherPort.write();
479  }
480  }
481 
483  now = yarp::os::Time::now();
484  if ((now - before) < period) {
485  double k = period-(now-before);
486  yarp::os::Time::delay(k);
487  }
488  }
489  yInfo("Server Intertial thread finished\n");
490 }
