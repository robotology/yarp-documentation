100 bool BatteryWrapper::read(yarp::os::ConnectionReader& connection)
101 {
102  yarp::os::Bottle in;
103  yarp::os::Bottle out;
104  bool ok = in.read(connection);
105  if (!ok) return false;
106 
107  // parse in, prepare out
108  int code = in.get(0).asVocab();
109  bool ret = false;
110  if (code == VOCAB_IBATTERY)
111  {
112  int cmd = in.get(1).asVocab();
113  if (cmd == VOCAB_BATTERY_INFO)
114  {
115  if (battery_p)
116  {
117  std::string info;
118  battery_p->getBatteryInfo(info);
119  out.addVocab(VOCAB_IS);
120  out.addVocab(cmd);
121  out.addString(info);
122  ret = true;
123  }
124  }
125  else
126  {
127  yError("Invalid vocab received in BatteryWrapper");
128  }
129  }
130  else
131  {
132  yError("Invalid vocab received in BatteryWrapper");
133  }
134 
135  if (!ret)
136  {
137  out.clear();
138  out.addVocab(VOCAB_FAILED);
139  }
140 
141  yarp::os::ConnectionWriter *returnToSender = connection.getWriter();
142  if (returnToSender != nullptr) {
143  out.write(*returnToSender);
144  }
145  return true;
146 }

220 void BatteryWrapper::run()
221 {
222  if (battery_p!=nullptr)
223  {
224  double charge = 0;
225  double voltage = 0;
226  double current = 0;
227  double temperature = 0;
228 
229  bool ret = true;
230  ret &= battery_p->getBatteryCharge(charge);
231  ret &= battery_p->getBatteryVoltage(voltage);
232  ret &= battery_p->getBatteryCurrent(current);
233  ret &= battery_p->getBatteryTemperature(temperature);
234 
235  if (ret)
236  {
237  lastStateStamp.update();
238  yarp::os::Bottle& b = streamingPort.prepare();
239  b.clear();
240  b.addFloat64(voltage);
241  b.addFloat64(current);
242  b.addFloat64(charge);
243  b.addFloat64(temperature);
244  streamingPort.setEnvelope(lastStateStamp);
245  streamingPort.write();
246  }
247  else
248  {
249  yError("BatteryWrapper: %s: Sensor returned error", sensorId.c_str());
250  }
251  }
252 }
