251  bool readFromSelector(const std::string& name) {
252  if (!selector) return false;
253  Bottle plugins = selector->getSelectedPlugins();
254  Bottle group = plugins.findGroup(name.c_str()).tail();
255  if (group.isNull()) {
256  yError("Cannot find \"%s\" plugin (not built in, and no .ini file found for it)", name.c_str());
257  yError("Check that YARP_DATA_DIRS leads to at least one directory with plugins/%s.ini or share/yarp/plugins/%s.ini in it", name.c_str(), name.c_str());
258  return false;
259  }
260  return readFromSearchable(group, name);
261  }
