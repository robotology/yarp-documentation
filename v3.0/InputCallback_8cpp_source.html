55 {

66 // Debug dropped frames using alternating 30x30 red and green squares
67 // at the bottom right of the image, see gazebo_yarp_plugins camera
68 // plugin.
69 #define DEBUG_SQUARES 0
70 #if DEBUG_SQUARES
71  int found = -1;
72  for (int i = 0; i < 10; ++i) {
73  unsigned char* pix = img.getPixelAddress(img.width()-15-(i*30),img.height()-15);
74  if(i % 2) {
75  // Check if this is a RED pixel (255,0,0)
76  // lossy carriers will change the value, so we look for a
77  // pixel close enough to the one we are looking for
78  if (pix[0] <= 5 && pix[1] >= 250 && pix[2] <= 5) {
79  found = i;
80  break;
81  }
82  } else {
83  // Check if this is a GREEN pixel (255,0,0)
84  if (pix[0] >= 250 && pix[1] <= 5 && pix[2] <= 5) {
85  found = i;
86  break;
87  }
88  }
89  }
90  if (found != expected) {
91  yWarning() << "InputCallback" << (eye==0?"left ":"right") << " expected" << expected << "found" << found << "next" << (found + 1) % 10;
92  }
93  expected = (found + 1) % 10;
94 #endif // DEBUG_SQUARES
95 
96  eyeRenderTexture->mutex.lock();
97 
98  if(eyeRenderTexture->ptr) {
99  unsigned int w = img.width();
100  unsigned int h = img.height();
101  unsigned int rs = img.getRowSize();
102  unsigned char *data = img.getRawImage();
103 
104  // update data directly on the mapped buffer
105  if (eyeRenderTexture->width == w && eyeRenderTexture->height == h) {
106  // Texture and image have the same size: no problems
107  memcpy(eyeRenderTexture->ptr, data, eyeRenderTexture->bufferSize);
108  } else if (eyeRenderTexture->width >= w && eyeRenderTexture->height >= h) {
109  // Texture is larger than image: image is centered in the texture
110  int x = (eyeRenderTexture->width - w)/2;
111  int y = (eyeRenderTexture->height - h)/2;
112  for (unsigned int i = 0; i < h; ++i) {
113  unsigned char* textureStart = eyeRenderTexture->ptr + (y+i)*eyeRenderTexture->rowSize + x*3;
114  unsigned char* dataStart = data + (i*rs);
115  memcpy(textureStart, dataStart, rs);
116  }
117  } else {
118  // Texture is smaller than image: image is cropped
119  int x = (w - eyeRenderTexture->width)/2;
120  int y = (h - eyeRenderTexture->height)/2;
121  for (unsigned int i = 0; i < eyeRenderTexture->width; ++i) {
122  unsigned char* textureStart = eyeRenderTexture->ptr + (y+i)*(i*eyeRenderTexture->rowSize) + x*3;
123  unsigned char* dataStart = data + (y+i)*rs + x*3;
124  memcpy(textureStart, dataStart, eyeRenderTexture->rowSize);
125  }
126  }
