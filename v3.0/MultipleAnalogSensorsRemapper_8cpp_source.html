99 // Return an empty list if the key is not found, and an error (false) if the key was found but it is not a list of strings
100 bool getVectorOfStringFromListInConfig(const std::string& key, const os::Searchable& config, std::vector<std::string> & vectorOfStrings)
101 {
102  yarp::os::Property prop;
103  prop.fromString(config.toString().c_str());
104  bool keyExists = prop.check(key.c_str());
105 
106  yarp::os::Bottle *propList=prop.find(key.c_str()).asList();
107  if (!propList && keyExists)
108  {
109  yError() <<"MultipleAnalogSensorsRemapper : Error parsing parameters: if present " << key << " should be followed by a list of strings.\n";
110  return false;
111  }
112 
113  if (!propList && !keyExists)
114  {
115  vectorOfStrings.resize(0);
116  return true;
117  }
118 
119  vectorOfStrings.resize(propList->size());
120  for (size_t ax=0; ax < propList->size(); ax++)
121  {
122  vectorOfStrings[ax] = propList->get(ax).asString().c_str();
123  }
124 
125  return true;
126 }

128 bool MultipleAnalogSensorsRemapper::parseOptions(const Property& prop)
129 {
130  bool ok = true;
131 
132  m_remappedSensors.resize(MAS_NrOfSensorTypes);
133 
134  for (size_t i = 0; i < MAS_NrOfSensorTypes; i++)
135  {
136  MAS_SensorType sensType = static_cast<MAS_SensorType>(i);
137  std::string optionName = MAS_getTagFromEnum(sensType) +"Names";
138  ok = getVectorOfStringFromListInConfig(optionName , prop, m_remappedSensors[i]);
139  if (!ok)
140  {
141  yError() << "MultipleAnalogSensorsRemapper: " << optionName << " should be followed by a list of string.";
142  return false;
143  }
144  }
145 
146  return ok;
147 }

154 template<typename Interface>
155 bool MultipleAnalogSensorsRemapper::genericAttachAll(const MAS_SensorType sensorType,
156  std::vector<Interface *>& subDeviceVec,
157  const PolyDriverList &polylist,
158  bool (Interface::*getNameMethodPtr)(size_t, std::string&) const,
159  size_t (Interface::*getNrOfSensorsMethodPtr)() const)
160 {
161  std::map<std::string, SensorInSubDevice> sensorLocationMap;
162 
163  subDeviceVec.resize(polylist.size());
164 
165  for(int p=0; p<polylist.size(); p++)
166  {
167  // If this fails it is ok, this just means that this devices does not expose this kind of sensors
168  polylist[p]->poly->view(subDeviceVec[p]);
169 
170  if (subDeviceVec[p])
171  {
172  size_t nrOfSensorsInSubDevice = MAS_CALL_MEMBER_FN(subDeviceVec[p], getNrOfSensorsMethodPtr)();
173  for (size_t s=0; s < nrOfSensorsInSubDevice; s++)
174  {
175  std::string name;
176  bool ok = MAS_CALL_MEMBER_FN(subDeviceVec[p], getNameMethodPtr)(s,name);
177  if (!ok)
178  {
179  yError() << "MultipleAnalogSensorsRemapper: Failure in getting a name in the device " << polylist[p]->key;
180  return false;
181  }
182 
183  // If the name is already in the map, raise an error
184  if (sensorLocationMap.find(name) != sensorLocationMap.end())
185  {
186  SensorInSubDevice deviceWithSameName = sensorLocationMap.find(name)->second;
187  yError() << "MultipleAnalogSensorsRemapper: sensor ambiguity: sensor with name " << name
188  << " present on both device " << polylist[p]->key << " " << polylist[deviceWithSameName.subDevice]->key;
189  return false;
190  }
191 
192  sensorLocationMap[name] = SensorInSubDevice(p, s);
193  }
194  }
195  }
196 
197  // Fill the indices map given the name of all the subdevices
198  std::vector<SensorInSubDevice>& sensIndicesMap = m_indicesMap[static_cast<size_t>(sensorType)];
199  sensIndicesMap.resize(m_remappedSensors[sensorType].size());
200  for(size_t i=0; i < m_remappedSensors[sensorType].size(); i++)
201  {
202  std::string name = m_remappedSensors[sensorType][i];
203  if (sensorLocationMap.find(name) == sensorLocationMap.end())
204  {
205  yError() << "MultipleAnalogSensorsRemapper: impossible to find sensor name " << name << ", exiting.";
206  return false;
207  }
208 
209  sensIndicesMap[i] = sensorLocationMap.find(name)->second;
210  }
211 
212  return true;
213 }

217 bool MultipleAnalogSensorsRemapper::attachAll(const PolyDriverList &polylist)
218 {
219  bool ok = true;
220  m_indicesMap.resize(MAS_NrOfSensorTypes);
221  ok = ok && genericAttachAll(ThreeAxisGyroscopes, m_iThreeAxisGyroscopes, polylist,
222  &IThreeAxisGyroscopes::getThreeAxisGyroscopeName, &IThreeAxisGyroscopes::getNrOfThreeAxisGyroscopes);
223  ok = ok && genericAttachAll(ThreeAxisLinearAccelerometers, m_iThreeAxisLinearAccelerometers, polylist,
224  &IThreeAxisLinearAccelerometers::getThreeAxisLinearAccelerometerName, &IThreeAxisLinearAccelerometers::getNrOfThreeAxisLinearAccelerometers);
225  ok = ok && genericAttachAll(ThreeAxisMagnetometers, m_iThreeAxisMagnetometers, polylist,
226  &IThreeAxisMagnetometers::getThreeAxisMagnetometerName, &IThreeAxisMagnetometers::getNrOfThreeAxisMagnetometers);
227  ok = ok && genericAttachAll(OrientationSensors, m_iOrientationSensors, polylist,
228  &IOrientationSensors::getOrientationSensorName, &IOrientationSensors::getNrOfOrientationSensors);
229  ok = ok && genericAttachAll(TemperatureSensors, m_iTemperatureSensors, polylist,
230  &ITemperatureSensors::getTemperatureSensorName, &ITemperatureSensors::getNrOfTemperatureSensors);
231  ok = ok && genericAttachAll(SixAxisForceTorqueSensors, m_iSixAxisForceTorqueSensors, polylist,
232  &ISixAxisForceTorqueSensors::getSixAxisForceTorqueSensorName, &ISixAxisForceTorqueSensors::getNrOfSixAxisForceTorqueSensors);
233  ok = ok && genericAttachAll(ContactLoadCellArrays, m_iContactLoadCellArrays, polylist,
234  &IContactLoadCellArrays::getContactLoadCellArrayName, &IContactLoadCellArrays::getNrOfContactLoadCellArrays);
235  ok = ok && genericAttachAll(EncoderArrays, m_iEncoderArrays, polylist,
236  &IEncoderArrays::getEncoderArrayName, &IEncoderArrays::getNrOfEncoderArrays);
237  ok = ok && genericAttachAll(SkinPatches, m_iSkinPatches, polylist,
238  &ISkinPatches::getSkinPatchName, &ISkinPatches::getNrOfSkinPatches);
239 
240  return ok;
