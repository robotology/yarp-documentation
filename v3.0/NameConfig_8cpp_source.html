50 bool NameConfig::fromString(const std::string& txt) {
51  address = Contact();
52  SplitString ss(txt.c_str());
53  if (ss.size()>=1) {
54  if (ss.get(0)[0]=='[') {
55  // use Property format
56  Property config;
57  config.fromConfig(txt.c_str());
58 
59  Bottle& b = config.findGroup("name");
60  if (b.isNull()) {
61  fprintf(stderr, "Cannot find yarp group in config file\n");
62  std::exit(1);
63  }
64  address = Contact(b.find("host").asString().c_str(),
65  b.find("port").asInt32());
66  mode = b.check("mode", Value("yarp")).asString().c_str();
67  return (address.getPort()!=0);
68  }
69  }
70 
71  if (ss.size()>=2) {
72  address = Contact(ss.get(0), NetType::toInt(ss.get(1)));
73  if (ss.size()>=3) {
74  mode = ss.get(2);
75  } else {
76  mode = "yarp";
77  }
78  if (mode=="ros") {
79  address.setCarrier("xmlrpc");
80  }
81  return true;
82  }
83  return false;
84 }

130 bool NameConfig::createPath(const std::string& fileName, int ignoreLevel) {
131  size_t index = fileName.rfind('/');
132  if (index==std::string::npos) {
133  index = fileName.rfind('\\');
134  if (index==std::string::npos) {
135  return false;
136  }
137  }
138  std::string base = fileName.substr(0, index);
139  if (yarp::os::stat((char*)base.c_str())<0) {
140  bool result = createPath(base, ignoreLevel-1);
141  if (result==false) {
142  return false;
143  }
144  }
145  if (ignoreLevel<=0) {
146  if (yarp::os::stat(fileName.c_str())<0) {
147  if (yarp::os::mkdir(fileName.c_str())>=0) {
148  return true;
149  }
150  return false;
151  }
152  }
153  return true;
154 }

219 std::string NameConfig::getHostName(bool prefer_loopback, const std::string& seed) {
220  // try to pick a good host identifier
221 
222  std::string result = "127.0.0.1";
223  bool loopback = true;
224  bool found = false;
225 
226  // Pick an IPv4 address.
227  // Prefer non-local addresses, then seed, then shorter addresses.
228  // Avoid IPv6.
229 #ifdef YARP_HAS_ACE
230  ACE_INET_Addr *ips = nullptr;
231  size_t count = 0;
232  char hostAddress[256];
233  if (ACE::get_ip_interfaces(count, ips)>=0) {
234  for (size_t i=0; i<count; i++) {
235  std::string ip = ips[i].get_host_addr(hostAddress, 256);
236 #else
237  int family, s;
238  char hostname[NI_MAXHOST];
239  std::string ip;
240  struct ifaddrs *ifaddr, *ifa;
241  if (yarp::os::impl::getifaddrs(&ifaddr) == -1) {
242  perror("getifaddrs in getIps");
243  std::exit(EXIT_FAILURE);
244  }
245  for (ifa = ifaddr; ifa != nullptr; ifa = ifa->ifa_next) {
246  if (ifa->ifa_addr == nullptr) continue;
247  family = ifa->ifa_addr->sa_family;
248  if (family == AF_INET || family == AF_INET6) {
249  s = yarp::os::impl::getnameinfo(ifa->ifa_addr,
250  (family == AF_INET) ?
251  sizeof(struct sockaddr_in) :
252  sizeof(struct sockaddr_in6),
253  hostname,
254  NI_MAXHOST,
255  nullptr,
256  0,
257  NI_NUMERICHOST);
258  if (s != 0) {
259  printf("getnameinfo() failed: %s\n", yarp::os::impl::gai_strerror(s));
260  std::exit(EXIT_FAILURE);
261  }
262  ip = std::string(hostname);
263 #endif
264 
265  YARP_DEBUG(Logger::get(), std::string("scanning network interface ") +
266  ip.c_str());
267 
268  if (ip.find(':')!=std::string::npos) continue;
269 
270  bool would_be_loopback = false;
271  if (ip == "127.0.0.1" || ip == "127.1.0.1" ||
272  ip == "127.0.1.1") {
273  would_be_loopback = true;
274  }
275 #ifdef YARP_HAS_ACE
276 #ifdef ACE_ADDR_HAS_LOOPBACK_METHOD
277  would_be_loopback = ips[i].is_loopback();
278 #endif
279 #endif
280 
281  // If we haven't any interface yet, take this one
282  if (!found) {
283  result = ip;
284  loopback = would_be_loopback;
285  found = true;
286  continue;
287  }
288 
289  // We have an interface
290 
291  // If this isn't the right kind of interface, skip it
292  if (would_be_loopback != prefer_loopback) continue;
293 
294  // This is the right kind of interface
295 
296  // If we haven't the right kind of interface yet, take it
297  if (prefer_loopback != loopback) {
298  result = ip;
299  loopback = would_be_loopback;
300  continue;
301  }
302 
303  // If it matches the seed interface, take it
304  if (ip==seed) {
305  result = ip;
306  loopback = would_be_loopback;
307  continue;
308  }
309 
310  // If it is shorter, and what we have isn't the seed, take it
311  if (ip.length()<result.length() && result!=seed) {
312  result = ip;
313  loopback = would_be_loopback;
314  continue;
315  }
316  }
317  }
318 #ifdef YARP_HAS_ACE
319  delete[] ips;
320 #else
321  freeifaddrs(ifaddr);
322 #endif
323 
324  return result.c_str();
325 }

328 bool NameConfig::isLocalName(const std::string& name) {
329  bool result = false;
330 
331 #if defined(YARP_HAS_ACE)
332  ACE_INET_Addr *ips = nullptr;
333  size_t count = 0;
334  if (ACE::get_ip_interfaces(count, ips)>=0) {
335  for (size_t i=0; i<count; i++) {
336  std::string ip = ips[i].get_host_addr();
337  if (ip==name) {
338  result = true;
339  break;
340  }
341  }
342  delete[] ips;
343  }
344 #elif defined(__unix__)
345 
349  char hostname[HOST_NAME_MAX];
350  yarp::os::impl::gethostname(hostname, HOST_NAME_MAX);
351  if (strcmp(hostname, name.c_str()) == 0) result = true;
352  if (!result) {
353  Bottle lst = getIpsAsBottle();
354  for (int i=0; i<lst.size(); i++) {
355  if (lst.get(i).asString()==name) {
356  result = true;
357  break;
358  }
359  }
360  }
361 #endif
362 
363  // just in case
364  if (name=="localhost"||name=="127.0.0.1") {
365  result = true;
366  }
367 
368  return result;
369 }
