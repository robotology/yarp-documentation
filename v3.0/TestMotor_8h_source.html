48  void update() {
49  if (posMode) return;
50  double now = yarp::os::Time::now();
51  if (last<0) last = now;
52  double dt = now-last;
53  for (int i=0; i<njoints; i++) {
54  pos[i] += vel[i]*dt;
55  }
56  last = now;
57  }
