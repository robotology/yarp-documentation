53 std::string TcpRosCarrier::getRosType(ConnectionState& proto) {
54  std::string typ;

58  typ = t.getName();

60  message_definition = t.readProperties().find("message_definition").asString();
61  user_type = typ;

68  wire_type = rtyp;
69  } else if (typ!="") {
70  rtyp = typ;
71  wire_type = typ;
72  }
73  }
74  Name n(proto.getRoute().getCarrierName() + "://test");
75  std::string mode = "topic";
76  std::string modeValue = n.getCarrierModifier("topic");
77  if (modeValue=="") {
78  mode = "service";

87 
88  dbg_printf("USER TYPE %s\n", user_type.c_str());

94 bool TcpRosCarrier::sendHeader(ConnectionState& proto) {
95  dbg_printf("Route is %s\n", proto.getRoute().toString().c_str());
96  Name n(proto.getRoute().getCarrierName() + "://test");
97  std::string mode = "topic";
98  std::string modeValue = n.getCarrierModifier("topic");
99  if (modeValue=="") {
100  mode = "service";
101  modeValue = n.getCarrierModifier("service");
102  isService = true;
103  }
104  if (modeValue=="") {
105  printf("*** no topic or service specified!\n");
106  mode = "topic";
107  modeValue = "notopic";
108  isService = false;
109  }
110  std::string rawValue = n.getCarrierModifier("raw");
111  if (rawValue=="2") {
112  raw = 2;
113  dbg_printf("ROS-native mode requested\n");
114  } else if (rawValue=="1") {
115  raw = 1;
116  dbg_printf("Raw mode requested\n");
117  } else if (rawValue=="0") {
118  raw = 0;
119  dbg_printf("Cooked mode requested\n");
120  }
121 
122  RosHeader header;
123  dbg_printf("Writing to %s\n", proto.getStreams().getRemoteAddress().toString().c_str());
124  dbg_printf("Writing from %s\n", proto.getStreams().getLocalAddress().toString().c_str());
125 
126  std::string rtyp = getRosType(proto);
127  if (rtyp!="") {
128  header.data["type"] = rtyp;
129  }
130  header.data[mode] = modeValue;
131  header.data["md5sum"] = (md5sum!="")?md5sum:"*";
132  if (message_definition!="") {
133  header.data["message_definition"] = message_definition;
134  }
135  NestedContact nc(proto.getRoute().getFromName());
136  header.data["callerid"] = nc.getNodeName();
137  header.data["persistent"] = "1";
138  string header_serial = header.writeHeader();
139  string header_len(4,'\0');
140  char *at = (char*)header_len.c_str();
141  RosHeader::appendInt32(at,header_serial.length());
142  dbg_printf("Writing %s -- %d bytes\n",
143  RosHeader::showMessage(header_len).c_str(),
144  (int)header_len.length());
145 
146  Bytes b1((char*)header_len.c_str(),header_len.length());
147  proto.os().write(b1);
148  dbg_printf("Writing %s -- %d bytes\n",
149  RosHeader::showMessage(header_serial).c_str(),
150  (int)header_serial.length());
151  Bytes b2((char*)header_serial.c_str(),header_serial.length());
152  proto.os().write(b2);
153 
154  return proto.os().isOk();
155 }

159  RosHeader header;
160 
161  char mlen[4];
162  Bytes mlen_buf(mlen,4);
163 
164  int res = proto.is().readFull(mlen_buf);
165  if (res<4) {
166  printf("Fail %s %d\n", __FILE__, __LINE__);
167  return false;
168  }
169  int len = NetType::netInt(mlen_buf);
170  dbg_printf("Len %d\n", len);
171  if (len>10000) {
172  printf("not ready for serious messages\n");
173  return false;
174  }
175  ManagedBytes m(len);
176  res = proto.is().readFull(m.bytes());
177  if (res!=len) {
178  printf("Fail %s %d\n", __FILE__, __LINE__);
179  return false;
180  }
181  header.readHeader(string(m.get(),m.length()));
182  dbg_printf("Message header: %s\n", header.toString().c_str());
183  std::string rosname;
184  if (header.data.find("type")!=header.data.end()) {
185  rosname = header.data["type"];
186  }
187  dbg_printf("<incoming> Type of data is [%s]s\n", rosname.c_str());
188  if (header.data.find("callerid")!=header.data.end()) {
189  string name = header.data["callerid"];
190  dbg_printf("<incoming> callerid is %s\n", name.c_str());
191  dbg_printf("Route was %s\n", proto.getRoute().toString().c_str());
192  Route route = proto.getRoute();
193  route.setToName(name);
194  proto.setRoute(route);
195  dbg_printf("Route is now %s\n", proto.getRoute().toString().c_str());
196  }
197 
198  if (!isService) {
199  isService = (header.data.find("request_type")!=header.data.end());
200  }
201  if (rosname!="" && (user_type != wire_type || user_type == "")) {
202  kind = TcpRosStream::rosToKind(rosname.c_str());
203  TcpRosStream::configureTwiddler(twiddler,kind.c_str(),rosname.c_str(),false,false);
204  translate = TCPROS_TRANSLATE_TWIDDLER;
205  } else {
206  rosname = "";
207  }
208  dbg_printf("tcpros %s mode\n", isService?"service":"topic");
209 
210  // we may be a pull stream
211  sender = isService;
212 
213  processRosHeader(header);
214 
215  auto* stream = new TcpRosStream(proto.giveStreams(),
216  sender,
217  sender,
218  isService,
219  raw,
220  rosname.c_str());
221 
222  if (stream==nullptr) { return false; }
223 
224  dbg_printf("Getting ready to hand off streams...\n");
225 
226  proto.takeStreams(stream);
227 
228  return proto.is().isOk();

250  return false;

270 
271  route = proto.getRoute();
272  if (header.data.find("callerid")!=header.data.end()) {
273  route.setFromName(header.data["callerid"]);

275  route.setFromName("tcpros");
276  }
277  proto.setRoute(route);
278 
279  // Let's just ignore everything that is sane and holy, and
280  // send the same header right back.
281  // **UPDATE** Oh, ok, let's modify the callerid. Begrudgingly.
282  NestedContact nc(proto.getRoute().getToName());
283  header.data["callerid"] = nc.getNodeName();

285  string header_serial = header.writeHeader();

291  (int)header_len.length());

295  dbg_printf("Writing %s -- %d bytes\n",
296  RosHeader::showMessage(header_serial).c_str(),
297  (int)header_serial.length());

299  proto.os().write(b2);
300 
301  if (header.data.find("probe")!=header.data.end()) {

304  }
305 
306 
307  if (!isService) {
308  isService = (header.data.find("service")!=header.data.end());
309  }

311  if (wire_type!="sensor_msgs/Image") { // currently using a custom method for images
312  kind = TcpRosStream::rosToKind(rosname.c_str());

314  translate = TCPROS_TRANSLATE_TWIDDLER;
315  }
316  } else {
317  rosname = "";
318  }

320 
321  processRosHeader(header);
322 
323  if (isService) {
324  auto* stream = new TcpRosStream(proto.giveStreams(),
325  sender,
326  false,
327  isService,
328  raw,
329  rosname.c_str());
330  if (stream==nullptr) { return false; }
331  proto.takeStreams(stream);

338 bool TcpRosCarrier::write(ConnectionState& proto, SizedWriter& writer) {
339  SizedWriter *flex_writer = &writer;
340 
341  if (raw!=2) {
342  // At startup, we check for what kind of messages are going
343  // through, and prepare an appropriate byte-rejiggering if
344  // needed.
345  if (translate==TCPROS_TRANSLATE_UNKNOWN) {
346  dbg_printf("* TCPROS_TRANSLATE_UNKNOWN\n");
347  FlexImage *img = nullptr;
348  if (user_type=="yarp/image"||user_type=="yarp/bottle") {
349  img = wi.checkForImage(writer);
350  }
351  if (img) {
352  translate = TCPROS_TRANSLATE_IMAGE;
353  std::string frame = "/frame";
354  ri.init(*img,frame);
355  } else {
356  if (WireBottle::extractBlobFromBottle(writer,wt)) {
357  translate = TCPROS_TRANSLATE_BOTTLE_BLOB;
358  } else {
359  translate = TCPROS_TRANSLATE_INHIBIT;
360  }
361  }
362  }
363  } else {
364  translate = TCPROS_TRANSLATE_INHIBIT;
365  }
366 
367  // Apply byte-rejiggering if needed.
368  switch (translate) {
369  case TCPROS_TRANSLATE_IMAGE:
370  {
371  dbg_printf("* TCPROS_TRANSLATE_IMAGE\n");
372  FlexImage *img = wi.checkForImage(writer);
373  if (img==nullptr) {
374  fprintf(stderr, "TCPROS Expected an image, but did not get one.\n");
375  return false;
376  }
377  ri.update(img,seq,Time::now()); // Time here is the timestamp of the ROS message, so Time::now(), the mutable one is correct.
378  seq++;
379  flex_writer = &ri;
380  }
381  break;
382  case TCPROS_TRANSLATE_BOTTLE_BLOB:
383  {
384  dbg_printf("* TCPROS_TRANSLATE_BOTTLE_BLOB\n");
385  if (!WireBottle::extractBlobFromBottle(writer,wt)) {
386  fprintf(stderr, "TCPROS Expected a bottle blob, but did not get one.\n");
387  return false;
388  }
389  flex_writer = &wt;
390  }
391  break;
392  case TCPROS_TRANSLATE_TWIDDLER:
393  {
394  dbg_printf("* TCPROS_TRANSLATE_TWIDDLER\n");
395  twiddler_output.attach(writer,twiddler);
396  if (twiddler_output.update()) {
397  flex_writer = &twiddler_output;
398  } else {
399  flex_writer = nullptr;
400  }
401  }
402  break;
403  case TCPROS_TRANSLATE_INHIBIT:
404  dbg_printf("* TCPROS_TRANSLATE_INHIBIT\n");
405  break;
406  default:
407  dbg_printf("* TCPROS_TRANSLATE_OTHER\n");
408  break;
409  }
410 
411  if (flex_writer == nullptr) {
412  return false;
413  }
414 
415  int len = 0;
416  for (size_t i=0; i<flex_writer->length(); i++) {
417  len += (int)flex_writer->length(i);
418  }
419  dbg_printf("Prepping to write %d blocks (%d bytes)\n",
420  (int)flex_writer->length(),
421  len);
422 
423  string header_len(4,'\0');
424  char *at = (char*)header_len.c_str();
425  RosHeader::appendInt32(at,len);
426  Bytes b1((char*)header_len.c_str(),header_len.length());
427  proto.os().write(b1);
428  flex_writer->write(proto.os());
429 
430  dbg_printf("done sending\n");
431 
432  if (isService) {
433  if (!sender) {
434  if (!persistent) {
435  proto.os().close();
436  }
437  }
438  }
439 
440  return proto.getStreams().isOk();
441 }
