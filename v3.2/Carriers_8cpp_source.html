32 std::string bytes_to_string(const Bytes& header)
33 {
34  std::string ret;
35  for (size_t i = 0; i < header.length(); i++) {
36  ret += NetType::toString(header.get()[i]);

38  }
39  ret += "[";

75 Carrier* Carriers::Private::chooseCarrier(const std::string& name,
76  bool load_if_needed,
77  bool return_template)
78 {
79  auto pos = name.find('+');
80  if (pos != std::string::npos) {
81  return chooseCarrier(name.substr(0, pos), load_if_needed, return_template);
82  }
83 
84  for (auto& delegate : delegates) {
85  Carrier& c = *delegate;
86  if (name == c.getName()) {
87  if (!return_template) {
88  return c.create();
89  }
90  return &c;
91  }
92  }
93 
94  if (load_if_needed) {
95  // ok, we didn't find a carrier, but we have a name.
96  // let's try to register it, and see if a dll is found.
97  if (NetworkBase::registerCarrier(name.c_str(), nullptr)) {
98  // We made progress, let's try again...
99  return Carriers::Private::chooseCarrier(name, false);
100  }
101  }
102 
103  YARP_SPRINTF1(Logger::get(),
104  error,
105  "Could not find carrier \"%s\"",
106  (!name.empty()) ? name.c_str() : "[bytes]");
107 
108  return nullptr;
109 }

111 Carrier* Carriers::Private::chooseCarrier(const Bytes& header,
112  bool load_if_needed)
113 {
114  for (auto& delegate : delegates) {
115  Carrier& c = *delegate;
116  if (c.checkHeader(header)) {
117  return c.create();
118  }
119  }
120 
121  if (load_if_needed) {
122  if (scanForCarrier(header)) {
123  // We made progress, let's try again...
124  return Carriers::Private::chooseCarrier(header, true);
125  }
126  }
127 
128  YARP_SPRINTF1(Logger::get(),
129  error,
130  "Could not find carrier for a connection starting with: %s",
131  bytes_to_string(header).c_str());
132 
133  return nullptr;
134 }

137 bool Carriers::Private::matchCarrier(const Bytes& header, Bottle& code)
138 {
139  size_t at = 0;
140  bool success = true;
141  bool done = false;
142  for (size_t i = 0; i < code.size() && !done; i++) {
143  Value& v = code.get(i);
144  if (v.isString()) {
145  std::string str = v.asString();
146  for (char j : str) {
147  if (header.length() <= at) {
148  success = false;
149  done = true;
150  break;
151  }
152  if (j != header.get()[at]) {
153  success = false;
154  done = true;
155  break;
156  }
157  at++;
158  }
159  } else {
160  at++;
161  }
162  }
163  return success;
164 }

166 bool Carriers::Private::checkForCarrier(const Bytes& header, Searchable& group)
167 {
168  Bottle code = group.findGroup("code").tail();
169  if (code.size() == 0) {
170  return false;
171  }
172  if (matchCarrier(header, code)) {
173  std::string name = group.find("name").asString();
174  if (NetworkBase::registerCarrier(name.c_str(), nullptr)) {
175  return true;
176  }
177  }
178  return false;
179 }

181 bool Carriers::Private::scanForCarrier(const Bytes& header)
182 {
183  YARP_SPRINTF0(Logger::get(),
184  debug,
185  "Scanning for a carrier by header.");
186  YarpPluginSelector selector;
187  selector.scan();
188  Bottle lst = selector.getSelectedPlugins();
189  for (size_t i = 0; i < lst.size(); i++) {
190  if (checkForCarrier(header, lst.get(i))) {
191  return true;
192  }
193  }
194  return false;

322 Bottle Carriers::listCarriers()
323 {
324  Carriers& instance = getInstance();
325  std::lock_guard<std::mutex> guard(Private::mutex);
326 
327  Bottle lst;
328  Property done;
329 
330  std::vector<Carrier*>& delegates = instance.mPriv->delegates;
331  for (auto& delegate : delegates) {
332  Carrier& c = *delegate;
333  lst.addString(c.getName());
334  done.put(c.getName(), 1);
335  }
336 
337  instance.mPriv->scan();
338  Bottle plugins = instance.mPriv->getSelectedPlugins();
339  for (size_t i = 0; i < plugins.size(); i++) {
340  Value& options = plugins.get(i);
341  std::string name = options.check("name", Value("untitled")).asString();
342  if (done.check(name)) {
343  continue;
344  }
345 
346  SharedLibraryFactory lib;
347  YarpPluginSettings settings;
348  settings.setSelector(*instance.mPriv);
349  settings.readFromSearchable(options, name);
350  settings.open(lib);
351  if (lib.getName().empty()) {
352  continue;
353  }
354  lst.addString(name);
355  done.put(name, 1);
356  }
357 
358  return lst;
359 }
