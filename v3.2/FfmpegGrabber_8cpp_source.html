195  bool getAudio(AVPacket& packet,Sound& sound)
196  {
197  int ct = 0;
198  int bytesRead = 0;
199  int bytesWritten = 0;
200  int gotFrame = 0;
201  while (bytesRead<packet.size) {
202  ct = audioBufferLen;
203  AVPacket tmp = packet;
204  tmp.data += bytesRead;
205  tmp.size -= bytesRead;
206  if (!pAudio) {
207  if (!(pAudio = av_frame_alloc())) {
208  fprintf(stderr, "out of memory\n");
209  ::exit(1);
210  }
211  } else {
212  av_frame_unref(pAudio);
213  }
214  int r = avcodec_decode_audio4(pCodecCtx, pAudio, &gotFrame, &packet);
215  ct = 0;
216  if (gotFrame) {
217  ct = av_samples_get_buffer_size(nullptr,
218  pCodecCtx->channels,
219  pFrame->nb_samples,
220  pCodecCtx->sample_fmt,
221  1);
222  }
223  if (r<0) {
224  printf("error decoding audio\n");
225  return false;
226  }
227  int num_channels = getChannels();
228  int num_rate = getRate();
229  //audioBufferAt += ct;
230  //audioBufferLen += ct;
231  bytesRead += r;
232  bytesWritten += ct;
233  if (bytesRead==packet.size) {
234  int num_samples = bytesWritten/(sizeof(int16_t)*num_channels);
235  sound.resize(num_samples,num_channels);
236  sound.setFrequency(num_rate);
237 
238  int idx = 0;
239  for (int i=0; i<num_samples; i++) {
240  for (int j=0; j<num_channels; j++) {
241  sound.set(audioBuffer[idx],i,j);
242  idx++;
243  }
244  }
245  }
246  }
247  return true;
248  }

456 bool FfmpegGrabber::open(yarp::os::Searchable & config)
457 {
458  std::string fname =
459  config.check("source",
460  Value("default.avi"),
461  "media file to read from").asString();
462 
463  if (config.check("loop","media should loop (default)")) {
464  shouldLoop = true;
465  }
466 
467  if (config.check("noloop","media should not loop")) {
468  shouldLoop = false;
469  }
470 
471  imageSync = false;
472  std::string sync =
473  config.check("sync",
474  Value("image"),
475  "sync on image or audio (if have to choose)?").asString();
476  imageSync = (sync=="image");
477 
478  needRateControl = true; // default for recorded media
479 
480  if (config.check("nodelay","media will play in simulated realtime unless this is present")) {
481  needRateControl = false;
482  }
483 
484  pace = config.check("pace",Value(1.0),
485  "simulated realtime multiplier factor (must be <1 right now)").asFloat64();
486 
487  // Register all formats and codecs
488  av_register_all();
489  avdevice_register_all();
490 
491  // Open video file
492  if (config.check("v4l","if present, read from video4linux") || config.check("v4l1","if present, read from video4linux") || config.check("v4l2","if present, read from video4linux2")) {
493  needRateControl = false; // reading from live media
494  if (!openV4L(config,&pFormatCtx,&pFormatCtx2)) {
495  printf("Could not open Video4Linux input\n");
496  return false;
497  }
498  } else if (config.check("ieee1394","if present, read from firewire")) {
499  needRateControl = false; // reading from live media
500  if (!openFirewire(config,&pFormatCtx)) {
501  printf("Could not open ieee1394 input\n");
502  return false;
503  }
504  } else {
505  if (!openFile(&pFormatCtx,fname.c_str())) {
506  printf("Could not open media file %s\n", fname.c_str());
507  return false; // Couldn't open file
508  }
509  }
510 
511 
512  // Retrieve stream information
513  if(avformat_find_stream_info(pFormatCtx, nullptr)<0) {
514  printf("Could not find stream information in %s\n", m_uri.c_str());
515  return false; // Couldn't find stream information
516  }
517 
518  // Dump information about file onto standard error
519  av_dump_format(pFormatCtx, 0, m_uri.c_str(), false);
520 
521  if (pFormatCtx2!=nullptr) {
522 
523  if(avformat_find_stream_info(pFormatCtx2, nullptr)<0) {
524  printf("Could not find stream information in %s\n", m_uri.c_str());
525  return false; // Couldn't find stream information
526  }
527 
528  // Dump information about file onto standard error
529  av_dump_format(pFormatCtx2, 0, m_uri.c_str(), false);
530  }
531 
532 
533  if (pFormatCtx2!=nullptr) {
534  pAudioFormatCtx = pFormatCtx2;
535  } else {
536  pAudioFormatCtx = pFormatCtx;
537  }
538 
539  yAssert(system_resource==nullptr);
540  system_resource = new FfmpegHelper;
541  yAssert(system_resource!=nullptr);
542  FfmpegHelper& helper = HELPER(system_resource);
543  DecoderState& videoDecoder = helper.videoDecoder;
544  DecoderState& audioDecoder = helper.audioDecoder;
545 
546 
547  // Find the first video stream
548  int videoStream = videoDecoder.getStream(pFormatCtx,
549  AVMEDIA_TYPE_VIDEO,
550  "video");
551  // Find the first audio stream
552  int audioStream = audioDecoder.getStream(pAudioFormatCtx,
553  AVMEDIA_TYPE_AUDIO,
554  "audio");
555 
556  if (videoStream==-1&&audioStream==-1) {
557  return false;
558  }
559 
560  _hasVideo = (videoStream!=-1);
561  _hasAudio = (audioStream!=-1);
562 
563  bool ok = true;
564  if (_hasVideo) {
565  ok = ok && videoDecoder.getCodec(pFormatCtx);
566  }
567  if (_hasAudio) {
568  ok = ok && audioDecoder.getCodec(pAudioFormatCtx);
569  }
570  if (!ok) return false;
571 
572  if (_hasVideo) {
573  ok = ok && videoDecoder.allocateImage();
574  }
575  if (_hasAudio) {
576  ok = ok && audioDecoder.allocateSound();
577  }
578  if (!ok) return false;
579 
580  if (_hasVideo) {
581  m_w = videoDecoder.getWidth();
582  m_h = videoDecoder.getHeight();
583  }
584  if (_hasAudio) {
585  m_channels = audioDecoder.getChannels();
586  m_rate = audioDecoder.getRate();
587  }
588  printf(" video size %dx%d, audio %dHz with %d channels, %s sync\n",
589  m_w, m_h,
590  m_rate, m_channels,
591  imageSync?"image":"audio");
592 
593  if (!(_hasVideo||_hasAudio)) {
594  return false;
595  }
596  active = true;
597  return true;
598 }

650 bool FfmpegGrabber::getAudioVisual(yarp::sig::ImageOf<yarp::sig::PixelRgb>& image,
651  yarp::sig::Sound& sound)
652 {
653 
654  FfmpegHelper& helper = HELPER(system_resource);
655  DecoderState& videoDecoder = helper.videoDecoder;
656  DecoderState& audioDecoder = helper.audioDecoder;
657 
658  bool tryAgain = false;
659  bool triedAgain = false;
660 
661  do {
662 
663  bool gotAudio = false;
664  bool gotVideo = false;
665  if (startTime<0.5) {
666  startTime = SystemClock::nowSystem();
667  }
668  double time_target = 0;
669  while(av_read_frame(pFormatCtx, &packet)>=0) {
670  // Is this a packet from the video stream?
671  bool done = false;
672  if (packet.stream_index==videoDecoder.getIndex()) {
673  done = videoDecoder.getVideo(packet);
674  image.resize(1,1);
675  if (done) {
676  //printf("got a video frame\n");
677  gotVideo = true;
678  }
679  } if (packet.stream_index==audioDecoder.getIndex()) {
680  done = audioDecoder.getAudio(packet,sound);
681  if (done) {
682  //printf("got an audio frame\n");
683  gotAudio = true;
684  }
685  }
686  AVRational& time_base = pFormatCtx->streams[packet.stream_index]->time_base;
687  double rbase = av_q2d(time_base);
688 
689  time_target = packet.pts*rbase;
690 
691  av_free_packet(&packet);
692  if (((imageSync?gotVideo:videoDecoder.haveFrame())||!_hasVideo)&&
693  ((imageSync?1:gotAudio)||!_hasAudio)) {
694  if (_hasVideo) {
695  videoDecoder.getVideo(image);
696  } else {
697  image.resize(0,0);
698  }
699  if (needRateControl) {
700  double now = (SystemClock::nowSystem()-startTime)*pace;
701  double delay = time_target-now;
702  if (delay>0) {
703  SystemClock::delaySystem(delay);
704  }
705  }
706 
707  if (!_hasAudio) {
708  sound.resize(0,0);
709  }
710  return true;
711  }
712  }
713 
714  tryAgain = !triedAgain;
715 
716  if (tryAgain) {
717  if (!shouldLoop) {
718  return false;
719  }
720  av_seek_frame(pFormatCtx,-1,0,AVSEEK_FLAG_BACKWARD);
721  startTime = SystemClock::nowSystem();
722  triedAgain = true;
723  }
724  } while (tryAgain);
725 
726  return false;
727 }
