288 void FakeLaser::run()
289 {
290  mutex.lock();
291  laser_data.clear();
292  double t = yarp::os::Time::now();
293  static double t_orig = yarp::os::Time::now();
294  double size = (t - (t_orig));
295 
296  static int test_count = 0;
297  static int test = 0;
298 
299  if (m_test_mode == USE_PATTERN)
300  {
301  for (int i = 0; i < sensorsNum; i++)
302  {
303  double value = 0;
304  if (test == 0)
305  { value = i / 100.0; }
306  else if (test == 1)
307  { value = size * 2; }
308  else if (test == 2)
309  {
310  if (i >= 0 && i <= 10) { value = 1.0 + i / 20.0; }
311  else if (i >= 90 && i <= 100) { value = 2.0 + (i - 90) / 20.0; }
312  else { value = min_distance; }
313  }
314 
315  if (value < min_distance) { value = min_distance; }
316  if (value > max_distance) { value = max_distance; }
317  laser_data.push_back(value);
318  }
319 
320  test_count++;
321  if (test_count == 60)
322  {
323  test_count = 0; test++; if (test > 2) { test = 0; }
324  t_orig = yarp::os::Time::now();
325  }
326  }
327  else if (m_test_mode == NO_OBSTACLES)
328  {
329  for (int i = 0; i < sensorsNum; i++)
330  {
331  laser_data.push_back(std::numeric_limits<double>::infinity());
332  }
333  }
334  else if (m_test_mode == USE_MAPFILE)
335  {
336  if (m_loc_mode == LOC_NOT_SET)
337  {
338  //do nothing
339  }
340  else if (m_loc_mode == LOC_FROM_PORT)
341  {
342  Bottle* b = m_loc_port->read(false);
343  if (b)
344  {
345  m_loc_x = b->get(0).asFloat64();
346  m_loc_y = b->get(1).asFloat64();
347  m_loc_t = b->get(2).asFloat64();
348  }
349  }
350  else if (m_loc_mode == LOC_FROM_CLIENT)
351  {
352  yarp::dev::Map2DLocation loc;
353  if (m_iLoc->getCurrentPosition(loc))
354  {
355  m_loc_x = loc.x;
356  m_loc_y = loc.y;
357  m_loc_t = loc.theta;
358  }
359  else
360  {
361  yError() << "Error occurred while getting current position from localization server";
362  }
363  }
364  else
365  {
366  yDebug() << "No localization mode selected. This branch should be not reachable.";
367  }
368 
369  for (int i = 0; i < sensorsNum; i++)
370  {
371  //compute the ray in the robot reference frame
372  double robot_curr_t = i*resolution + min_angle;
373  double robot_curr_x = max_distance * cos(robot_curr_t*DEG2RAD);
374  double robot_curr_y = max_distance * sin(robot_curr_t*DEG2RAD);
375 
376  //transforms the ray from the robot to the world reference frame
377  MapGrid2D::XYWorld ray_world;
378  ray_world.x = robot_curr_x*cos(m_loc_t*DEG2RAD) - robot_curr_y*sin(m_loc_t*DEG2RAD) + m_loc_x;
379  ray_world.y = robot_curr_x*sin(m_loc_t*DEG2RAD) + robot_curr_y*cos(m_loc_t*DEG2RAD) + m_loc_y;
380  MapGrid2D::XYCell src = m_map.world2Cell(MapGrid2D::XYWorld(m_loc_x, m_loc_y));
381  MapGrid2D::XYCell dst = m_map.world2Cell(ray_world);
382  double distance = checkStraightLine(src,dst);
383  laser_data.push_back(distance + (*m_dis)(*m_gen));
384  }
385  }
386 
387  mutex.unlock();
388  return;
389 }

391 double FakeLaser::checkStraightLine(MapGrid2D::XYCell src, MapGrid2D::XYCell dst)
392 {
393  MapGrid2D::XYCell src_final = src;
394 
395  //here using the fast Bresenham algorithm
396  int dx = abs(dst.x - src.x);
397  int dy = abs(dst.y - src.y);
398  int err = dx - dy;
399 
400  int sx;
401  int sy;
402  if (src.x < dst.x) { sx = 1; } else { sx = -1; }
403  if (src.y < dst.y) { sy = 1; } else { sy = -1; }
404 
405  while (true)
406  {
407  //if (m_map.isFree(src) == false)
408  if (m_map.isWall(src))
409  {
410  yarp::dev::MapGrid2D::XYWorld world_start = m_map.cell2World(src);
411  yarp::dev::MapGrid2D::XYWorld world_end = m_map.cell2World(src_final);
412  return sqrt(pow(world_start.x - world_end.x, 2) + pow(world_start.y - world_end.y, 2));
413  }
414  if (src.x == dst.x && src.y == dst.y) break;
415  int e2 = err * 2;
416  if (e2 > -dy)
417  {
418  err = err - dy;
419  src.x += sx;
420  }
421  if (e2 < dx)
422  {
423  err = err + dx;
424  src.y += sy;
425  }
426  }
427  return std::numeric_limits<double>::infinity();
428 }
