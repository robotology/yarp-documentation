72 void StateExtendedInputPort::onRead(yarp::dev::impl::jointData &v)
73 {
74  now=Time::now();
75  mutex.lock();
76 
77  if (count>0)
78  {
79  double tmpDT=now-prev;
80  deltaT+=tmpDT;
81  if (tmpDT>deltaTMax)
82  deltaTMax=tmpDT;
83  if (tmpDT<deltaTMin)
84  deltaTMin=tmpDT;
85  }
86 
87  prev=now;
88  count++;
89 
90  valid=true;
91  last=v;
92  getEnvelope(lastStamp);
93  //check that timestamp are available
94  if (!lastStamp.isValid())
95  lastStamp.update(now);
96  mutex.unlock();
97 }

103 bool StateExtendedInputPort::getLastSingle(int j, int field, double *data, Stamp &stamp, double &localArrivalTime)
104 {
105  mutex.lock();
106  bool ret = valid;
107  if (ret)
108  {
109  switch(field)
110  {
111  case VOCAB_ENCODER:
112  *data = last.jointPosition[j];
113  ret = last.jointPosition_isValid;
114  break;
115 
116  case VOCAB_ENCODER_SPEED:
117  ret = last.jointVelocity_isValid;
118  *data = last.jointVelocity[j];
119  break;
120 
121  case VOCAB_ENCODER_ACCELERATION:
122  ret = last.jointAcceleration_isValid;
123  *data = last.jointAcceleration[j];
124  break;
125 
126  case VOCAB_MOTOR_ENCODER:
127  ret = last.motorPosition_isValid;
128  *data = last.motorPosition[j];
129  break;
130 
131  case VOCAB_MOTOR_ENCODER_SPEED:
132  ret = last.motorVelocity_isValid;
133  *data = last.motorVelocity[j];
134  break;
135 
136  case VOCAB_MOTOR_ENCODER_ACCELERATION:
137  ret = last.motorAcceleration_isValid;
138  *data = last.motorAcceleration[j];
139  break;
140 
141  case VOCAB_TRQ:
142  ret = last.torque_isValid;
143  *data = last.torque[j];
144  break;
145 
146  case VOCAB_PWMCONTROL_PWM_OUTPUT:
147  ret = last.pwmDutycycle_isValid;
148  *data = last.pwmDutycycle[j];
149  break;
150 
151  case VOCAB_AMP_CURRENT:
152  ret = last.current_isValid;
153  *data = last.current[j];
154  break;
155 
156  default:
157  yError() << "RemoteControlBoard internal error while reading data. Cannot get 'single' data of type " << yarp::os::Vocab::decode(field);
158  break;
159  }
160 
161  localArrivalTime=now;
162  stamp = lastStamp;
163  if (ret && ( (Time::now()-localArrivalTime) > timeout) )
164  ret = false;
165  }
166  mutex.unlock();
167 
168  return ret;
169 }
