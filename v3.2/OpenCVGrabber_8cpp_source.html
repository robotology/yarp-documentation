68 bool OpenCVGrabber::open(Searchable & config) {
69  m_saidSize = false;
70  m_saidResize = false;
71  m_transpose = false;
72  m_flip_x = false;
73  m_flip_y = false;
74 
75  // Are we capturing from a file or a camera ?
76  std::string file = config.check("movie", Value(""),
77  "if present, read from specified file rather than camera").asString();
78  fromFile = (file!="");
79  if (fromFile) {
80 
81  // Try to open a capture object for the file
82  m_cap.open(file.c_str());
83  if (!m_cap.isOpened()) {
84  yError("Unable to open file '%s' for capture!", file.c_str());
85  return false;
86  }
87 
88  // Should we loop?
89  m_loop = config.check("loop","if present, loop movie");
90 
91  } else {
92 
93  m_loop = false;
94 
95  int camera_idx =
96  config.check("camera",
97  Value(CV_CAP_ANY),
98  "if present, read from camera identified by this index").asInt32();
99 
100  // Try to open a capture object for the first camera
101  m_cap.open(camera_idx);
102  if (!m_cap.isOpened()) {
103  yError("Unable to open camera for capture!");
104  return false;
105  }
106  else
107  {
108  yInfo("Capturing from camera: %d",camera_idx);
109  }
110 
111  if ( config.check("framerate","if present, specifies desired camera device framerate") ) {
112  double m_fps = config.check("framerate", Value(-1)).asFloat64();
113  m_cap.set(CV_CAP_PROP_FPS, m_fps);
114  }
115 
116  if (config.check("flip_x", "if present, flip the image along the x-axis")) m_flip_x = true;
117  if (config.check("flip_y", "if present, flip the image along the y-axis")) m_flip_y = true;
118  if (config.check("transpose", "if present, rotate the image along of 90 degrees")) m_transpose = true;
119  }
120 
121 
122  // Extract the desired image size from the configuration if
123  // present, otherwise query the capture device
124  if (config.check("width","if present, specifies desired image width")) {
125  m_w = config.check("width", Value(0)).asInt32();
126  if (!fromFile && m_w>0) {
127  m_cap.set(CV_CAP_PROP_FRAME_WIDTH, m_w);
128  }
129  } else {
130  m_w = (size_t)m_cap.get(CV_CAP_PROP_FRAME_WIDTH);
131  }
132 
133  if (config.check("height","if present, specifies desired image height")) {
134  m_h = config.check("height", Value(0)).asInt32();
135  if (!fromFile && m_h>0) {
136  m_cap.set(CV_CAP_PROP_FRAME_HEIGHT, m_h);
137  }
138  } else {
139  m_h = (size_t)m_cap.get(CV_CAP_PROP_FRAME_HEIGHT);
140  }
141 
142  // Ignore capture properties - they are unreliable
143 
144  yInfo("OpenCVGrabber opened");
145  // Success!
146 
147  // save our configuration for future reference
148  m_config.fromString(config.toString());
149 
150  return true;
151 
152 }

181 bool OpenCVGrabber::sendImage(const cv::Mat & frame, ImageOf<PixelRgb> & image)
182 {
183  // Resize the output image, this should not result in new
184  // memory allocation if the image is already the correct size
185  image.resize(frame.cols, frame.rows);
186 
187  if (!m_saidSize) {
188  yDebug("Received image of size %zux%zu\n", image.width(), image.height());
189  m_saidSize = true;
190  }
191 
192  // create the timestamp
193  m_laststamp.update();
194 
195  // Convert to RGB color space
196  cv::Mat frame_rgb;
197  cv::cvtColor(frame, frame_rgb, cv::COLOR_BGR2RGB);
198 
199  // Copy the captured image to the output image
200  memcpy(image.getRawImage(), frame_rgb.data, sizeof(unsigned char) * frame_rgb.rows * frame_rgb.cols * frame_rgb.channels());
201 
202  if (m_w == 0) {
203  m_w = image.width();
204  }
205  if (m_h == 0) {
206  m_h = image.height();
207  }
208  if (fromFile) {
209  if (m_w>0 && m_h>0) {
210  if (image.width() != m_w || image.height() != m_h) {
211  if (!m_saidResize) {
212  yDebug("Software scaling from %zux%zu to %zux%zu", image.width(), image.height(), m_w, m_h);
213  m_saidResize = true;
214  }
215  image.copy(image, m_w, m_h);
216  }
217  }
218  }
219 
220  DBG yDebug("%zu by %zu image\n", image.width(), image.height());
221 
222  return true;
223 
224 }

226 bool OpenCVGrabber::getImage(ImageOf<PixelRgb> & image) {
227 
228  // Must have a capture object
229  if (!m_cap.isOpened()) {
230  image.zero();
231  return false;
232  }
233 
234  // Grab and retrieve a frame
235  cv::Mat frame;
236  m_cap.read(frame);
237 
238  if (frame.empty() && m_loop) {
239  bool ok = open(m_config);
240  if (!ok) return false;
241  m_cap.read(frame);
242  }
243 
244  if (frame.empty()) {
245  image.zero();
246  return false;
247  }
248 
249  if (m_transpose)
250  {
251  cv::transpose(frame, frame);
252  }
253 
254  if (m_flip_x && m_flip_y)
255  {
256  cv::flip(frame, frame, -1);
257  }
258  else if (m_flip_x)
259  {
260  cv::flip(frame, frame, 0);
261  }
262  else if (m_flip_y)
263  {
264  cv::flip(frame, frame, 1);
265  }
266 
267  return sendImage(frame, image);
268 }
