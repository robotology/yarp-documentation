253 bool fakeDepthCameraDriver::getDepthImage(ImageOf<PixelFloat>& depthImage, Stamp* timeStamp)
254 {
255  if (!image->getImage(imageof)) {return false;}
256  depthImage.resize(imageof);
257  for (size_t i = 0; i < imageof.width(); i++)
258  {
259  for (size_t j = 0; j < imageof.height(); j++)
260  {
261  PixelRgb pix = (*(PixelRgb*)imageof.getPixelAddress(i, j));
262  *(PixelFloat*)depthImage.getPixelAddress(i, j) = (float(pix.b) / 255.0)/3.0 + (float(pix.g) / 255.0) / 3.0 + (float(pix.r) / 255.0) / 3.0;
263  }
264  }
265  if(timeStamp)
266  timeStamp->update(yarp::os::Time::now());
267  return true;
268 }
