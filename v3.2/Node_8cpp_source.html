42  void report(const PortInfo& info) override
43  {
44  if (info.tag == PortInfo::PORTINFO_CONNECTION) {
45  NameClient& nic = NameClient::getNameClient();

202  void clear()
203  {
204  if (!mutex.try_lock()) {
205  return;
206  }
207  while (name_cache.begin() != name_cache.end()) {
208  Contactable* c = name_cache.begin()->first;
209  if (c != nullptr) {
210  mutex.unlock();
211  c->interrupt();
212  c->close();
213  mutex.lock();
214  // Close will remove the Contactable from the map only the first
215  // time that a node is found (for example if "/foo+@/node" and
216  // "/foo-@node" are registered, only the first time that "/node"
217  // is found it is removed automatically. The second time it must
218  // be removed manually.
219  if (!name_cache.empty() && name_cache.begin()->first == c) {
220  name_cache.erase(name_cache.begin());
221  }
222  }
223  }
224  mutex.unlock();

248  void getBusInfo(NodeArgs& na)
249  {
250  unsigned int opaque_id = 1;
251  ROSReport report;
252  Value v;
253  Bottle* connections = v.asList();
254 
255  mutex.lock();
256  for (auto& it : by_part_name) {
257  NodeItem& item = it.second;
258  if (!(item.isSubscriber() || item.isPublisher())) {
259  continue;
260  }
261  item.update();
262  item.contactable->getReport(report);
263  }
264  mutex.unlock();
265 
266  for (std::multimap<std::string, std::string>::const_iterator it = report.outgoingURIs.begin(); it != report.outgoingURIs.end(); ++it) {
267  Bottle& lst = connections->addList();
268  lst.addInt32(opaque_id); // connectionId
269  lst.addString(it->second);
270  lst.addString("o");
271  lst.addString("TCPROS");
272  NestedContact nc(it->first);
273  lst.addString(toRosName(nc.getNestedName()));
274  opaque_id++;
275  }
276 
277  for (std::multimap<std::string, std::string>::const_iterator it = report.incomingURIs.begin(); it != report.incomingURIs.end(); ++it) {
278  Bottle& lst = connections->addList();
279  lst.addInt32(opaque_id); // connectionId
280  lst.addString(it->second);
281  lst.addString("i");
282  lst.addString("TCPROS");
283  NestedContact nc(it->first);
284  lst.addString(toRosName(nc.getNestedName()));
285  opaque_id++;
286  }
287 
288  if (connections->size() == 0) {
289  connections->addList(); // add empty list
290  }
291 
292  na.reply = v;
293  na.success();
294  }

454 void yarp::os::Node::Helper::remove(Contactable& contactable)
455 {
456  mutex.lock();
457  NodeItem item = name_cache[&contactable];
458  name_cache.erase(&contactable);
459  std::string nestedName = item.nc.getNestedName();
460  for (auto it = by_part_name.begin(); it != by_part_name.end(); ++it) {
461  if (it->first == nestedName && it->second.contactable->where().toString() == contactable.where().toString()) {
462  by_part_name.erase(it);
463  break;
464  }
465  }
466  std::string category = item.nc.getCategory();
467  for (auto it = by_category.begin(); it != by_category.end(); ++it) {
468  if (it->first == category && it->second.contactable->where().toString() == contactable.where().toString()) {
469  by_category.erase(it);
470  break;
471  }
472  }
473  mutex.unlock();

490 bool yarp::os::Node::Helper::read(ConnectionReader& reader)
491 {
492  if (!reader.isValid()) {
493  return false;
494  }
495  NodeArgs na;
496  na.request.read(reader);
497  //printf("NODE API for %s received %s\n",
498  //name.c_str(),
499  //na.request.toString().c_str());
500  std::string key = na.request.get(0).asString();
501  na.args = na.request.tail().tail();
502  if (key == "getBusStats") {
503  getBusStats(na);
504  } else if (key == "getBusInfo") {
505  getBusInfo(na);
506  } else if (key == "getMasterUri") {
507  getMasterUri(na);
508  } else if (key == "shutdown") {
509  shutdown(na);
510  } else if (key == "getPid") {
511  getPid(na);
512  } else if (key == "getSubscriptions") {
513  getSubscriptions(na);
514  } else if (key == "getPublications") {
515  getPublications(na);
516  } else if (key == "paramUpdate") {
517  paramUpdate(na);
518  } else if (key == "publisherUpdate") {
519  publisherUpdate(na);
520  } else if (key == "requestTopic") {
521  requestTopic(na);
522  } else {
523  na.error("I have no idea what you are talking about");
524  }
525  if (na.should_drop) {
526  reader.requestDrop(); // ROS likes to close down.
527  }
528  if (reader.getWriter() != nullptr) {
529  Bottle full;
530  full.addInt32(na.code);
531  full.addString(na.msg);
532  full.add(na.reply);
533  //printf("NODE %s <<< %s\n",
534  //name.c_str(),
535  //full.toString().c_str());
536  full.write(*reader.getWriter());
537  }
538  return true;
539 }
