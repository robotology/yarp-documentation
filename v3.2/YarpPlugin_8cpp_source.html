30 bool YarpPluginSettings::subopen(SharedLibraryFactory& factory,
31  const std::string& dll_name,
32  const std::string& fn_name)
33 {
34  YARP_SPRINTF2(impl::Logger::get(), debug, "Trying plugin [dll: %s] [fn: %s]", dll_name.c_str(), fn_name.c_str());
35  bool ok = factory.open(dll_name.c_str(), fn_name.c_str());
36  if (verbose) {
37  fprintf(stderr, "Trying to find library '%s' containing function '%s' -- %s\n", dll_name.c_str(), fn_name.c_str(), ok ? "found" : "fail");
38  }
39  if (ok) {
40  YARP_SPRINTF2(impl::Logger::get(), debug, "Found plugin [dll: %s] [fn: %s]", dll_name.c_str(), fn_name.c_str());
41  this->dll_name = dll_name;
42  this->fn_name = fn_name;
43  } else if (verbose || (factory.getStatus() != SharedLibraryFactory::STATUS_LIBRARY_NOT_FOUND)) {
44  fprintf(stderr, "error while opening %s:\n %s\n", dll_name.c_str(), factory.getError().c_str());
45  }
46  return ok;
47 }

49 bool YarpPluginSettings::open(SharedLibraryFactory& factory)
50 {
51  YARP_SPRINTF3(impl::Logger::get(), debug, "Plugin [name: %s] [dll: %s] [fn: %s]", name.c_str(), dll_name.c_str(), fn_name.c_str());
52  if (selector != nullptr && !name.empty()) {
53  // we may have a YARP-specific search path available,
54  // and a proper name for the DLL
55  Bottle paths = selector->getSearchPath();
56  for (size_t i = 0; i < paths.size(); i++) {
57  Searchable& options = paths.get(i);
58  std::string path = options.find("path").asString();
59  std::string ext = options.find("extension").asString();
60  std::string basename = (dll_name.find('.') != std::string::npos) ? name : dll_name;
61  std::string fn = (fn_name.empty()) ? name : fn_name;
62 
63  std::string fullpath;
64 
65 #if defined(_MSC_VER) && !defined(NDEBUG)
66  // MSVC DEBUG build: try debug name before basic name
67  fullpath = std::string(path).append("/").append(basename).append("d").append(ext);
68  if (subopen(factory, fullpath, fn))
69  return true;
70 #endif // defined(_MSC_VER) && !defined(NDEBUG)
71 
72  // Basic name
73  fullpath = std::string(path).append("/").append(basename).append(ext);
74  if (subopen(factory, fullpath, fn)) {
75  return true;
76  }
77 
78 #if defined(_MSC_VER) && defined(NDEBUG)
79  // MSVC RELEASE build: try debug name after basic name
80  fullpath = std::string(path).append("/").append(basename).append("d").append(ext);
81  if (subopen(factory, fullpath, fn))
82  return true;
83 #endif // defined(_MSC_VER) && defined(NDEBUG)
84 
85 
86 #ifdef CMAKE_INTDIR
87  // On multi-config system, try to find the plugin in the
88  // current config subdirectory
89 
90 # if defined(_MSC_VER) && !defined(NDEBUG)
91  // MSVC DEBUG build: try debug name before basic name
92  fullpath = std::string(path).append("/" CMAKE_INTDIR "/").append(dll_name).append("d").append(ext);
93  if (subopen(factory, fullpath, fn))
94  return true;
95 # endif // defined(_MSC_VER) && !defined(NDEBUG)
96 
97  // Basic name
98  fullpath = std::string(path).append("/" CMAKE_INTDIR "/").append(dll_name).append(ext);
99  if (subopen(factory, fullpath, fn))
100  return true;
101 
102 # if defined(_MSC_VER) && defined(NDEBUG)
103  // MSVC RELEASE build: try debug name after basic name
104  fullpath = std::string(path).append("/" CMAKE_INTDIR "/").append(dll_name).append("d").append(ext);
105  if (subopen(factory, fullpath, fn))
106  return true;
107 # endif // defined(_MSC_VER) && defined(NDEBUG)
108 
109 #endif // CMAKE_INTDIR
110  }
111  }
112  if (!dll_name.empty() || !fn_name.empty()) {
113  return open(factory, dll_name, fn_name);
114  }
115  return factory.open((std::string("yarp_") + name).c_str(),
116  (fn_name.empty()) ? name.c_str() : fn_name.c_str());
117 }

119 void YarpPluginSettings::reportStatus(SharedLibraryFactory& factory) const
120 {
121  int problem = factory.getStatus();
122  if (problem == 0) {
123  return;
124  }
125  switch (problem) {
126  case SharedLibraryFactory::STATUS_LIBRARY_NOT_LOADED:
127  if (verbose) {
128  fprintf(stderr, "Cannot load plugin from shared library (%s)\n", dll_name.c_str());
129  fprintf(stderr, "(%s)\n", factory.getError().c_str());
130  }
131  break;
132  case SharedLibraryFactory::STATUS_LIBRARY_NOT_FOUND:
133  fprintf(stderr, "Cannot load plugin from shared library (%s)\n", dll_name.c_str());
134  fprintf(stderr, "(%s)\n", factory.getError().c_str());
135  break;
136  case SharedLibraryFactory::STATUS_FACTORY_NOT_FOUND:
137  fprintf(stderr, "cannot find YARP hook in shared library (%s:%s)\n", dll_name.c_str(), fn_name.c_str());
138  fprintf(stderr, "(%s)\n", factory.getError().c_str());
139  break;
140  case SharedLibraryFactory::STATUS_FACTORY_NOT_FUNCTIONAL:
141  fprintf(stderr, "YARP hook in shared library misbehaved (%s:%s)\n", dll_name.c_str(), fn_name.c_str());
142  fprintf(stderr, "(the library may be too old/new and need to be recompiled to match YARP version)\n");
143  fprintf(stderr, "(%s)\n", factory.getError().c_str());
144  break;
145  default:
146  fprintf(stderr, "Unknown error (%s:%s)\n", dll_name.c_str(), fn_name.c_str());
147  fprintf(stderr, "(%s)\n", factory.getError().c_str());
148  break;
149  }
150 }

158 void YarpPluginSelector::scan()
159 {
160  // This method needs to be accessed by one thread only
161  std::lock_guard<std::mutex> guard(mutex);
162 
163  // If it was scanned in the last 5 seconds, there is no need to scan again
164  bool need_scan = true;
165  if (config.check("last_update_time")) {
166  if (SystemClock::nowSystem() - config.find("last_update_time").asFloat64() < 5) {
167  need_scan = false;
168  }
169  }
170  if (!need_scan) {
171  return;
172  }
173 
174  YARP_SPRINTF0(Logger::get(),
175  debug,
176  "Scanning. I'm scanning. I hope you like scanning too.");
177 
178  // Search plugins directories
179  ResourceFinder& rf = ResourceFinder::getResourceFinderSingleton();
180  if (!rf.isConfigured()) {
181  rf.configure(0, nullptr);
182  }
183  rf.setQuiet(true);
184  Bottle plugin_paths = rf.findPaths("plugins");
185  if (plugin_paths.size() == 0) {
186  plugin_paths = rf.findPaths("share/yarp/plugins");
187  }
188 
189  // Search .ini files in plugins directories
190  config.clear();
191  if (plugin_paths.size() > 0) {
192  for (size_t i = 0; i < plugin_paths.size(); i++) {
193  std::string target = plugin_paths.get(i).asString();
194  YARP_SPRINTF1(Logger::get(),
195  debug,
196  "Loading configuration files related to plugins from %s.",
197  target.c_str());
198  config.fromConfigDir(target, "inifile", false);
199  }
200  } else {
201  YARP_SPRINTF0(Logger::get(),
202  debug,
203  "Plugin directory not found");
204  }
205 
206  // Read the .ini files and populate the lists
207  plugins.clear();
208  search_path.clear();
209  Bottle inilst = config.findGroup("inifile").tail();
210  for (size_t i = 0; i < inilst.size(); i++) {
211  std::string inifile = inilst.get(i).asString();
212  Bottle inigroup = config.findGroup(inifile);
213  Bottle lst = inigroup.findGroup("plugin").tail();
214  for (size_t i = 0; i < lst.size(); i++) {
215  std::string plugin_name = lst.get(i).asString();
216  Bottle group = inigroup.findGroup(plugin_name);
217  group.add(Value::makeValue(std::string("(inifile \"") + inifile + "\")"));
218  if (select(group)) {
219  plugins.addList() = group;
220  }
221  }
222  lst = inigroup.findGroup("search").tail();
223  for (size_t i = 0; i < lst.size(); i++) {
224  std::string search_name = lst.get(i).asString();
225  Bottle group = inigroup.findGroup(search_name);
226  search_path.addList() = group;
227  }
228  }
229 
230  // Update last_update_time
231  config.put("last_update_time", SystemClock::nowSystem());
232 }
