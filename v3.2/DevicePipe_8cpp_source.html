43 bool DevicePipe::open(const char *key, PolyDriver& poly,
44  yarp::os::Searchable& config, const char *comment) {
45 
46  Value *name;
47  if (config.check(key,name,comment)) {
48  if (name->isString()) {
49  // maybe user isn't doing nested configuration
50  yarp::os::Property p;
51  p.setMonitor(config.getMonitor(),
52  name->toString().c_str()); // pass on any monitoring
53  p.fromString(config.toString());
54  p.put("device",name->toString());
55  p.unput("subdevice");
56  p.unput("wrapped");
57  poly.open(p);
58  } else {
59  Bottle subdevice = config.findGroup(key).tail();
60  poly.open(subdevice);
61  }
62  if (!poly.isValid()) {
63  printf("cannot make <%s>\n", name->toString().c_str());
64  return false;
65  }
66  } else {
67  printf("\"--%s <name>\" not set\n", key);
68  return false;
69  }
70  return true;
71 }
