31 void FallbackNameServer::run()
32 {
33  DgramTwoWayStream send;
34  send.join(getAddress(), true);
35  listen.join(getAddress(), false);
36 
37  YARP_DEBUG(Logger::get(), "Fallback server running");
38  while (listen.isOk() && send.isOk() && !closed) {
39  YARP_DEBUG(Logger::get(), "Fallback server waiting");
40  std::string msg;
41  listen.beginPacket();
42  msg = listen.readLine();
43  listen.endPacket();
44  YARP_DEBUG(Logger::get(), "Fallback server got something");
45  if (listen.isOk() && !closed) {
46  YARP_DEBUG(Logger::get(), std::string("Fallback server got ") + msg);
47  if (msg.find("NAME_SERVER ") == 0) {
48  Contact addr;
49  std::string result = owner.apply(msg, addr);
50  send.beginPacket();
51  send.writeLine(result.c_str(), (int)result.length());
52  send.flush();
53  send.endPacket();
54  }
55  }
56  }
