56 /* This routine will be called by the PortAudio engine when audio is needed.
57 ** It may be called at interrupt level on some machines so don't do anything
58 ** that could mess up the system like calling malloc() or free().
59 */
60 static int bufferIOCallback( const void *inputBuffer, void *outputBuffer,
61  unsigned long framesPerBuffer,
62  const PaStreamCallbackTimeInfo* timeInfo,
63  PaStreamCallbackFlags statusFlags,
64  void *userData )
65 {
66  auto* dataBuffers = static_cast<circularDataBuffers*>(userData);
67  CircularAudioBuffer_16t *playdata = dataBuffers->playData;
68  CircularAudioBuffer_16t *recdata = dataBuffers->recData;
69  int num_rec_channels = dataBuffers->numRecChannels;
70  int num_play_channels = dataBuffers->numPlayChannels;
71  int finished = paComplete;
72 
73  if (dataBuffers->canRec)
74  {
75  const auto* rptr = (const SAMPLE*)inputBuffer;
76  unsigned int framesToCalc;
77  unsigned int i;
78  size_t framesLeft = (recdata->getMaxSize().getSamples()* recdata->getMaxSize().getChannels()) -
79  (recdata->size().getSamples() * recdata->size().getChannels());
80 
81  (void) outputBuffer; // just to prevent unused variable warnings
82  (void) timeInfo;
83  (void) statusFlags;
84  (void) userData;
85 
86  if( framesLeft/ num_rec_channels < framesPerBuffer )
87  {
88  framesToCalc = framesLeft/ num_rec_channels;
89 #ifdef STOP_REC_ON_EMPTY_BUFFER
90  //if we return paComplete, then the callback is not called anymore.
91  //method Pa_IsStreamActive() will return 1.
92  //user needs to call Pa_StopStream() before starting a new recording session
93  finished = paComplete;
94 #else
95  finished = paContinue;
96 #endif
97  }
98  else
99  {
100  framesToCalc = framesPerBuffer;
101  //if we return paContinue, then the callback will be invoked again later
102  //method Pa_IsStreamActive() will return 0
103  finished = paContinue;
104  }
105 
106  if( inputBuffer == nullptr )
107  {
108  for( i=0; i<framesToCalc; i++ )
109  {
110  recdata->write(0); // left
111  if(num_rec_channels == 2 ) recdata->write(0); // right
112  }
113  }
114  else
115  {
116 #if 0
117  yDebug() << "Writing" << framesToCalc*2*2 << "bytes in the circular buffer";
118 #endif
119  for( i=0; i<framesToCalc; i++ )
120  {
121  recdata->write(*rptr++); // left
122  if(num_rec_channels == 2 ) recdata->write(*rptr++); // right
123  }
124  }
125  //note: you can record or play but not simultaneously (for now)
126  return finished;
127  }
128 
129  if (dataBuffers->canPlay)
130  {
131  auto* wptr = (SAMPLE*)outputBuffer;
132  unsigned int i;
133 
134  size_t framesLeft = playdata->size().getSamples()* playdata->size().getChannels();
135 
136  (void) inputBuffer; // just to prevent unused variable warnings
137  (void) timeInfo;
138  (void) statusFlags;
139  (void) userData;
140 
141  if( framesLeft/ num_play_channels < framesPerBuffer )
142  {
143  // final buffer
144  for( i=0; i<framesLeft/ num_play_channels; i++ )
145  {
146  *wptr++ = playdata->read(); // left
147  if( num_play_channels == 2 ) *wptr++ = playdata->read(); // right
148  for (int chs=2; chs<num_play_channels; chs++) playdata->read(); //remove all additional channels > 2
149  }
150  for( ; i<framesPerBuffer; i++ )
151  {
152  *wptr++ = 0; // left
153  if(num_play_channels == 2 ) *wptr++ = 0; // right
154  }
155 #ifdef STOP_PLAY_ON_EMPTY_BUFFER
156  //if we return paComplete, then the callback is not called anymore.
157  //method Pa_IsStreamActive() will return 1.
158  //user needs to call Pa_StopStream() before starting a new recording session
159  finished = paComplete;
160 #else
161  finished = paContinue;
162 #endif
163  }
164  else
165  {
166 #if 1
167  yDebug() << "Reading" << framesPerBuffer*2 << "bytes from the circular buffer";
168 #endif
169  for( i=0; i<framesPerBuffer; i++ )
170  {
171  *wptr++ = playdata->read(); // left
172  if( num_play_channels == 2 ) *wptr++ = playdata->read(); // right
173  for (int chs=2; chs<num_play_channels; chs++) playdata->read(); //remove all additional channels > 2
174  }
175  //if we return paContinue, then the callback will be invoked again later
176  //method Pa_IsStreamActive() will return 0
177  finished = paContinue;
178  }
179  //note: you can record or play but not simultaneously (for now)
180  return finished;
181  }
182 
183  printf("No read/write operations requested, aborting\n");
184  return paAbort;
185 }

289  yInfo("Device number %d", inputParameters.device);

292  if ((Pa_GetDeviceInfo( inputParameters.device ))!=nullptr) {

306  wantWrite?(&outputParameters):nullptr,

308  DEFAULT_FRAMES_PER_BUFFER,
309  paClipOff,
310  bufferIOCallback,
311  &dataBuffers );
312 
313  if( err != paNoError )
314  {
315  fprintf( stderr, "An error occurred while using the portaudio stream\n" );
316  fprintf( stderr, "Error number: %d\n", err );
317  fprintf( stderr, "Error message: %s\n", Pa_GetErrorText( err ) );
318  }
319 
320  //start the thread
321  pThread.stream = stream;
322  pThread.start();
323 
324  return (err==paNoError);

400  }

402  size_t buff_size = 0;
403  int buff_size_wdt = 0;

406  buff_size = dataBuffers.recData->size().getSamples();
407 
408  if (buff_size_wdt > 100)
409  {

411  {
412  yError() << "PortAudioDeviceDriver::getSound() Buffer size is still zero after 100 iterations, returning";
413  return false;
414  }
415  else
416  {
417  yDebug() << "PortAudioDeviceDriver::getSound() Buffer size is " << buff_size << "/" << this->numSamples <<" after 100 iterations";
418  if (m_getSoundIsNotBlocking)
419  {
420  yError() << "PortAudioDeviceDriver::getSound() is in not-blocking mode, returning";
421  return false;
422  }
423  }
424  }
425  buff_size_wdt++;
426  yarp::os::SystemClock::delaySystem(SLEEP_TIME);
427  }
428  while (buff_size < this->numSamples);
429 
430  buff_size_wdt = 0;
431 
432  if (sound.getChannels()!=this->m_numRecordChannels && sound.getSamples() != this->numSamples)
433  {
434  sound.resize(this->numSamples,this->m_numRecordChannels);
435  }
436  sound.setFrequency(this->m_frequency);
437 
438  for (size_t i=0; i<this->numSamples; i++)
439  for (size_t j=0; j<this->m_numRecordChannels; j++)
440  {
441  SAMPLE s = dataBuffers.recData->read();
442  sound.set(s,i,j);
443  }
444  return true;

565  chans << "channels, " << freq << " Hz";
566  this->close();
567  m_driverConfig.playChannels = (int)(chans);
568  m_driverConfig.rate = (int)(freq);
569  bool ok = open(m_driverConfig);
570  if (ok == false)

573  return false;
574  }
575  }
576 
577  if (renderMode == RENDER_IMMEDIATE)
578  return immediateSound(sound);
579  else if (renderMode == RENDER_APPEND)
