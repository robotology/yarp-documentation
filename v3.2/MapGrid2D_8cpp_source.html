44 bool MapGrid2D::isIdenticalTo(const MapGrid2D& other) const
45 {
46  if (m_map_name != other.m_map_name) return false;
47  if (m_origin.x != other.m_origin.x) return false;
48  if (m_origin.y != other.m_origin.y) return false;
49  if (m_origin.theta != other.m_origin.theta) return false;
50  if (m_resolution != other.m_resolution) return false;
51  if (m_width != other.width()) return false;
52  if (m_height != other.height()) return false;
53  for (size_t y = 0; y < m_height; y++) for (size_t x = 0; x < m_width; x++)
54  if (m_map_occupancy.safePixel(x, y) != other.m_map_occupancy.safePixel(x, y)) return false;
55  for (size_t y = 0; y < m_height; y++) for (size_t x = 0; x < m_width; x++)
56  if (m_map_flags.safePixel(x, y) != other.m_map_flags.safePixel(x, y)) return false;
57  return true;
58 }

174 bool MapGrid2D::enlargeObstacles(double size)
175 {
176  if (size <= 0)
177  {
178  for (size_t y = 0; y < m_height; y++)
179  {
180  for (size_t x = 0; x < m_width; x++)
181  {
182  if (this->m_map_flags.safePixel(x, y) == MapGrid2D::map_flags::MAP_CELL_ENLARGED_OBSTACLE)
183  {
184  this->m_map_flags.safePixel(x, y) = MapGrid2D::map_flags::MAP_CELL_FREE;
185  }
186  }
187  }
188  return true;
189  }
190  auto repeat_num = (size_t)(std::ceil(size/m_resolution));
191  for (size_t repeat = 0; repeat < repeat_num; repeat++)
192  {
193  //contains the cells to be enlarged;
194  std::vector<XYCell> list_of_cells;
195  for (size_t y = 0; y < m_height; y++)
196  {
197  for (size_t x = 0; x < m_width; x++)
198  {
199  //this check could be optimized...
200  if (this->m_map_flags.safePixel(x, y) == MAP_CELL_KEEP_OUT ||
201  this->m_map_flags.safePixel(x, y) == MAP_CELL_ENLARGED_OBSTACLE ||
202  this->m_map_flags.safePixel(x, y) == MAP_CELL_WALL ||
203  this->m_map_flags.safePixel(x, y) == MAP_CELL_UNKNOWN ||
204  this->m_map_flags.safePixel(x, y) == MAP_CELL_TEMPORARY_OBSTACLE)
205  {
206  list_of_cells.emplace_back(x, y);
207  }
208  }
209  }
210 
211  //process each cell of the list and enlarges it
212  for (auto& list_of_cell : list_of_cells)
213  {
214  enlargeCell(list_of_cell);
215  }
216  }
217  return true;
218 }

220 void MapGrid2D::enlargeCell(XYCell cell)
221 {
222  int i = cell.x;
223  int j = cell.y;
224  int il = cell.x - 1>0 ? cell.x - 1 : 0;
225  int ir = cell.x + 1<(int)(m_width) - 1 ? cell.x + 1 : (int)(m_width)-1;
226  int ju = cell.y - 1>0 ? cell.y - 1 : 0;
227  int jd = cell.y + 1<(int)(m_height) - 1 ? cell.y + 1 : (int)(m_height)-1;
228 
229  if (m_map_flags.pixel(il, j) == MAP_CELL_FREE) m_map_flags.pixel(il, j) = MAP_CELL_ENLARGED_OBSTACLE;
230  if (m_map_flags.pixel(ir, j) == MAP_CELL_FREE) m_map_flags.pixel(ir, j) = MAP_CELL_ENLARGED_OBSTACLE;
231  if (m_map_flags.pixel(i, ju) == MAP_CELL_FREE) m_map_flags.pixel(i, ju) = MAP_CELL_ENLARGED_OBSTACLE;
232  if (m_map_flags.pixel(i, jd) == MAP_CELL_FREE) m_map_flags.pixel(i, jd) = MAP_CELL_ENLARGED_OBSTACLE;
233  if (m_map_flags.pixel(il, ju) == MAP_CELL_FREE) m_map_flags.pixel(il, ju) = MAP_CELL_ENLARGED_OBSTACLE;
234  if (m_map_flags.pixel(il, jd) == MAP_CELL_FREE) m_map_flags.pixel(il, jd) = MAP_CELL_ENLARGED_OBSTACLE;
235  if (m_map_flags.pixel(ir, ju) == MAP_CELL_FREE) m_map_flags.pixel(ir, ju) = MAP_CELL_ENLARGED_OBSTACLE;
236  if (m_map_flags.pixel(ir, jd) == MAP_CELL_FREE) m_map_flags.pixel(ir, jd) = MAP_CELL_ENLARGED_OBSTACLE;
237 }

487 bool MapGrid2D::loadFromFile(std::string map_file_with_path)
488 {
489  Property mapfile;
490  string mapfile_path = extractPathFromFile(map_file_with_path);
491  if (mapfile.fromConfigFile(map_file_with_path) == false)
492  {
493  yError() << "Unable to open .map description file:" << map_file_with_path;
494  return false;
495  }
496 
497  if (mapfile.check("MapName") ==false)
498  {
499  yError() << "Unable to find 'MapName' parameter inside:" << map_file_with_path;
500  return false;
501  }
502  m_map_name = mapfile.find("MapName").asString();
503 
504  bool YarpMapDataFound = false;
505  string ppm_flg_filename;
506  if (mapfile.check("YarpMapData") == false)
507  {
508  yWarning() << "Unable to find 'YarpMapData' parameter inside:" << map_file_with_path;
509  YarpMapDataFound = false;
510  }
511  else
512  {
513  ppm_flg_filename = mapfile.find("YarpMapData").asString();
514  YarpMapDataFound = true;
515  }
516 
517  bool RosMapDataFound = false;
518  string yaml_filename;
519  if (mapfile.check("RosMapData") == false)
520  {
521  yWarning() << "Unable to find 'RosMapData' parameter inside:" << map_file_with_path;
522  RosMapDataFound = false;
523  }
524  else
525  {
526  yaml_filename = mapfile.find("RosMapData").asString();
527  RosMapDataFound = true;
528  }
529 
530  m_width = -1;
531  m_height = -1;
532  string ppm_flg_filename_with_path = mapfile_path + ppm_flg_filename;
533  string yaml_filename_with_path = mapfile_path + yaml_filename;
534  if (YarpMapDataFound && RosMapDataFound)
535  {
536  return this->loadMapYarpAndRos(ppm_flg_filename_with_path, yaml_filename_with_path);
537  }
538  else if (!YarpMapDataFound && RosMapDataFound)
539  {
540  return this->loadMapROSOnly(yaml_filename_with_path);
541  }
542  else if (YarpMapDataFound && !RosMapDataFound)
543  {
544  return this->loadMapYarpOnly(ppm_flg_filename_with_path);
545  }
546  else
547  {
548  yError() << "Critical error: unable to find neither 'RosMapData' nor 'YarpMapData' inside:" << map_file_with_path;
549  return false;
550  }
551  return true;
552 }

554 MapGrid2D::CellData MapGrid2D::PixelToCellData(const yarp::sig::PixelRgb& pixin) const
555 {
556  if (pixin.r == 0 && pixin.g == 0 && pixin.b == 0) return MAP_CELL_WALL;

643  if (right > (int)this->width()) return false;

814 MapGrid2D::XYWorld MapGrid2D::cell2World(MapGrid2D::XYCell cell) const
815 {
816  //convert a cell (from the upper-left corner) to the map reference frame (located in m_origin, measured in meters)
817  //beware: the location of m_origin is referred to the lower-left corner (ROS convention)
818  MapGrid2D::XYWorld v;
819  v.x = double(cell.x)*this->m_resolution;
820  v.y = double(cell.y)*this->m_resolution;
821  v.x = + v.x + m_origin.x + 0 * this->m_resolution;
822  v.y = - v.y + m_origin.y + (m_height-1)*this->m_resolution;
823  return v;
824 }
