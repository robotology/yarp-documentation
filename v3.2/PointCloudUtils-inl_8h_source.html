13 yarp::sig::PointCloud<T1> yarp::sig::utils::depthRgbToPC(const yarp::sig::ImageOf<yarp::sig::PixelFloat>& depth,

15  const yarp::sig::IntrinsicParams& intrinsic)
16 {
17  yAssert(depth.width() != 0);
18  yAssert(depth.height() != 0);
19  yAssert(depth.width() == color.width());
20  yAssert(depth.height() == color.height());
21  size_t w = depth.width();
22  size_t h = depth.height();
23  yarp::sig::PointCloud<T1> pointCloud;
24  pointCloud.resize(w, h);
25 
26  for (size_t u = 0; u < w; ++u) {
27  for (size_t v = 0; v < h; ++v) {
28  // Depth
29  // De-projection equation (pinhole model):
30  // x = (u - ppx)/ fx * z
31  // y = (v - ppy)/ fy * z
32  // z = z
33  pointCloud(u,v).x = (u - intrinsic.principalPointX)/intrinsic.focalLengthX*depth.pixel(u,v);
34  pointCloud(u,v).y = (v - intrinsic.principalPointY)/intrinsic.focalLengthY*depth.pixel(u,v);
35  pointCloud(u,v).z = depth.pixel(u,v);
36 
37  if (std::is_same<T1, DataXYZRGBA>::value) {
38  if (std::is_same<T2, PixelRgb>::value ||
39  std::is_same<T2, PixelBgr>::value ||
40  std::is_same<T2, PixelRgba>::value ||
41  std::is_same<T2, PixelBgra>::value) {
42  // Color
43  pointCloud(u,v).r = color.pixel(u,v).r;
44  pointCloud(u,v).g = color.pixel(u,v).g;
45  pointCloud(u,v).b = color.pixel(u,v).b;
46  if (std::is_same<T2, PixelRgba>::value ||
47  std::is_same<T2, PixelBgra>::value) {
48  pointCloud(u,v).a = color.pixel(u,v).a;
49  }
50  }
51  }
52 
53 
54  }
55  }
56  return pointCloud;
57 }
