174 void Logger::show(std::uint32_t level, const std::string& txt)
175 {
176  std::uint32_t inLevel = level;
177  //fprintf(stderr, "level %d txt %s\n", level, txt.c_str());
178  if (verbose > 0) {
179  level = 10000;
180  }
181  if (verbose < 0) {
182  level = 0;
183  }
184  if (stream == nullptr) {
185  stream = stderr;
186  if (NetworkBase::getEnvironment("YARP_LOGGER_STREAM") == "stdout") {
187  stream = stdout;
188  }
189  }
190  if (parent == nullptr) {
191  if (level >= low) {
192  if (inLevel <= YARP_LM_DEBUG) {
193  fprintf(stream, "%s(%04lx): %s\n", prefix.c_str(), ThreadImpl::getKeyOfCaller(), txt.c_str());
194  } else {
195  fprintf(stream, "%s: %s\n", prefix.c_str(), txt.c_str());
196  }
197  fflush(stream);
198  }
199  } else {
200  std::string more(prefix);
201  more += ": ";
202  more += txt;
203  parent->show(inLevel, more);
204  }
205 }
