41 {
42  int maxsize;
43  int start;

47 public:
48  inline bool isFull()
49  {
50  return (end + 1) % maxsize == start;
51  }
52 
53  inline const byte* getRawData()
54  {
55  return elems;
56  }
57 
58  inline bool isEmpty()
59  {
60  return end == start;
61  }
62 
63  inline bool write_elem(byte elem)
64  {
65  elems[end] = elem;
66  end = (end + 1) % maxsize;
67  if (end == start)
68  {
69  yError("rpLidar buffer overrun!");
70  start = (start + 1) % maxsize; // full, overwrite
71  return false;
72  }
73  return true;
74  }
75 
76  inline bool write_elems(byte* elems, int size)
77  {
78  for (int i = 0; i < size; i++)
79  {
80  if (write_elem(elems[i]) == false) return false;
81  }
82  return true;
83  }
84 
85  inline int size()
86  {
87  int i;
88  if (end>start)
89  i = end - start;
90  else if (end == start)
91  i = 0;
92  else
93  i = maxsize - start + end;
94  return i;
95  }
96 
97  inline bool read_elem(byte* elem)
98  {
99  if (end == start)
100  {
101  yError("rpLidar buffer underrun!");
102  return false;
103  }
104  *elem = elems[start];
105  start = (start + 1) % maxsize;
106  return true;
107  }
108 
109  inline void throw_away_elem()
110  {
111  start = (start + 1) % maxsize;
112  }
113 
114  inline void throw_away_elems(int size)
115  {
116  start = (start + size) % maxsize;
117  }
118 
119  inline byte select_elem(int offset)
120  {
121  return elems[(start+offset) % maxsize];
122  }
123 
124  inline void select_elems(byte* elems, int size)
125  {
126  for (int i = 0; i < size; i++)
127  {
128  elems[i] = select_elem(i);
129  }
130  }
131 
132  inline bool read_elems(byte* elems, int size)
133  {
134  for (int i = 0; i < size; i++)
135  {
136  if (read_elem(&elems[i]) == false) return false;
137  }
138  return true;
139  }
140 
141  inline unsigned int getMaxSize()
142  {

144  }
