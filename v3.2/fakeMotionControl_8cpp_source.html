32 void FakeMotionControl::run()
33 {
34  for (int i=0;i <_njoints ;i++)
35  {
36  if (_controlModes[i] == VOCAB_CM_VELOCITY)
37  {
38  //increment joint position
39  if (this->_command_speeds[i]!=0)
40  {
41  double elapsed = yarp::os::Time::now()-prev_time;
42  double increment = _command_speeds[i]*elapsed;
43  pos[i]+=increment;
44  }
45 
46  //velocity watchdog
47  if (yarp::os::Time::now()-last_velocity_command[i]>=VELOCITY_WATCHDOG)
48  {
49  this->_command_speeds[i]=0.0;
50  }
51  }
52  else if (_controlModes[i] == VOCAB_CM_PWM)
53  {
54  //increment joint position
55  if (this->refpwm[i]!=0)
56  {
57  double elapsed = yarp::os::Time::now()-prev_time;
58  double increment = refpwm[i]*elapsed*PWM_CONSTANT;
59  pos[i]+=increment;
60  }
61 
62  //pwm watchdog
63  if (yarp::os::Time::now()-last_pwm_command[i]>=OPENLOOP_WATCHDOG)
64  {
65  this->refpwm[i]=0.0;
66  }
67  }
68  else if (_controlModes[i] == VOCAB_CM_POSITION_DIRECT)
69  {
70  pos[i] = _posDir_references[i];
71  }
72  else if (_controlModes[i] == VOCAB_CM_POSITION)
73  {
74  pos[i] = _posCtrl_references[i];
75  //do something with _ref_speeds[i];
76  }
77  else if (_controlModes[i] == VOCAB_CM_IDLE)
78  {
79  //do nothing
80  }
81  else if (_controlModes[i] == VOCAB_CM_MIXED)
82  {
83  //not yet implemented
84  }
85  else
86  {
87  //unsupported control mode
88  yWarning() << "Unsupported control mode, joint " << i << " " << yarp::os::Vocab::decode(_controlModes[i]);
89  }
90  }
91  prev_time = yarp::os::Time::now();
92 }
