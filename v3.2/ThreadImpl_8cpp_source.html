157 int ThreadImpl::join(double seconds)
158 {
159  closing = true;
160  if (needJoin) {
161  if (seconds > 0) {
162  if (!initWasSuccessful) {
163  // join called before start completed
164  YARP_ERROR(Logger::get(), std::string("Tried to join a thread before starting it"));
165  return -1;
166  }
167  synchro.waitWithTimeout(seconds);
168  if (active) {
169  return -1;
170  }
171  }
172 
173  int result = -1;
174  if (thread.joinable()) {
175  thread.join();
176  result = 0;
177  }
178 
179  needJoin = false;
180  active = false;
181  while (synchro.check()) {
182  }
183  return result;
184  }
185  return 0;
186 }

242 bool ThreadImpl::start()
243 {
244  join();
245  closing = false;
246  initWasSuccessful = false;
247  beforeStart();
248  thread = std::thread(theExecutiveBranch, (void*)this);
249  int result = thread.joinable() ? 0 : 1;
250  if (result == 0) {
251  // we must, at some point in the future, join the thread
252  needJoin = true;
253 
254  // the thread started correctly, wait for the initialization
255  YARP_DEBUG(Logger::get(), std::string("Child thread initializing"));
256  synchroWait();
257  initWasSuccessful = true;
258  if (opened) {
259  ++threadCount;
260  YARP_DEBUG(Logger::get(), "Child thread initialized ok");
261  afterStart(true);
262  return true;
263  }
264  YARP_DEBUG(Logger::get(), "Child thread did not initialize ok");
265  //wait for the thread to really exit
266  ThreadImpl::join(-1);
267  }
268  //the thread did not start, call afterStart() to warn the user
269  char tmp[80];
270  sprintf(tmp, "%d", result);
271  YARP_ERROR(Logger::get(), std::string("A thread failed to start with error code: ") + std::string(tmp));
272  afterStart(false);
273  return false;
274 }

306 int ThreadImpl::setPriority(int priority, int policy)
307 {
308  if (priority == -1) {
309  priority = defaultPriority;
310  policy = defaultPolicy;
311  } else {
312  defaultPriority = priority;
313  defaultPolicy = policy;
314  }
315  if (active && priority != -1) {
316 #if defined(YARP_HAS_ACE)
317  if (std::is_same<std::thread::native_handle_type, ACE_hthread_t>::value) {
318  return ACE_Thread::setprio(thread.native_handle(), priority, policy);
319  }
320  YARP_ERROR(Logger::get(), "Cannot set priority without ACE");
321 #elif defined(__unix__)
322  if (std::is_same<std::thread::native_handle_type, pthread_t>::value) {
323  struct sched_param thread_param;
324  thread_param.sched_priority = priority;
325  int ret = pthread_setschedparam(thread.native_handle(), policy, &thread_param);
326  return (ret != 0) ? -1 : 0;
327  } else {
328  YARP_ERROR(Logger::get(), "Cannot set priority without ACE");
329  }
330 #else
331  YARP_ERROR(Logger::get(), "Cannot set priority without ACE");
332 #endif
333  }
334  return 0;
335 }
