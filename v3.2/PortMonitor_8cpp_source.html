29 bool PortMonitor::configure(yarp::os::ConnectionState& proto)
30 {
31  portName = proto.getRoute().getToName();
32  sourceName = proto.getRoute().getFromName();
33  group = getPeers().add(portName,this);
34  if (!group) return false;
35 
36  Property options;
37  options.fromString(proto.getSenderSpecifier());
38  options.put("source", sourceName);
39  options.put("destination", portName);
40  options.put("sender_side",
41  (proto.getContactable()->getName() == sourceName) ? 1 : 0);
42  options.put("receiver_side",
43  (proto.getContactable()->getName() == portName) ? 1 : 0);
44  options.put("carrier", proto.getRoute().getCarrierName());
45  return configureFromProperty(options);
46 }

48 bool PortMonitor::configureFromProperty(yarp::os::Property& options) {
49  if(binder) delete binder;
50  binder = nullptr;
51 
52  std::string script = options.check("type", Value("lua")).asString();
53  std::string filename = options.check("file", Value("modifier")).asString();
54  std::string constraint = options.check("constraint", Value("")).asString();
55  // context is used to find the script files
56  std::string context = options.check("context", Value("")).asString();
57 
58  // check which monitor should be used
59  if((binder = MonitorBinding::create(script.c_str())) == nullptr)
60  {
61  yError(R"(Currently only 'lua' script and 'dll' object is supported by portmonitor)");
62  return false;
63  }
64 
65  // set the acceptance constraint
66  binder->setAcceptConstraint(constraint.c_str());
67 
68  std::string strFile = filename;
69 
70  if(script != "dll")
71  {
72  yarp::os::ResourceFinder rf;
73  rf.setDefaultContext(context.c_str());
74  rf.configure(0, nullptr);
75  strFile = rf.findFile(filename);
76  if(strFile == "")
77  strFile = rf.findFile(filename+".lua");
78  }
79 
80  // provide some useful information for the monitor object
81  // which can be accessed in the create() callback.
82  Property info;
83  info.clear();
84  info.put("filename", strFile);
85  info.put("type", script);
86  info.put("source", options.find("source").asString());
87  info.put("destination", options.find("destination").asString());
88  info.put("sender_side", options.find("sender_side").asInt32());
89  info.put("receiver_side",options.find("receiver_side").asInt32());
90  info.put("carrier", options.find("carrier").asString());
91 
92  PortMonitor::lock();
93  bReady = binder->load(info);
94  PortMonitor::unlock();
95  return bReady;
96 }

115 yarp::os::ConnectionReader& PortMonitor::modifyIncomingData(yarp::os::ConnectionReader& reader)
116 {
117  if(!bReady) return reader;
118 
119  // When we are here,
120  // the incoming data should be accessed using localReader.
121  // The reader passed to this function is infact empty.
122  // first check if we need to call the update callback
123  if(!binder->hasUpdate()) {
124  localReader->setParentConnectionReader(&reader);
125  return *localReader;
126  }
127 
128  PortMonitor::lock();
129  yarp::os::Things thing;
130  thing.setConnectionReader(*localReader);
131  yarp::os::Things& result = binder->updateData(thing);
132  PortMonitor::unlock();
133  con.reset();
134  if(result.write(con.getWriter())) {
135  con.getReader().setParentConnectionReader(&reader);
136  return con.getReader();
137  }
138  return *localReader;

141 bool PortMonitor::acceptIncomingData(yarp::os::ConnectionReader& reader)
142 {
143  if(!bReady) return false;
144 
145  bool result = false;
146  localReader = &reader;
147  // If no accept callback avoid calling the binder
148  if(binder->hasAccept())
149  {
150  PortMonitor::lock();
151  Things thing;
152  // set the reference connection reader
153  thing.setConnectionReader(reader);
154  result = binder->acceptData(thing);
155  PortMonitor::unlock();
156  if(!result)
157  return false;
158 
159  // When data is read here using the reader passed to this functions,
160  // then it won't be available for modifyIncomingData(). Thus, we write
161  // it to a dummy connection and pass it to the modifyOutgoingData() using
162  // localReader.
163  // localReader points to a connection reader which contains
164  // either the original or modified data.
165  if(thing.hasBeenRead()) {
166  con.reset();
167  if(thing.write(con.getWriter()))
168  localReader = &con.getReader();
169  }
170  }
171 
172  if(group!=nullptr) {
173  getPeers().lock();
174  result = group->acceptIncomingData(this);
175  getPeers().unlock();
176  }
177  return result;
178 }
