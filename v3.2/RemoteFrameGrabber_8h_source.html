400  bool getImage(yarp::sig::ImageOf<yarp::sig::PixelRgb>& image) override {
401  mutex.lock();
402  if(no_stream == true)
403  {
404  image.zero();
405  mutex.unlock();
406  return false;
407  }
408 
409  if (reader.read(true)!=NULL) {
410  image = *(reader.lastRead());
411  lastHeight = image.height();
412  lastWidth = image.width();
413  mutex.unlock();
414  return true;
415  }
416  mutex.unlock();
417  return false;

420  bool getImageCrop(cropType_id_t cropType, yarp::sig::VectorOf<std::pair<int, int> > vertices, yarp::sig::ImageOf<yarp::sig::PixelRgb>& image) override
421  {
422  yarp::os::Bottle cmd, response;
423  cmd.addVocab(VOCAB_FRAMEGRABBER_IMAGE);
424  cmd.addVocab(VOCAB_GET);
425  cmd.addVocab(VOCAB_CROP);
426  cmd.addInt32(cropType);
427  yarp::os::Bottle & list = cmd.addList();
428  for(size_t i=0; i<vertices.size(); i++)
429  {
430  list.addInt32(vertices[i].first);
431  list.addInt32(vertices[i].second);
432  }
433  port.write(cmd,response);
434 
435  // Parse the response
436  image.zero();
437  if( (response.get(0).asVocab() != VOCAB_CROP) || (response.size() != 5) || (!response.get(4).isBlob()))
438  {
439  yError() << "getImageCrop: malformed response message. Size is " << response.size();
440  return false;
441  }
442 
443  image.resize(response.get(2).asInt32(), response.get(3).asInt32());
444  unsigned char *pixelOut = image.getRawImage();
445 
446  if(response.get(4).asBlob())
447  memcpy(pixelOut, response.get(4).asBlob(), (size_t) image.getRawImageSize());
448 
449  return true;
450  }
