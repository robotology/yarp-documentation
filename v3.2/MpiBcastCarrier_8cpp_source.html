62 void MpiBcastCarrier::prepareDisconnect() {
63  comm->sema.wait();
64  #ifdef MPI_DEBUG
65  printf("[MpiBcastCarrier @ %s] Disconnect : %s\n", name.c_str(), other.c_str());
66  #endif
67  int cmd = CMD_DISCONNECT;
68  MPI_Bcast(&cmd, 1, MPI_INT, 0,comm->comm);
69  int length = other.length() + name.length() + 3;
70  char* remote_c = new char[length];
71  strcpy(remote_c, (other+"<-"+name).c_str());
72  MPI_Bcast(&length, 1, MPI_INT, 0,comm->comm);
73  MPI_Bcast(remote_c, length, MPI_CHAR, 0,comm->comm);
74  delete [] remote_c;
75  comm->disconnect(false);
76  comm->sema.post();
