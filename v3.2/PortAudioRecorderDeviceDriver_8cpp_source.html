57 /* This routine will be called by the PortAudio engine when audio is needed.
58 ** It may be called at interrupt level on some machines so don't do anything
59 ** that could mess up the system like calling malloc() or free().
60 */
61 static int bufferIOCallback( const void *inputBuffer, void *outputBuffer,
62  unsigned long framesPerBuffer,
63  const PaStreamCallbackTimeInfo* timeInfo,
64  PaStreamCallbackFlags statusFlags,
65  void *userData )
66 {
67  CircularAudioBuffer_16t *recdata = (CircularAudioBuffer_16t*)(userData);
68  int num_rec_channels = recdata->getMaxSize().getChannels();
69  int finished = paComplete;
70 
71  if (1)
72  {
73  const auto* rptr = (const SAMPLE*)inputBuffer;
74  unsigned int framesToCalc;
75  unsigned int i;
76  size_t framesLeft = (recdata->getMaxSize().getSamples()* recdata->getMaxSize().getChannels()) -
77  (recdata->size().getSamples() * recdata->size().getChannels());
78 
79  (void) outputBuffer; // just to prevent unused variable warnings
80  (void) timeInfo;
81  (void) statusFlags;
82  (void) userData;
83 
84  if( framesLeft/ num_rec_channels < framesPerBuffer )
85  {
86  framesToCalc = framesLeft/ num_rec_channels;
87 #ifdef STOP_REC_ON_EMPTY_BUFFER
88  //if we return paComplete, then the callback is not called anymore.
89  //method Pa_IsStreamActive() will return 1.
90  //user needs to call Pa_StopStream() before starting a new recording session
91  finished = paComplete;
92 #else
93  finished = paContinue;
94 #endif
95  }
96  else
97  {
98  framesToCalc = framesPerBuffer;
99  //if we return paContinue, then the callback will be invoked again later
100  //method Pa_IsStreamActive() will return 0
101  finished = paContinue;
102  }
103 
104  if( inputBuffer == nullptr )
105  {
106  for( i=0; i<framesToCalc; i++ )
107  {
108  recdata->write(0); // left
109  if(num_rec_channels == 2 ) recdata->write(0); // right
110  }
111  }
112  else
113  {
114 #if 0
115  yDebug() << "Writing" << framesToCalc*2*2 << "bytes in the circular buffer";
116 #endif
117  for( i=0; i<framesToCalc; i++ )
118  {
119  recdata->write(*rptr++); // left
120  if(num_rec_channels == 2 ) recdata->write(*rptr++); // right
121  }
122  }
123  return finished;
124  }
125 
126  printf("No write operations requested, aborting\n");
127  return paAbort;
128 }

160  if (m_config.cfg_recChannels == 0) m_config.cfg_recChannels = DEFAULT_NUM_CHANNELS;

170 
171  m_err = Pa_Initialize();
172  if(m_err != paNoError )

174  yError("portaudio system failed to initialize");

177

179  yInfo("Device number %d", m_inputParameters.device);
180  m_inputParameters.channelCount = m_config.cfg_recChannels;
181  m_inputParameters.sampleFormat = PA_SAMPLE_TYPE;
182  if ((Pa_GetDeviceInfo(m_inputParameters.device ))!=nullptr) {
183  m_inputParameters.suggestedLatency = Pa_GetDeviceInfo(m_inputParameters.device )->defaultLowInputLatency;
184  }
185  m_inputParameters.hostApiSpecificStreamInfo = nullptr;
186 
187  m_err = Pa_OpenStream(
188  &m_stream,
189  &m_inputParameters,
190  nullptr,
191  m_config.cfg_rate,
192  DEFAULT_FRAMES_PER_BUFFER,
193  paClipOff,
194  bufferIOCallback,
195  m_recDataBuffer);
196 
197  if(m_err != paNoError )
198  {
199  yError("An error occurred while using the portaudio stream\n" );
200  yError("Error number: %d\n", m_err );
201  yError("Error message: %s\n", Pa_GetErrorText(m_err ) );
202  }
203 
204  //start the thread
205  bool ret = this->start();
206  YARP_UNUSED(ret);
207 
208  return (m_err==paNoError);
209 }

288 bool PortAudioRecorderDeviceDriver::getSound(yarp::sig::Sound& sound, size_t min_number_of_samples, size_t max_number_of_samples, double max_samples_timeout_s)
289 {
290  //check for something_to_record
291  {
292  #ifdef AUTOMATIC_REC_START
293  if (m_isRecording == false)
294  {
295  this->startRecording();
296  }
297  #else
298  double debug_time = yarp::os::Time::now();
299  while (m_isRecording == false)
300  {
301  if (yarp::os::Time::now() - debug_time > 5.0)
302  {
303  yInfo() << "getSound() is currently waiting. Use ::startRecording() to start the audio stream";
304  debug_time = yarp::os::Time::now();
305  }
306  yarp::os::SystemClock::delaySystem(SLEEP_TIME);
307  }
308  #endif
309  }
310 
311  //prevents simultaneous start/stop/reset etc.
312  //std::lock_guard<std::mutex> lock(m_mutex); //This must be used carefully
313 
314  //check on input parameters
315  if (max_number_of_samples < min_number_of_samples)
316  {
317  yError() << "max_number_of_samples must be greater than min_number_of_samples!";
318  return false;
319  }
320  if (max_number_of_samples > this->m_config.cfg_samples)
321  {
322  yWarning() << "max_number_of_samples bigger than the internal audio buffer! It will be truncated to:" << this->m_config.cfg_samples;
323  max_number_of_samples = this->m_config.cfg_samples;
324  }
325 
326  //wait until the desired number of samples are obtained
327  size_t buff_size = 0;
328  double start_time = yarp::os::Time::now();
329  double debug_time = yarp::os::Time::now();
330  do
331  {
332  buff_size = m_recDataBuffer->size().getSamples();
333  if (buff_size >= max_number_of_samples) { break; }
334  if (buff_size >= min_number_of_samples && yarp::os::Time::now() - start_time > max_samples_timeout_s) { break; }
335  if (m_isRecording == false) { break; }
336 
337  if (yarp::os::Time::now() - debug_time > 1.0)
338  {
339  debug_time = yarp::os::Time::now();
340  yDebug() << "PortAudioRecorderDeviceDriver::getSound() Buffer size is " << buff_size << "/" << max_number_of_samples << " after 1s";
341  }
342 
343  yarp::os::SystemClock::delaySystem(SLEEP_TIME);
344  }
345  while (true);
346 
347  //prepare the sound data struct
348  size_t samples_to_be_copied = buff_size;
349  if (samples_to_be_copied > max_number_of_samples) samples_to_be_copied = max_number_of_samples;
350  if (sound.getChannels()!=this->m_config.cfg_recChannels && sound.getSamples() != samples_to_be_copied)
351  {
352  sound.resize(samples_to_be_copied, this->m_config.cfg_recChannels);
353  }
354  sound.setFrequency(this->m_config.cfg_rate);
355 
356  //fill the sound data struct, reading samples from the circular buffer
357  for (size_t i=0; i< samples_to_be_copied; i++)
358  for (size_t j=0; j<this->m_config.cfg_recChannels; j++)
359  {
360  SAMPLE s = m_recDataBuffer->read();
361  sound.set(s,i,j);
362  }
363  return true;
364 }

376 void PortAudioRecorderDeviceDriver::run()
377 {
378  while(this->isStopping()==false)
379  {
380  if (m_isRecording)
381  {
382  while( ( m_err = Pa_IsStreamActive(m_stream) ) == 1 )
383  {
384  yarp::os::SystemClock::delaySystem(SLEEP_TIME);
385  }
386 
387  if (m_err == 0)
388  {
389  Pa_StopStream(m_stream);
390  yDebug() << "The recording stream has been stopped";
391  m_isRecording = false;
392  }
393  if(m_err < 0 )
394  {
395  handleError();
396  return;
397  }
398  }
399 
400  yarp::os::Time::delay(SLEEP_TIME);
401  }
402  return;
403 }
