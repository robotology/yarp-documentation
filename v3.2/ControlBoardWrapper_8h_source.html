90 public:

94  int *subdev_jointsVectorLen; // number of joints belonging to each subdevice
95  int **jointNumbers;
96  int **modes;

98  yarp::dev::impl::SubDevice **subdevices_p;
99 
100  MultiJointData() :
101  deviceNum(0),
102  maxJointsNumForDevice(0),
103  subdev_jointsVectorLen(nullptr),
104  jointNumbers(nullptr),
105  modes(nullptr),
106  values(nullptr),
107  subdevices_p(nullptr)
108 {}
109 
110  void resize(int _deviceNum, int _maxJointsNumForDevice, yarp::dev::impl::WrappedDevice *_device)
111  {
112  deviceNum = _deviceNum;
113  maxJointsNumForDevice = _maxJointsNumForDevice;
114  subdev_jointsVectorLen = new int [deviceNum];
115  jointNumbers = new int *[deviceNum]; // alloc a vector of pointers
116  jointNumbers[0] = new int[deviceNum * _maxJointsNumForDevice]; // alloc real memory for data
117 
118  modes = new int *[deviceNum]; // alloc a vector of pointers
119  modes[0] = new int[deviceNum * _maxJointsNumForDevice]; // alloc real memory for data
120 
121  values = new double *[deviceNum]; // alloc a vector of pointers
122  values[0] = new double[deviceNum * _maxJointsNumForDevice]; // alloc real memory for data
123 
124  subdevices_p = new yarp::dev::impl::SubDevice *[deviceNum];
125  subdevices_p[0] = _device->getSubdevice(0);
126 
127  for (int i = 1; i < deviceNum; i++)
128  {
129  jointNumbers[i] = jointNumbers[i-1] + _maxJointsNumForDevice; // set pointer to correct location
130  values [i] = values[i-1] + _maxJointsNumForDevice; // set pointer to correct location
131  modes [i] = modes[i-1] + _maxJointsNumForDevice; // set pointer to correct location
132  subdevices_p[i] = _device->getSubdevice(i);
133  }
134  }
135 
136  void destroy()
137  {
138  // release matrix memory
139  delete[] jointNumbers[0];
140  delete[] values[0];
141  delete[] modes[0];
142 
143  // release vector of pointers
144  delete[] jointNumbers;
145  delete[] values;
146  delete[] modes;
147 
148  // delete other vectors
149  delete[] subdev_jointsVectorLen;
150  delete[] subdevices_p;
151  }
152 };
