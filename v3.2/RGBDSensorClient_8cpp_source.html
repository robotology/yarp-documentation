79 {
80  Bottle &rosGroup = config.findGroup("ROS");
81  if(rosGroup.isNull())
82  {
83  if(verbose >= 3)
84  yInfo() << "RGBDSensorClient: ROS configuration parameters are not set, skipping ROS topic initialization.";
85  use_ROS = false;
86  }
87  else
88  {
89  if(verbose >= 2)
90  yWarning() << "RGBDSensorClient: ROS topic support is not yet implemented";
91  use_ROS = false;
92  }
93 
94  if(!use_ROS) // default
95  {
96  // Parse LOCAL port names
97  // TBD: check if user types '...' as port name, how to create RPC port names
98  if (!config.check("localImagePort", "full name of the port for streaming color image"))
99  {
100  yError() << "RGBDSensorClient: missing 'localImagePort' parameter. Check you configuration file; it must be like:";
101  yError() << " localImagePort: Full name of the local port to open, e.g. /myApp/image_camera";
102  return false;
103  }
104  else
105  {
106  local_colorFrame_StreamingPort_name = config.find("localImagePort").asString();
107  }
108 
109  if (!config.check("localDepthPort", "full name of the port for streaming depth image"))
110  {
111  yError() << "RGBDSensorClient: missing 'localDepthPort' parameter. Check you configuration file; it must be like:";
112  yError() << " localDepthPort: Full name of the local port to open, e.g. /myApp/depth_camera";
113  return false;
114  }
115  else
116  {
117  local_depthFrame_StreamingPort_name = config.find("localDepthPort").asString();
118  }
119 
120  // Parse REMOTE port names
121  if (!config.check("remoteImagePort", "full name of the port for streaming color image"))
122  {
123  yError() << "RGBDSensorClient: missing 'remoteImagePort' parameter. Check you configuration file; it must be like:";
124  yError() << " remoteImagePort: Full name of the port to read color images from, e.g. /robotName/image_camera";
125  return false;
126  }
127  else
128  {
129  remote_colorFrame_StreamingPort_name = config.find("remoteImagePort").asString();

137  }
138  else
139  {
140  remote_depthFrame_StreamingPort_name = config.find("remoteDepthPort").asString();
141  }
142 
143  // Single RPC port
144  if (!config.check("localRpcPort", "full name of the port for streaming depth image"))
145  {
146  yError() << "RGBDSensorClient: missing 'localRpcPort' parameter. Check you configuration file; it must be like:";

148  return false;

150  else
151  {
152  local_rpcPort_name = config.find("localRpcPort").asString();
153  }

156  {
157  yError() << "RGBDSensorClient: missing 'remoteRpcPort' parameter. Check you configuration file; it must be like:";
158  yError() << " remoteRpcPort: Full name of the remote RPC port, e.g. /robotName/RGBD/rpc";
159  return false;
160  }
161  else
162  {
163  remote_rpcPort_name = config.find("remoteRpcPort").asString();
164  }
165 
166 
167  /*
168  * When using multiple RPC ports
169  *
170  local_colorFrame_rpcPort_Name = local_colorFrame_StreamingPort_Name + "/rpc:i";
171  remote_colorFrame_rpcPort_Name = remote_colorFrame_StreamingPort_Name + "/rpc:i";
172  local_depthFrame_rpcPort_Name = local_depthFrame_StreamingPort_Name + "/rpc:i";
173  remote_depthFrame_rpcPort_Name = remote_depthFrame_StreamingPort_Name + "/rpc:i";
174 
175  */
176  }
177 
178  yarp::os::Bottle &ROS_parameters = config.findGroup("ROS");
179  if(!ROS_parameters.isNull())

186  {
187  if(verbose >= 5)
188  yInfo() << "RGBDSensorClient: 'ROS' group was NOT found in config file, skipping ROS specific parameters.";
189  }
190 
191  return true;

194 bool RGBDSensorClient::initialize_YARP(yarp::os::Searchable &config)
195 {
196  bool ret;
197 
198  // Opening Streaming ports
199  ret = colorFrame_StreamingPort.open(local_colorFrame_StreamingPort_name);
200  ret &= depthFrame_StreamingPort.open(local_depthFrame_StreamingPort_name);
201 
202  if(!ret)
203  {
204  yError() << sensorId << " cannot open local streaming ports.";
205  colorFrame_StreamingPort.close();
206  depthFrame_StreamingPort.close();
207  }
208 
209  if(! yarp::os::Network::connect(remote_colorFrame_StreamingPort_name, colorFrame_StreamingPort.getName(), "udp") )
210  {
211  yError() << colorFrame_StreamingPort.getName() << " cannot connect to remote port " << remote_colorFrame_StreamingPort_name;
212  return false;
213  }
214 
215  if(! yarp::os::Network::connect(remote_depthFrame_StreamingPort_name, depthFrame_StreamingPort.getName(), "udp") )
216  {
217  yError() << depthFrame_StreamingPort.getName() << " cannot connect to remote port " << remote_depthFrame_StreamingPort_name;
218  return false;
219  }
220 
221 
222  // Single RPC port
223  ret = rpcPort.open(local_rpcPort_name);
224 
225  if(!ret)
226  {
227  yError() << sensorId << " cannot open local RPC port " << local_rpcPort_name;
228  colorFrame_StreamingPort.close();
229  depthFrame_StreamingPort.close();
230  rpcPort.close();
231  }
232 
233  if(! rpcPort.addOutput(remote_rpcPort_name) )
234  {
235  yError() << sensorId << " cannot connect to port " << remote_rpcPort_name;
236  colorFrame_StreamingPort.close();
237  depthFrame_StreamingPort.close();
238  rpcPort.close();
239  return false;
240  }
241 
242  // Check protocol version
243  yarp::os::Bottle cmd, response;
244  cmd.addVocab(VOCAB_RGBD_SENSOR);
245  cmd.addVocab(VOCAB_GET);
246  cmd.addVocab(VOCAB_RGBD_PROTOCOL_VERSION);
247  rpcPort.write(cmd, response);
248  int major = response.get(3).asInt32();
249  int minor = response.get(4).asInt32();
250 
251  if(major != RGBD_INTERFACE_PROTOCOL_VERSION_MAJOR)
252  {
253  yError() << "Major protocol number does not match, please verify client and server are updated. \
254  Expected: " << RGBD_INTERFACE_PROTOCOL_VERSION_MAJOR << " received: " << major;
255  return false;
256  }
257 
258 
259  if(minor != RGBD_INTERFACE_PROTOCOL_VERSION_MINOR)
260  {
261  yWarning() << "Minor protocol number does not match, please verify client and server are updated.\
262  Expected: " << RGBD_INTERFACE_PROTOCOL_VERSION_MINOR << " received: " << minor;
263  }
264 
265  /*
266  * Multiple RPC ports
267  *
268  ret &= colorFrame_rpcPort.open(local_colorFrame_rpcPort_Name.c_str() );
269  ret &= depthFrame_rpcPort.open(local_depthFrame_rpcPort_Name.c_str() );
270 
271  if(!ret)
272  yError() << "sensorId cannot open ports";
273 
274  // doing connections: How to correctly handle YARP_PORT_PREFIX for remote port names??
275  if(! colorFrame_rpcPort.addOutput(remote_colorFrame_rpcPort_Name.c_str()) ) // This will handle local port names only
276  {
277  yError() << sensorId << " cannot add output " << remote_colorFrame_rpcPort_Name;
278  return false;
279  }
280 
281  if(! depthFrame_rpcPort.addOutput(remote_depthFrame_rpcPort_Name.c_str()) ) // This will handle local port names only
282  {
283  yError() << sensorId << " cannot add output " << remote_depthFrame_rpcPort_Name;
284  return false;
285  }
286  */
287 
288 
289  streamingReader->attach(&colorFrame_StreamingPort, &depthFrame_StreamingPort);
290 
291  return true;
292 }

330 {
