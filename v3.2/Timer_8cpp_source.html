175 void TimerSingleton::run()
176 {
177  mu.lock();
178  for (auto t : timers) {
179  MonoThreadTimer& timer = *t.second;
180  YarpTimerEvent tEvent = timer.getEventNow();
181  if (timer.m_active && tEvent.currentReal > tEvent.currentExpected) {
182  timer.m_active = timer.step(tEvent, false);
183  timer.m_lastReal = tEvent.currentReal;
184  }
185  }
186  mu.unlock();
187 }

266 YarpTimerEvent yarp::os::Timer::PrivateImpl::getEventNow(unsigned int iteration)
267 {
268  YarpTimerEvent event;
269 
270  event.currentReal = yarp::os::Time::now();
271  event.currentExpected = m_startStamp + iteration * m_settings.period;
272  event.lastExpected = event.currentExpected - m_settings.period;
273  event.lastReal = m_lastReal;
274  event.lastDuration = event.currentReal - m_lastReal;
275  event.runCount = iteration;
276  return event;
277 }

279 bool yarp::os::Timer::PrivateImpl::runTimer(unsigned int iteration, YarpTimerEvent event)
280 {
281  if (m_mutex != nullptr) {
282  m_mutex->lock();
283  }
284 
285  bool ret = m_callback(event);
286 
287  if (m_mutex != nullptr) {
288  m_mutex->unlock();
289  }
290 
291  if (!ret) {
292  return false;
293  }
294 
295  m_lastReal = event.currentReal;
296 
297  double timerAge = (yarp::os::Time::now() - m_startStamp);
298 
299  //totalRunCount == 0 ----> infinite run count. follows the run count of the timer
300  bool stop(m_settings.totalRunCount != 0 && m_settings.totalRunCount <= iteration);
301 
302  //totalTime == 0 ----> infinite time. follows the age check for the timer
303  stop |= m_settings.totalTime > 0.00001 && (m_settings.totalTime - timerAge) < m_settings.tolerance;
304 
305  return !stop;
306 }
