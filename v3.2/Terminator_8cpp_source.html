23 bool Terminator::terminateByName(const char* name)
24 {
25  if (name == nullptr) {
26  return false;
27  }
28 
29  std::string s(name);
30 
31  if (s.find("/quit") == std::string::npos) {
32  // name doesn't include /quit
33  // old mechanism won't work, let's try new
34  PortCommand pc('\0', "i");
35  NetworkBase::sendMessage(s, pc, true);
36  return true;
37  }
38 
39  if (name[0] != '/') {
40  s.clear();
41  s += "/";
42  s += name;
43  }
44 
45  Bottle cmd("quit");
46  Bottle reply;
47  Contact c = NetworkBase::queryName(s);

54  style.carrier = "text_ack";
55  style.expectReply = true;
