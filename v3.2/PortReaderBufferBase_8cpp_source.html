382 PortReader* PortReaderBufferBase::readBase(bool& missed, bool cleanup)
383 {
384  missed = false;
385  if (mPriv->period < 0 || cleanup) {
386  mPriv->contentSema.wait();
387  } else {
388  bool ok = false;
389  double now = SystemClock::nowSystem();
390  double target = now + mPriv->period;
391  if (mPriv->last_recv > 0) {
392  target = mPriv->last_recv + mPriv->period;
393  }
394  double diff = target - now;
395  if (diff > 0) {
396  ok = mPriv->contentSema.waitWithTimeout(diff);
397  } else {
398  ok = mPriv->contentSema.check();
399  if (ok) {
400  mPriv->contentSema.wait();
401  }
402  }
403  if (!ok) {
404  missed = true;
405  if (mPriv->last_recv > 0) {
406  mPriv->last_recv += mPriv->period;
407  }
408  return nullptr;
409  }
410  now = SystemClock::nowSystem();
411  if (mPriv->last_recv < 0) {
412  mPriv->last_recv = now;
413  } else {
414  diff = target - now;
415  if (diff > 0) {
416  SystemClock::delaySystem(diff);
417  }
418  mPriv->last_recv = target;
419  }
420  }
421  mPriv->stateMutex.lock();
422  PortReaderPacket* readerPacket = mPriv->getContent();
423  PortReader* reader = nullptr;
424  if (readerPacket != nullptr) {
425  PortReader* external = readerPacket->getExternal();
426  if (external == nullptr) {
427  reader = readerPacket->getReader();
428  } else {
429  reader = external;
430  }
431  }
432  mPriv->stateMutex.unlock();
433  if (reader != nullptr) {
434  mPriv->consumeSema.post();
435  }
436  return reader;
437 }

440 bool PortReaderBufferBase::read(ConnectionReader& connection)
441 {
442  if (connection.getReference() != nullptr) {
443  //printf("REF %ld %d\n", (long int)connection.getReference(),
444  // connection.isValid());
445  return acceptObjectBase(connection.getReference(), nullptr);
446  }
447 
448  if (mPriv->replier != nullptr) {
449  if (connection.getWriter() != nullptr) {
450  return mPriv->replier->read(connection);
451  }
452  }
453  PortReaderPacket* reader = nullptr;
454  while (reader == nullptr) {
455  mPriv->stateMutex.lock();
456  reader = mPriv->get();
457  if ((reader != nullptr) && reader->getReader() == nullptr) {
458  PortReader* next = create();
459  yAssert(next != nullptr);
460  reader->setReader(next);
461  }
462 
463  mPriv->stateMutex.unlock();
464  if (reader == nullptr) {
465  mPriv->consumeSema.wait();
466  }
467  }
468  bool ok = false;
469  if (connection.isValid()) {
470  yAssert(reader->getReader() != nullptr);
471  ok = reader->getReader()->read(connection);
472  reader->setEnvelope(connection.readEnvelope());
473  } else {
474  // this is a disconnection
475  // don't talk to this port ever again
476  mPriv->port = nullptr;
477  }
478  if (ok) {
479  mPriv->stateMutex.lock();
480  bool pruned = false;
481  if (mPriv->ct > 0 && mPriv->prune) {
482  PortReaderPacket* readerPacket = mPriv->dropContent();
483  pruned = (readerPacket != nullptr);
484  }
485  //mPriv->configure(reader, false, true);
486  mPriv->pool.addActivePacket(reader);
487  mPriv->ct++;
488  mPriv->stateMutex.unlock();
489  if (!pruned) {
490  mPriv->contentSema.post();
491  }
492  //YARP_ERROR(Logger::get(), ">>>>>>>>>>>>>>>>> adding data");
493  } else {
494  mPriv->stateMutex.lock();
495  mPriv->pool.addInactivePacket(reader);
496  mPriv->stateMutex.unlock();
497  //YARP_ERROR(Logger::get(), ">>>>>>>>>>>>>>>>> skipping data");
498 
499  // important to give reader a shot anyway, allowing proper closing
500  YARP_DEBUG(Logger::get(), "giving PortReaderBuffer chance to close");
501  mPriv->contentSema.post();
502  }
503  return ok;
504 }
