123 bool ServerSerial::open(Searchable& prop)
124 {
125  verb = (prop.check("verbose",Value(0),"Specifies if the device is in verbose mode (0/1).").asInt32())>0;
126  if (verb)
127  yInfo("running with verbose output\n");
128 
129  Value *name;
130  if (prop.check("subdevice",name,"name of specific control device to wrap")) {
131  yDebug("Subdevice %s\n", name->toString().c_str());
132  if (name->isString()) {
133  // maybe user isn't doing nested configuration
134  Property p;
135  p.setMonitor(prop.getMonitor(),
136  "subdevice"); // pass on any monitoring
137  p.fromString(prop.toString());
138  p.put("device",name->toString());
139  poly.open(p);
140  } else {
141  Bottle subdevice = prop.findGroup("subdevice").tail();
142  poly.open(subdevice);
143  }
144  if (!poly.isValid()) {
145  yError("cannot make <%s>\n", name->toString().c_str());
146  }
147  } else {
148  yError("\"--subdevice <name>\" not set for serial\n");
149  return false;
150  }
151 
152  if (!poly.isValid()) {
153  return false;
154  }
155 
156  std::string rootName =
157  prop.check("name",Value("/serial"),
158  "prefix for port names").asString();
159 
160  command_buffer.attach(toDevice);
161  reply_buffer.attach(fromDevice);
162 
163  command_buffer.useCallback(callback_impl);
164 
165  toDevice.open(rootName+"/in");
166  fromDevice.open(rootName+"/out");
167 
168 
169 
170  if (poly.isValid())
171  poly.view(serial);
172 
173  if(serial != nullptr) {
174  start();
175  return true;
176  }
177 
178  yError("subdevice <%s> doesn't look like a serial port (no appropriate interfaces were acquired)\n",
179  name->toString().c_str());
180 
181  return false;
182 }
