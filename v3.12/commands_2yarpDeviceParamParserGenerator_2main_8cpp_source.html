38bool ParamsFilesGenerator::nested_sections_found()
39{
40 bool b = true;
41 for (auto param : m_params)
42 {
43 //aaa::bbb is not nested (size == 1)
44 //aaa::bbb::ccc is nested (size == 2)
45 if (param.getListOfGroups().size() > 1)
46 b = false;
47 }
48 return !b;
49}
