159 void Odometry2D_nws_yarp::run()
160 {
161  if (m_odometry2D_interface!=nullptr)
162  {
163  yarp::os::Stamp timeStamp(static_cast<int>(m_stampCount++), yarp::os::Time::now());
164  yarp::dev::OdometryData odometryData;
165  m_odometry2D_interface->getOdometry(odometryData);
166  if (m_port_odometry.getOutputCount()>0)
167  {
168  m_port_odometry.setEnvelope(timeStamp);
169  yarp::dev::OdometryData &odometryDataFromPort = m_port_odometry.prepare();
170  odometryDataFromPort.odom_x= odometryData.odom_x; //position in the odom reference frame
171  odometryDataFromPort.odom_y= odometryData.odom_y;
172  odometryDataFromPort.odom_theta= odometryData.odom_theta;
173  odometryDataFromPort.base_vel_x= odometryData.base_vel_x; //velocity in the robot reference frame
174  odometryDataFromPort.base_vel_y= odometryData.base_vel_y;
175  odometryDataFromPort.base_vel_theta= odometryData.base_vel_theta;
176  odometryDataFromPort.odom_vel_x= odometryData.odom_vel_x; //velocity in the odom reference frame
177  odometryDataFromPort.odom_vel_y= odometryData.odom_vel_y;
178  odometryDataFromPort.odom_vel_theta= odometryData.odom_vel_theta;
179  m_port_odometry.write();
180  }
181 
182  if (m_port_odometer.getOutputCount()>0)
183  {
184  m_port_odometer.setEnvelope(timeStamp);
185  yarp::os::Bottle &odometer_data = m_port_odometer.prepare();
186  odometer_data.clear();
187  double traveled_distance = sqrt((odometryData.odom_vel_x - m_oldOdometryData.odom_x) *
188  (odometryData.odom_vel_x - m_oldOdometryData.odom_x) +
189  (odometryData.odom_vel_y - m_oldOdometryData.odom_y) *
190  (odometryData.odom_vel_y - m_oldOdometryData.odom_y));
191  double traveled_angle = odometryData.odom_theta - m_oldOdometryData.odom_theta;
192  odometer_data.addFloat64(traveled_distance);
193  odometer_data.addFloat64(traveled_angle);
194  m_port_odometer.write();
195  }
196 
197 
198  if (m_port_velocity.getOutputCount()>0)
199  {
200  m_port_velocity.setEnvelope(timeStamp);
201  yarp::os::Bottle &velocityData = m_port_velocity.prepare();
202  velocityData.clear();
203  velocityData.addFloat64(sqrt(odometryData.odom_vel_x * odometryData.odom_vel_x +
204  odometryData.odom_vel_y * odometryData.odom_vel_y));
205  velocityData.addFloat64(odometryData.base_vel_theta);
206  m_port_velocity.write();
207  }
208  } else {
209  yCError(ODOMETRY2D_NWS_YARP) << "no yarp::dev::Nav2D::IOdometry2D interface found";
210  }
211 }
