92 bool Navigation2DClient::getCurrentPosition(Map2DLocation& loc, yarp::sig::Matrix& cov)
93 {
94  yarp::os::Bottle b;
95  yarp::os::Bottle resp;
96 
97  b.addVocab32(VOCAB_INAVIGATION);
98  b.addVocab32(VOCAB_NAV_GET_CURRENT_POSCOV);
99 
100  bool ret = m_rpc_port_to_localization_server.write(b, resp);
101  if (ret)
102  {
103  if (resp.get(0).asVocab32() != VOCAB_OK || resp.size() != 6)
104  {
105  yCError(NAVIGATION2DCLIENT) << "getCurrentPosition() received error from localization server";
106  return false;
107  }
108  else
109  {
110  if (cov.rows() != 3 || cov.cols() != 3)
111  {
112  yCDebug(NAVIGATION2DCLIENT) << "Performance warning: covariance matrix is not (3,3), resizing...";
113  cov.resize(3, 3);
114  }
115  loc.map_id = resp.get(1).asString();
116  loc.x = resp.get(2).asFloat64();
117  loc.y = resp.get(3).asFloat64();
118  loc.theta = resp.get(4).asFloat64();
119  Bottle* mc = resp.get(5).asList();
120  if (mc == nullptr) {
121  return false;
122  }
123  for (size_t i = 0; i < 3; i++) { for (size_t j = 0; j < 3; j++) { cov[i][j] = mc->get(i * 3 + j).asFloat64(); } }
124  return true;
125  }
126  }
127  else
128  {
129  yCError(NAVIGATION2DCLIENT) << "getCurrentPosition() error on writing on rpc port";
130  return false;
131  }
132  return true;
