27 bool Navigation2DClient::storeLocation(std::string location_name, Map2DLocation loc)
28 {
29  yarp::os::Bottle b;
30  yarp::os::Bottle resp;
31 
32  b.addVocab32(VOCAB_INAVIGATION);
33  b.addVocab32(VOCAB_NAV_STORE_X);
34  b.addVocab32(VOCAB_NAV_LOCATION);
35  b.addString(location_name);
36  b.addString(loc.map_id);
37  b.addFloat64(loc.x);
38  b.addFloat64(loc.y);
39  b.addFloat64(loc.theta);
40 
41  bool ret = m_rpc_port_to_Map2DServer.write(b, resp);
42  if (ret)
43  {
44  if (resp.get(0).asVocab32() != VOCAB_OK)
45  {
46  yCError(NAVIGATION2DCLIENT) << "storeLocation() received error from locations server";
47  return false;
48  }
49  }
50  else
51  {
52  yCError(NAVIGATION2DCLIENT) << "storeLocation() error on writing on rpc port";
53  return false;
54  }
55  return true;
