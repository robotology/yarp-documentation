27 bool Map2DClient::open(yarp::os::Searchable &config)
28 {
29  yCWarning(MAP2DCLIENT) << "The 'Map2DClient' device is deprecated in favour of 'map2D_nwc_yarp'.";
30  yCWarning(MAP2DCLIENT) << "The old device is no longer supported, and it will be deprecated in YARP 3.7 and removed in YARP 4.";
31  yCWarning(MAP2DCLIENT) << "Please update your scripts.";
32 
33  m_local_name.clear();
34  m_map_server.clear();
35 
36  m_local_name = config.find("local").asString();
37  m_map_server = config.find("remote").asString();
38  m_carrier = config.check("carrier", yarp::os::Value("tcp"), "the carrier used for the connection with the server").asString();
39 
40  if (m_local_name == "")
41  {
42  yCError(MAP2DCLIENT, "open() error you have to provide valid local name");
43  return false;
44  }
45  if (m_map_server == "")
46  {
47  yCError(MAP2DCLIENT, "open() error you have to provide valid remote name");
48  return false;
49  }
50 
51  std::string local_rpc1 = m_local_name;
52  local_rpc1 += "/mapClient_rpc";
53 
54  std::string remote_rpc1 = m_map_server;
55  remote_rpc1 += "/rpc";
56 
57  if (!m_rpcPort_to_Map2DServer.open(local_rpc1))
58  {
59  yCError(MAP2DCLIENT, "open() error could not open rpc port %s, check network", local_rpc1.c_str());
60  return false;
61  }
62 
63  bool ok=false;
64  ok=Network::connect(local_rpc1, remote_rpc1, m_carrier);
65  if (!ok)
66  {
67  yCError(MAP2DCLIENT, "open() error could not connect to %s", remote_rpc1.c_str());
68  return false;
69  }
70 
71  return true;
72 }

74 bool Map2DClient::store_map(const MapGrid2D& map)
75 {
76  yarp::os::Bottle b;
77  yarp::os::Bottle resp;
78 
79  b.addVocab32(VOCAB_IMAP);
80  b.addVocab32(VOCAB_IMAP_SET_MAP);
81  yarp::os::Bottle& mapbot = b.addList();
82  MapGrid2D maptemp = map;
83  if (Property::copyPortable(maptemp, mapbot) == false)
84  {
85  yCError(MAP2DCLIENT) << "store_map() failed copyPortable()";
86  return false;
87  }
88  //yCDebug(MAP2DCLIENT) << b.toString();
89  bool ret = m_rpcPort_to_Map2DServer.write(b, resp);
90  if (ret)
91  {
92  if (resp.get(0).asVocab32() != VOCAB_IMAP_OK)
93  {
94  yCError(MAP2DCLIENT) << "store_map() received error from server";
95  return false;
96  }
97  }
98  else
99  {
100  yCError(MAP2DCLIENT) << "store_map() error on writing on rpc port";
101  return false;
102  }
103  return true;
