182 bool navigation2DServer::parse_respond_vocab(const yarp::os::Bottle& command, yarp::os::Bottle& reply)
183 {
184  if (command.get(0).isVocab32() == false)
185  {
186  yCError(NAVIGATION2DSERVER) << "General error in navigation2DServer::parse_respond_vocab()";
187  return false;
188  }
189 
190  if (command.get(0).asVocab32() != VOCAB_INAVIGATION ||
191  command.get(1).isVocab32() == false)
192  {
193  yCError(NAVIGATION2DSERVER) << "Invalid vocab received";
194  reply.addVocab32(VOCAB_ERR);
195  return true;
196  }
197 
198  int request = command.get(1).asVocab32();
199  if (request == VOCAB_NAV_GOTOABS)
200  {
201  Map2DLocation loc;
202  loc.map_id = command.get(2).asString();
203  loc.x = command.get(3).asFloat64();
204  loc.y = command.get(4).asFloat64();
205  loc.theta = command.get(5).asFloat64();
206  bool ret = iNav_target->gotoTargetByAbsoluteLocation(loc);
207  if (ret)
208  {
209  clear_current_goal_name();
210  reply.addVocab32(VOCAB_OK);
211  }
212  else
213  {
214  yCError(NAVIGATION2DSERVER) << "gotoTargetByAbsoluteLocation() failed";
215  reply.addVocab32(VOCAB_ERR);
216  }
217  }
218  else if (request == VOCAB_NAV_GOTOABS_AND_NAME)
219  {
220  Map2DLocation loc;
221  loc.map_id = command.get(2).asString();
222  loc.x = command.get(3).asFloat64();
223  loc.y = command.get(4).asFloat64();
224  loc.theta = command.get(5).asFloat64();
225  std::string location_name = command.get(6).asString();
226  bool ret = iNav_target->gotoTargetByAbsoluteLocation(loc);
227  if (ret)
228  {
229  set_current_goal_name(location_name);
230  reply.addVocab32(VOCAB_OK);
231  }
232  else
233  {
234  yCError(NAVIGATION2DSERVER) << "gotoTargetByAbsoluteLocation() failed";
235  reply.addVocab32(VOCAB_ERR);
236  }
237  }
238  else if (request == VOCAB_NAV_RECOMPUTE_PATH)
239  {
240  bool ret = iNav_ctrl->recomputeCurrentNavigationPath();
241  if (ret)
242  {
243  reply.addVocab32(VOCAB_OK);
244  }
245  else
246  {
247  yCError(NAVIGATION2DSERVER) << "recomputeCurrentNavigationPath() failed";
248  reply.addVocab32(VOCAB_ERR);
249  }
250  reply.addVocab32(VOCAB_OK);
251  }
252  else if (request == VOCAB_NAV_GOTOREL)
253  {
254  if (command.size() == 5)
255  {
256  double x = command.get(2).asFloat64();
257  double y = command.get(3).asFloat64();
258  double theta = command.get(4).asFloat64();
259  bool ret = iNav_target->gotoTargetByRelativeLocation(x, y, theta);
260  if (ret)
261  {
262  clear_current_goal_name();
263  reply.addVocab32(VOCAB_OK);
264  }
265  else
266  {
267  yCError(NAVIGATION2DSERVER) << "gotoTargetByRelativeLocation() failed";
268  reply.addVocab32(VOCAB_ERR);
269  }
270  }
271  else if (command.size() == 4)
272  {
273  double x = command.get(2).asFloat64();
274  double y = command.get(3).asFloat64();
275  bool ret = iNav_target->gotoTargetByRelativeLocation(x, y);
276  if (ret)
277  {
278  clear_current_goal_name();
279  reply.addVocab32(VOCAB_OK);
280  }
281  else
282  {
283  yCError(NAVIGATION2DSERVER) << "gotoTargetByRelativeLocation() failed";
284  reply.addVocab32(VOCAB_ERR);
285  }
286  }
287  else
288  {
289  yCError(NAVIGATION2DSERVER) << "Invalid number of params";
290  reply.addVocab32(VOCAB_ERR);
291  }
292  }
293  else if (request == VOCAB_NAV_GET_NAME_TARGET)
294  {
295  reply.addVocab32(VOCAB_OK);
296  reply.addString(m_current_goal_name);
297  }
298  else if (request == VOCAB_NAV_GET_NAVIGATION_STATUS)
299  {
300  yarp::dev::Nav2D::NavigationStatusEnum nav_status = yarp::dev::Nav2D::navigation_status_error;
301  bool ret = iNav_ctrl->getNavigationStatus(nav_status);
302  if (ret)
303  {
304  reply.addVocab32(VOCAB_OK);
305  reply.addInt32(nav_status);
306  }
307  else
308  {
309  yCError(NAVIGATION2DSERVER) << "getNavigationStatus() failed";
310  reply.addVocab32(VOCAB_ERR);
311  }
312  }
313  else if (request == VOCAB_NAV_STOP)
314  {
315  bool ret = iNav_ctrl->stopNavigation();
316  if (ret)
317  {
318  clear_current_goal_name();
319  reply.addVocab32(VOCAB_OK);
320  }
321  else
322  {
323  yCError(NAVIGATION2DSERVER) << "stopNavigation() failed";
324  reply.addVocab32(VOCAB_ERR);
325  }
326  }
327  else if (request == VOCAB_NAV_SUSPEND)
328  {
329  double time = -1;
330  if (command.size() > 1)
331  {
332  time = command.get(1).asFloat64();
333  bool ret = iNav_ctrl->suspendNavigation(time);
334  if (ret)
335  {
336  reply.addVocab32(VOCAB_OK);
337  }
338  else
339  {
340  yCError(NAVIGATION2DSERVER) << "suspendNavigation() failed";
341  reply.addVocab32(VOCAB_ERR);
342  }
343  }
344  else
345  {
346  bool ret = iNav_ctrl->suspendNavigation();
347  if (ret)
348  {
349  reply.addVocab32(VOCAB_OK);
350  }
351  else
352  {
353  yCError(NAVIGATION2DSERVER) << "suspendNavigation() failed";
354  reply.addVocab32(VOCAB_ERR);
355  }
356  }
357  }
358  else if (request == VOCAB_NAV_RESUME)
359  {
360  bool ret = iNav_ctrl->resumeNavigation();
361  if (ret)
362  {
363  reply.addVocab32(VOCAB_OK);
364  }
365  else
366  {
367  yCError(NAVIGATION2DSERVER) << "resumeNavigation failed()";
368  reply.addVocab32(VOCAB_ERR);
369  }
370  }
371  else if (request == VOCAB_NAV_GET_NAVIGATION_WAYPOINTS)
372  {
373  Map2DPath locs;
374  bool ret = iNav_ctrl->getAllNavigationWaypoints((yarp::dev::Nav2D::TrajectoryTypeEnum)(command.get(2).asInt32()), locs);
375  if (ret)
376  {
377  reply.addVocab32(VOCAB_OK);
378  Bottle& waypoints = reply.addList();
379  for (auto it = locs.begin(); it!=locs.end(); it++)
380  {
381  Bottle& the_waypoint = waypoints.addList();
382  the_waypoint.addString(it->map_id);
383  the_waypoint.addFloat64(it->x);
384  the_waypoint.addFloat64(it->y);
385  the_waypoint.addFloat64(it->theta);
386  }
387  }
388  else
389  {
390  yCError(NAVIGATION2DSERVER) << "getAllNavigationWaypoints() failed";
391  reply.addVocab32(VOCAB_ERR);
392  }
393  }
394  else if (request == VOCAB_NAV_GET_CURRENT_WAYPOINT)
395  {
396  Map2DLocation loc;
397  bool ret = iNav_ctrl->getCurrentNavigationWaypoint(loc);
398  if (ret)
399  {
400  reply.addVocab32(VOCAB_OK);
401  reply.addString(loc.map_id);
402  reply.addFloat64(loc.x);
403  reply.addFloat64(loc.y);
404  reply.addFloat64(loc.theta);
405  }
406  else
407  {
408  yCError(NAVIGATION2DSERVER) << "getCurrentNavigationWaypoint() failed";
409  reply.addVocab32(VOCAB_ERR);
410  }
411  }
412  else if (request == VOCAB_NAV_GET_NAV_MAP)
413  {
414  MapGrid2D map;
415  if (iNav_ctrl->getCurrentNavigationMap((yarp::dev::Nav2D::NavigationMapTypeEnum)(command.get(2).asInt32()), map))
416  {
417  reply.addVocab32(VOCAB_OK);
418  yarp::os::Bottle& mapbot = reply.addList();
419  Property::copyPortable(map, mapbot);
420  }
421  else
422  {
423  yCError(NAVIGATION2DSERVER) << "getCurrentNavigationMap() failed";
424  reply.addVocab32(VOCAB_ERR);
425  }
426  }
427  else if (request == VOCAB_NAV_GET_ABS_TARGET)
428  {
429  Map2DLocation loc;
430  bool ret;
431  ret = iNav_target->getAbsoluteLocationOfCurrentTarget(loc);
432  if (ret)
433  {
434  reply.addVocab32(VOCAB_OK);
435  reply.addString(loc.map_id);
436  reply.addFloat64(loc.x);
437  reply.addFloat64(loc.y);
438  reply.addFloat64(loc.theta);
439  }
440  else
441  {
442  yCError(NAVIGATION2DSERVER) << "getAbsoluteLocationOfCurrentTarget() failed";
443  reply.addVocab32(VOCAB_ERR);
444  }
445  }
446  else if (request == VOCAB_NAV_GET_REL_TARGET)
447  {
448  Map2DLocation loc;
449  bool ret;
450  ret = iNav_target->getRelativeLocationOfCurrentTarget(loc.x, loc.y, loc.theta);
451  if (ret)
452  {
453  reply.addVocab32(VOCAB_OK);
454  reply.addFloat64(loc.x);
455  reply.addFloat64(loc.y);
456  reply.addFloat64(loc.theta);
457  }
458  else
459  {
460  yCError(NAVIGATION2DSERVER) << "getRelativeLocationOfCurrentTarget() failed";
461  reply.addVocab32(VOCAB_ERR);
462  }
463  }
464  else
465  {
466  yCError(NAVIGATION2DSERVER) << "Invalid vocab received:" << yarp::os::Vocab32::decode(request);
467  reply.addVocab32(VOCAB_ERR);
468  }
469 
470  return true;

473 bool navigation2DServer::read(yarp::os::ConnectionReader& connection)
474 {
475  yarp::os::Bottle command;
476  yarp::os::Bottle reply;
477  bool ok = command.read(connection);
478  if (!ok) {
479  return false;
480  }
481  reply.clear();
482 
483  //^^^^^^^^^^^^^^^^^ STRING SECTION
484  if (command.get(0).isString())
485  {
486  parse_respond_string(command, reply);
487  }
488  //^^^^^^^^^^^^^^^^^ VOCAB SECTION
489  else if (command.get(0).isVocab32())
490  {
491  parse_respond_vocab(command, reply);
492  }
493  else
494  {
495  yCError(NAVIGATION2DSERVER) << "Invalid command type";
496  reply.addVocab32(VOCAB_ERR);
497  }
498 
499  yarp::os::ConnectionWriter *returnToSender = connection.getWriter();
500  if (returnToSender != nullptr)
501  {
502  reply.write(*returnToSender);
503  }
504 
505  return true;
506 }
