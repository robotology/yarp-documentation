160 bool Navigation2DClient::parse_respond_string(const yarp::os::Bottle& command, yarp::os::Bottle& reply)
161 {
162  if (command.get(0).isString() == false)
163  {
164  yCError(NAVIGATION2DCLIENT) << "General error in parse_respond_string";
165  return false;
166  }
167 
168  if (command.get(0).asString() == "help")
169  {
170  reply.addVocab32("many");
171  reply.addString("Available commands are:");
172  reply.addString("goto <locationName>");
173  //reply.addString("gotoAbs <x> <y> <angle in degrees>");
174  //reply.addString("gotoRel <x> <y> <angle in degrees>");
175  reply.addString("store_location <location_name> <map_id> <x> <y> <y>");
176  reply.addString("store_current_location <location_name>");
177  reply.addString("delete_location <location_name>");
178  reply.addString("clear_all_locations");
179  reply.addString("get_last_target");
180  reply.addString("get_location_list");
181  reply.addString("get_navigation_status");
182  reply.addString("stop_loc");
183  reply.addString("start_loc");
184  reply.addString("stop_nav");
185  reply.addString("pause_nav");
186  reply.addString("resume_nav");
187  reply.addString("get_current_loc");
188  reply.addString("initLoc <map_name> <x> <y> <angle in degrees>");
189  }
190  else if (command.get(0).asString() == "store_current_location")
191  {
192  bool ret = this->storeCurrentPosition(command.get(1).asString());
193  if (ret)
194  {
195  reply.addString("store_current_location done");
196  }
197  else
198  {
199  reply.addString("store_current_location failed");
200  }
201  }
202  else if (command.get(0).asString() == "gotoAbs")
203  {
204  Map2DLocation loc;
205  loc.map_id = command.get(1).asString();
206  loc.x = command.get(2).asFloat64();
207  loc.y = command.get(3).asFloat64();
208  if (command.size() == 5)
209  {
210  loc.theta = command.get(4).asFloat64();
211  }
212  else
213  {
214  loc.theta = nan("");
215  }
216 
217  bool ret = this->gotoTargetByAbsoluteLocation(loc);
218  if (ret)
219  {
220  reply.addString("gotoTargetByAbsoluteLocation() executed successfully");
221  }
222  else
223  {
224  reply.addString("gotoTargetByAbsoluteLocation() returned an error");
225  }
226  }
227 
228  else if (command.get(0).asString() == "gotoRel")
229  {
230  yarp::sig::Vector v;
231  double x = command.get(1).asFloat64();
232  double y = command.get(2).asFloat64();
233  bool ret;
234  if (command.size() == 4)
235  {
236  double t = command.get(3).asFloat64();
237  ret = this->gotoTargetByRelativeLocation(x, y, t);
238  }
239  else
240  {
241  ret = this->gotoTargetByRelativeLocation(x, y);
242  }
243 
244  if (ret)
245  {
246  reply.addString("gotoTargetByRelativeLocation() executed successfully");
247  }
248  else
249  {
250  reply.addString("gotoTargetByRelativeLocation() returned an error");
251  }
252  }
253  else if (command.get(0).asString() == "get_location_list")
254  {
255  std::vector<std::string> locations;
256  bool ret = getLocationsList(locations);
257  if (ret)
258  {
259  for (size_t i=0; i < locations.size(); i++)
260  {
261  reply.addString(locations[i]);
262  }
263  }
264  else
265  {
266  reply.addString("get_location_list failed");
267  }
268  }
269  else if (command.get(0).asString() == "get_navigation_status")
270  {
271  yarp::dev::Nav2D::NavigationStatusEnum ss;
272  bool ret = this->getNavigationStatus(ss);
273  if (ret)
274  {
275  std::string s = yarp::dev::Nav2D::INavigation2DHelpers::statusToString(ss);
276  reply.addString(s.c_str());
277  }
278  else
279  {
280  reply.addString("getNavigationStatus() failed");
281  }
282  }
283  else if (command.get(0).isString() && command.get(0).asString() == "get_current_loc")
284  {
285  {
286  Map2DLocation curr_loc;
287  bool ret1 = this->getCurrentPosition(curr_loc);
288  if (ret1)
289  {
290  std::string s = std::string("Current Location is: ") + curr_loc.toString();
291  reply.addString(s);
292  }
293  else
294  {
295  reply.addString("getCurrentPosition(curr_loc) failed");
296  }
297  }
298  {
299  Map2DLocation curr_loc;
300  Matrix cov;
301  bool ret2 = this->getCurrentPosition(curr_loc, cov);
302  if (ret2)
303  {
304  std::string s = std::string("Current Location with covariance is: ") + curr_loc.toString() + "\n" + cov.toString();
305  reply.addString(s);
306  }
307  else
308  {
309  reply.addString("getCurrentPosition(curr_loc, covariance) failed");
310  }
311  }
312  }
313  else if (command.get(0).isString() && command.get(0).asString() == "initLoc")
314  {
315  Map2DLocation init_loc;
316  bool ret = false;
317  if (command.size() == 5)
318  {
319  init_loc.map_id = command.get(1).asString();
320  init_loc.x = command.get(2).asFloat64();
321  init_loc.y = command.get(3).asFloat64();
322  init_loc.theta = command.get(4).asFloat64();
323  ret = this->setInitialPose(init_loc);
324  }
325  else if (command.size() == 6)
326  {
327  init_loc.map_id = command.get(1).asString();
328  init_loc.x = command.get(2).asFloat64();
329  init_loc.y = command.get(3).asFloat64();
330  init_loc.theta = command.get(4).asFloat64();
331  Bottle* b= command.get(5).asList();
332  if (b && b->size()==9)
333  {
334  yarp::sig::Matrix cov(3,3);
335  for (size_t i = 0; i < 3; i++) { for (size_t j = 0; j < 3; j++) { cov[i][j] = b->get(i * 3 + j).asFloat64(); } }
336  ret = this->setInitialPose(init_loc, cov);
337  } else {
338  ret = false;
339  }
340  }
341 
342  if (ret)
343  {
344  std::string s = std::string("Localization initialized to: ") + init_loc.toString();
345  reply.addString(s);
346  }
347  else
348  {
349  reply.addString("setInitialPose() failed");
350  }
351  }
352  else if (command.get(0).asString() == "store_location")
353  {
354  if (command.size() != 6)
355  {
356  reply.addString("store_location failed (invalid params)");
357  }
358  else
359  {
360  Map2DLocation loc;
361  loc.map_id = command.get(2).asString();
362  loc.x = command.get(3).asFloat64();
363  loc.y = command.get(4).asFloat64();
364  loc.theta = command.get(5).asFloat64();
365  bool ret = this->storeLocation(command.get(1).asString(), loc);
366  if (ret)
367  {
368  reply.addString("store_location done");
369  }
370  else
371  {
372  reply.addString("store_location failed");
373  }
374  }
375  }
376  else if (command.get(0).asString() == "delete_location")
377  {
378  bool ret = this->deleteLocation(command.get(1).asString());
379  if (ret)
380  {
381  reply.addString("delete_location done");
382  }
383  else
384  {
385  reply.addString("delete_location failed");
386  }
387  }
388  else if (command.get(0).asString() == "clear_all_locations")
389  {
390  std::vector<std::string> locations;
391  bool ret = getLocationsList(locations);
392  if (ret)
393  {
394  for (size_t i = 0; i < locations.size(); i++)
395  {
396  bool ret = this->deleteLocation(locations[i]);
397  if (ret == false)
398  {
399  reply.addString("clear_all_locations failed");
400  }
401  }
402  reply.addString("clear_all_locations done");
403  }
404  else
405  {
406  reply.addString("clear_all_locations failed");
407  }
408  }
409  else if (command.get(0).asString() == "goto")
410  {
411  bool ret = this->gotoTargetByLocationName(command.get(1).asString());
412  if (ret)
413  {
414  reply.addString("goto done");
415  }
416  else
417  {
418  reply.addString("goto failed");
419  }
420 
421  }
422  else if (command.get(0).asString() == "get_last_target")
423  {
424  std::string last_target;
425  bool b = this->getNameOfCurrentTarget(last_target);
426  if (b)
427  {
428  reply.addString(last_target);
429  }
430  else
431  {
432  yCError(NAVIGATION2DCLIENT) << "get_last_target failed: goto <location_name> target not found.";
433  reply.addString("not found");
434  }
435  }
436  else if (command.get(0).asString() == "stop_nav")
437  {
438  this->stopNavigation();
439  reply.addString("Stopping movement.");
440  }
441  else if (command.get(0).asString() == "stop_loc")
442  {
443  this->stopLocalizationService();
444  reply.addString("Stopping localization service.");
445  }
446  else if (command.get(0).asString() == "pause_nav")
447  {
448  double time = -1;
449  if (command.size() > 1) {
450  time = command.get(1).asFloat64();
451  }
452  this->suspendNavigation(time);
453  reply.addString("Pausing.");
454  }
455  else if (command.get(0).asString() == "resume_nav")
456  {
457  this->resumeNavigation();
458  reply.addString("Resuming.");
459  }
460  else if (command.get(0).asString() == "start_loc")
461  {
462  this->startLocalizationService();
463  reply.addString("Starting localization service.");
464  }
465  else
466  {
467  yCError(NAVIGATION2DCLIENT) << "Unknown command";
468  reply.addVocab32(VOCAB_ERR);
469  }
470  return true;
471 }

575 bool Navigation2DClient::gotoTargetByLocationName(std::string location_name)
576 {
577  Map2DLocation loc;
578  Map2DArea area;
579 
580  //first of all, ask to the location server if location_name exists as a location_name...
581  bool found = this->getLocation(location_name, loc);
582 
583  //...if found, ok...otherwise check if location_name is an area name instead...
584  if (found == false)
585  {
586  found = this->getArea(location_name, area);
587  if (found)
588  {
589  area.getRandomLocation(loc);
590  }
591  }
592 
593  //...if it is neither a location, nor an area then quit...
594  if (found == false)
595  {
596  yCError(NAVIGATION2DCLIENT) << "Location not found, stopping navigation";
597  this->stopNavigation();
598  return false;
599  }
600 
601  //...otherwise we can go to the found/computed location!
602  yarp::os::Bottle b;
603  yarp::os::Bottle resp;
604 
605  b.addVocab32(VOCAB_INAVIGATION);
606  b.addVocab32(VOCAB_NAV_GOTOABS_AND_NAME);
607  b.addString(loc.map_id);
608  b.addFloat64(loc.x);
609  b.addFloat64(loc.y);
610  b.addFloat64(loc.theta);
611  b.addString(location_name);
612  bool ret = m_rpc_port_to_navigation_server.write(b, resp);
613  if (ret)
614  {
615  if (resp.get(0).asVocab32() != VOCAB_OK)
616  {
617  yCError(NAVIGATION2DCLIENT) << "gotoTargetByLocationName() received error from navigation server";
618  return false;
619  }
620  }
621  else
622  {
623  yCError(NAVIGATION2DCLIENT) << "gotoTargetByLocationName() error on writing on rpc port";
624  return false;
625  }
626 
627  return true;
628 }

662 {

666  yarp::os::Bottle resp_loc;

670  b_nav.addVocab32(VOCAB_NAV_GET_CURRENT_POS);
671  bool ret_nav = m_rpc_port_to_localization_server.write(b_nav, resp_nav);
672  if (ret_nav)
673  {
674  if (resp_nav.get(0).asVocab32() != VOCAB_OK || resp_nav.size()!=5)
675  {
676  yCError(NAVIGATION2DCLIENT) << "storeCurrentPosition() received error from localization server";
677  return false;
678  }
679  else
680  {
681  loc.map_id = resp_nav.get(1).asString();
682  loc.x = resp_nav.get(2).asFloat64();
683  loc.y = resp_nav.get(3).asFloat64();
684  loc.theta = resp_nav.get(4).asFloat64();
685  }
686  }
687  else
688  {
689  yCError(NAVIGATION2DCLIENT) << "storeCurrentPosition() error on writing on rpc port";
690  return false;
691  }
