77

150 void Odometry2D_nws_ros::run()
151 {
152  if (m_odometry2D_interface!=nullptr)
153  {
154  yarp::os::Stamp timeStamp(static_cast<int>(m_stampCount++), yarp::os::Time::now());
155  yarp::dev::OdometryData odometryData;
156  m_odometry2D_interface->getOdometry(odometryData);
157 
158  if (1)
159  {
160  yarp::rosmsg::nav_msgs::Odometry& rosData = rosPublisherPort_odometry.prepare();
161  rosData.header.seq = timeStamp.getCount();
162  rosData.header.stamp = timeStamp.getTime();
163  rosData.header.frame_id = m_odomFrame;
164  rosData.child_frame_id = m_baseFrame;
165 
166  rosData.pose.pose.position.x = odometryData.odom_x;
167  rosData.pose.pose.position.y = odometryData.odom_y;
168  rosData.pose.pose.position.z = 0.0;
169  yarp::rosmsg::geometry_msgs::Quaternion odom_quat;
170  double halfYaw = odometryData.odom_theta * DEG2RAD * 0.5;
171  double cosYaw = cos(halfYaw);
172  double sinYaw = sin(halfYaw);
173  odom_quat.x = 0;
174  odom_quat.y = 0;
175  odom_quat.z = sinYaw;
176  odom_quat.w = cosYaw;
177  rosData.pose.pose.orientation = odom_quat;
178  rosData.twist.twist.linear.x = odometryData.base_vel_x;
179  rosData.twist.twist.linear.y = odometryData.base_vel_y;
180  rosData.twist.twist.linear.z = 0;
181  rosData.twist.twist.angular.x = 0;
182  rosData.twist.twist.angular.y = 0;
183  rosData.twist.twist.angular.z = odometryData.base_vel_theta * DEG2RAD;
184  rosPublisherPort_odometry.write();
185  }
186 
187  if (m_enable_publish_tf)
188  {
189  yarp::rosmsg::tf2_msgs::TFMessage& rosData = rosPublisherPort_tf.prepare();
190  yarp::rosmsg::geometry_msgs::TransformStamped transform;
191  transform.header.frame_id = m_odomFrame;
192  transform.child_frame_id = m_baseFrame;
193  transform.header.seq = timeStamp.getCount();
194  transform.header.stamp = timeStamp.getTime();
195  double halfYaw = odometryData.odom_theta * DEG2RAD * 0.5;
196  double cosYaw = cos(halfYaw);
197  double sinYaw = sin(halfYaw);
198  transform.transform.rotation.x = 0;
199  transform.transform.rotation.y = 0;
200  transform.transform.rotation.z = sinYaw;
201  transform.transform.rotation.w = cosYaw;
202  transform.transform.translation.x = odometryData.odom_x;
203  transform.transform.translation.y = odometryData.odom_y;
204  transform.transform.translation.z = 0;
205  if (rosData.transforms.size() == 0)
206  {
207  rosData.transforms.push_back(transform);
208  }
209  else if (rosData.transforms.size() == 1)
210  {
211  rosData.transforms[0] = transform;
212  }
213  else
214  {
215  yCWarning(ODOMETRY2D_NWS_ROS) << "Size of /tf topic should be 1, instead it is:" << rosData.transforms.size();
216  }
217  rosPublisherPort_tf.write();
218  }
219 
220  } else{
221  yCError(ODOMETRY2D_NWS_ROS) << "the interface is not valid";
222  }
223 }
