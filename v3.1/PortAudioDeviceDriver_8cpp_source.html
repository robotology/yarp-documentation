34 /* This routine will be called by the PortAudio engine when audio is needed.
35 ** It may be called at interrupt level on some machines so don't do anything
36 ** that could mess up the system like calling malloc() or free().
37 */
38 static int bufferIOCallback( const void *inputBuffer, void *outputBuffer,
39  unsigned long framesPerBuffer,
40  const PaStreamCallbackTimeInfo* timeInfo,
41  PaStreamCallbackFlags statusFlags,
42  void *userData )
43 {
44  auto* dataBuffers = static_cast<circularDataBuffers*>(userData);
45  circularBuffer *playdata = dataBuffers->playData;
46  circularBuffer *recdata = dataBuffers->recData;
47  int num_channels = dataBuffers->numChannels;
48  int finished = paComplete;
49 
50  if (dataBuffers->canRec)
51  {
52  const auto* rptr = (const SAMPLE*)inputBuffer;
53  unsigned int framesToCalc;
54  unsigned int i;
55  unsigned long framesLeft = recdata->getMaxSize()-recdata->size();
56 
57  (void) outputBuffer; // just to prevent unused variable warnings
58  (void) timeInfo;
59  (void) statusFlags;
60  (void) userData;
61 
62  if( framesLeft/num_channels < framesPerBuffer )
63  {
64  framesToCalc = framesLeft/num_channels;
65  finished = paComplete;
66  }
67  else
68  {
69  framesToCalc = framesPerBuffer;
70  finished = paContinue;
71  }
72 
73  if( inputBuffer == nullptr )
74  {
75  for( i=0; i<framesToCalc; i++ )
76  {
77  recdata->write(0); // left
78  if( num_channels == 2 ) recdata->write(0); // right
79  }
80  }
81  else
82  {
83  for( i=0; i<framesToCalc; i++ )
84  {
85  recdata->write(*rptr++); // left
86  if( num_channels == 2 ) recdata->write(*rptr++); // right
87  }
88  }
89  //note: you can record or play but not simultaneously (for now)
90  return finished;
91  }
92 
93  if (dataBuffers->canPlay)
94  {
95  auto* wptr = (SAMPLE*)outputBuffer;
96  unsigned int i;
97 
98  unsigned int framesLeft = playdata->size();
99 
100  (void) inputBuffer; // just to prevent unused variable warnings
101  (void) timeInfo;
102  (void) statusFlags;
103  (void) userData;
104 
105  if( framesLeft/num_channels < framesPerBuffer )
106  {
107  // final buffer
108  for( i=0; i<framesLeft/num_channels; i++ )
109  {
110  *wptr++ = playdata->read(); // left
111  if( num_channels == 2 ) *wptr++ = playdata->read(); // right
112  }
113  for( ; i<framesPerBuffer; i++ )
114  {
115  *wptr++ = 0; // left
116  if( num_channels == 2 ) *wptr++ = 0; // right
117  }
118  finished = paComplete;
119  }
120  else
121  {
122  for( i=0; i<framesPerBuffer; i++ )
123  {
124  *wptr++ = playdata->read(); // left
125  if( num_channels == 2 ) *wptr++ = playdata->read(); // right
126  }
127  finished = paContinue;
128  }
129  //note: you can record or play but not simultaneously (for now)
130  return finished;
131  }
132 
133  printf("No read/write operations requested, aborting\n");
134  return paAbort;
135 }

161 bool PortAudioDeviceDriver::open(yarp::os::Searchable& config)
162 {
163  driverConfig.rate = config.check("rate",Value(0),"audio sample rate (0=automatic)").asInt32();
164  driverConfig.samples = config.check("samples",Value(0),"number of samples per network packet (0=automatic). For chunks of 1 second of recording set samples=rate. Channels number is handled internally.").asInt32();
165  driverConfig.channels = config.check("channels",Value(0),"number of audio channels (0=automatic, max is 2)").asInt32();
166  driverConfig.wantRead = (bool)config.check("read","if present, just deal with reading audio (microphone)");
167  driverConfig.wantWrite = (bool)config.check("write","if present, just deal with writing audio (speaker)");
168  driverConfig.deviceNumber = config.check("id",Value(-1),"which portaudio index to use (-1=automatic)").asInt32();
169 
170  if (!(driverConfig.wantRead||driverConfig.wantWrite))
171  {
172  driverConfig.wantRead = driverConfig.wantWrite = true;
173  }
174 
175  if (config.check("loopback","if present, send audio read from microphone immediately back to speaker"))
176  {
177  printf ("WARN: loopback not yet implemented\n");
178  loopBack = true;
179  }
180 
181  if (config.check("render_mode_append"))
182  {
183  renderMode = RENDER_APPEND;
184  }
185  if (config.check("render_mode_immediate"))
186  {
187  renderMode = RENDER_IMMEDIATE;
188  }
189 
190  return open(driverConfig);
191 }

193 bool PortAudioDeviceDriver::open(PortAudioDeviceDriverSettings& config)
194 {
195  int rate = config.rate;
196  int samples = config.samples;
197  int channels = config.channels;
198  bool wantRead = config.wantRead;
199  bool wantWrite = config.wantWrite;
200  int deviceNumber = config.deviceNumber;
201 
202  if (channels==0) channels = DEFAULT_NUM_CHANNELS;
203  numChannels = channels;
204  if (rate==0) rate = DEFAULT_SAMPLE_RATE;
205  frequency = rate;
206 
207  if (samples==0)
208  numSamples = frequency; // by default let's stream chunks of 1 second
209  else
210  numSamples = samples;
211 
212  //buffer.allocate(num_samples*num_channels*sizeof(SAMPLE));
213  numBytes = numSamples * sizeof(SAMPLE) * numChannels;
214  int twiceTheBuffer = numBytes * 2;
215  dataBuffers.numChannels=numChannels;
216  if (dataBuffers.playData==nullptr)
217  dataBuffers.playData = new circularBuffer(twiceTheBuffer);
218  if (dataBuffers.recData==nullptr)
219  dataBuffers.recData = new circularBuffer(twiceTheBuffer);
220  if (wantRead) dataBuffers.canRec = true;
221  if (wantWrite) dataBuffers.canPlay = true;
222 
223  err = Pa_Initialize();
224  if( err != paNoError ) {
225  printf("portaudio system failed to initialize\n");
226  return false;
227  }
228 
229  inputParameters.device = (deviceNumber==-1)?Pa_GetDefaultInputDevice():deviceNumber;
230  printf("Device number %d\n", inputParameters.device);
231  inputParameters.channelCount = numChannels;
232  inputParameters.sampleFormat = PA_SAMPLE_TYPE;
233  if ((Pa_GetDeviceInfo( inputParameters.device ))!=nullptr) {
234  inputParameters.suggestedLatency = Pa_GetDeviceInfo( inputParameters.device )->defaultLowInputLatency;
235  }
236  inputParameters.hostApiSpecificStreamInfo = nullptr;
237 
238  outputParameters.device = (deviceNumber==-1)?Pa_GetDefaultOutputDevice():deviceNumber;
239  outputParameters.channelCount = numChannels;
240  outputParameters.sampleFormat = PA_SAMPLE_TYPE;
241  outputParameters.suggestedLatency = Pa_GetDeviceInfo( outputParameters.device )->defaultLowOutputLatency;
242  outputParameters.hostApiSpecificStreamInfo = nullptr;
243 
244  err = Pa_OpenStream(
245  &stream,
246  wantRead?(&inputParameters):nullptr,
247  wantWrite?(&outputParameters):nullptr,
248  frequency,
249  DEFAULT_FRAMES_PER_BUFFER,
250  paClipOff,
251  bufferIOCallback,
252  &dataBuffers );
253 
254  if( err != paNoError )
255  {
256  fprintf( stderr, "An error occurred while using the portaudio stream\n" );
257  fprintf( stderr, "Error number: %d\n", err );
258  fprintf( stderr, "Error message: %s\n", Pa_GetErrorText( err ) );
259  }
260 
261  //start the thread
262  pThread.stream = stream;
263  pThread.start();
264 
265  return (err==paNoError);
266 }

336 bool PortAudioDeviceDriver::getSound(yarp::sig::Sound& sound)
337 {
338  if (pThread.something_to_record == false)
339  {
340  this->startRecording();
341  }
342 
343  int buff_size = 0;
344  static int buff_size_wdt = 0;
345  while (buff_size<this->numSamples*this->numChannels)
346  {
347  buff_size = dataBuffers.recData->size();
348  if (buff_size == 0 && buff_size_wdt++ == 100) break;
349  yarp::os::SystemClock::delaySystem(SLEEP_TIME);
350  }
351  buff_size_wdt = 0;
352 
353  if (sound.getChannels()!=this->numChannels && sound.getSamples() != this->numSamples)
354  {
355  sound.resize(this->numSamples,this->numChannels);
356  }
357  sound.setFrequency(this->driverConfig.rate);
358 
359  for (int i=0; i<this->numSamples; i++)
360  for (int j=0; j<this->numChannels; j++)
361  {
362  SAMPLE s = dataBuffers.recData->read();
363  sound.set(s,i,j);
364  }
365  return true;
366 }

446 bool PortAudioDeviceDriver::renderSound(yarp::sig::Sound& sound)
447 {
448  int freq = sound.getFrequency();
449  int chans = sound.getChannels();
450  if (freq != this->frequency ||
451  chans != this->numChannels)
452  {
453  //wait for current playback to finish
454  while (Pa_IsStreamStopped(stream )==0)
455  {
456  yarp::os::Time::delay(SLEEP_TIME);
457  }
458 
459  //reset the driver
460  printf("***** driver configuration changed, resetting *****\n");
461  this->close();
462  driverConfig.channels = chans;
463  driverConfig.rate = freq;
464  bool ok = open(driverConfig);
465  if (ok == false)
466  {
467  printf("error occurred during driver reconfiguration, aborting\n");
468  return false;
469  }
470  }
471 
472  if (renderMode == RENDER_IMMEDIATE)
473  return immediateSound(sound);
474  else if (renderMode == RENDER_APPEND)
475  return appendSound(sound);
476 
477  return false;
478 }
