149  int status = 0; /* return value: 0 = success */
150  zfp_type type; /* array scalar type */
151  zfp_field* field; /* array meta data */
152  zfp_stream* zfp; /* compressed stream */
153  size_t bufsize; /* byte size of compressed buffer */
154  bitstream* stream; /* bit stream to write to or read from */
155 
156  type = zfp_type_float;
157  field = zfp_field_2d(array, type, nx, ny);
158 
159  /* allocate meta data for a compressed stream */
160  zfp = zfp_stream_open(NULL);
161 
162  /* set compression mode and parameters via one of three functions */
163  /* zfp_stream_set_rate(zfp, rate, type, 3, 0); */
164  /* zfp_stream_set_precision(zfp, precision); */
165  zfp_stream_set_accuracy(zfp, tolerance);
166

168  bufsize = zfp_stream_maximum_size(zfp, field);
169 
170  resizeV(buffer,bufsize);
171 
172  /* associate bit stream with allocated buffer */
173  stream = stream_open(buffer, bufsize);
174  zfp_stream_set_bit_stream(zfp, stream);
175  zfp_stream_rewind(zfp);
176 
177  /* compress entire array */
178  /* compress array and output compressed stream */
179  zfpsize = zfp_compress(zfp, field);
180  if (!zfpsize) {
181  fprintf(stderr, "compression failed\n");
182  status = 1;
183  }
184 
185  resizeF(compressed,zfpsize);
186  memcpy(compressed,(float*) stream_data(zfp->stream),zfpsize);
187 
188  /* clean up */
189  zfp_field_free(field);
190  zfp_stream_close(zfp);
191  stream_close(stream);
