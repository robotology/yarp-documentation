26 void finalizeMPI() {
27  if (MpiControl) {
28  MpiControl->finalize();
29  delete MpiControl;
30  MpiControl = nullptr;
31  }
32  int ct = 0;
33  int finalized;
34  while (ct < 5) {
35  sleep(1);
36  MPI_Finalized(&finalized);
37  if (finalized) {
38  return;
39  }
40  ct++;
41  }
42  yError("MpiControlThread: Finalizing MPI failed! Calling MPI_Abort");
43  MPI_Abort(MPI_COMM_WORLD,1);
44 }

52 bool MpiControlThread::threadInit() {
53  // We have to finalize MPI at process termination
54  atexit(finalizeMPI);
55 
56  #ifdef MPI_DEBUG
57  printf("[MpiControl] Initialize\n");
58  #endif
59 
60  int provided;
61  // We need full multithread support for MPI
62  int requested = MPI_THREAD_MULTIPLE;
63  // Passing NULL for argc/argv pointers is fine for MPI-2
64  int err = MPI_Init_thread(nullptr, nullptr, requested , &provided);
65  if (err != MPI_SUCCESS ) {
66  yError("MpiControlThread: Couldn't initialize MPI\n");
67  return false;
68  }
69 
70  if (provided >= requested) {
71  return true;
72  }
73  else {
74  MPI_Finalize();
75  yError("MpiControlThread: MPI implementation doesn't provide required thread safety: requested %s, provided %s", NetType::toString(requested).c_str(), NetType::toString(provided).c_str());
76  return false;
77  }
78 }
