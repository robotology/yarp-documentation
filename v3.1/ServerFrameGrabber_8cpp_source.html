57 bool ServerFrameGrabber::open(yarp::os::Searchable& config) {
58  if (active) {
59  yError("Did you just try to open the same ServerFrameGrabber twice?\n");
60  return false;
61  }
62 
63  // for AV, control whether output goes on a single port or multiple
64  bool separatePorts = false;
65 
66  p.setReader(*this);
67 
68  yarp::os::Value *name;
69 
70  if (config.check("subdevice",name,"name (or nested configuration) of device to wrap")) {
71  if (name->isString()) {
72  // maybe user isn't doing nested configuration
73  yarp::os::Property p;
74  p.setMonitor(config.getMonitor(),
75  name->toString().c_str()); // pass on any monitoring
76  p.fromString(config.toString());
77  p.put("device",name->toString());
78  p.unput("subdevice");
79  poly.open(p);
80  } else {
81  Bottle subdevice = config.findGroup("subdevice").tail();
82  poly.open(subdevice);
83  }
84  if (!poly.isValid()) {
85  //yError("cannot make <%s>\n", name->toString().c_str());
86  return false;
87  }
88  } else {
89  yError("\"--subdevice <name>\" not set for server_framegrabber\n");
90  return false;
91  }
92  if (poly.isValid()) {
93  IAudioVisualStream *str;
94  poly.view(str);
95  bool a = true;
96  bool v = true;
97  bool vraw = true;
98  if (str!=nullptr) {
99  a = str->hasAudio();
100  v = str->hasVideo();
101  vraw = str->hasRawVideo();
102  }
103  if (v) {
104  poly.view(fgImage);
105  }
106  if (vraw) {
107  poly.view(fgImageRaw);
108  }
109  if (a) {
110  poly.view(fgSound);
111  }
112  if (a&&v) {
113  poly.view(fgAv);
114  }
115  poly.view(fgCtrl);
116  if(fgCtrl)
117  ifgCtrl_Parser.configure(fgCtrl);
118  poly.view(fgTimed);
119  poly.view(rgbVis_p);
120 
121  bool conf = rgbParser.configure(rgbVis_p);
122 
123  if(!conf)
124  {
125  yWarning() << "ServerFrameGrabber: error configuring interfaces for parsers";
126  }
127  }
128 
129  canDrop = !config.check("no_drop","if present, use strict policy for sending data");
130  addStamp = config.check("stamp","if present, add timestamps to data");
131 
132  p.promiseType(Type::byName("yarp/image")); // TODO: reflect audio options
133  p.setWriteOnly();
134  p.open(config.check("name",Value("/grabber"),
135  "name of port to send data on").asString());
136 
137  /*
138  double framerate=0;
139  if (config.check("framerate", name,
140  "maximum rate in Hz to read from subdevice")) {
141  framerate=name->asFloat64();
142  }
143  */
144 
145  if (fgAv&&
146  !config.check("shared-ports",
147  "If present, send audio and images on same port")) {
148  separatePorts = true;
149  yAssert(p2==nullptr);
150  p2 = new Port;
151  yAssert(p2!=nullptr);
152  p2->open(config.check("name2",Value("/grabber2"),
153  "Name of second port to send data on, when audio and images sent separately").asString());
154  }
155 
156  if (fgAv!=nullptr) {
157  if (separatePorts) {
158  yAssert(p2!=nullptr);
159  thread.attach(new DataWriter2<yarp::sig::ImageOf<yarp::sig::PixelRgb>, yarp::sig::Sound>(p,*p2,*this,canDrop,addStamp));
160  } else {
161  thread.attach(new DataWriter<ImageRgbSound>(p,*this,canDrop,
162  addStamp));
163  }
164  } else if (fgImage!=nullptr) {
165  thread.attach(new DataWriter<yarp::sig::ImageOf<yarp::sig::PixelRgb> >(p,*this,canDrop,addStamp,fgTimed));
166  } else if (fgImageRaw!=nullptr) {
167  thread.attach(new DataWriter<yarp::sig::ImageOf<yarp::sig::PixelMono> >(p,*this,canDrop,addStamp,fgTimed));
168  } else if (fgSound!=nullptr) {
169  thread.attach(new DataWriter<yarp::sig::Sound>(p,*this,canDrop));
170  } else {
171  yError("subdevice <%s> doesn't look like a framegrabber\n",
172  name->toString().c_str());
173  return false;
174  }
175 
176  singleThreaded =
177  config.check("single_threaded",
178  "if present, operate in single threaded mode")!=0;
179  thread.open(config.check("framerate",Value("0"),
180  "maximum rate in Hz to read from subdevice").asFloat64(),
181  singleThreaded);
182  active = true;
183

185  DeviceResponder::makeUsage();
186  addUsage("[set] [bri] $fBrightness", "set brightness");
187  addUsage("[set] [expo] $fExposure", "set exposure");
188  addUsage("[set] [shar] $fSharpness", "set sharpness");
189  addUsage("[set] [whit] $fBlue $fRed", "set white balance");

198  addUsage("[get] [expo]", "get exposure");

200  addUsage("[get] [whit]", "get white balance");

214 bool ServerFrameGrabber::respond(const yarp::os::Bottle& cmd,
215  yarp::os::Bottle& response) {
216  int code = cmd.get(0).asVocab();
217 
218  auto* fgCtrlDC1394=dynamic_cast<IFrameGrabberControlsDC1394*>(fgCtrl);
219 
220  switch (code)
221  {
222  // first check if requests are coming from new iFrameGrabberControl2 interface and process them
223  case VOCAB_FRAMEGRABBER_CONTROL:
224  {
225  return ifgCtrl_Parser.respond(cmd, response); // I don't like all those returns everywhere!!! :-(
226  } break;
227  case VOCAB_RGB_VISUAL_PARAMS:
228  {
229  return rgbParser.respond(cmd,response);
230  } break;
232  // DC1394 COMMANDS
234  case VOCAB_FRAMEGRABBER_CONTROL_DC1394:
235  {
236  if (fgCtrlDC1394)
237  {
238  int codeDC1394 = cmd.get(1).asVocab();
239  switch(codeDC1394)
240  {
241  case VOCAB_DRGETMSK: // VOCAB_DRGETMSK 12
242  response.addInt32(int(fgCtrlDC1394->getVideoModeMaskDC1394()));
243  return true;
244  case VOCAB_DRGETVMD: // VOCAB_DRGETVMD 13
245  response.addInt32(int(fgCtrlDC1394->getVideoModeDC1394()));
246  return true;
247  case VOCAB_DRSETVMD: // VOCAB_DRSETVMD 14
248  response.addInt32(int(fgCtrlDC1394->setVideoModeDC1394(cmd.get(1).asInt32())));
249  return true;
250  case VOCAB_DRGETFPM: // VOCAB_DRGETFPM 15
251  response.addInt32(int(fgCtrlDC1394->getFPSMaskDC1394()));
252  return true;
253  case VOCAB_DRGETFPS: // VOCAB_DRGETFPS 16
254  response.addInt32(int(fgCtrlDC1394->getFPSDC1394()));
255  return true;
256  case VOCAB_DRSETFPS: // VOCAB_DRSETFPS 17
257  response.addInt32(int(fgCtrlDC1394->setFPSDC1394(cmd.get(1).asInt32())));
258  return true;
259 
260  case VOCAB_DRGETISO: // VOCAB_DRGETISO 18
261  response.addInt32(int(fgCtrlDC1394->getISOSpeedDC1394()));
262  return true;
263  case VOCAB_DRSETISO: // VOCAB_DRSETISO 19
264  response.addInt32(int(fgCtrlDC1394->setISOSpeedDC1394(cmd.get(1).asInt32())));
265  return true;
266 
267  case VOCAB_DRGETCCM: // VOCAB_DRGETCCM 20
268  response.addInt32(int(fgCtrlDC1394->getColorCodingMaskDC1394(cmd.get(1).asInt32())));
269  return true;
270  case VOCAB_DRGETCOD: // VOCAB_DRGETCOD 21
271  response.addInt32(int(fgCtrlDC1394->getColorCodingDC1394()));
272  return true;
273  case VOCAB_DRSETCOD: // VOCAB_DRSETCOD 22
274  response.addInt32(int(fgCtrlDC1394->setColorCodingDC1394(cmd.get(1).asInt32())));
275  return true;
276  case VOCAB_DRGETF7M: // VOCAB_DRGETF7M 25
277  {
278  unsigned int xstep,ystep,xdim,ydim,xoffstep,yoffstep;
279  fgCtrlDC1394->getFormat7MaxWindowDC1394(xdim,ydim,xstep,ystep,xoffstep,yoffstep);
280  response.addInt32(xdim);
281  response.addInt32(ydim);
282  response.addInt32(xstep);
283  response.addInt32(ystep);
284  response.addInt32(xoffstep);
285  response.addInt32(yoffstep);
286  }
287  return true;
288  case VOCAB_DRGETWF7: // VOCAB_DRGETWF7 26
289  {
290  unsigned int xdim,ydim;
291  int x0,y0;
292  fgCtrlDC1394->getFormat7WindowDC1394(xdim,ydim,x0,y0);
293  response.addInt32(xdim);
294  response.addInt32(ydim);
295  response.addInt32(x0);
296  response.addInt32(y0);
297  }
298  return true;
299  case VOCAB_DRSETWF7: // VOCAB_DRSETWF7 27
300  response.addInt32(int(fgCtrlDC1394->setFormat7WindowDC1394(cmd.get(1).asInt32(),cmd.get(2).asInt32(),cmd.get(3).asInt32(),cmd.get(4).asInt32())));
301  return true;
302  case VOCAB_DRSETOPM: // VOCAB_DRSETOPM 28
303  response.addInt32(int(fgCtrlDC1394->setOperationModeDC1394(cmd.get(1).asInt32()!=0)));
304  return true;
305  case VOCAB_DRGETOPM: // VOCAB_DRGETOPM 29
306  response.addInt32(fgCtrlDC1394->getOperationModeDC1394());
307  return true;
308 
309  case VOCAB_DRSETTXM: // VOCAB_DRSETTXM 30
310  response.addInt32(int(fgCtrlDC1394->setTransmissionDC1394(cmd.get(1).asInt32()!=0)));
311  return true;
312  case VOCAB_DRGETTXM: // VOCAB_DRGETTXM 31
313  response.addInt32(fgCtrlDC1394->getTransmissionDC1394());
314  return true;
315  /*
316  case VOCAB_DRSETBAY: // VOCAB_DRSETBAY 32
317  response.addInt32(int(fgCtrlDC1394->setBayerDC1394(bool(cmd.get(1).asInt32()))));
318  return true;
319  case VOCAB_DRGETBAY: // VOCAB_DRGETBAY 33
320  response.addInt32(fgCtrlDC1394->getBayerDC1394());
321  return true;
322  */
323  case VOCAB_DRSETBCS: // VOCAB_DRSETBCS 34
324  response.addInt32(int(fgCtrlDC1394->setBroadcastDC1394(cmd.get(1).asInt32()!=0)));
325  return true;
326  case VOCAB_DRSETDEF: // VOCAB_DRSETDEF 35
327  response.addInt32(int(fgCtrlDC1394->setDefaultsDC1394()));
328  return true;
329  case VOCAB_DRSETRST: // VOCAB_DRSETRST 36
330  response.addInt32(int(fgCtrlDC1394->setResetDC1394()));
331  return true;
332  case VOCAB_DRSETPWR: // VOCAB_DRSETPWR 37
333  response.addInt32(int(fgCtrlDC1394->setPowerDC1394(cmd.get(1).asInt32()!=0)));
334  return true;
335  case VOCAB_DRSETCAP: // VOCAB_DRSETCAP 38
336  response.addInt32(int(fgCtrlDC1394->setCaptureDC1394(cmd.get(1).asInt32()!=0)));
337  return true;
338  case VOCAB_DRSETBPP: // VOCAB_DRSETCAP 39
339  response.addInt32(int(fgCtrlDC1394->setBytesPerPacketDC1394(cmd.get(1).asInt32())));
340  return true;
341  case VOCAB_DRGETBPP: // VOCAB_DRGETTXM 40
342  response.addInt32(fgCtrlDC1394->getBytesPerPacketDC1394());
343  return true;
344  default:
345  return DeviceResponder::respond(cmd,response);
346  }
347  }
348  }
349  }
350  yError() << "ServerFrameGrabber: command not recognized" << cmd.toString();
351  return DeviceResponder::respond(cmd,response);
352 }
