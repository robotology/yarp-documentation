53 bool MpiCarrier::sendHeader(ConnectionState& proto) {
54  // Send the "magic number" for this carrier
55  ManagedBytes header(8);
56  getHeader(header.bytes());
57  proto.os().write(header.bytes());
58  if (!proto.os().isOk()) return false;
59 
60  // Now we can do whatever we want, as long as somehow
61  // we also send the name of the originating port
62 
63  name = proto.getRoute().getFromName();
64  other = proto.getRoute().getToName();
65  Bytes b2((char*)name.c_str(),name.length());
66  proto.os().write(b2);
67  proto.os().write('\r');
68  proto.os().write('\n');
69 
70  // Sender
71  route = name + "->" + other;
72 
73  createStream(true);
74 
75  if (!MpiControl) return false;
76  if (! MpiControl->isRunning())
77  return false;
78  comm->openPort();
79  char* port = comm->port_name;
80  char* uid = comm->unique_id;
81 
82  #ifdef MPI_DEBUG
83  printf("[MpiCarrier @ %s] setting up MpiPort '%s'\n", route.c_str(), port);
84  #endif
85 
86  Bytes b4(uid,strlen(uid));
87  proto.os().write(b4);
88  proto.os().write('\r');
89  proto.os().write('\n');
90 
91  Bytes b3(port,strlen(port));
92  proto.os().write(b3);
93  proto.os().write('\r');
94  proto.os().write('\n');
95  proto.os().flush();
96 
97 
98  #ifdef MPI_DEBUG
99  printf("[MpiCarrier @ %s] Header sent\n", route.c_str());
100  #endif
101 
102  return proto.os().isOk();
103 }
