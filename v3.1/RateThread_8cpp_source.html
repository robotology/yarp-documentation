219 bool RateThreadWrapper::open(double framerate, bool polling)
220 {
221  double period = 0.0;
222  if (framerate>0) {
223  period=(1.0/framerate);
224  YARP_SPRINTF2(Logger::get(), info,
225  "Setting framerate to: %.0lf[Hz] (thread period %f[s])\n",
226  framerate, period);
227  } else {
228  YARP_SPRINTF0(Logger::get(), info,
229  "No framerate specified, polling the device");
230  period=0.0; //continuous
231  }
232  PeriodicThread::setPeriod(period);
233  if (!polling) {
234  start();
235  }
236  return true;
237 }
