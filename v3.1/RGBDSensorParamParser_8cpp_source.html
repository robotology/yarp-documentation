16 static bool checkParam(const Bottle& input, RGBDSensorParamParser::RGBDParam& param, bool& found)
17 {
18  bool ret = false;
19  Bottle bt=input.findGroup(param.name).tail(); // the first element is the name of the parameter
20 
21  if (!bt.isNull())
22  {
23  Bottle* b;
24  if (param.size>1 && bt.size()==1)
25  {
26  b = bt.get(0).asList();
27  }
28  else
29  b = &bt;
30  if (b->isNull())
31  {
32  yError()<<"RGBDSensorParamParser: check"<<param.name<<"in config file";
33  return false;
34  }
35  if (b->size() != param.size)
36  {
37  yError() << "RGBDSensorParamParser: parameter" << param.name << "size should be" << param.size;
38  return false;
39  }
40  param.val.resize(param.size);
41  for (size_t i=0;i<b->size();i++)
42  {
43  ret = true;
44  param.val[i] = b->get(i);
45  found = true;
46  }
47  }
48  else
49  {
50  ret = true;
51  found = false;
52  }
53  return ret;
54 }

56 static bool checkParam(const Bottle& settings, const Bottle& description, RGBDSensorParamParser::RGBDParam &param)
57 {
58  bool ret1, ret2, ret3;
59 
60  ret3 = true;
61  ret1 = checkParam(settings, param, param.isSetting); // look for settings
62  ret2 = checkParam(description, param, param.isDescription);// look for HW_DESCRIPTION
63 
64  if ( (param.isSetting) && (param.isDescription) )
65  {
66  yError() << "Setting " << param.name << " can either be a 'SETTING' or 'HW_DESCRIPTION', not both. Fix the config file. \
67  Look for documentation online.";
68  ret3 = false;
69  }
70  return (ret1 && ret2 && ret3);
71 }

74 static bool parseIntrinsic(const Searchable& config, const string& groupName, RGBDSensorParamParser::IntrinsicParams &params)
75 {
76 
77  pair<string, double*> realparam;
78  vector<pair<string, double*> > realParams;
79  size_t i;
80 
81  if (!config.check(groupName))
82  {
83  if (params.isOptional)
84  {
85  return true;
86  }
87  else
88  {
89  return false;
90  }
91  }
92  Bottle& intrinsic = config.findGroup(groupName);
93 
94  realparam.first = "focalLengthX"; realparam.second = &params.focalLengthX; realParams.push_back(realparam);
95  realparam.first = "focalLengthY"; realparam.second = &params.focalLengthY; realParams.push_back(realparam);
96  realparam.first = "principalPointX"; realparam.second = &params.principalPointX; realParams.push_back(realparam);
97  realparam.first = "principalPointY"; realparam.second = &params.principalPointY; realParams.push_back(realparam);
98 
99  for(i = 0; i < realParams.size(); i++)
100  {
101  if (!intrinsic.check(realParams[i].first))
102  {
103  yError() << "RGBDSensorParamParser: missing" << realParams[i].first << "param in" << groupName << "group in the configuration file";
104  return false;
105  }
106 
107  *(realParams[i].second) = intrinsic.find(realParams[i].first).asFloat64();
108  }
109 
110  if (!intrinsic.check("distortionModel"))
111  {
112  yError() << "missing distortionModel param in configuration";
113  return false;
114  }
115 
116  if (!config.check(intrinsic.find("distortionModel").asString()))
117  {
118  yError() << "missing" << intrinsic.find("distortionModel").asString() << "group in configuration file";
119  return false;
120  }
121 
122  Bottle& distortion = config.findGroup(intrinsic.find("distortionModel").asString());
123 
124  if (!distortion.check("name"))
125  {
126  yError() << "RGBDSensorParamParser: missing name param in" << config.find("distortionModel").asString() << "group in configuration file";
127  return false;
128  }
129  if (distortion.find("name").asString() != "plumb_bob")
130  {
131  yError() << "RGBDSensorParamParser: only plumb_bob distortion model is supported at the moment";
132  return false;
133  }
134 
135  realParams.clear();
136  realparam.first = "k1"; realparam.second = &params.distortionModel.k1; realParams.push_back(realparam);
137  realparam.first = "k2"; realparam.second = &params.distortionModel.k2; realParams.push_back(realparam);
138  realparam.first = "t1"; realparam.second = &params.distortionModel.t1; realParams.push_back(realparam);
139  realparam.first = "t2"; realparam.second = &params.distortionModel.t2; realParams.push_back(realparam);
140  realparam.first = "k3"; realparam.second = &params.distortionModel.k3; realParams.push_back(realparam);
141 
142  for(i = 0; i < realParams.size(); i++)
143  {
144  if (!distortion.check(realParams[i].first))
145  {
146  yError() << "RGBDSensorParamParser: missing" << realParams[i].first << "param in" << intrinsic.find("distortionModel").asString() << "group in the configuration file";
147  return false;
148  }
149  *(realParams[i].second) = distortion.find(realParams[i].first).asFloat64();
150  }
151 
152  return true;
153 }

155 bool RGBDSensorParamParser::parseParam(Searchable &config, std::vector<RGBDParam*>& params)
156 {
157  bool ret = true;
158 
159  if (!config.check("SETTINGS"))
160  {
161  yError() << "RGBDSensorParamParser: missing SETTINGS section on the configuration file";
162  return false;
163  }
164 
165  Bottle& settings = config.findGroup("SETTINGS");
166 
167  if (!config.check("HW_DESCRIPTION"))
168  {
169  yError() << "RGBDSensorParamParser: missing HW_DESCRIPTION section on the configuration file";
170  return false;
171  }
172 
173  Bottle& description = config.findGroup("HW_DESCRIPTION");
174 
175 
176  for (auto& v: params)
177  {
178  if (!checkParam(settings, description, *v) ) {ret = false;}
179  }
180 
181  if (!ret)
182  {
183  yError() << "depthCamera driver input file not correct, please fix it!";
184  return false;
185  }
186 
187  if (!parseIntrinsic(config, "RGB_INTRINSIC_PARAMETERS", rgbIntrinsic))
188  {
189  yError() << "RGBDSensorParamParser: incomplete or missing RGB_INTRINSIC_PARAMETERS section on the configuration file";
190  return false;
191  }
192 
193 
194  if (!parseIntrinsic(config, "DEPTH_INTRINSIC_PARAMETERS", depthIntrinsic))
195  {
196  yError() << "RGBDSensorParamParser: incomplete or missing DEPTH_INTRINSIC_PARAMETERS section on the configuration file";
197  return false;
198  }
199 
200  if (!config.check("EXTRINSIC_PARAMETERS"))
201  {
202  if (!isOptionalExtrinsic)
203  {
204  yError() << "RGBDSensorParamParser: missing EXTRINSIC_PARAMETERS section on the configuration file";
205  return false;
206  }
207  else
208  {
209  return true;
210  }
211  }
212  else
213  {
214 
215  Bottle& extrinsic = config.findGroup("EXTRINSIC_PARAMETERS");
216 
217  if (!extrinsic.check("transformation"))
218  {
219  yError() << "RGBDSensorParamParser: missing transformation parameter under EXTRINSIC_PARAMETERS group in configuration file";
220  return false;
221  }
222 
223  Bottle transformation = extrinsic.findGroup("transformation").tail();
224  Bottle* tf;
225 
226  if (transformation.size()==1)
227  {
228  tf = transformation.get(0).asList();
229  }
230  else
231  {
232  tf=&transformation;
233  }
234  if (!(tf->size() == 4*4))
235  {
236  yError() << "RGBDSensorParamParser: the size of the transformation matrix is wrong";
237  return false;
238  }
239 
240  for(int i = 0; i < 4; i++)
241  {
242  for(int j = 0; j < 4; j++)
243  {
244  int k = i*4+j;
245  Value& v = tf->get(k);
246  if (!v.isFloat64())
247  {
248  yError() << "wrong data format on transformation matrix (position" << k << ")";
249  return false;
250  }
251  transformationMatrix[i][j] = v.asFloat64();
252  }
253  }
254  }
255 
256  return ret;
257 }
