81 bool BoschIMU::open(yarp::os::Searchable& config)
82 {
83  //debug
84  yTrace("Parameters are:\n\t%s", config.toString().c_str());
85 
86  if(!config.check("comport") && !config.check("i2c"))
87  {
88  yError() << "BoshImu: Params 'comport' and 'i2c' not found";
89  return false;
90  }
91 
92  if (config.check("comport") && config.check("i2c"))
93  {
94  yError() << "BoshImu: Params 'comport' and 'i2c' both specified";
95  return false;
96  }
97 
98  i2c_flag = config.check("i2c");
99 
100  readFunc = i2c_flag ? &BoschIMU::sendReadCommandI2c : &BoschIMU::sendReadCommandSer;
101 
102  // In case of reading through serial the first two bytes of the response are the ack, so
103  // they need to be discarded when publishing the data.
104  responseOffset = i2c_flag ? 0 : 2;
105 
106  if (i2c_flag)
107  {
108  if (!config.find("i2c").isString())
109  {
110  yError()<<"BoshImu: i2c param malformed, it should be a string, aborting.";
111  return false;
112  }
113 
114  std::string i2cDevFile = config.find("i2c").asString();
115  fd = ::open(i2cDevFile.c_str(), O_RDWR);
116 
117  if (fd < 0)
118  {
119  yError("BoshImu: can't open %s, %s", i2cDevFile.c_str(), strerror(errno));
120  return false;
121  }
122 
123  if (::ioctl(fd, I2C_SLAVE, i2cAddrA) < 0)
124  {
125  yError("BoshImu: ioctl failed on %s, %s", i2cDevFile.c_str(), strerror(errno));
126  return false;
127  }
128 
129  }
130  else
131  {
132  fd = ::open(config.find("comport").toString().c_str(), O_RDWR | O_NOCTTY );
133  if (fd < 0) {
134  yError("can't open %s, %s", config.find("comport").toString().c_str(), strerror(errno));
135  return false;
136  }
137  //Get the current options for the port...
138  struct termios options;
139  tcgetattr(fd, &options);
140 
141  cfmakeraw(&options);
142 
143  //set the baud rate to 115200
144  int baudRate = B115200;
145  cfsetospeed(&options, baudRate);
146  cfsetispeed(&options, baudRate);
147 
148  //set the number of data bits.
149  options.c_cflag &= ~CSIZE; // Mask the character size bits
150  options.c_cflag |= CS8;
151 
152  //set the number of stop bits to 1
153  options.c_cflag &= ~CSTOPB;
154 
155  //Set parity to None
156  options.c_cflag &=~PARENB;
157 
158  //set for non-canonical (raw processing, no echo, etc.)
159  // options.c_iflag = IGNPAR; // ignore parity check
160  options.c_oflag = 0; // raw output
161  options.c_lflag = 0; // raw input
162 
163  // SET NOT BLOCKING READ
164  options.c_cc[VMIN] = 0; // block reading until RX x characters. If x = 0, it is non-blocking.
165  options.c_cc[VTIME] = 2; // Inter-Character Timer -- i.e. timeout= x*.1 s
166 
167  //Set local mode and enable the receiver
168  options.c_cflag |= (CLOCAL | CREAD);
169 
170  tcflush(fd, TCIOFLUSH);
171 
172  //Set the new options for the port...
173  if ( tcsetattr(fd, TCSANOW, &options) != 0)
174  {
175  yError("Configuring comport failed");
176  return false;
177  }
178 
179  }
180 
181  nChannels = config.check("channels", Value(12)).asInt32();
182 
183  double period = config.check("period",Value(10),"Thread period in ms").asInt32() / 1000.0;
184  setPeriod(period);
185 
186  return PeriodicThread::start();
187 }

197 bool BoschIMU::checkReadResponse(unsigned char* response)
198 {
199  if(response[0] == (unsigned char) REPLY_HEAD)
200  {
201  return true;
202  }
203 
204  if(response[0] == (unsigned char) ERROR_HEAD)
205  {
206  if(response[1] != REGISTER_NOT_READY) // if error is 0x07, do not print error messages
207  {
208  yError("Bosch BNO055 IMU - Inertial sensor didn't understand the command. \n\
209  If this error happens more than once in a row, please check serial communication is working fine and it isn't affected by electrical disturbance.");
210  }
211  errorCounter[response[1]]++;
212  readSysError();
213  return false;
214  }
215 
216  errorCounter[0]++;
217  yError("Bosch BNO055 IMU - Received unknown response message. \n\
218  If this error happens more than once in a row, please check serial communication is working fine and it isn't affected by electrical disturbance.");
219  dropGarbage();
220  readSysError();
221  return false;
222 }

250 bool BoschIMU::sendReadCommandSer(unsigned char register_add, int len, unsigned char* buf, std::string comment)
251 {
252  int command_len;
253  int nbytes_w;
254 
255  //
256  // Create a READ message
257  //
258  bool success = false;
259  for(int trials=0; (trials<3) && (success==false); trials++)
260  {
261  totMessagesRead++;
262  command_len = 4;
263  command[0]= START_BYTE; // start byte
264  command[1]= READ_CMD; // read operation
265  command[2]= register_add; // register to read
266  command[3]= len; // length in bytes
267 
268 // printf("> READ_COMMAND: %s ... ", comment.c_str());
269 // printf("\nCommand is:\n");
270 // printBuffer(command, command_len);
271 
272  nbytes_w = ::write(fd, (void*)command, command_len);
273 
274  if(nbytes_w != command_len)
275  {
276  yError() << "BoschIMU device cannot correctly send the message: " << comment;
277  // DO NOT return here. If something was sent, then the imu will reply with a message
278  // even an error message. I have to parse it before proceeding and not leave garbage behind.
279  }
280  // Read the write reply
281  memset(buf, 0x00, 20);
282  int readbytes = readBytes(buf, RESP_HEADER_SIZE);
283  if(readbytes != RESP_HEADER_SIZE)
284  {
285  yError("Expected %d bytes, read %d instead\n", RESP_HEADER_SIZE, readbytes);
286  success = false;
287  }
288  else if(!checkReadResponse(buf))
289  {
290  success = false;
291  yarp::os::SystemClock::delaySystem(0.002);
292  }
293  else
294  {
295  success = true;
296 // printf("> SUCCESS!\n"); fflush(stdout);
297 
298  // Read the data payload
299  readBytes(&buf[2], (int) buf[1]);
300 // printf("\tReply is:\n");
301 // printBuffer(buf, buf[1]+2);
302 // printf("***************\n");
303  }
304  }
305 // if(!success)
306 // yError("> FAILED reading %s!\n", comment.c_str());
307  return success;
308 }

311 {
312  int command_len = 4+len;
313  int nbytes_w;
314 
315  command[0]= START_BYTE; // start byte
316  command[1]= WRITE_CMD; // read operation
317  command[2]= register_add; // operation mode register
318  command[3]= (unsigned char) len; // length 1 byte
319  for(int i=0; i<len; i++)
320  command[4+i] = cmd[i]; // data
321 
322 // printf("> WRITE_COMMAND: %s ... ", comment.c_str());
323 // printf("\nCommand is:\n");
324 // printBuffer(command, command_len);
325 
326  nbytes_w = ::write(fd, (void*)command, command_len);

331  // even an error message. I have to parse it before proceeding and not leave garbage behind.
332  }
333 
334  // Read the write reply
335  memset(response, 0x00, 20);
336  readBytes(response, 2);
337  if(!checkWriteResponse(response))
338  {
339 // printf("> FAILED!\n"); fflush(stdout);
340  yError() << "FAILED writing " << comment;

577 void BoschIMU::run()
578 {
579  timeStamp = yarp::os::SystemClock::nowSystem();
580 
581  int16_t raw_data[16];
582 
583  // In order to avoid zeros when a single read from a sensor is missing,
584  // initialize the new measure to be equal to the previous one
585  data_tmp = data;
586 
587 
588  if (!(this->*readFunc)(REG_ACC_DATA, 32, response, "Read all"))
589  {
590  yError()<<"BoshImu: failed to read all the data";
591  errs++;
592  }
593  else
594  {
595  // Correctly construct int16 data
596  for(int i=0; i<16; i++)
597  {
598  raw_data[i] = response[responseOffset+1+i*2] << 8 | response[responseOffset+i*2];
599  }
600 
601  // Get quaternion
602  quaternion_tmp = quaternion;
603  quaternion_tmp.w() = ((double)raw_data[12]) / (2 << 13);
604  quaternion_tmp.x() = ((double)raw_data[13]) / (2 << 13);
605  quaternion_tmp.y() = ((double)raw_data[14]) / (2 << 13);
606  quaternion_tmp.z() = ((double)raw_data[15]) / (2 << 13);
607 
608  // Convert to RPY angles
609  RPY_angle.resize(3);
610  RPY_angle = yarp::math::dcm2rpy(quaternion.toRotationMatrix4x4());
611  data_tmp[0] = RPY_angle[0] * 180 / M_PI;
612  data_tmp[1] = RPY_angle[1] * 180 / M_PI;
613  data_tmp[2] = RPY_angle[2] * 180 / M_PI;
614 
615  // Fill in accel values
616  data_tmp[3] = (double)raw_data[0] / 100.0;
617  data_tmp[4] = (double)raw_data[1] / 100.0;
618  data_tmp[5] = (double)raw_data[2] / 100.0;
619 
620 
621  // Fill in Gyro values
622  data_tmp[6] = (double)raw_data[6] / 100.0;
623  data_tmp[7] = (double)raw_data[7] / 100.0;
624  data_tmp[8] = (double)raw_data[8] / 100.0;
625 
626  // Fill in Magnetometer values
627  data_tmp[9] = (double)raw_data[3] / 100.0;
628  data_tmp[10] = (double)raw_data[4] / 100.0;
629  data_tmp[11] = (double)raw_data[5] / 100.0;
630  }
631 
632  // Protect only this section in order to avoid slow race conditions when gathering this data
633  {
634  LockGuard guard(mutex);
635  data = data_tmp;
636  quaternion = quaternion_tmp;
637  }
638 
639  if (timeStamp > timeLastReport + TIME_REPORT_INTERVAL) {
640  // if almost 1 errors occourred in last interval, then print report
641  if(errs != 0)
642  {
643  yDebug(" IMUBOSCH periodic error report of last %d sec:", TIME_REPORT_INTERVAL);
644  yDebug("\t errors while reading data: %d", errs);
645  }
646 
647  errs = 0;
648  timeLastReport=timeStamp;
649  }
650 }
