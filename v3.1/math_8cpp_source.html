644 Vector yarp::math::dcm2axis(const Matrix &R)
645 {
646  yAssert((R.rows()>=3) && (R.cols()>=3));
647 
648  Vector v(4);
649  v[0]=R(2,1)-R(1,2);
650  v[1]=R(0,2)-R(2,0);
651  v[2]=R(1,0)-R(0,1);
652  v[3]=0.0;
653  double r=yarp::math::norm(v);
654  double theta=atan2(0.5*r,0.5*(R(0,0)+R(1,1)+R(2,2)-1));
655 
656  if (r<1e-9)
657  {
658  // if we enter here, then
659  // R is symmetric; this can
660  // happen only if the rotation
661  // angle is 0 (R=I) or 180 degrees
662  Matrix A=R.submatrix(0,2,0,2);
663  Matrix U(3,3), V(3,3);
664  Vector S(3);
665 
666  // A=I+sin(theta)*S+(1-cos(theta))*S^2
667  // where S is the skew matrix.
668  // Given a point x, A*x is the rotated one,
669  // hence if Ax=x then x belongs to the rotation
670  // axis. We have therefore to find the kernel of
671  // the linear application (A-I).
672  SVD(A-eye(3,3),U,S,V);
673 
674  v[0]=V(0,2);
675  v[1]=V(1,2);
676  v[2]=V(2,2);
677  r=yarp::math::norm(v);
678  }
679 
680  v=(1.0/r)*v;
681  v[3]=theta;
682 
683  return v;
684 }

686 Matrix yarp::math::axis2dcm(const Vector &v)
687 {
688  yAssert(v.length()>=4);
689 
690  Matrix R=eye(4,4);
691 
692  double theta=v[3];
693  if (theta==0.0)
694  return R;
695 
696  double c=cos(theta);
697  double s=sin(theta);
698  double C=1.0-c;
699 
700  double xs =v[0]*s;
701  double ys =v[1]*s;
702  double zs =v[2]*s;
703  double xC =v[0]*C;
704  double yC =v[1]*C;
705  double zC =v[2]*C;
706  double xyC=v[0]*yC;
707  double yzC=v[1]*zC;
708  double zxC=v[2]*xC;
709 
710  R(0,0)=v[0]*xC+c;
711  R(0,1)=xyC-zs;
712  R(0,2)=zxC+ys;
713  R(1,0)=xyC+zs;
714  R(1,1)=v[1]*yC+c;
715  R(1,2)=yzC-xs;
716  R(2,0)=zxC-ys;
717  R(2,1)=yzC+xs;
718  R(2,2)=v[2]*zC+c;
719 
720  return R;
721 }

723 Vector yarp::math::dcm2euler(const Matrix &R)
724 {
725  yAssert((R.rows()>=3) && (R.cols()>=3));
726 
727  Vector v(3);
728  bool singularity=false;
729  if (R(2,2)<1.0)
730  {
731  if (R(2,2)>-1.0)
732  {
733  v[0]=atan2(R(1,2),R(0,2));
734  v[1]=acos(R(2,2));
735  v[2]=atan2(R(2,1),-R(2,0));
736  }
737  else
738  {
739  // Not a unique solution: gamma-alpha=atan2(R10,R11)
740  singularity=true;
741  v[0]=-atan2(R(1,0),R(1,1));
742  v[1]=M_PI;
743  v[2]=0.0;
744  }
745  }
746  else
747  {
748  // Not a unique solution: gamma+alpha=atan2(R10,R11)
749  singularity=true;
