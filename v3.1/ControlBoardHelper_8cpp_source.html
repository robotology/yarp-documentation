162 ControlBoardHelper::ControlBoardHelper(int n, const int *aMap, const double *angToEncs, const double *zs, const double *newtons, const double *amps, const double *volts, const double *dutycycles, const double *kbemf, const double *ktau)
163 {
164  yAssert(n>=0); // if number of joints is negative complain!
165  yAssert(aMap!=nullptr); // at least the axisMap is required
166  mPriv = new PrivateUnitsHandler(n);
167 
168  memcpy(mPriv->axisMap, aMap, sizeof(int)*n);
169 
170  if (zs!=nullptr) memcpy(mPriv->position_zeros, zs, sizeof(double)*n);
171  if (angToEncs!=nullptr) memcpy(mPriv->angleToEncoders, angToEncs, sizeof(double)*n);
172  if (newtons!=nullptr) memcpy(mPriv->newtonsToSensors, newtons, sizeof(double)*n);
173  if (amps!=nullptr) memcpy(mPriv->ampereToSensors, amps, sizeof(double)*n);
174  if (volts!=nullptr) memcpy(mPriv->voltToSensors, volts, sizeof(double)*n);
175  if (dutycycles != nullptr) memcpy(mPriv->dutycycleToPWMs, dutycycles, sizeof(double)*n);
176  if (kbemf != nullptr) memcpy(mPriv->bemfToRaws, kbemf, sizeof(double)*n);
177  if (ktau != nullptr) memcpy(mPriv->ktauToRaws, ktau, sizeof(double)*n);
178 
179  // invert the axis map
180  memset (mPriv->invAxisMap, 0, sizeof(int) * n);
181  int i;
182  for (i = 0; i < n; i++)
183  {
184  int j;
185  for (j = 0; j < n; j++)
186  {
187  if (mPriv->axisMap[j] == i)
188  {
189  mPriv->invAxisMap[i] = j;
190  break;
191  }
192  }
193  }
194 }

743 double ControlBoardHelper::get_pidoutput_conversion_factor_user2raw(const yarp::dev::PidControlTypeEnum& pidtype, int j)

751  case PidOutputUnitsEnum::VELOCITY_METRIC: output_conversion_factor = mPriv->angleToEncoders[j]; break;

759 double ControlBoardHelper::get_pidfeedback_conversion_factor_user2raw(const yarp::dev::PidControlTypeEnum& pidtype, int j)

778 void ControlBoardHelper::convert_pid_to_user(const yarp::dev::PidControlTypeEnum& pidtype, const Pid &in_raw, int j_raw, Pid &out_usr, int &k_usr)
779 {
780  ControlBoardHelper* cb_helper = this;
781  k_usr = cb_helper->toUser(j_raw);
782  out_usr = in_raw;
783 
784  //feedback is at the denominator, i.e. kp=[PWM/deg]
785  double feedback_conversion_units_user2raw = get_pidfeedback_conversion_factor_user2raw(pidtype, k_usr);
786  out_usr.kp = out_usr.kp / feedback_conversion_units_user2raw;
787  out_usr.ki = out_usr.ki / feedback_conversion_units_user2raw;
788  out_usr.kd = out_usr.kd / feedback_conversion_units_user2raw;
