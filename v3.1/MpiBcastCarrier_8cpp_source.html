61 void MpiBcastCarrier::prepareDisconnect() {
62  comm->sema.wait();
63  #ifdef MPI_DEBUG
64  printf("[MpiBcastCarrier @ %s] Disconnect : %s\n", name.c_str(), other.c_str());
65  #endif
66  int cmd = CMD_DISCONNECT;
67  MPI_Bcast(&cmd, 1, MPI_INT, 0,comm->comm);
68  int length = other.length() + name.length() + 3;
69  char* remote_c = new char[length];
70  strcpy(remote_c, (other+"<-"+name).c_str());
71  MPI_Bcast(&length, 1, MPI_INT, 0,comm->comm);
72  MPI_Bcast(remote_c, length, MPI_CHAR, 0,comm->comm);
73  delete [] remote_c;
74  comm->disconnect(false);
75  comm->sema.post();
