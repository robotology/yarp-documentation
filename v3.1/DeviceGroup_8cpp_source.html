66  void update() {
67  mutex.lock();
68  std::vector<PolyDriver *>& lst = drivers;
69  for (unsigned int i=0; i<lst.size(); i++) {
70  if (needDrive[i]) {
71  IService *service;
72  lst[i]->view(service);
73  if (service!=nullptr) {
74  service->updateService();
75  }
76  }
77  }
78  mutex.unlock();
79  }

92  bool add(const std::string& name, yarp::os::Searchable& config) {
93  //printf("ADDING %s\n", config.toString().c_str());
94  auto* pd = new PolyDriver();
95  yAssert(pd!=nullptr);
96  bool result = pd->open(config);
97  if (!result) {
98  delete pd;
99  return false;
100  }
101  drivers.push_back(pd);
102  names.push_back(name);
103  IService *service = nullptr;
104  pd->view(service);
105  bool backgrounded = true;
106  if (service!=nullptr) {
107  backgrounded = service->startService();
108  if (backgrounded) {
109  // we don't need to poll this, so forget about the
110  // service interface
111  printf("group: service backgrounded\n");
112  service = nullptr;
113  }
114  }
115  needDrive.push_back(!backgrounded);
116  needDriveSummary = needDriveSummary || (!backgrounded);
117  Drivers::factory().add(new DriverLinkCreator(name,*pd));
118  return true;
119  }
