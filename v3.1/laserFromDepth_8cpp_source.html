44 bool LaserFromDepth::open(yarp::os::Searchable& config)
45 {
46  Property subConfig;
47  m_info = "LaserFromDepth device";
48  m_device_status = DEVICE_OK_STANBY;
49 
50 #ifdef LASER_DEBUG
51  yDebug("%s\n", config.toString().c_str());
52 #endif
53 
54  m_min_distance = 0.1; //m
55  m_max_distance = 2.5; //m
56  bool br = config.check("SUBDEVICE");
57  if (br != false)
58  {
59  yarp::os::Searchable& general_config = config.findGroup("SUBDEVICE");
60  m_clip_max_enable = general_config.check("clip_max");
61  m_clip_min_enable = general_config.check("clip_min");
62  if (m_clip_max_enable) { m_max_distance = general_config.find("clip_max").asFloat64(); }
63  if (m_clip_min_enable) { m_min_distance = general_config.find("clip_min").asFloat64(); }
64  m_do_not_clip_infinity_enable = (general_config.find("allow_infinity").asInt32()!=0);
65  }
66  else
67  {
68  yError() << "Missing SUBDEVICE section";
69  return false;
70  }
71  bool bs = config.check("SKIP");
72  if (bs != false)
73  {
74  yarp::os::Searchable& skip_config = config.findGroup("SKIP");
75  Bottle mins = skip_config.findGroup("min");
76  Bottle maxs = skip_config.findGroup("max");
77  size_t s_mins = mins.size();
78  size_t s_maxs = mins.size();
79  if (s_mins == s_maxs && s_maxs > 1 )
80  {
81  for (size_t s = 1; s < s_maxs; s++)
82  {
83  Range_t range;
84  range.max = maxs.get(s).asFloat64();
85  range.min = mins.get(s).asFloat64();
86  if (range.max >= 0 && range.max <= 360 &&
87  range.min >= 0 && range.min <= 360 &&
88  range.max > range.min)
89  {
90  m_range_skip_vector.push_back(range);
91  }
92  else
93  {
94  yError() << "Invalid range in SKIP section";
95  return false;
96  }
97  }
98  }
99 
100  }
101 
102  Property prop;
103  if(!config.check("RGBD_SENSOR_CLIENT"))
104  {
105  yError() << "missing RGBD_SENSOR_CLIENT section in configuration file!";
106  return false;
107  }
108  prop.fromString(config.findGroup("RGBD_SENSOR_CLIENT").toString());
109  prop.put("device", "RGBDSensorClient");
110 
111  driver.open(prop);
112  if (!driver.isValid())
113  {
114  yError("Error opening PolyDriver check parameters");
115  return false;
116  }
117  driver.view(iRGBD);
118  if (!iRGBD)
119  {
120  yError("Error opening iRGBD interface. Device not available");
121  return false;
122  }
123 
124  m_depth_width = iRGBD->getDepthWidth();
125  m_depth_height = iRGBD->getDepthHeight();
126  double hfov = 0;
127  double vfov = 0;
128  iRGBD->getDepthFOV(hfov, vfov);
129  m_sensorsNum = m_depth_width;
130  m_resolution = hfov / m_depth_width;
131  m_laser_data.resize(m_sensorsNum, 0.0);
132  m_max_angle = +hfov / 2;
133  m_min_angle = -hfov / 2;
134  PeriodicThread::start();
135 
136  yInfo("Sensor ready");
137  return true;
138 }

219 bool LaserFromDepth::getLaserMeasurement(std::vector<LaserMeasurementData> &data)
220 {
221  LockGuard guard(mutex);
222 #ifdef LASER_DEBUG
223  //yDebug("data: %s\n", laser_data.toString().c_str());
224 #endif
225  size_t size = m_laser_data.size();
226  data.resize(size);
227  if (m_max_angle < m_min_angle) { yError() << "getLaserMeasurement failed"; return false; }
228  double laser_angle_of_view = m_max_angle - m_min_angle;
229  for (size_t i = 0; i < size; i++)
230  {
231  double angle = (i / double(size)*laser_angle_of_view + m_min_angle)* DEG2RAD;
232  data[i].set_polar(m_laser_data[i], angle);
233  }
234  m_device_status = yarp::dev::IRangefinder2D::DEVICE_OK_IN_USE;
235  return true;

254 void LaserFromDepth::run()
255 {
256 #ifdef DEBUG_TIMING
257  double t1 = yarp::os::Time::now();
258 #endif
259  LockGuard guard(mutex);
260 
261  iRGBD->getDepthImage(m_depth_image);
262  if (m_depth_image.getRawImage()==nullptr)
263  {
264  yDebug()<<"invalid image received";
265  return;
266  }
267 
268  if (m_depth_image.width()!=m_depth_width ||
269  m_depth_image.height()!=m_depth_height)
270  {
271  yDebug()<<"invalid image size";
272  return;
273  }
274 
275 
276  auto* pointer = (float*)m_depth_image.getPixelAddress(0, m_depth_height / 2);
277  double angle, distance, infinity, angleShift;
278  size_t i;
279 
280  infinity = std::numeric_limits<double>::infinity();
281  angleShift = m_sensorsNum * m_resolution / 2;
282 
283  for (int elem = 0; elem < m_sensorsNum; elem++)
284  {
285 
286  angle = elem * m_resolution; //deg
287 
288  //the 1 / cos(blabla) distortion simulate the way RGBD devices calculate the distance..
289  distance = *(pointer + elem);
290  distance /= cos((angle - angleShift) * DEG2RAD); //m
291 
292  if (m_clip_min_enable && distance < m_min_distance)
293  {
294  distance = m_max_distance;
295  }
296 
297  if (m_clip_max_enable &&
298  distance > m_max_distance &&
299  !m_do_not_clip_infinity_enable &&
300  distance <= infinity)
301  {
302  distance = m_max_distance;
303  }
304 
305  for (i = 0; i < m_range_skip_vector.size(); i++)
306  {
307  if (angle > m_range_skip_vector[i].min && angle < m_range_skip_vector[i].max)
308  {
309  distance = infinity;
310  }
311  }
312 
313  m_laser_data[m_sensorsNum - 1 - elem] = distance;
314  }
315 
316  return;
317 }
