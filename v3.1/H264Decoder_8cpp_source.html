145 GstFlowReturn new_sample(GstAppSink *appsink, gpointer user_data)
146 {
147 #ifdef debug_time
148  static bool isFirst = true;
149  double start_time = Time::now();
150  double end_time=0;
151 
152  static double last_call;
153  static double sumOf_timeBetweenCalls = 0;
154  static double sumOf_timeOfNewSampleFunc = 0;
155  static uint32_t count=0;
156  #define MAX_COUNT 100
157 
158 
159  if(!isFirst)
160  sumOf_timeBetweenCalls+=(start_time -last_call);
161 
162  last_call = start_time;
163 
164 
165 #endif
166 
167  auto* dec_data = (data_for_gst_callback*)user_data;
168 
169  GstSample *sample = nullptr;
170  g_signal_emit_by_name (appsink, "pull-sample", &sample, NULL);
171  if(!sample)
172  {
173  g_print("GSTREAMER: could not take a sample!\n");
174  return GST_FLOW_OK;
175  }
176 
177  GstCaps *caps = gst_sample_get_caps (sample);
178  if(!caps)
179  {
180  g_print("GSTREAMER: could not get caps of sample!\n");
181  return GST_FLOW_ERROR;
182  }
183  GstStructure *struc = gst_caps_get_structure(caps, 0);
184  if(!struc)
185  {
186  g_print("GSTREAMER: could not get struct of caps!\n");
187  return GST_FLOW_ERROR;
188  }
189  gint width, height;
190  gboolean res;
191  res = gst_structure_get_int(struc, "width", &width);
192  if(!res)
193  {
194  g_print("GSTREAMER: could not get width!\n");
195  return GST_FLOW_ERROR;
196  }
197 
198  res = gst_structure_get_int(struc, "height", &height);
199  if(!res)
200  {
201  g_print("GSTREAMER: could not get height!\n");
202  return GST_FLOW_ERROR;
203  }
204  //g_print("Image has size %d x %d", width, height);
205 
206  GstBuffer *buffer = gst_sample_get_buffer(sample);
207  GstMapInfo map;
208  if(!gst_buffer_map(buffer, &map, GST_MAP_READ))
209  {
210  g_print("GSTREAMER: could not get map!\n");
211  return GST_FLOW_ERROR;
212  }
213  //HERE I GET MY IMAGE!!!!
214  //DO SOMETHING...
215  //ImageOf<PixelRgb> &yframebuff = yarp_stuff_ptr->yport_ptr->prepare();
216  dec_data->m->lock();
217  dec_data->isNew = true;
218  dec_data->img->resize(width, height);
219 
220  unsigned char *ydata_ptr = dec_data->img->getRawImage();
221  memcpy(ydata_ptr, map.data, width*height*3);
222 
223  dec_data->m->unlock();
224  gst_buffer_unmap(buffer, &map);
225 
226  gst_sample_unref(sample);
227  if(dec_data->isReq)
228  dec_data->s->post();
229 
230 
231 #ifdef debug_time
232  end_time = Time::now();
233  sumOf_timeOfNewSampleFunc += (end_time-start_time);
234  count++;
235  isFirst=false;
236 
237  if(count>=MAX_COUNT)
238  {
239  g_print("On %d times: NewSampleFunc is long %.6f sec and sleeps %.6f sec\n",
240  MAX_COUNT, (sumOf_timeOfNewSampleFunc/MAX_COUNT), (sumOf_timeBetweenCalls/MAX_COUNT) );
241  count = 0;
242  isFirst = true;
243  sumOf_timeBetweenCalls = 0;
244  sumOf_timeOfNewSampleFunc = 0;
245  }
246 
247 
248 #endif
249 
250 
251  return GST_FLOW_OK;
252
