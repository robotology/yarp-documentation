36 bool RosLookup::lookupCore(const std::string& name) {
37  Bottle req, reply;
38  req.addString("lookupNode");
39  req.addString("dummy_id");
40  req.addString(name);
41  rpc(getRosCoreAddress(), "xmlrpc", req, reply, verbose);
42  if (reply.get(0).asInt32()!=1) {
43  fprintf(stderr, "Failure: %s\n", reply.toString().c_str());
44  return false;
45  }
46  std::string url = reply.get(2).asString();
47  std::string::size_type break1 = url.find("://",0);
48  if (break1==std::string::npos) {
49  fprintf(stderr, "url not understood: %s\n", url.c_str());
50  return false;
51  }
52  std::string::size_type break2 = url.find(':',break1+3);
53  if (break2==std::string::npos) {
54  fprintf(stderr, "url not understood: %s\n", url.c_str());
55  return false;
56  }
57  std::string::size_type break3 = url.find('/',break2+1);
58  if (break3==std::string::npos) {
59  fprintf(stderr, "url not understood: %s\n", url.c_str());
60  return false;
61  }
62  hostname = url.substr(break1+3,break2-break1-3);
63  Value vportnum;
64  vportnum.fromString(url.substr(break2+1,break3-break2-1).c_str());
65  portnum = vportnum.asInt32();
66  if (verbose) printf("%s\n", reply.toString().c_str());
67  valid = (portnum!=0);
68  rpc(getRosCoreAddress(), "xmlrpc", req, reply, verbose);
69  return valid;
70 }

100  return false;
101  }
102  Value hostname2 = pref->get(1);

105  portnum = portnum2.asInt32();

107  if (verbose) {

109  hostname.c_str(), portnum);
