59 boolean fill_net_input_buffer (j_decompress_ptr cinfo)
60 {
61  // The whole JPEG data is expected to reside in the supplied memory
62  // buffer, so any request for more data beyond the given buffer size
63  // is treated as an error.
64  auto* mybuffer = (JOCTET *) cinfo->client_data;
65  fprintf(stderr, "JPEG data unusually large\n");
66  // Insert a fake EOI marker
67  mybuffer[0] = (JOCTET) 0xFF;
68  mybuffer[1] = (JOCTET) JPEG_EOI;
69  cinfo->src->next_input_byte = mybuffer;
70  cinfo->src->bytes_in_buffer = 2;
71  return TRUE;
72 }

80 void skip_net_input_data (j_decompress_ptr cinfo, long num_bytes)
81 {
82  auto src = (net_src_ptr) cinfo->src;
83 
84  if (num_bytes > 0) {
85  while (num_bytes > (long) src->bytes_in_buffer) {
86  num_bytes -= (long) src->bytes_in_buffer;
87  (void) (*src->fill_input_buffer) (cinfo);
88  }
89  src->next_input_byte += (size_t) num_bytes;
90  src->bytes_in_buffer -= (size_t) num_bytes;
91  }

142  bool decompress(const Bytes& cimg, FlexImage& img) {
143  bool debug = false;
144 
145  if (!active) {
146  init();
147  active = true;
148  }
149  cinfo.client_data = &error_buffer;
150  cinfo.err = jpeg_std_error(&jerr.pub);
151  jerr.pub.error_exit = net_error_exit;
152 
153  if (setjmp(jerr.setjmp_buffer)) {
154  jpeg_finish_decompress(&cinfo);
155  return false;
156  }
157 
158  jpeg_net_src(&cinfo,(char*)cimg.get(),cimg.length());
159  jpeg_save_markers(&cinfo, JPEG_COM, 0xFFFF);
160  jpeg_read_header(&cinfo, TRUE);
161  jpeg_calc_output_dimensions(&cinfo);
162 
163  if(cinfo.jpeg_color_space == JCS_GRAYSCALE) {
164  img.setPixelCode(VOCAB_PIXEL_MONO);
165  }
166  else
167  {
168  img.setPixelCode(VOCAB_PIXEL_RGB);
169  }
170 
171  if (debug) printf("Got image %dx%d\n", cinfo.output_width, cinfo.output_height);
172  img.resize(cinfo.output_width,cinfo.output_height);
173  jpeg_start_decompress(&cinfo);
174  //int row_stride = cinfo.output_width * cinfo.output_components;
175 
176  int at = 0;
177  while (cinfo.output_scanline < cinfo.output_height) {
178  JSAMPLE *lines[1];
179  lines[0] = (JSAMPLE*)(img.getPixelAddress(0,at));
180  jpeg_read_scanlines(&cinfo, lines, 1);
181  at++;
182  }
183  if(readEnvelopeCallback && cinfo.marker_list && cinfo.marker_list->data_length > 0) {
184  Bytes envelope(reinterpret_cast<char*>(cinfo.marker_list->data), cinfo.marker_list->data_length);
185  readEnvelopeCallback(readEnvelopeCallbackData, envelope);
186  }
187  if (debug) printf("Read image!\n");
188  jpeg_finish_decompress(&cinfo);
189  return true;
190  }

193  jpeg_destroy_decompress(&cinfo);
194  }

198  fini();
199  active = false;
200  }
