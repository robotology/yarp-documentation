233  bool readFromSelector(const std::string& name)
234  {
235  if (!selector)
236  return false;
237  Bottle plugins = selector->getSelectedPlugins();
238  Bottle group = plugins.findGroup(name.c_str()).tail();
239  if (group.isNull()) {
240  yError("Cannot find \"%s\" plugin (not built in, and no .ini file found for it)\n"
241  "Check that YARP_DATA_DIRS leads to at least one directory with plugins/%s.ini "
242  "or share/yarp/plugins/%s.ini in it",
243  name.c_str(),
244  name.c_str(),
245  name.c_str());
246  return false;
247  }
248  return readFromSearchable(group, name);
249  }
