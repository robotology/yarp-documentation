39 void InputPortProcessor::onRead(yarp::sig::Vector &v)
40 {
41  now=Time::now();
42  mutex.lock();
43 
44  if (count>0)
45  {
46  double tmpDT=now-prev;
47  deltaT+=tmpDT;
48  if (tmpDT>deltaTMax)
49  deltaTMax=tmpDT;
50  if (tmpDT<deltaTMin)
51  deltaTMin=tmpDT;
52 
53  //compare network time
54  if (tmpDT*1000<ANALOG_TIMEOUT)
55  {
56  state=IAnalogSensor::AS_OK;
57  }
58  else
59  {
60  state=IAnalogSensor::AS_TIMEOUT;
61  }
62  }
63 
64  prev=now;
65  count++;
66 
67  lastVector=v;
68  Stamp newStamp;
69  getEnvelope(newStamp);
70 
71  //initialialization (first received data)
72  if (lastStamp.isValid()==false)
73  {
74  lastStamp = newStamp;
75  }
76 
77  //now compare timestamps
78  if ((1000*(newStamp.getTime()-lastStamp.getTime()))<ANALOG_TIMEOUT)
79  {
80  state=IAnalogSensor::AS_OK;
81  }
82  else
83  {
84  state=IAnalogSensor::AS_TIMEOUT;
85  }
86  lastStamp = newStamp;
87 
88  mutex.unlock();
89 }

176 bool yarp::dev::AnalogSensorClient::open(yarp::os::Searchable &config)
177 {
178  std::string carrier = config.check("carrier", Value("udp"), "default carrier for streaming robot state").asString();
179 
180  local.clear();
181  remote.clear();
182 
183  local = config.find("local").asString();
184  remote = config.find("remote").asString();
185 
186  if (local=="")
187  {
188  yError("AnalogSensorClient::open() error you have to provide valid local name");
189  return false;
190  }
191  if (remote=="")
192  {
193  yError("AnalogSensorClient::open() error you have to provide valid remote name\n");
194  return false;
195  }
196 
197  std::string local_rpc = local;
198  local_rpc += "/rpc:o";
199  std::string remote_rpc = remote;
200  remote_rpc += "/rpc:i";
201 
202  if (!inputPort.open(local))
203  {
204  yError("AnalogSensorClient::open() error could not open port %s, check network", local.c_str());
205  return false;
206  }
207  inputPort.useCallback();
208 
209  if (!rpcPort.open(local_rpc))
210  {
211  yError("AnalogSensorClient::open() error could not open rpc port %s, check network", local_rpc.c_str());
212  return false;
213  }
214 
215  bool ok=Network::connect(remote.c_str(), local.c_str(), carrier.c_str());
216  if (!ok)
217  {
218  yError("AnalogSensorClient::open() error could not connect to %s", remote.c_str());
219  return false;
220  }
221 
222  ok=Network::connect(local_rpc, remote_rpc);
223  if (!ok)
224  {
225  yError("AnalogSensorClient::open() error could not connect to %s\n", remote_rpc.c_str());
226  return false;
227  }
228 
229  return true;
230 }
