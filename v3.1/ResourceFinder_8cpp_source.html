35 static std::string getPwd() {
36  std::string result;
37  int len = 5;
38  char *buf = nullptr;
39  while (true) {
40  if (buf!=nullptr) delete[] buf;
41  buf = new char[len];
42  if (!buf) break;
43  char *dir = yarp::os::getcwd(buf, len);
44  if (dir) {
45  result = dir;
46  break;
47  }
48  if (errno!=ERANGE) break;
49  len *= 2;
50  }
51  if (buf!=nullptr) delete[] buf;
52  buf = nullptr;
53  return result;
54 }

57 static Bottle parsePaths(const std::string& txt) {
58  if (txt.empty()) {
59  return Bottle();
60  }
61  char slash = NetworkBase::getDirectorySeparator()[0];
62  char sep = NetworkBase::getPathSeparator()[0];
63  Bottle result;
64  const char *at = txt.c_str();
65  int slash_tweak = 0;
66  int len = 0;
67  for (char ch : txt) {
68  if (ch==sep) {
69  result.addString(std::string(at, len-slash_tweak));
70  at += len+1;
71  len = 0;
72  slash_tweak = 0;
73  continue;
74  }
75  slash_tweak = (ch==slash && len>0)?1:0;
76  len++;
77  }
78  if (len>0) {
79  result.addString(std::string(at, len-slash_tweak));
80  }
81  return result;
82 }

85 static void appendResourceType(std::string& path,
86  const std::string& resourceType) {
87  if (resourceType=="") return;

89  if (path.length()>0) {

94  path += resourceType;

146  static std::string extractPath(const char *fname)
147  {
148  std::string s{fname};
149  auto n = s.rfind('/');
150 #if defined(_WIN32)
151  if (n == std::string::npos) {
152  n = s.rfind('\\');
153  }
154 #endif
155  if (n != std::string::npos) {
156  return s.substr(0,n);
157  }
158  return {};
159  }

231  bool isRooted(const std::string& str) {
232  if (isAbsolute(str)) return true;
233  if (str.length()>=2) {
234  if (str[0]=='.'&&(str[1]=='/'||str[1]=='\\')) {
235  return true;
236  }
237  } else if (str==".") {
238  return true;
239  }

243  std::string getPath(const std::string& base1,
244  const std::string& base2,
245  const std::string& base3,
246  const std::string& name) {
247  if (isAbsolute(name)) {
248  return name;
249  }
250 
251  std::string s;
252  std::string slash = NetworkBase::getDirectorySeparator();
253 
254  if (base1!="") {
255  s = base1;
256  s = s + slash;
257  }
258 
259  if (isRooted(base2)) {
260  s = base2;
261  } else {
262  s = s + base2;
263  }
264  if (base2!="") {
265  s = s + slash;
266  }
267 
268  if (isRooted(base3)) {
269  s = base3;
270  } else {
271  s = s + base3;
272  }
273  if (base3!="") {
274  s = s + slash;
275  }
276 
277  s = s + name;
278 
279  return s;
280  }

282  std::string check(const std::string& base1,
283  const std::string& base2,
284  const std::string& base3,
285  const std::string& name,
286  bool isDir,
287  const Bottle& doc,
288  const char *doc2) {
289  std::string s = getPath(base1, base2, base3, name);
290 
291  // check cache first
292  Bottle *prev = cache.find(s).asList();
293  if (prev!=nullptr) {
294  double t = prev->get(0).asFloat64();
295  int flag = prev->get(1).asInt32();
296  if (SystemClock::nowSystem()-t<RESOURCE_FINDER_CACHE_TIME) {
297  if (flag) return s;
298  return {};
299  }
300  }
301 
302  if (verbose) {
303  std::string base = doc.toString();
304  fprintf(RTARGET, "||| checking [%s] (%s%s%s)\n", s.c_str(),
305  base.c_str(),
306  (base.length()==0) ? "" : " ",
307  doc2);
308  }
309  bool ok = exists(s.c_str(), isDir);
310  Value status;
311  yAssert(status.asList());
312  status.asList()->addFloat64(SystemClock::nowSystem());
313  status.asList()->addInt32(ok?1:0);
314  cache.put(s, status);
315  if (ok) {
316  if (verbose) {
317  fprintf(RTARGET, "||| found %s\n", s.c_str());
318  }
319  return s;
320  }
321  return {};
322  }

390  void findFileBase(Property& config, const std::string& name,
391  bool isDir,
392  Bottle& output, const ResourceFinderOptions& opts) {
393  Bottle doc;
394  size_t prelen = output.size();
395  findFileBaseInner(config, name, isDir, true, output, opts, doc, nullptr);
396  if (output.size()!=prelen) return;
397  bool justTop = (opts.duplicateFilesPolicy==ResourceFinderOptions::First);
398  if (justTop) {
399  if (canShowErrors(opts)) {
400  fprintf(RTARGET, "||| did not find %s\n", name.c_str());

412  void findFileBaseInner(Property& config, const std::string& name,
413  bool isDir, bool allowPathd,
414  Bottle& output, const ResourceFinderOptions& opts,
415  const Bottle& predoc, const char *reason) {
416  Bottle doc;
417  if (verbose) {
418  doc = predoc;
419  if (reason) {
420  doc.addString(reason);
421  }
422  }
423  ResourceFinderOptions::SearchLocations locs = opts.searchLocations;
424  ResourceFinderOptions::SearchFlavor flavor = opts.searchFlavor;
425  std::string resourceType = opts.resourceType;
426 
427  bool justTop = (opts.duplicateFilesPolicy==ResourceFinderOptions::First);
428 
429  // check current directory
430  if (locs & ResourceFinderOptions::Directory) {
431  if (name==""&&isDir) {
432  addString(output, getPwd());
433  if (justTop) return;
434  }
435  std::string str = check(getPwd(), resourceType, "", name, isDir, doc, "pwd");
436  if (str!="") {
437  if (mainActive) useNearMain = true;
438  addString(output, str);
439  if (justTop) return;
440  }
441  }
442 
443  if ((locs & ResourceFinderOptions::NearMainConfig) && useNearMain) {
444  if (configFilePath!="") {
445  std::string str = check(configFilePath, resourceType, "", name, isDir, doc, "defaultConfigFile path");
446  if (str!="") {
447  addString(output, str);
448  if (justTop) return;
449  }
450  }
451  }
452 
453  if (locs & ResourceFinderOptions::Robot) {
454  std::string slash = NetworkBase::getDirectorySeparator();
455  bool found = false;
456  std::string robot = NetworkBase::getEnvironment("YARP_ROBOT_NAME",
457  &found);
458  if (!found) robot = "default";
459 
460  // Nested search to locate robot directory
461  Bottle paths;
462  ResourceFinderOptions opts2;
463  opts2.searchLocations = (ResourceFinderOptions::SearchLocations)(ResourceFinderOptions::User | ResourceFinderOptions::Sysadmin | ResourceFinderOptions::Installed);
464  opts2.resourceType = "robots";
465  opts2.duplicateFilesPolicy = ResourceFinderOptions::All;
466  findFileBaseInner(config, robot, true, allowPathd, paths, opts2, doc, "robot");
467  appendResourceType(paths, resourceType);
468  for (size_t j=0; j<paths.size(); j++) {
469  std::string str = check(paths.get(j).asString(),
470  "", "",
471  name, isDir, doc, "robot");
472  if (str!="") {
473  addString(output, str);
474  if (justTop) return;
475  }
476  }
477  }
478 
479  if ((locs & ResourceFinderOptions::Context) && !useNearMain) {
480  for (size_t i=0; i<apps.size(); i++) {
481  std::string app = apps.get(i).asString();
482 
483  // New context still apparently applies only to "applications"
484  // which means we need to restrict our attention to "app"
485  // directories.
486 
487  // Nested search to locate context directory
488  Bottle paths;
489  ResourceFinderOptions opts2;
490  prependResourceType(app, "contexts");
491  opts2.searchLocations = (ResourceFinderOptions::SearchLocations)ResourceFinderOptions::Default;
492  opts2.duplicateFilesPolicy = ResourceFinderOptions::All;
493  findFileBaseInner(config, app, true, allowPathd, paths, opts2, doc, "context");
494  appendResourceType(paths, resourceType);
495  for (size_t j=0; j<paths.size(); j++) {
496  std::string str = check(paths.get(j).asString(), "", "",
497  name, isDir, doc, "context");
498  if (str!="") {
499  addString(output, str);
500  if (justTop) return;
501  }
502  }
503  }
504  }
505 
506  // check YARP_CONFIG_HOME
507  if ((locs & ResourceFinderOptions::User) &&
508  (flavor & ResourceFinderOptions::ConfigLike)) {
509  std::string home = ResourceFinder::getConfigHomeNoCreate();
510  if (home!="") {
511  appendResourceType(home, resourceType);
512  std::string str = check(home, "", "", name, isDir,
513  doc, "YARP_CONFIG_HOME");
514  if (str!="") {
515  addString(output, str);
516  if (justTop) return;
517  }
518  }
519  }
520 
521  // check YARP_DATA_HOME
522  if ((locs & ResourceFinderOptions::User) &&
523  (flavor & ResourceFinderOptions::DataLike)) {
524  std::string home = ResourceFinder::getDataHomeNoCreate();
525  if (home!="") {
526  appendResourceType(home, resourceType);
527  std::string str = check(home, "", "", name, isDir,
528  doc, "YARP_DATA_HOME");
529  if (str!="") {
530  addString(output, str);
531  if (justTop) return;
532  }
533  }
534  }
535 
536  // check YARP_CONFIG_DIRS
537  if (locs & ResourceFinderOptions::Sysadmin) {
538  Bottle dirs = ResourceFinder::getConfigDirs();
539  appendResourceType(dirs, resourceType);
540  for (size_t i=0; i<dirs.size(); i++) {
541  std::string str = check(dirs.get(i).asString(),
542  "", "", name, isDir,
543  doc, "YARP_CONFIG_DIRS");
544  if (str!="") {
545  addString(output, str);
546  if (justTop) return;
547  }
548  }
549  }
550 
551  // check YARP_DATA_DIRS
552  if (locs & ResourceFinderOptions::Installed) {
553  Bottle dirs = ResourceFinder::getDataDirs();
554  appendResourceType(dirs, resourceType);
555  for (size_t i=0; i<dirs.size(); i++) {
556  std::string str = check(dirs.get(i).asString(),
557  "", "", name, isDir,
558  doc, "YARP_DATA_DIRS");
559  if (str!="") {
560  addString(output, str);
561  if (justTop) return;
562  }
563  }
564  }
565 
566  if (allowPathd && (locs & ResourceFinderOptions::Installed)) {
567  // Nested search to locate path.d directories
568  Bottle pathds;
569  ResourceFinderOptions opts2;
570  opts2.searchLocations = (ResourceFinderOptions::SearchLocations)(opts.searchLocations & ResourceFinderOptions::Installed);
571  opts2.resourceType = "config";
572  findFileBaseInner(config, "path.d", true, false, pathds, opts2, doc, "path.d");
573 
574  for (size_t i=0; i<pathds.size(); i++) {
575  // check /.../path.d/*
576  // this directory is expected to contain *.ini files like this:
577  // [search BUNDLE_NAME]
578  // path /PATH1 /PATH2
579  // for example:
580  // [search icub]
581  // path /usr/share/iCub
582  Property pathd;
583  pathd.fromConfigFile(pathds.get(i).asString());
584  Bottle sections = pathd.findGroup("search").tail();
585  for (size_t i=0; i<sections.size(); i++) {
586  std::string search_name = sections.get(i).asString();
587  Bottle group = pathd.findGroup(search_name);
588  Bottle paths = group.findGroup("path").tail();
589  appendResourceType(paths, resourceType);
590  for (size_t j=0; j<paths.size(); j++) {
591  std::string str = check(paths.get(j).asString(), "", "",
592  name, isDir, doc, "yarp.d");
593  if (str!="") {
594  addString(output, str);
595  if (justTop) return;
596  }
597  }
598  }
599  }
600  }
601  }
