33 void addHeapOperation(const char *act) {
34  if (heap_count_active) {
35  heap_count_mutex->lock();
36  if (!heap_expect_ops) {
37  heap_count_mutex->unlock();
38  fprintf(stderr, "Unexpected '%s' heap operation detected.\n", act);
39  yarp_print_trace(stderr, __FILE__, __LINE__);
40  heap_count_mutex->lock();
41  }
42  heap_count_ops++;
43  heap_count_mutex->unlock();
44  }
45 }

136 int UnitTest::run(int argc, char *argv[])
137 {
138  static bool ran = false;
139 
140  if (argc==0) {
141  runTests();
142  ran = true;
143  } else {
144  std::string name = getName();
145  bool onList = false;
146  for (int i=0; i<argc; i++) {
147  if (name == std::string(argv[i])) {
148  onList = true;
149  break;
150  }
151  }
152  if (onList) {
153  runTests();
154  ran = true;
155  }
156  }
157 
158  // Run sub tests
159  for (auto& subTest : subTests) {
160  hasProblem += subTest->run(argc, argv);
161  }
162 
163  if (parent != nullptr) {
164  return hasProblem;
165  }
166 
167  if (hasProblem) {
168  report(0, "A PROBLEM WAS ENCOUNTERED");
169  return 1;
170  }
171 
172  if (!ran) {
173  report(0, "no test found");
174  return 2;
175  }
176 
177  report(0, "no problems reported");
178  return 0;
179 }

316 int UnitTest::heapMonitorOps()
317 {
318 #ifdef YARP_TEST_HEAP
319  heap_count_mutex->lock();
320  int diff = heap_count_ops;
321  heap_count_ops = 0;
322  heap_count_mutex->unlock();
323  if (!heap_expect_ops) {
324  checkEqual(0, diff, "heap operator count");
325  }
326  return diff;
327 #else
328  return -1;
329 #endif
330 }

332 int UnitTest::heapMonitorEnd()
333 {
334 #ifdef YARP_TEST_HEAP
335  if (!heap_count_mutex) return 0;
336  heap_count_mutex->lock();
337  heap_count_active = false;
338  int diff = heap_count_ops;
339  heap_count_ops = 0;
340  heap_count_mutex->unlock();
341  if (!heap_expect_ops) {
342  checkEqual(0, diff, "heap operator final count");
343  }
344  delete heap_count_mutex;
345  heap_count_mutex = nullptr;
346  return diff;
347 #else
348  return -1;
349 #endif
350 }
