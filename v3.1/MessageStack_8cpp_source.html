67  void stack(PortWriter& msg, const std::string& tag) {
68  mutex.lock();
69  msgs.emplace_back();
70  if (tag!="") {
71  Bottle b;
72  b.read(msg);
73  Bottle& back = msgs.back();
74  back.clear();
75  back.addString(tag);
76  back.append(b);
77  } else {
78  msgs.back().read(msg);
79  }
80  if (available_threads==0) {
81  if (threads.size()<max_threads || max_threads == 0) {
82  available_threads++;
83  threads.push_back(new MessageStackThread(*this));
84  threads.back()->start();
85  }
86  }
87  available_threads--;
88  mutex.unlock();
89  produce.post();
90  }
91 
92  bool process() {
93  produce.wait();
94  if (!active) return false;
95  mutex.lock();
96  Bottle b = msgs.front();
97  msgs.pop_front();
98  mutex.unlock();
99  DummyConnector con;
100  b.write(con.getWriter());
101  owner.read(con.getReader());
102  mutex.lock();
103  available_threads++;
104  mutex.unlock();
105  return active;
106  }
