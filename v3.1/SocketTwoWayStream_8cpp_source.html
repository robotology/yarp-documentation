31 int SocketTwoWayStream::open(const Contact& address) {
32  if (address.getPort()==-1) {
33  return -1;
34  }
35  std::string host = address.getHost();
36  yarp::os::impl::TcpConnector connector;
37 #ifdef YARP_HAS_ACE
38  if (address.getHost() == "localhost") {
39  // ACE does not like localhost. At all.
40  NameConfig config;
41  host = config.getHostName(true);
42  }
43  ACE_INET_Addr addr(address.getPort(), host.c_str());
44  YARP_timeval openTimeout;
45  YARP_timeval *timeout = nullptr;
46  if (address.hasTimeout()) {
47  openTimeout.set(address.getTimeout());
48  timeout = &openTimeout;
49  }
50  int result = connector.connect(stream, addr, timeout, ACE_Addr::sap_any, 1);
51 #else
52  int result;
53 
54  if (address.hasTimeout())
55  {
56  YARP_timeval timeout;
57  /* set timeout seconds and microseconds */
58  timeout.tv_sec = (int) address.getTimeout();
59  timeout.tv_usec = (address.getTimeout() - (int) address.getTimeout()) * 1000000;
60  result = connector.connect(stream, address, &timeout);
61  }
62  else
63  {
64  result = connector.connect(stream, address, nullptr);
65  }
66 
67 
68 #endif
69  if (result>=0) {
70  happy = true;
71  } else {
72  YARP_SPRINTF2(Logger::get(),
73  debug,
74  "TCP connection to tcp://%s:%d failed to open",
75  host.c_str(),
76  address.getPort());
77  }
78  updateAddresses();
79  return result;
80 }

91 void SocketTwoWayStream::updateAddresses() {
92  int one = 1;
93  stream.set_option(IPPROTO_TCP, TCP_NODELAY, &one, sizeof(int));
94 #ifdef YARP_HAS_ACE
95  ACE_INET_Addr local, remote;
96  stream.get_local_addr(local);
97  stream.get_remote_addr(remote);
98  char localHostAddress[256];
99  char remoteHostAddress[256];
100  local.get_host_addr(localHostAddress, 256);
101  remote.get_host_addr(remoteHostAddress, 256);
102  localAddress = Contact(localHostAddress, local.get_port_number());
103  remoteAddress = Contact(remoteHostAddress, remote.get_port_number());
104 #else
105  struct sockaddr local;
106  struct sockaddr remote;
107  memset(&local, 0, sizeof(local));
108  memset(&remote, 0, sizeof(remote));
109  stream.get_local_addr(local);
110  stream.get_remote_addr(remote);
111  if (local.sa_family == AF_INET || local.sa_family == AF_INET6) {
112  char* localHostAddress = new char[local.sa_family == AF_INET ? INET_ADDRSTRLEN : INET6_ADDRSTRLEN];
113  char* remoteHostAddress = new char[remote.sa_family == AF_INET ? INET_ADDRSTRLEN : INET6_ADDRSTRLEN];
114  const char* ret = nullptr;
115  ret = inet_ntop(local.sa_family,
116  (local.sa_family == AF_INET ? reinterpret_cast<void*>(&reinterpret_cast<struct sockaddr_in*>(&local)->sin_addr):
117  reinterpret_cast<void*>(&reinterpret_cast<struct sockaddr_in6*>(&local)->sin6_addr)),
118  localHostAddress,
119  (local.sa_family == AF_INET ? INET_ADDRSTRLEN:
120  INET6_ADDRSTRLEN));
121  if (ret) {
122  localAddress = Contact(localHostAddress, ntohs(reinterpret_cast<struct sockaddr_in*>(&local)->sin_port));
123  } else {
124  YARP_ERROR(Logger::get(), "SocketTwoWayStream::updateAddresses failed getting local address");
125  }
126  ret = inet_ntop(remote.sa_family,
127  (remote.sa_family == AF_INET ? reinterpret_cast<void*>(&reinterpret_cast<struct sockaddr_in*>(&remote)->sin_addr):
128  reinterpret_cast<void*>(&reinterpret_cast<struct sockaddr_in6*>(&remote)->sin6_addr)),
129  remoteHostAddress,
130  (remote.sa_family == AF_INET ? INET_ADDRSTRLEN:
131  INET6_ADDRSTRLEN));
132  if (ret) {
133  remoteAddress = Contact(remoteHostAddress, ntohs(reinterpret_cast<struct sockaddr_in*>(&remote)->sin_port));
134  } else {
135  YARP_ERROR(Logger::get(), "SocketTwoWayStream::updateAddresses failed getting local address");
136  }
137  delete[] localHostAddress;
138  delete[] remoteHostAddress;
139  } else {
140  YARP_ERROR(Logger::get(), "Unknown address type");
141  }
142 #endif
143 }
