44  void stop() {
45  double delay = 0.1;
46  while (!done.check()) {
47  if (delay>1) {
48  worked = false;
49  break;
50  }
51  // Always use SystemClock for this delay
52  yarp::os::SystemClock::delaySystem(delay);
53  delay *= 2;
54  }
55  if (delay<=1) {
56  worked = true;
57  }
58  slave.close();
59  }
