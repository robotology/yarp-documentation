79 /* Make the table for a fast CRC. */
80 static void make_crc_table() {
81  unsigned long c;
82  int n, k;
83 
84  for (n = 0; n < 256; n++) {
85  c = (unsigned long) n;
86  for (k = 0; k < 8; k++) {
87  if (c & 1)
88  c = 0xedb88320L ^ (c >> 1);
89  else
90  c = c >> 1;
91  }
92  crc_table[n] = c;
93  }
94  crc_table_computed = 1;
95 }

97 /* Update a running CRC with the bytes buf[0..len-1]--the CRC
98  should be initialized to all 1's, and the transmitted value
99  is the 1's complement of the final running CRC (see the
100  crc() routine below)). */
101 
102 static unsigned long update_crc(unsigned long crc, unsigned char *buf,
103  size_t len) {
104 
105  unsigned long c = crc;
106  size_t n;
107 
108  if (!crc_table_computed)
109  make_crc_table();
110  for (n = 0; n < len; n++) {
111  c = crc_table[(c ^ buf[n]) & 0xff] ^ (c >> 8);
112  }
113  return c;
114 }
