138 Contact Contact::fromString(const std::string& txt)
139 {
140  std::string str(txt);
141  Contact c;
142  std::string::size_type start = 0;
143  std::string::size_type base = str.find("://");
144  std::string::size_type offset = 2;
145  if (base==std::string::npos) {
146  base = str.find(":/");
147  offset = 1;
148  }
149  if (base==std::string::npos) {
150  if (str.length()>0 && str[0] == '/') {
151  base = 0;
152  offset = 0;
153  }
154  }
155  if (base!=std::string::npos) {
156  c.mPriv->carrier = str.substr(0, base);
157  start = base+offset;
158  // check if we have a direct machine:NNN syntax
159  std::string::size_type colon = std::string::npos;
160  int mode = 0;
161  int nums = 0;
162  std::string::size_type i;
163  for (i=start+1; i<str.length(); i++) {
164  char ch = str[i];
165  if (ch==':') {
166  if (mode == 0) {
167  colon = i;
168  mode = 1;
169  continue;
170  } else {
171  mode = -1;
172  break;
173  }
174  }
175  if (ch=='/') {
176  break;
177  }
178  if (mode==1) {
179  if (ch>='0'&&ch<='9') {
180  nums++;
181  continue;
182  } else {
183  mode = -1;
184  break;
185  }
186  }
187  }
188  if (mode==1 && nums>=1) {
189  // yes, machine:nnn
190  if (c.mPriv->carrier.empty()) {
191  c.mPriv->carrier = "tcp";
192  }
193  c.mPriv->hostname = str.substr(start+1, colon-start-1);
194  c.mPriv->port = atoi(str.substr(colon+1, nums).c_str());
195  start = i;
196  }
197  }
198  std::string rname = str.substr(start);
199  if (rname!="/") {
200  c.mPriv->regName = rname;
201  }
202  return c;
203 }

341 std::string Contact::convertHostToIp(const char *name)
342 {
343 #if defined(YARP_HAS_ACE)
344  ACE_INET_Addr addr((u_short)0, name);
345  char ipstr[256];
346  addr.get_host_addr(ipstr, sizeof(ipstr));
347 
348 #else
349  char ipstr[INET6_ADDRSTRLEN];
350  int status;
351  struct addrinfo hints, *res, *p;
352 
353  memset(&hints, 0, sizeof hints); // make sure the struct is empty
354  hints.ai_family = AF_UNSPEC; // don't care IPv4 or IPv6
355  hints.ai_socktype = SOCK_STREAM; // TCP stream sockets
356  hints.ai_flags = AI_PASSIVE; // fill in my IP for me
357 
358  if ((status = yarp::os::impl::getaddrinfo(name, "http", &hints, &res)) != 0) {
359  fprintf(stderr, "getaddrinfo error: %s\n", yarp::os::impl::gai_strerror(status));
360  std::exit(1);
361  }
362 
363  for(p = res; p != nullptr; p = p->ai_next) {
364  void *addr;
365 
366  if (p->ai_family == AF_INET) { // IPv4
367  auto *ipv4 = (struct sockaddr_in *)p->ai_addr;
368  addr = &(ipv4->sin_addr);
369  } else { // IPv6
370  auto *ipv6 = (struct sockaddr_in6 *)p->ai_addr;
371  addr = &(ipv6->sin6_addr);
372  }
373 
374  // convert the IP to a string and print it:
375  inet_ntop(p->ai_family, addr, ipstr, sizeof ipstr);
376  }
377  yarp::os::impl::freeaddrinfo(res);
378 #endif
379 
380  if (NameConfig::isLocalName(ipstr)) {
381  return NameConfig::getHostName();
382  }
383  return ipstr;
384 }
