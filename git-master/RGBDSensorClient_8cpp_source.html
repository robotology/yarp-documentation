39bool RGBDSensorClient::open(yarp::os::Searchable& config)
40{
41 if (!parseParams(config)) { return false; }
42
43 bool ret = false;
44
45 // Opening Streaming ports
46 ret = colorFrame_StreamingPort.open(m_localImagePort);
47 ret &= depthFrame_StreamingPort.open(m_localDepthPort);
48
49 if (!ret)
50 {
51 yCError(RGBDSENSORCLIENT) << " cannot open local streaming ports: " << m_localImagePort << " or " << m_localDepthPort;
52 colorFrame_StreamingPort.close();
53 depthFrame_StreamingPort.close();
54 }
55
56 if (!yarp::os::Network::connect(m_remoteImagePort, colorFrame_StreamingPort.getName(), m_ImageCarrier))
57 {
58 yCError(RGBDSENSORCLIENT) << colorFrame_StreamingPort.getName() << " cannot connect to remote port " << m_remoteImagePort << "with carrier " << m_ImageCarrier;
59 return false;
60 }
61
62 if (!yarp::os::Network::connect(m_remoteDepthPort, depthFrame_StreamingPort.getName(), m_DepthCarrier))
63 {
64 yCError(RGBDSENSORCLIENT) << depthFrame_StreamingPort.getName() << " cannot connect to remote port " << m_remoteDepthPort << "with carrier " << m_DepthCarrier;
65 return false;
66 }
67
68
69 // Single RPC port
70 ret = rpcPort.open(m_localRpcPort);
71
72 if (!ret)
73 {
74 yCError(RGBDSENSORCLIENT) << " cannot open local RPC port " << m_localRpcPort;
75 colorFrame_StreamingPort.close();
76 depthFrame_StreamingPort.close();
77 rpcPort.close();
78 }
79
80 if (!rpcPort.addOutput(m_remoteRpcPort))
81 {
82 yCError(RGBDSENSORCLIENT) << " cannot connect to port " << m_remoteRpcPort;
83 colorFrame_StreamingPort.close();
84 depthFrame_StreamingPort.close();
85 rpcPort.close();
86 return false;
87 }
88
89 // Check protocol version
90 yarp::os::Bottle cmd;
91 yarp::os::Bottle response;
92 cmd.addVocab32(VOCAB_RGBD_SENSOR);
93 cmd.addVocab32(VOCAB_GET);
94 cmd.addVocab32(VOCAB_RGBD_PROTOCOL_VERSION);
95 rpcPort.write(cmd, response);
96 int major = response.get(3).asInt32();
97 int minor = response.get(4).asInt32();
98
99 if (major != RGBD_INTERFACE_PROTOCOL_VERSION_MAJOR)
100 {
101 yCError(RGBDSENSORCLIENT) << "Major protocol number does not match, please verify client and server are updated. \
102 Expected: " << RGBD_INTERFACE_PROTOCOL_VERSION_MAJOR << " received: " << major;
103 return false;
104 }
105
106
107 if (minor != RGBD_INTERFACE_PROTOCOL_VERSION_MINOR)
108 {
109 yCWarning(RGBDSENSORCLIENT) << "Minor protocol number does not match, please verify client and server are updated.\
110 Expected: " << RGBD_INTERFACE_PROTOCOL_VERSION_MINOR << " received: " << minor;
111 }
112
113 streamingReader->attach(&colorFrame_StreamingPort, &depthFrame_StreamingPort);
114
115 return true;
116}

143bool RGBDSensorClient::getExtrinsicParam(yarp::sig::Matrix &extrinsic)
144{
145 yarp::os::Bottle cmd;
146 yarp::os::Bottle response;
147 cmd.addVocab32(VOCAB_RGBD_SENSOR);
148 cmd.addVocab32(VOCAB_GET);
149 cmd.addVocab32(VOCAB_EXTRINSIC_PARAM);
150 rpcPort.write(cmd, response);
151
152 // Minimal check on response, we suppose the response is always correctly formatted
153 if((response.get(0).asVocab32()) == VOCAB_FAILED)
154 {
155 extrinsic.zero();
156 return false;
157 }
158
159 return Property::copyPortable(response.get(3), extrinsic); // will it really work??
160}
