9bool GPTDevice::open(yarp::os::Searchable &config)
10{
11 // Azure settings
12 azure_api_version = config.check("api_version", yarp::os::Value("2023-05-15")).asString();
13 azure_deployment_id = std::getenv("DEPLOYMENT_ID");
14 azure_resource = std::getenv("AZURE_RESOURCE");
15
16 if (!azure_resource)
17 {
18 yWarning() << "Could not read env variable AZURE_RESOURCE. Device set in offline mode";
19 m_offline = true;
20 return true;
21 }
22
23 if (!oai.auth.SetAzureKeyEnv("AZURE_API_KEY"))
24 {
25 yWarning() << "Invalid or no azure key provided. Device set in offline mode.";
26 m_offline = true;
27 }
28
29 return true;
30}

32bool GPTDevice::ask(const std::string &question, std::string &oAnswer)
33{
34 // Adding prompt to conversation
35 m_convo->AddUserData(question);
36
37 if (m_offline)
38 {
39 yWarning() << "Device in offline mode";
40 return false;
41 }
42
43 // Asking gpt for an answer
44 try
45 {
46 liboai::Response res = oai.Azure->create_chat_completion(
47 azure_resource, azure_deployment_id, azure_api_version,
48 *m_convo);
49 m_convo->Update(res);
50 }
51 catch (const std::exception &e)
52 {
53 yError() << e.what() << '\n';
54 return false;
55 }
56
57 oAnswer = m_convo->GetLastResponse();
58 return true;
59}

85bool GPTDevice::readPrompt(std::string &oPrompt)
86{
87 auto &convo_json = m_convo->GetJSON();
88 for (auto &message : convo_json["messages"])
89 {
90 if (message["role"] == "system")
91 {
92 oPrompt = message["content"];
93 return true;

95 }

102bool GPTDevice::getConversation(std::vector<std::pair<Author, Content>> &oConversation)
103{
104 std::vector<std::pair<std::string, std::string>> conversation;
105
106 auto &convo_json = m_convo->GetJSON();
107
108 if (convo_json["messages"].empty())
109 {
110 yWarning() << "Conversation is empty!";
111 return false;
112 }
113
114 for (auto &message : convo_json["messages"])
115 {
116 std::string role = message["role"].get<std::string>();
117 std::string content = message["content"].get<std::string>();
118
119 conversation.push_back(std::make_pair(role, content));
120 }
121
122 oConversation = conversation;
123 return true;
124}
