74bool navigation2D_nws_yarp::open(Searchable& config)
75{
76 Property params;
77 params.fromString(config.toString().c_str());
78 yCDebug(NAVIGATION2D_NWS_YARP) << "Configuration: \n" << config.toString().c_str();
79
80 if (config.check("GENERAL") == false)
81 {
82 yCWarning(NAVIGATION2D_NWS_YARP) << "Missing GENERAL group, assuming default options";
83 }
84
85 Bottle& general_group = config.findGroup("GENERAL");
86 if (!general_group.check("period"))
87 {
88 yCInfo(NAVIGATION2D_NWS_YARP) << "Missing 'period' parameter. Using default value: 0.010";
89 m_period = 0.010;
90 }
91 else
92 {
93 m_period = config.find("period").asFloat64();
94 }
95
96 if (!general_group.check("name"))
97 {
98 yCInfo(NAVIGATION2D_NWS_YARP) << "Missing 'name' parameter. Using default value: " << m_local_name;
99 }
100 else
101 {
102 m_local_name = config.find("name").asString();
103 if (m_local_name.c_str()[0] != '/') { yCError(NAVIGATION2D_NWS_YARP) << "Missing '/' in name parameter"; return false; }
104 yCInfo(NAVIGATION2D_NWS_YARP) << "Using local name:" << m_local_name;
105 }
106 m_rpcPortName = m_local_name + "/rpc";
107 m_statusPortName = m_local_name + "/status:o";
108
109 if (!initialize_YARP(config))
110 {
111 yCError(NAVIGATION2D_NWS_YARP) << "Error initializing YARP ports";
112 return false;
113 }
114
115 yCInfo(NAVIGATION2D_NWS_YARP) << "Waiting for device to attach";
116
117 return true;
118}

164void navigation2D_nws_yarp::run()
165{
166 bool ok = iNav_ctrl->getNavigationStatus(m_navigation_status);
167
168 double m_stats_time_curr = yarp::os::Time::now();
169
170 std::string a_navigation_status = getStatusAsString(m_navigation_status);
171
172 if(a_navigation_status != m_prev_navigation_status
173 || m_stats_time_curr - m_stats_time_last > 5.0)
174 {
175 if (!ok)
176 {
177 yCError(NAVIGATION2D_NWS_YARP, "Unable to get Navigation Status!\n");
178 }
179 else
180 {
181 yCInfo(NAVIGATION2D_NWS_YARP) << "Running, ALL ok. Navigation status:" << a_navigation_status;
182 updateStatusPort(a_navigation_status);
183 m_prev_navigation_status = a_navigation_status;
184 }
185 m_stats_time_last = yarp::os::Time::now();
186 }

189std::string navigation2D_nws_yarp::getStatusAsString(NavigationStatusEnum status)
190{
191 if (status == navigation_status_idle) {
192 return std::string("navigation_status_idle");
193 } else if (status == navigation_status_moving) {
194 return std::string("navigation_status_moving");
195 } else if (status == navigation_status_waiting_obstacle) {
196 return std::string("navigation_status_waiting_obstacle");
197 } else if (status == navigation_status_goal_reached) {
198 return std::string("navigation_status_goal_reached");
199 } else if (status == navigation_status_aborted) {
200 return std::string("navigation_status_aborted");
201 } else if (status == navigation_status_failing) {
202 return std::string("navigation_status_failing");
203 } else if (status == navigation_status_paused) {
204 return std::string("navigation_status_paused");
205 } else if (status == navigation_status_preparing_before_move) {
206 return std::string("navigation_status_preparing_before_move");
207 } else if (status == navigation_status_thinking) {
208 return std::string("navigation_status_thinking");
209 } else if (status == navigation_status_error) {
210 return std::string("navigation_status_error");
211 }
212 return std::string("navigation_status_error");
