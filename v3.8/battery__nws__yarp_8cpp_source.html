66bool Battery_nws_yarp::read(yarp::os::ConnectionReader& connection)
67{
68 yarp::os::Bottle in;
69 yarp::os::Bottle out;
70 bool ok = in.read(connection);
71 if (!ok) {
72 return false;
73 }
74
75 // parse in, prepare out
76 int code = in.get(0).asVocab32();
77 bool ret = false;
78 if (code == VOCAB_IBATTERY)
79 {
80 int cmd = in.get(1).asVocab32();
81 if (cmd == VOCAB_BATTERY_INFO)
82 {
83 if (m_ibattery_p)
84 {
85 std::string info;
86 m_ibattery_p->getBatteryInfo(info);
87 out.addVocab32(VOCAB_IS);
88 out.addVocab32(cmd);
89 out.addString(info);
90 ret = true;
91 }
92 }
93 else
94 {
95 yCError(BATTERYWRAPPER, "Invalid vocab received");
96 }
97 }
98 else
99 {
100 yCError(BATTERYWRAPPER, "Invalid vocab received");
101 }
102
103 if (!ret)
104 {
105 out.clear();
106 out.addVocab32(VOCAB_FAILED);
107 }
108
109 yarp::os::ConnectionWriter *returnToSender = connection.getWriter();
110 if (returnToSender != nullptr)
111 {
112 out.write(*returnToSender);
113 }
114 return true;
115}

122bool Battery_nws_yarp::open(yarp::os::Searchable &config)
123{
124 Property params;
125 params.fromString(config.toString());
126
127 if (!config.check("period"))
128 {
129 m_period = 1.0;
130 yCWarning(BATTERYWRAPPER) << "Missing 'period' parameter. Assuming default value 1.0 s";
131 }
132 else
133 {
134 m_period = config.find("period").asFloat32();
135 }
136 yCInfo(BATTERYWRAPPER) << "Using period:" << m_period << "s";
137
138 if (!config.check("quitPortName"))
139 {
140 m_quitPortName = config.find("quitPortName").asString();
141 }
142
143 if (!config.check("name"))
144 {
145 yCError(BATTERYWRAPPER) << "Missing 'name' parameter. Check you configuration file; it must be like:";
146 yCError(BATTERYWRAPPER) << "--name: prefix of the ports opened by the device, e.g. /robotName/battery1";
147 yCError(BATTERYWRAPPER) << "/data:o and /rpc:i are automatically appended by the wrapper at the end";
148 return false;
149 }
150 else
151 {
152 m_streamingPortName = config.find("name").asString() + "/data:o";
153 m_rpcPortName = config.find("name").asString() + "/rpc:i";
154 }
155
156 m_enable_shutdown = config.check("enable_shutdown", Value(0), "enable/disable the automatic shutdown").asBool();
157 m_enable_log = config.check("enable_log", Value(0), "enable/disable log to file").asBool();
158
159 if(!initialize_YARP(config))
160 {
161 yCError(BATTERYWRAPPER) << m_sensorId << "Error initializing YARP ports";
162 return false;
163 }
164
165 if (m_enable_log)
166 {
167 yCInfo(BATTERYWRAPPER, "writing to log file batteryLog.txt");
168 m_logFile = fopen("batteryLog.txt", "w");
169 }
170
171 return true;
172}

194void Battery_nws_yarp::run()
195{
196 if (m_ibattery_p!=nullptr)
197 {
198 m_log_buffer[0] = 0;
199
200 //acquire data from the wrapped device
201 bool ret_sts, ret_chg, ret_vlt, ret_cur, ret_tmp;
202 {
203 double tmp;
204 ret_chg = m_ibattery_p->getBatteryCharge(tmp);
205 if (ret_chg) {
206 m_battery_charge = tmp;
207 }
208 }
209 {
210 double tmp;
211 ret_vlt = m_ibattery_p->getBatteryVoltage(tmp);
212 if (ret_vlt) {
213 m_battery_voltage = tmp;
214 }
215 }
216 {
217 double tmp;
218 ret_cur = m_ibattery_p->getBatteryCurrent(tmp);
219 if (ret_cur) {
220 m_battery_current = tmp;
221 }
222 }
223 {
224 double tmp;
225 ret_tmp = m_ibattery_p->getBatteryTemperature(tmp);
226 if (ret_tmp) {
227 m_battery_temperature = tmp;
228 }
229 }
230 {
231 IBattery::Battery_status tmp;
232 ret_sts = m_ibattery_p->getBatteryStatus(tmp);
233 if (ret_sts) {
234 m_battery_status = tmp;
235 }
236 }
237
238 if (ret_sts)
239 {
240 m_lastStateStamp.update();
241 yarp::os::Bottle& b = m_streamingPort.prepare();
242 b.clear();
243 b.addFloat64(m_battery_voltage); //0
244 b.addFloat64(m_battery_current); //1
245 b.addFloat64(m_battery_charge); //2
246 b.addFloat64(m_battery_temperature); //3
247 b.addInt32(m_battery_status); //4
248 m_streamingPort.setEnvelope(m_lastStateStamp);
249 m_streamingPort.write();
250
251 // if the battery is not charging, checks its status of charge
252 if (m_battery_status > 0.4) {
253 check_battery_status(m_battery_charge);
254 }
255
256 // save data to file
257 if (m_enable_log)
258 {
259 time_t rawtime;
260 struct tm * timeinfo;
261 time(&rawtime);
262 timeinfo = localtime(&rawtime);
263 char* battery_timestamp = asctime(timeinfo);
264 std::snprintf(m_log_buffer, 1024, "battery status: %+6.1fA % 6.1fV charge:% 6.1f%% time: %s", m_battery_current, m_battery_voltage, m_battery_charge, battery_timestamp);
265 fprintf(m_logFile, "%s", m_log_buffer);
266 }
267 }
268 else
269 {
270 yCError(BATTERYWRAPPER, "BatteryWrapper: %s: Sensor returned error", m_sensorId.c_str());
271 }
272 }

339void Battery_nws_yarp::check_battery_status(double battery_charge)
340{
341 static bool notify_15 = true;
342 static bool notify_12 = true;
343 static bool notify_10 = true;
344 static bool notify_0 = true;
345
346 if (battery_charge > 20)
347 {
348 notify_15 = true;
349 notify_12 = true;
350 notify_10 = true;
351 notify_0 = true;
352 }
353
354 if (battery_charge < 5)
355 {
356 if (notify_0)
357 {
358 if (m_enable_shutdown)
359 {
360 emergency_shutdown("CRITICAL WARNING: battery charge below critical level 5%. The robot will be stopped and the system will shutdown in 2mins.");
361 if (m_quitPortName != "") { stop_robot(m_quitPortName); }
362 notify_0 = false;
363 }
364 else
365 {
366 notify_message("CRITICAL WARNING: battery charge reached critical level 5%, but the emergency shutodown is currently disabled!");
367 notify_0 = false;
368 }
369 }
370 }
371 else if (battery_charge < 10)
372 {
373 if (notify_10) { notify_message("WARNING: battery charge below 10%"); notify_10 = false; }
374 }
375 else if (battery_charge < 12)
376 {
377 if (notify_12) { notify_message("WARNING: battery charge below 12%"); notify_12 = false; }
378 }
379 else if (battery_charge < 15)
380 {
381 if (notify_15) { notify_message("WARNING: battery charge below 15%"); notify_15 = false; }
382 }
383}
