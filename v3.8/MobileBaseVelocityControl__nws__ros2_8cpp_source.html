30bool MobileBaseVelocityControl_nws_ros2::open(yarp::os::Searchable& config)
31{
32 // node_name check
33 if (!config.check("node_name")) {
34 yCError(MOBVEL_NWS_ROS2) << "missing node_name parameter";
35 return false;
36 }
37 m_node_name = config.find("node_name").asString();
38
39 // topic_name name check
40 if (!config.check("topic_name")) {
41 yCError(MOBVEL_NWS_ROS2) << "missing topic_name parameter";
42 return false;
43 }
44 m_topic_name = config.find("topic_name").asString();
45
46 m_node = NodeCreator::createNode(m_node_name);
47 m_ros2_subscriber = m_node->create_subscription<geometry_msgs::msg::Twist>(m_topic_name,
48 10,
49 std::bind(&MobileBaseVelocityControl_nws_ros2::twist_callback, this, _1));
50
51 if (!m_ros2_subscriber)
52 {
53 yCError(MOBVEL_NWS_ROS2) << " opening " << m_topic_name << " Topic, check your ROS2 network configuration\n";
54 return false;
55 }
56
57 m_spinner = new Ros2Spinner(m_node);
58 m_spinner->start();
59
60 yCInfo(MOBVEL_NWS_ROS2) << "Waiting for device to attach";
61
62 return true;
63}
