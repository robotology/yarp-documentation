85template <class ROS_MSG>
86bool GenericSensor_nwc_ros2<ROS_MSG>::open(yarp::os::Searchable & config)
87{
88 if (!config.check("topic_name")) {
89 yCError(GENERICSENSOR_NWC_ROS2, "Missing topic_name parameter, exiting.");
90 return false;
91 }
92
93 if (!config.check("node_name"))
94 {
95 yCError(GENERICSENSOR_NWC_ROS2) << "Missing node_name parameter";
96 return false;
97 }
98
99 if (!config.check("sensor_name")) {
100 yCError(GENERICSENSOR_NWC_ROS2, "Missing sensor_name parameter, exiting.");
101 return false;
102 }
103
104 m_rosNodeName = config.find("node_name").asString();
105 if (m_rosNodeName.c_str()[0] == '/') {
106 yCError(GENERICSENSOR_NWC_ROS2) << "node name cannot begin with /";
107 return false;
108 }
109
110 m_subscriptionName = config.find("topic_name").asString();
111 if (m_subscriptionName.c_str()[0] != '/') {
112 yCError(GENERICSENSOR_NWC_ROS2) << "Missing '/' in topic_name parameter";
113 return false;
114 }
115
116 m_sensorName = config.find("sensor_name").asString();
117
118 m_node = NodeCreator::createNode(m_rosNodeName);
119 m_subscription = m_node->create_subscription<ROS_MSG>(m_subscriptionName, rclcpp::QoS(10),
120 std::bind(&GenericSensor_nwc_ros2<ROS_MSG>::subscription_callback,
121 this, std::placeholders::_1));
122
123 if (m_node == nullptr) {
124 yCError(GENERICSENSOR_NWC_ROS2) << "Opening " << m_rosNodeName << " Node creation failed, check your yarp-ROS network configuration\n";
125 return false;
126 }
127
128 if (m_subscription == nullptr) {
129 yCError(GENERICSENSOR_NWC_ROS2) << "Opening " << m_subscriptionName << " Topic failed, check your yarp-ROS network configuration\n";
130 return false;
131 }
132
133 m_internalStatus = yarp::dev::MAS_status::MAS_WAITING_FOR_FIRST_READ;
134 m_spinner = new Ros2Spinner(m_node);
135 m_spinner->start();
136
137 yCInfo(GENERICSENSOR_NWC_ROS2) << "Device opened";
138
139 return true;
140}

146 delete m_spinner;
147 yCInfo(GENERICSENSOR_NWC_ROS2) << "Closed";
148 return true;
149}
