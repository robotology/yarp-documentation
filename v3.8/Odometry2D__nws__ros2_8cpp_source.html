65bool Odometry2D_nws_ros2::open(yarp::os::Searchable &config)
66{
67 if (!config.check("period")) {
68 yCWarning(ODOMETRY2D_NWS_ROS2) << "missing 'period' parameter, using default value of" << DEFAULT_THREAD_PERIOD;
69 } else {
70 m_period = config.find("period").asFloat64();
71 }
72
73 if (!config.check("node_name")) {
74 yCError(ODOMETRY2D_NWS_ROS2) << "missing node_name parameter";
75 return false;
76 }
77 m_nodeName = config.find("node_name").asString();
78 if (m_nodeName[0] == '/') {
79 yCError(ODOMETRY2D_NWS_ROS2) << "node_name parameter cannot begin with '/'";
80 return false;
81 }
82
83 if (!config.check("topic_name")) {
84 yCError(ODOMETRY2D_NWS_ROS2) << "missing topic_name parameter";
85 return false;
86 }
87 m_topicName = config.find("topic_name").asString();
88 if (m_topicName[0] != '/') {
89 yCError(ODOMETRY2D_NWS_ROS2) << "missing initial / in topic_name parameter";
90 return false;
91 }
92
93 if (!config.check("odom_frame")) {
94 yCError(ODOMETRY2D_NWS_ROS2) << "missing odom_frame parameter";
95 return false;
96 }
97 m_odomFrame = config.find("odom_frame").asString();
98
99
100 if (!config.check("base_frame")) {
101 yCError(ODOMETRY2D_NWS_ROS2) << "missing base_frame parameter";
102 return false;
103 }
104 m_baseFrame = config.find("base_frame").asString();
105
106 rclcpp::NodeOptions node_options;
107 node_options.allow_undeclared_parameters(true);
108 node_options.automatically_declare_parameters_from_overrides(true);
109 m_node = NodeCreator::createNode(m_nodeName, node_options);
110 if (m_node == nullptr) {
111 yCError(ODOMETRY2D_NWS_ROS2) << " opening " << m_nodeName << " Node, check your yarp-ROS2 network configuration\n";
112 return false;
113 }
114
115 rclcpp::Parameter simTime( "use_sim_time", rclcpp::ParameterValue( true ) );
116 m_node->set_parameter( simTime );
117 const std::string m_tf_topic ="/tf";
118 m_publisher_tf = m_node->create_publisher<tf2_msgs::msg::TFMessage>(m_tf_topic, 10);
119
120 m_ros2Publisher_odometry = m_node->create_publisher<nav_msgs::msg::Odometry>(m_topicName, 10);
121
122 yCInfo(ODOMETRY2D_NWS_ROS2) << "Waiting for device to attach";
123 return true;
124}

130void Odometry2D_nws_ros2::run()
131{
132
133 if (m_odometry2D_interface!=nullptr && m_ros2Publisher_odometry && m_publisher_tf) {
134 yarp::dev::OdometryData odometryData;
135 double synchronized_timestamp = 0;
136 m_odometry2D_interface->getOdometry(odometryData, &synchronized_timestamp);
137
138 if (std::isnan(synchronized_timestamp) == false)
139 {
140 m_timeStamp.update(synchronized_timestamp);
141 }
142 else
143 {
144 m_timeStamp.update(yarp::os::Time::now());
145 }
146
147 nav_msgs::msg::Odometry odometryMsg;
148 odometryMsg.header.frame_id = m_odomFrame;
149 odometryMsg.child_frame_id = m_baseFrame;
150
151 odometryMsg.pose.pose.position.x = odometryData.odom_x;
152 odometryMsg.pose.pose.position.y = odometryData.odom_y;
153 odometryMsg.pose.pose.position.z = 0.0;
154 geometry_msgs::msg::Quaternion odom_quat;
155 double halfYaw = odometryData.odom_theta * DEG2RAD * 0.5;
156 double cosYaw = cos(halfYaw);
157 double sinYaw = sin(halfYaw);
158 odom_quat.x = 0;
159 odom_quat.y = 0;
160 odom_quat.z = sinYaw;
161 odom_quat.w = cosYaw;
162 odometryMsg.pose.pose.orientation = odom_quat;
163 odometryMsg.twist.twist.linear.x = odometryData.base_vel_x;
164 odometryMsg.twist.twist.linear.y = odometryData.base_vel_y;
165 odometryMsg.twist.twist.linear.z = 0;
166 odometryMsg.twist.twist.angular.x = 0;
167 odometryMsg.twist.twist.angular.y = 0;
168 odometryMsg.twist.twist.angular.z = odometryData.base_vel_theta * DEG2RAD;
169
170 // tf publisher
171 tf2_msgs::msg::TFMessage rosData;
172
173 geometry_msgs::msg::TransformStamped tsData;
174 tsData.child_frame_id = m_baseFrame;
175 tsData.header.frame_id = m_odomFrame;
176
177 tsData.transform.rotation.x = 0;
178 tsData.transform.rotation.y = 0;
179 tsData.transform.rotation.z = sinYaw;
180 tsData.transform.rotation.w = cosYaw;
181 tsData.transform.translation.x = odometryData.odom_x;
182 tsData.transform.translation.y = odometryData.odom_y;
183 tsData.transform.translation.z = 0;
184
185 odometryMsg.header.stamp.sec = int(m_timeStamp.getTime());
186 odometryMsg.header.stamp.nanosec = int(1000000000UL * (m_timeStamp.getTime() - int(m_timeStamp.getTime())));
187
188 tsData.header.stamp.sec = int(m_timeStamp.getTime());
189 tsData.header.stamp.nanosec = int(1000000000UL * (m_timeStamp.getTime() - int(m_timeStamp.getTime())));
190
191 if (rosData.transforms.size() == 0)
192 {
193 rosData.transforms.push_back(tsData);
194 }
195 else if (rosData.transforms.size() == 1)
196 {
197 rosData.transforms[0] = tsData;
198 }
199 else
200 {
201 yCWarning(ODOMETRY2D_NWS_ROS2) << "Size of /tf topic should be 1, instead it is:" << rosData.transforms.size();
202 }
203
204
205 m_ros2Publisher_odometry->publish(odometryMsg);
206
207 m_publisher_tf->publish(rosData);
208
209
210 } else{
211 yCError(ODOMETRY2D_NWS_ROS2) << "the interface is not valid";
212 }
213}
