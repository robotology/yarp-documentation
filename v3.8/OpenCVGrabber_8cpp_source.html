56bool OpenCVGrabber::open(Searchable & config) {
57 m_saidSize = false;
58 m_saidResize = false;
59 m_transpose = false;
60 m_flip_x = false;
61 m_flip_y = false;
62
63 // Are we capturing from a file or a camera ?
64 std::string file = config.check("movie", Value(""),
65 "if present, read from specified file rather than camera").asString();
66 fromFile = (file!="");
67 if (fromFile) {
68
69 // Try to open a capture object for the file
70 m_cap.open(file.c_str());
71 if (!m_cap.isOpened()) {
72 yCError(OPENCVGRABBER, "Unable to open file '%s' for capture!", file.c_str());
73 return false;
74 }
75
76 // Should we loop?
77 m_loop = config.check("loop","if present, loop movie");
78
79 } else {
80
81 m_loop = false;
82 int camera_idx =
83 config.check("camera",
84 Value(cv::VideoCaptureAPIs::CAP_ANY),
85 "if present, read from camera identified by this index").asInt32();
86 // Try to open a capture object for the first camera
87 m_cap.open(camera_idx);
88 if (!m_cap.isOpened()) {
89 yCError(OPENCVGRABBER, "Unable to open camera for capture!");
90 return false;
91 }
92 else
93 {
94 yCInfo(OPENCVGRABBER, "Capturing from camera: %d",camera_idx);
95 }
96
97 if ( config.check("framerate","if present, specifies desired camera device framerate") ) {
98 double m_fps = config.check("framerate", Value(-1)).asFloat64();
99 m_cap.set(cv::VideoCaptureProperties::CAP_PROP_FPS, m_fps);
100 }
101 }
102
103 if (config.check("flip_x", "if present, flip the image along the x-axis")) {
104 m_flip_x = true;
105 }
106
107 if (config.check("flip_y", "if present, flip the image along the y-axis")) {
108 m_flip_y = true;
109 }
110
111 if (config.check("transpose", "if present, rotate the image along of 90 degrees")) {
112 m_transpose = true;
113 }
114
115 // Extract the desired image size from the configuration if
116 // present, otherwise query the capture device
117 if (config.check("width","if present, specifies desired image width")) {
118 m_w = config.check("width", Value(0)).asInt32();
119 if (!fromFile && m_w>0) {
120 m_cap.set(cv::VideoCaptureProperties::CAP_PROP_FRAME_WIDTH, m_w);
121 }
122 } else {
123 m_w = m_cap.get(cv::VideoCaptureProperties::CAP_PROP_FRAME_WIDTH);
124 }
125
126 if (config.check("height","if present, specifies desired image height")) {
127 m_h = config.check("height", Value(0)).asInt32();
128 if (!fromFile && m_h>0) {
129 m_cap.set(cv::VideoCaptureProperties::CAP_PROP_FRAME_HEIGHT, m_h);
130 }
131 } else {
132 m_h = m_cap.get(cv::VideoCaptureProperties::CAP_PROP_FRAME_HEIGHT);
133 }
134
135 // Ignore capture properties - they are unreliable
136
137 yCInfo(OPENCVGRABBER, "OpenCVGrabber opened");
138 // Success!
139
140 // save our configuration for future reference
141 m_config.fromString(config.toString());
142
143 return true;
144
145}

174bool OpenCVGrabber::getImage(ImageOf<PixelRgb> & image) {
175 // Must have a capture object
176 if (!m_cap.isOpened()) {
177 image.zero();
178 return false;
179 }
180
181 // Callers may have not initialized the image dimensions (may happen if this device is not wrapped)
182 if (static_cast<int>(image.width()) != m_w || static_cast<int>(image.height()) != m_h) {
183 image.resize(m_w, m_h);
184 }
185
186 // Grab and retrieve a frame,
187 cv::Mat bgr;
188 m_cap.read(bgr);
189
190 if (bgr.empty() && m_loop) {
191 bool ok = open(m_config);
192 if (!ok) {
193 return false;
194 }
195 m_cap.read(bgr);
196 }
197
198 if (bgr.empty()) {
199 image.zero();
200 return false;
201 }
202
203 // Memory allocation occurs at the YARP image object
204 cv::Mat frame(image.height(), image.width(), CV_8UC3, image.getRawImage(), image.getRowSize());
205
206 // Comply with the expected pixel color of the output YARP frame
207 cv::cvtColor(bgr, frame, cv::COLOR_BGR2RGB);
208
209 if (m_transpose) {
210 cv::Mat transposed;
211 cv::transpose(frame, transposed);
212 image.resize(transposed.cols, transposed.rows); // erases previous content
213 frame = cv::Mat(image.height(), image.width(), CV_8UC3, image.getRawImage(), image.getRowSize());
214 transposed.copyTo(frame);
215 }
216
217 if (m_flip_x && m_flip_y) {
218 cv::flip(frame, frame, -1);
219 } else if (m_flip_x) {
220 cv::flip(frame, frame, 0);
221 } else if (m_flip_y) {
222 cv::flip(frame, frame, 1);
223 }
224
225 if (!m_saidSize) {
226 yCDebug(OPENCVGRABBER, "Received image of size %dx%d", frame.cols, frame.rows);
227 m_saidSize = true;
228 }
229
230 // create the timestamp
231 m_laststamp.update();
232
233 if (m_w == 0) {
234 m_w = frame.cols;
235 }
236
237 if (m_h == 0) {
238 m_h = frame.rows;
239 }
240
241 if (fromFile && (frame.cols != (!m_transpose ? m_w : m_h) || frame.rows != (!m_transpose ? m_h : m_w))) {
242 if (!m_saidResize) {
243 yCDebug(OPENCVGRABBER, "Software scaling from %dx%d to %dx%d", frame.cols, frame.rows, m_w, m_h);
244 m_saidResize = true;
245 }
246
247 cv::Mat resized;
248 cv::resize(frame, resized, {m_w, m_h});
249 image.resize(resized.cols, resized.rows); // erases previous content
250 frame = cv::Mat(image.height(), image.width(), CV_8UC3, image.getRawImage(), image.getRowSize());
251 resized.copyTo(frame);
252 }
253
254 yCTrace(OPENCVGRABBER, "%d by %d image", frame.cols, frame.rows);
255
256 return true;
257}
