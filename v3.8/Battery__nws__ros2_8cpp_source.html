58bool Battery_nws_ros2::open(yarp::os::Searchable &config)
59{
60 if (!config.check("period")) {
61 yCWarning(BATTERY_NWS_ROS2) << "missing 'period' parameter, using default value of" << DEFAULT_THREAD_PERIOD;
62 } else {
63 m_period = config.find("period").asFloat64();
64 }
65
66 if (!config.check("node_name")) {
67 yCError(BATTERY_NWS_ROS2) << "missing node_name parameter";
68 return false;
69 }
70 m_nodeName = config.find("node_name").asString();
71 if (m_nodeName[0] == '/') {
72 yCError(BATTERY_NWS_ROS2) << "node_name parameter cannot begin with '/'";
73 return false;
74 }
75
76 if (!config.check("topic_name")) {
77 yCError(BATTERY_NWS_ROS2) << "missing topic_name parameter";
78 return false;
79 }
80 m_topicName = config.find("topic_name").asString();
81 if (m_topicName[0] != '/') {
82 yCError(BATTERY_NWS_ROS2) << "missing initial / in topic_name parameter";
83 return false;
84 }
85
86 rclcpp::NodeOptions node_options;
87 node_options.allow_undeclared_parameters(true);
88 node_options.automatically_declare_parameters_from_overrides(true);
89 m_node = NodeCreator::createNode(m_nodeName, node_options);
90 if (m_node == nullptr) {
91 yCError(BATTERY_NWS_ROS2) << " opening " << m_nodeName << " Node, check your yarp-ROS2 network configuration\n";
92 return false;
93 }
94
95 m_ros2Publisher = m_node->create_publisher<sensor_msgs::msg::BatteryState>(m_topicName, 10);
96 return true;
97}

103void Battery_nws_ros2::run()
104{
105 if (m_battery_interface==nullptr)
106 {
107 yCError(BATTERY_NWS_ROS2) << "the interface is not valid";
108 }
109 else if (!m_ros2Publisher)
110 {
111 yCError(BATTERY_NWS_ROS2) << "the publisher is not ready";
112 }
113 else
114 {
115 double voltage=0;
116 double current=0;
117 double charge=0;
118 double temperature=0;
119 yarp::dev::IBattery::Battery_status status;
120 std::string battery_info;
121 m_battery_interface->getBatteryVoltage(voltage);
122 m_battery_interface->getBatteryCurrent(current);
123 m_battery_interface->getBatteryCharge(charge);
124 m_battery_interface->getBatteryStatus(status);
125 m_battery_interface->getBatteryTemperature(temperature);
126 m_battery_interface->getBatteryInfo(battery_info);
127
128 m_timeStamp.update(yarp::os::Time::now());
129
130 battMsg.voltage = voltage;
131 battMsg.current = current;
132 battMsg.temperature = temperature;
133 battMsg.charge = std::numeric_limits<double>::quiet_NaN();//std::nan("");
134 // battMsg.capacity = std::nan("");
135 // battMsg.design_capacity = std::nan("");
136 battMsg.percentage = charge;
137 battMsg.power_supply_status = 0;
138 battMsg.power_supply_health = 0;
139 battMsg.power_supply_technology = 0;
140 battMsg.present=true;
141 battMsg.location="";
142 battMsg.serial_number="";
143
144 battMsg.header.frame_id = "" ;
145 battMsg.header.stamp.sec = int(m_timeStamp.getTime());
146 battMsg.header.stamp.nanosec = int(1000000000UL * (m_timeStamp.getTime() - int(m_timeStamp.getTime())));
147
148 m_ros2Publisher->publish(battMsg);
149 }
150}
