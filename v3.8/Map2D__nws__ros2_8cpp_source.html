68bool Map2D_nws_ros2::open(yarp::os::Searchable &config)
69{
70 Property params;
71 params.fromString(config.toString());
72
73 if (!config.check("name"))
74 {
75 yCWarning(MAP2D_NWS_ROS2) << "Missing name parameter. Using:" << m_name;
76 }
77 else
78 {
79 m_name = config.find("name").asString();
80 if(m_name[0] == '/'){
81 yCError(MAP2D_NWS_ROS2) << "Nws name parameter cannot begin with an initial /";
82 return false;
83 }
84 }
85
86 m_rpcPortName = "/"+m_name+"/rpc";
87
88 //open rpc port
89 if (!m_rpcPort.open(m_rpcPortName))
90 {
91 yCError(MAP2D_NWS_ROS2, "Failed to open port %s", m_rpcPortName.c_str());
92 return false;
93 }
94 m_rpcPort.setReader(*this);
95
96 //ROS2 configuration
97 if(config.check("getmap")) m_getMapName = config.find("getmap").asString();
98 if(config.check("getmapbyname")) m_getMapByNameName = config.find("getmapbyname").asString();
99 if(config.check("roscmdparser")) m_rosCmdParserName = config.find("roscmdparser").asString();
100 if(config.check("markers_pub")) m_markersName = config.find("markers_pub").asString();
101 if (!config.check("node_name")) {
102 yCWarning(MAP2D_NWS_ROS2) << "Missing node_name parameter. Using:" << m_name;
103 m_nodeName = m_name;
104 }
105 else
106 {
107 m_nodeName = config.find("node_name").asString();
108 }
109 if(m_nodeName[0] == '/'){
110 yCError(MAP2D_NWS_ROS2) << "node_name cannot begin with an initial /";
111 return false;
112 }
113 m_node = NodeCreator::createNode(m_nodeName);
114 rmw_qos_profile_t qos;
115 qos.history = RMW_QOS_POLICY_HISTORY_SYSTEM_DEFAULT;
116 qos.depth=10;
117 rmw_time_t time;
118 time.sec=10000;
119 time.nsec = 0;
120 qos.deadline= time;
121 qos.lifespan=time;
122 qos.liveliness_lease_duration=time;
123 qos.reliability = RMW_QOS_POLICY_RELIABILITY_SYSTEM_DEFAULT;
124 qos.durability = RMW_QOS_POLICY_DURABILITY_SYSTEM_DEFAULT;
125 qos.liveliness = RMW_QOS_POLICY_LIVELINESS_SYSTEM_DEFAULT;
126 qos.avoid_ros_namespace_conventions = true;
127 m_ros2Service_getMap = m_node->create_service<nav_msgs::srv::GetMap>(m_getMapName,
128 std::bind(&Map2D_nws_ros2::getMapCallback,this,_1,_2,_3),qos );
129 m_ros2Service_getMapByName = m_node->create_service<map2d_nws_ros2_msgs::srv::GetMapByName>(m_getMapByNameName,
130 std::bind(&Map2D_nws_ros2::getMapByNameCallback,this,_1,_2,_3));
131 m_ros2Service_rosCmdParser = m_node->create_service<test_msgs::srv::BasicTypes>(m_rosCmdParserName,
132 std::bind(&Map2D_nws_ros2::rosCmdParserCallback,this,_1,_2,_3));
133
134 yCInfo(MAP2D_NWS_ROS2) << "Waiting for device to attach";
135 start();
136
137 return true;
138}

192bool Map2D_nws_ros2::updateVizMarkers()
193{
194 if (!m_ros2Publisher_markers)
195 {
196 m_ros2Publisher_markers = m_node->create_publisher<visualization_msgs::msg::MarkerArray>(m_markersName, 10);
197 }
198 builtin_interfaces::msg::Duration dur;
199 dur.sec = 0xFFFFFFFF;
200 double yarpTimeStamp = yarp::os::Time::now();
201 uint64_t time;
202 uint64_t sec_part;
203 rclcpp::Time ret;
204 time = (uint64_t)(yarpTimeStamp * 1000000000UL);
205 sec_part = (time / 1000000000UL);
206 if (sec_part > std::numeric_limits<unsigned int>::max())
207 {
208 yCWarning(MAP2D_NWS_ROS2) << "Timestamp exceeded the 64 bit representation, resetting it to 0";
209 sec_part = 0;
210 }
211 visualization_msgs::msg::Marker marker;
212 builtin_interfaces::msg::Time tt;
213 yarp::sig::Vector rpy(3);
214 yarp::math::Quaternion q;
215 visualization_msgs::msg::MarkerArray markers;
216
217 std::vector<std::string> locations;
218 int count = 1;
219 m_iMap2D->getLocationsList(locations);
220 for (auto it : locations)
221 {
222 yarp::dev::Nav2D::Map2DLocation loc;
223 m_iMap2D->getLocation(it, loc);
224
225 if(loc.map_id != m_currentMapName && m_currentMapName != "none")
226 {
227 continue;
228 }
229
230 rpy[0] = 0; //x
231 rpy[1] = 0; //y
232 rpy[2] = loc.theta / 180 * M_PI; //z
233 yarp::sig::Matrix m = yarp::math::rpy2dcm(rpy);
234 q.fromRotationMatrix(m);
235
236 marker.header.frame_id = "map";
237 tt.sec = (yarp::os::NetUint32) sec_part;;
238 marker.header.stamp = tt;
239 marker.ns = m_markersName+"_ns";
240 marker.id = count;
241 marker.type = visualization_msgs::msg::Marker::ARROW;//yarp::rosmsg::visualization_msgs::Marker::ARROW;
242 marker.action = visualization_msgs::msg::Marker::ADD;//yarp::rosmsg::visualization_msgs::Marker::ADD;
243 marker.pose.position.x = loc.x;
244 marker.pose.position.y = loc.y;
245 marker.pose.position.z = 0;
246 marker.pose.orientation.x = q.x();
247 marker.pose.orientation.y = q.y();
248 marker.pose.orientation.z = q.z();
249 marker.pose.orientation.w = q.w();
250 marker.scale.x = 1;
251 marker.scale.y = 0.1;
252 marker.scale.z = 0.1;
253 marker.color.a = 1.0;
254 marker.color.r = 0.0;
255 marker.color.g = 1.0;
256 marker.color.b = 0.0;
257 marker.lifetime = dur;
258 marker.text = it;
259 markers.markers.push_back(marker);
260 count++;
261 }
262
263 m_ros2Publisher_markers->publish(markers);
264 return true;
265}

304 nav_msgs::msg::MapMetaData metaToGo;

316 mapToGo.info.width = theMap.width();
317
318 double DEG2RAD = M_PI/180.0;
319 double tmp=0;
320 theMap.getResolution(tmp);
321 mapToGo.info.resolution=tmp;
322 double x, y, t;
323 theMap.getOrigin(x,y,t);
324 mapToGo.info.origin.position.x=x;
325 mapToGo.info.origin.position.y=y;
326 yarp::math::Quaternion q;

328 v[0]=0; v[1]=0; v[2]=1; v[3]=t*DEG2RAD;
329 q.fromAxisAngle(v);
330 mapToGo.info.origin.orientation.x = q.x();
331 mapToGo.info.origin.orientation.y = q.y();
332 mapToGo.info.origin.orientation.z = q.z();
333 mapToGo.info.origin.orientation.w = q.w();
334 mapToGo.data.resize(theMap.width()*theMap.height());

336 yarp::dev::Nav2D::XYCell cell;
337 for (cell.y=theMap.height(); cell.y-- > 0;)
338 {
339 for (cell.x=0; cell.x<theMap.width(); cell.x++)
340 {
341 theMap.getOccupancyData(cell,tmp);
342 mapToGo.data[index++]=(int)tmp;
343 }
344 }
345
346 response->map = mapToGo;
347 m_currentMapName = request->name;
348
349 metaToGo.map_load_time = mapToGo.info.map_load_time;
