78bool AudioPlayerWrapper::read(yarp::os::ConnectionReader& connection)
79{
80 yarp::os::Bottle command;
81 yarp::os::Bottle reply;
82 bool ok = command.read(connection);
83 if (!ok) {
84 return false;
85 }
86 reply.clear();
87
88 if (command.get(0).asString() == "start")
89 {
90 m_irender->startPlayback();
91 m_irender->isPlaying(m_isPlaying);
92 reply.addVocab32(VOCAB_OK);
93 }
94 else if (command.get(0).asString() == "stop")
95 {
96 m_irender->stopPlayback();
97 m_irender->isPlaying(m_isPlaying);
98 reply.addVocab32(VOCAB_OK);
99 }
100 else if (command.get(0).asString() == "sw_audio_gain")
101 {
102 double val = command.get(1).asFloat64();
103 if (val>=0)
104 {
105 m_irender->setSWGain(val);
106 reply.addVocab32(VOCAB_OK);
107 }
108 else
109 {
110 yCError(AUDIOPLAYERWRAPPER) << "Invalid audio gain";
111 reply.addVocab32(VOCAB_ERR);
112 }
113 }
114 else if (command.get(0).asString() == "hw_audio_gain")
115 {
116 double val = command.get(1).asFloat64();
117 if (val >= 0)
118 {
119 m_irender->setHWGain(val);
120 reply.addVocab32(VOCAB_OK);
121 }
122 else
123 {
124 yCError(AUDIOPLAYERWRAPPER) << "Invalid audio gain";
125 reply.addVocab32(VOCAB_ERR);
126 }
127 }
128 else if (command.get(0).asString() == "clear")
129 {
130 m_irender->resetPlaybackAudioBuffer();
131 reply.addVocab32(VOCAB_OK);
132 }
133 else if (command.get(0).asString() == "help")
134 {
135 reply.addVocab32("many");
136 reply.addString("start");
137 reply.addString("stop");
138 reply.addString("clear");
139 reply.addString("sw_audio_gain <gain>");
140 reply.addString("hw_audio_gain <gain>");
141 }
142 else
143 {
144 yCError(AUDIOPLAYERWRAPPER) << "Invalid command";
145 reply.addVocab32(VOCAB_ERR);
146 }
147
148 yarp::os::ConnectionWriter *returnToSender = connection.getWriter();
149 if (returnToSender != nullptr)
150 {
151 reply.write(*returnToSender);
152 }
153 return true;
154}

161bool AudioPlayerWrapper::open(yarp::os::Searchable &config)
162{
163 m_config.fromString(config.toString());
164
165 if (config.check("debug"))
166 {
167 m_debug_enabled = true;
168 }
169
170 if (config.check("period"))
171 {
172 m_period = config.find("period").asFloat64();
173 }
174
175 std::string name = "/audioPlayerWrapper";
176 if (config.check("name"))
177 {
178 name = config.find("name").asString();
179 }
180 m_audioInPortName = name + "/audio:i";
181 m_rpcPortName = name + "/rpc:i";
182 m_statusPortName = name + "/status:o";
183
184 if(!initialize_YARP(config) )
185 {
186 yCError(AUDIOPLAYERWRAPPER) << "Error initializing YARP ports";
187 return false;
188 }
189
190 if (config.check("playback_network_buffer_size"))
191 {
192 m_buffer_delay = config.find("playback_network_buffer_size").asFloat64();
193 }
194 yCInfo(AUDIOPLAYERWRAPPER) << "Using a 'playback_network_buffer_size' of" << m_buffer_delay << "s";
195 yCInfo(AUDIOPLAYERWRAPPER) << "Increase this value to robustify the real-time audio stream (it will increase latency too)";
196
197 if(config.check("subdevice"))
198 {
199 Property p;
200 p.fromString(config.toString(), false);
201 p.put("device", config.find("subdevice").asString());
202
203 if(!m_driver.open(p) || !m_driver.isValid())
204 {
205 yCError(AUDIOPLAYERWRAPPER) << "Failed to open subdevice.. check params";
206 return false;
207 }
208 if(!attach(&m_driver))
209 {
210 yCError(AUDIOPLAYERWRAPPER) << "Failed to open subdevice.. check params";
211 return false;
212 }
213 m_isDeviceOwned = true;
214 if (m_irender == nullptr)
215 {
216 yCError(AUDIOPLAYERWRAPPER, "m_irender is null\n");
217 return false;
218 }
219 }
220
221 return true;
222}

267void AudioPlayerWrapper::run()
268{
269 double current_time = yarp::os::Time::now();
270
271 Sound* s = m_audioInPort.read(false);
272 if (s != nullptr)
273 {
274 if (m_debug_enabled)
275 {
276 yCDebug(AUDIOPLAYERWRAPPER) << "Received sound of:" << s->getSamples() << " samples";
277 }
278
279 scheduled_sound_type ss;
280#if 1
281 //This is simple, but we don't know how big the sound is...
282 ss.scheduled_time = current_time + m_buffer_delay;
283#elif 0
284 //This is ok, but it doesn't work if the sounds have different durations...
285 ss.scheduled_time = current_time + 5.0 * s.getDuration();
286#else
287 ss.scheduled_time = current_time + m_buffer_delay > 5.0 * s.getDuration() ? (m_buffer_delay) : (5.0 * s.getDuration());
288#endif
289 ss.sound_data = *s;
290 m_sound_buffer.push(ss);
291 }
292
293 if (!m_sound_buffer.empty() && current_time > m_sound_buffer.front().scheduled_time)
294 {
295 m_irender->renderSound(m_sound_buffer.front().sound_data);
296 m_sound_buffer.pop();
297 }
298
299 m_irender->getPlaybackAudioBufferCurrentSize(m_current_buffer_size);
300 if (m_debug_enabled)
301 {
302 static double printer_wdt = yarp::os::Time::now();
303 if (yarp::os::Time::now() - printer_wdt > 1.0)
304 {
305 yCDebug(AUDIOPLAYERWRAPPER) << m_current_buffer_size.getSamples() << "/" << m_max_buffer_size.getSamples() << "samples";
306 printer_wdt = yarp::os::Time::now();
307 }
308 }
309
310 m_irender->isPlaying(m_isPlaying);
311
312 //status port
313 yarp::dev::AudioPlayerStatus status;
314 status.enabled = m_isPlaying;
315 status.current_buffer_size = m_current_buffer_size.getSamples();
316 status.max_buffer_size = m_max_buffer_size.getSamples();
317 m_statusPort.write(status);
318}
