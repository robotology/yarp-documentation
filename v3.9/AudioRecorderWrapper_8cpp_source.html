37bool AudioRecorderWrapper::open(yarp::os::Searchable& config)
38{
39 yCWarning(AUDIORECORDERWRAPPER) << "The 'AudioRecorderWrapper' device is deprecated in favour of 'audioRecorder_nws_yarp'.";
40 yCWarning(AUDIORECORDERWRAPPER) << "The old device is no longer supported, and it will be removed in next YARP releases.";
41 yCWarning(AUDIORECORDERWRAPPER) << "Please update your scripts.";
42
43 m_config.fromString(config.toString());
44
45 if (config.check("period"))
46 {
47 m_period = config.find("period").asFloat64();
48 }
49
50 // Get parameter samples_over_network
51 if (config.check("min_samples_over_network"))
52 {
53 m_min_number_of_samples_over_network = config.find("min_samples_over_network").asInt64();
54 }
55 if (config.check("max_samples_over_network"))
56 {
57 m_max_number_of_samples_over_network = config.find("max_samples_over_network").asInt64();
58 }
59 yCInfo(AUDIORECORDERWRAPPER) << "Wrapper configured to produce packets with the following size (in samples): " <<
60 m_min_number_of_samples_over_network << " < samples < " << m_max_number_of_samples_over_network;
61
62 // Get parameter samples_over_network
63 if (config.check("max_samples_timeout"))
64 {
65 m_getSound_timeout = config.find("max_samples_timeout").asFloat64();
66 }
67 yCInfo(AUDIORECORDERWRAPPER) << "Wrapper configured with max_samples_timeout: " << m_getSound_timeout << "s";
68
69 // Set the streaming port
70 std::string portname = "/audioRecorderWrapper";
71 if (config.check("name"))
72 {
73 portname= config.find("name").asString();
74 }
75 if (m_streamingPort.open(portname + "/audio:o") == false)
76 {
77 yCError(AUDIORECORDERWRAPPER) << "Unable to open port" << portname + "/audio:o";
78 return false;
79 }
80 if (m_statusPort.open(portname + "/status:o") == false)
81 {
82 yCError(AUDIORECORDERWRAPPER) << "Unable to open port" << portname + "/status:o";
83 return false;
84 }
85 // Set the RPC port
86 if (m_rpcPort.open(portname + "/rpc") == false)
87 {
88 yCError(AUDIORECORDERWRAPPER) << "Unable to open port" << portname + "/rpc";
89 return false;
90 }
91 m_rpcPort.setReader(*this);
92
93 // Subdevice check and initialization
94 if (config.check("subdevice"))
95 {
96 yarp::os::Property p;
97 p.fromString(config.toString(), false);
98 p.put("device", config.find("subdevice").asString());
99
100 if (!m_driver.open(p) || !m_driver.isValid())
101 {
102 yCError(AUDIORECORDERWRAPPER) << "Failed to open subdevice.. check params";
103 return false;
104 }
105
106 if (!attach(&m_driver))
107 {
108 yCError(AUDIORECORDERWRAPPER) << "Failed to open subdevice.. check params";
109 return false;
110 }
111 m_isDeviceOwned = true;
112 }
113
114 return true;
115}

141bool AudioRecorderWrapper::read(yarp::os::ConnectionReader& connection)
142{
143 yarp::os::Bottle command;
144 yarp::os::Bottle reply;
145 bool ok = command.read(connection);
146 if (!ok) {
147 return false;
148 }
149 reply.clear();
150
151 if (command.get(0).asString()=="start")
152 {
153 m_mic->startRecording();
154 m_mic->isRecording(m_isRecording);
155 reply.addVocab32(VOCAB_OK);
156 }
157 else if (command.get(0).asString() == "stop")
158 {
159 m_mic->stopRecording();
160 m_mic->isRecording(m_isRecording);
161 reply.addVocab32(VOCAB_OK);
162 }
163 else if (command.get(0).asString() == "sw_audio_gain")
164 {
165 double val = command.get(1).asFloat64();
166 if (val >= 0)
167 {
168 m_mic->setSWGain(val);
169 reply.addVocab32(VOCAB_OK);
170 }
171 else
172 {
173 yCError(AUDIORECORDERWRAPPER) << "Invalid audio gain";
174 reply.addVocab32(VOCAB_ERR);
175 }
176 }
177 else if (command.get(0).asString() == "hw_audio_gain")
178 {
179 double val = command.get(1).asFloat64();
180 if (val >= 0)
181 {
182 m_mic->setHWGain(val);
183 reply.addVocab32(VOCAB_OK);
184 }
185 else
186 {
187 yCError(AUDIORECORDERWRAPPER) << "Invalid audio gain";
188 reply.addVocab32(VOCAB_ERR);
189 }
190 }
191 else if (command.get(0).asString() == "clear")
192 {
193 m_mic->resetRecordingAudioBuffer();
194 reply.addVocab32(VOCAB_OK);
195 }
196 else if (command.get(0).asString() == "help")
197 {
198 reply.addVocab32("many");
199 reply.addString("start");
200 reply.addString("stop");
201 reply.addString("clear");
202 reply.addString("sw_audio_gain <gain>");
203 reply.addString("hw_audio_gain <gain>");
204 }
205 else
206 {
207 yCError(AUDIORECORDERWRAPPER) << "Invalid command";
208 reply.addVocab32(VOCAB_ERR);
209 }
210
211 yarp::os::ConnectionWriter *returnToSender = connection.getWriter();
212 if (returnToSender != nullptr)
213 {
214 reply.write(*returnToSender);
215 }
216 return true;
217}

219bool AudioRecorderWrapper::attach(PolyDriver* driver)
220{
221 if (driver->isValid())
222 {
223 driver->view(m_mic);
224 }
225
226 if (nullptr == m_mic)
227 {
228 yCError(AUDIORECORDERWRAPPER, "Subdevice passed to attach method is invalid");
229 return false;
230 }
231
232 m_dataThread = new AudioRecorderDataThread(this);
233 m_statusThread = new AudioRecorderStatusThread(this);
234 m_dataThread->setPeriod(m_period);
235 m_dataThread->start();
236 m_statusThread->start();
237
238 // Wait a little and then start if requested
239 if (m_config.check("start")) {
240 yCDebug(AUDIORECORDERWRAPPER) << "Auto start requested";
241 yarp::os::SystemClock::delaySystem(1);
242 m_mic->startRecording();
243 m_mic->isRecording(m_isRecording);
244 }
245
246 return true;
247}

268void AudioRecorderStatusThread::run()
269{
270 yarp::dev::AudioBufferSize device_buffer_current_size;
271 yarp::dev::AudioBufferSize device_buffer_max_size;
272 m_ARW->m_mic->getRecordingAudioBufferCurrentSize(device_buffer_current_size);
273 m_ARW->m_mic->getRecordingAudioBufferMaxSize(device_buffer_max_size);
274
275 if (m_ARW->m_debug_enabled)
276 {
277 static double printer_wdt = yarp::os::Time::now();
278 if (yarp::os::Time::now() - printer_wdt > 1.0)
279 {
280 yCDebug(AUDIORECORDERWRAPPER) << device_buffer_current_size.getSamples() << "/" << device_buffer_max_size.getSamples() << "samples";
281 printer_wdt = yarp::os::Time::now();
282 }
283 }
284
285
286 m_ARW->m_mic->isRecording(m_ARW->m_isRecording);
287
288 //status port
289 yarp::dev::AudioRecorderStatus status;
290 status.enabled = m_ARW->m_isRecording;
291 status.current_buffer_size = device_buffer_current_size.getSamples();
292 status.max_buffer_size = device_buffer_max_size.getSamples();
293 m_ARW->m_statusPort.write(status);
294}

296void AudioRecorderDataThread::run()
297{
298 if (0)
299 {
300 //debug: print the duration of the thread
301 double current_time = yarp::os::Time::now();
302 yCDebug(AUDIORECORDERWRAPPER) << (current_time - m_ARW->m_debug_last_time);
303 m_ARW->m_debug_last_time = current_time;
304 }
305
306 if (m_ARW->m_mic == nullptr)
307 {
308 yCError(AUDIORECORDERWRAPPER) << "The IAudioGrabberSound interface is not available yet!";
309 return;
310 }
311
312#ifdef PRINT_DEBUG_MESSAGES
313 {
314 audio_buffer_size buf_max;
315 audio_buffer_size buf_cur;
316 mic->getRecordingAudioBufferMaxSize(buf_max);
317 mic->getRecordingAudioBufferCurrentSize(buf_cur);
318 yCDebug(AUDIORECORDERWRAPPER) << "BEFORE Buffer status:" << buf_cur.getBytes() << "/" << buf_max.getBytes() << "bytes";
319 }
320#endif
321
322 yarp::sig::Sound snd;
323 m_ARW->m_mic->getSound(snd, m_ARW->m_min_number_of_samples_over_network, m_ARW->m_max_number_of_samples_over_network, m_ARW->m_getSound_timeout);
324
325 if (snd.getSamples() < m_ARW->m_min_number_of_samples_over_network ||
326 snd.getSamples() < m_ARW->m_max_number_of_samples_over_network)
327 {
328 yCWarning(AUDIORECORDERWRAPPER) << "subdevice->getSound() is not producing sounds of the requested size ("
329 << m_ARW->m_min_number_of_samples_over_network << "<"
330 << snd.getSamples() << "<"
331 << m_ARW->m_max_number_of_samples_over_network << ") failed";
332 }
333
334#ifdef PRINT_DEBUG_MESSAGES
335 {
336 audio_buffer_size buf_max;
337 audio_buffer_size buf_cur;
338 mic->getRecordingAudioBufferMaxSize(buf_max);
339 mic->getRecordingAudioBufferCurrentSize(buf_cur);
340 yCDebug(AUDIORECORDERWRAPPER) << "AFTER Buffer status:" << buf_cur.getBytes() << "/" << buf_max.getBytes() << "bytes";
341 }
342#endif
343#ifdef PRINT_DEBUG_MESSAGES
344 yCDebug(AUDIORECORDERWRAPPER) << "Sound size:" << snd.getSamples() * snd.getChannels() * snd.getBytesPerSample() << " bytes";
345 yCDebug(AUDIORECORDERWRAPPER);
346#endif
347
348 //prepare the timestamp
349 m_ARW->m_stamp.update();
350 m_ARW->m_streamingPort.setEnvelope(m_ARW->m_stamp);
351
352 //check before sending data
353 if (snd.getSamples() == 0)
354 {
355 yCError(AUDIORECORDERWRAPPER) << "Subdevice produced sound of 0 samples!";
356 return;
357 }
358 if (snd.getChannels() == 0)
359 {
360 yCError(AUDIORECORDERWRAPPER) << "Subdevice produced sound of 0 channels!";
361 return;
362 }
363 if (snd.getFrequency() == 0)
364 {
365 yCError(AUDIORECORDERWRAPPER) << "Subdevice produced sound with 0 frequency!";
366 return;
367 }
368
369 //send data
370 m_ARW->m_streamingPort.write(snd);
371}
