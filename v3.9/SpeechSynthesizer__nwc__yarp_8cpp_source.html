26bool SpeechSynthesizer_nwc_yarp::open(yarp::os::Searchable& config)
27{
28 m_local_name.clear();
29 m_thrift_server_name.clear();
30
31 if(!config.check("local"))
32 {
33 yCError(SPEECHSYNTH_NWC) << "No local name specified";
34
35 return false;
36 }
37 m_local_name = config.find("local").asString();
38
39 if(!config.check("remote"))
40 {
41 yCError(SPEECHSYNTH_NWC) << "No remote name specified";
42
43 return false;
44 }
45 m_thrift_server_name = config.find("remote").asString();
46
47 std::string thriftClientPortName = m_local_name + "/thrift:c";
48 std::string thriftServerPortName = m_thrift_server_name + "/rpc";
49
50 if(!m_thriftClientPort.open(thriftClientPortName))
51 {
52 yCError(SPEECHSYNTH_NWC) << "open() failed: could not open" << thriftClientPortName <<"check the network";
53
54 return false;
55 }
56
57 if(!yarp::os::Network::connect(thriftClientPortName, thriftServerPortName))
58 {
59 yCError(SPEECHSYNTH_NWC) << "Network::connect() failed: could not connect" << thriftClientPortName << "with" << thriftServerPortName;
60
61 return false;
62 }
63
64 if(!m_thriftClient.yarp().attachAsClient(m_thriftClientPort))
65 {
66 yCError(SPEECHSYNTH_NWC) << "Error! Cannot attach the m_thriftClientPort (" << thriftClientPortName << ") port as a client";
67
68 return false;
69 }
70
71 return true;
72}
