56bool fakeIMU::open(yarp::os::Searchable &config)
57{
58 double period;
59 if( config.check("period")) {
60 period = config.find("period").asInt32() / 1000.0;
61 setPeriod(period);
62 } else {
63 yCInfo(FAKEIMU) << "Using default period of " << DEFAULT_PERIOD << " s";
64 }
65 if (config.check("sensorName")) {
66 m_sensorName = config.find("sensorName").asString();
67 }
68 yCInfo(FAKEIMU) << "sensorName: " << m_sensorName;
69
70 constantValue = config.check("constantValue");
71
72 start();
73 return true;
74}

88void fakeIMU::run()
89{
90 static double count=10;
91
92 rpy[0] = 0;
93 rpy[1] = count * 3.14/180;
94 rpy[2] = 0;
95
96 dcm = rpy2dcm(rpy);
97 accels = gravity * dcm;
98
99 lastStamp.update();
100
101 dummy_value = count;
102 if (!constantValue) {
103 count++;
104 }
105
106 if (count >= 360) {
107 count = 0;
108 }
109}
