61static void checkRemapper(yarp::dev::PolyDriver & ddRemapper, int rand, size_t nrOfRemappedAxes)
62{
63 IPositionControl *pos = nullptr;
64 REQUIRE(ddRemapper.view(pos)); // interface position correctly opened
65 REQUIRE(pos);
66
67 int axes = 0;
68 CHECK(pos->getAxes(&axes)); // getAxes returned correctly
69 CHECK((size_t) axes == nrOfRemappedAxes); // remapper seems functional
70
71 IPositionDirect *posdir = nullptr;
72 REQUIRE(ddRemapper.view(posdir)); // direct position interface correctly opened
73 REQUIRE(posdir);
74
75 IEncoders * encs = nullptr;
76 REQUIRE(ddRemapper.view(encs)); // encoders interface correctly opened
77 REQUIRE(encs);
78
79 IControlMode *ctrlmode = nullptr;
80 REQUIRE(ddRemapper.view(ctrlmode)); // control mode interface correctly opened
81 REQUIRE(ctrlmode);
82
83 // Vector used for setting/getting data from the controlboard
84 std::vector<double> setPosition(nrOfRemappedAxes,-10),
85 setRefSpeeds(nrOfRemappedAxes,-15),
86 readedPosition(nrOfRemappedAxes,-20),
87 readedEncoders(nrOfRemappedAxes,-30);
88
89 for(size_t i=0; i < nrOfRemappedAxes; i++)
90 {
91 setPosition[i] = i*100.0+50.0 + rand;
92 setRefSpeeds[i] = i*10.0+5 + rand;
93 readedPosition[i] = -100 + rand;
94 }
95
96 // Set the control mode in position direct
97 std::vector<int> settedControlMode(nrOfRemappedAxes,VOCAB_CM_POSITION_DIRECT);
98 std::vector<int> readedControlMode(nrOfRemappedAxes,VOCAB_CM_POSITION);
99
100 CHECK(ctrlmode->setControlModes(settedControlMode.data())); // setControlModes correctly called
101
102 // Check that the read control mode is actually position direct
103 // Let's try 10 times because if the remapper is using some remotecontrolboards,
104 // it is possible that this return false if it is called before the first message
105 // has been received from the controlboard_nws_yarp
106 bool getControlModesOk = false;
107 for(int wait=0; wait < 20 && !getControlModesOk; wait++)
108 {
109 getControlModesOk = ctrlmode->getControlModes(readedControlMode.data());
110 yarp::os::Time::delay(0.005);
111 }
112 CHECK(getControlModesOk); // getControlModes correctly called
113
114 for(size_t i=0; i < nrOfRemappedAxes; i++)
115 {
116 CHECK(settedControlMode[i] == readedControlMode[i]); // Setted control mode and readed control mode match
117 }
118
119 // Test position direct methods
120
121 // Set position
122 CHECK(posdir->setPositions(setPosition.data())); // setPositions correctly called
123
124 // Set also the speeds in the mean time, so we are sure that we don't get
125 // spurios successful set/get of position because of intermediate buffers
126 CHECK(pos->setRefSpeeds(setRefSpeeds.data())); // setRefSpeeds correctly called
127
128 // Wait some time to make sure that the vector has been correctly propagated
129 // back and forth
130 yarp::os::Time::delay(0.1);
131
132 // Read position
133 CHECK(posdir->getRefPositions(readedPosition.data())); // getRefPositions correctly called
134
135 // Check that the two vector match
136 for(size_t i=0; i < nrOfRemappedAxes; i++)
137 {
138 CHECK(setPosition[i] == readedPosition[i]); // Setted position and readed ref position match
139 }
140
141 // Do a similar test for the encoders
142 // in fakeMotionControl their value is the one setted with setPosition
143 CHECK(encs->getEncoders(readedEncoders.data())); // getEncoders correctly called
144
145 // Check that the two vector match
146 for(size_t i=0; i < nrOfRemappedAxes; i++)
147 {
148 CHECK(setPosition[i] == readedEncoders[i]); // Setted position and readed encoders match
149 }
150}

153TEST_CASE("dev::ControlBoardRemapperTest", "[yarp::dev]")
154{
155 YARP_REQUIRE_PLUGIN("fakeMotionControl", "device");
156 YARP_REQUIRE_PLUGIN("controlboardremapper", "device");
157 YARP_REQUIRE_PLUGIN("controlBoard_nws_yarp", "device");
158 YARP_REQUIRE_PLUGIN("remotecontrolboardremapper", "device");
159
160 Network::setLocalMode(true);
161
162 SECTION("Test the controlboard remapper")
163 {
164 // We first allocate three fakeMotionControl boards
165 // and their wrappers that we will remap using the remapper
166 std::vector<PolyDriver *> fmcbs;
167 std::vector<PolyDriver *> wrappers;
168 fmcbs.resize(3);
169 wrappers.resize(3);
170
171 std::vector<int> fmcbsSizes;
172 fmcbsSizes.push_back(2);
173 fmcbsSizes.push_back(3);
174 fmcbsSizes.push_back(4);
175
176 std::vector<std::string> fmcbsNames;
177 fmcbsNames.push_back("fakeControlBoardA");
178 fmcbsNames.push_back("fakeControlBoardB");
179 fmcbsNames.push_back("fakeControlBoardC");
180
181 std::vector<std::string> wrapperNetworks;
182 wrapperNetworks.push_back("net_a");
183 wrapperNetworks.push_back("net_b");
184 wrapperNetworks.push_back("net_c");
185
186
187 for(int i=0; i < 3; i++)
188 {
189 fmcbs[i] = new PolyDriver();
190
191 Property p;
192
193 if(i==0) { p.fromConfig(fmcA_file_content); }
194 if(i==1) { p.fromConfig(fmcB_file_content); }
195 if(i==2) { p.fromConfig(fmcC_file_content); }
196
197 REQUIRE(fmcbs[i]->open(p)); // fakeMotionControlBoard open reported successful
198
199 IPositionControl *pos = nullptr;
200 REQUIRE(fmcbs[i]->view(pos)); // interface position correctly opened
201 REQUIRE(pos);
202
203 int axes = 0;
204 pos->getAxes(&axes);
205 CHECK(axes == fmcbsSizes[i]); // fakeMotionControlBoard seems functional
206
207 // Open the wrapper
208 wrappers[i] = new PolyDriver();
209
210 if(i==0) { p.fromConfig(wrapperA_file_content); }
211 if(i==1) { p.fromConfig(wrapperB_file_content); }
212 if(i==2) { p.fromConfig(wrapperC_file_content); }
213
214 REQUIRE(wrappers[i]->open(p)); // controlBoard_nws_yarp open reported successful
215
216 yarp::dev::IMultipleWrapper *iwrap = nullptr;
217 REQUIRE(wrappers[i]->view(iwrap)); // interface for multiple wrapper correctly opened for the controlBoard_nws_yarp
218 REQUIRE(iwrap);
219
220 PolyDriverList pdList;
221 pdList.push(fmcbs[i],wrapperNetworks[i].c_str());
222
223 CHECK(iwrap->attachAll(pdList)); // controlBoard_nws_yarp attached successfully to the device
224 }
225
226 // Create a list containing all the fake controlboards
227 yarp::dev::PolyDriverList fmcList;
228
229 for(int i=0; i < 3; i++)
230 {
231 fmcList.push(fmcbs[i],fmcbsNames[i].c_str());
232 }
233
234 // Open a controlboardremapper with the wrong axisName,
235 // and make sure that if fails during attachAll
236 PolyDriver ddRemapperWN;
237 Property pRemapperWN;
238 pRemapperWN.put("device","controlboardremapper");
239 pRemapperWN.addGroup("axesNames");
240 Bottle & axesListWN = pRemapperWN.findGroup("axesNames").addList();
241 axesListWN.addString("axisA1");
242 axesListWN.addString("axisB1");
243 axesListWN.addString("axisC1");
244 axesListWN.addString("axisB3");
245 axesListWN.addString("axisC3");
246 axesListWN.addString("axisA2");
247 axesListWN.addString("thisIsAnAxisNameThatDoNotExist");
248
249 REQUIRE(ddRemapperWN.open(pRemapperWN)); // controlboardremapper with wrong names open reported successful
250
251 yarp::dev::IMultipleWrapper *imultwrapWN = nullptr;
252 REQUIRE(ddRemapperWN.view(imultwrapWN)); // interface for multiple wrapper with wrong names correctly opened
253 REQUIRE(imultwrapWN);
254
255 CHECK_FALSE(imultwrapWN->attachAll(fmcList)); // attachAll for controlboardremapper with wrong names successful
256
257 // Make sure that a controlboard in which attachAll is not successfull
258 // closes correctly
259 CHECK(ddRemapperWN.close()); // controlboardremapper with wrong names close was successful
260
261 // Open the controlboardremapper
262 PolyDriver ddRemapper;
263 Property pRemapper;
264 pRemapper.put("device","controlboardremapper");
265 pRemapper.addGroup("axesNames");
266 Bottle & axesList = pRemapper.findGroup("axesNames").addList();
267 axesList.addString("axisA1");
268 axesList.addString("axisB1");
269 axesList.addString("axisC1");
270 axesList.addString("axisB3");
271 axesList.addString("axisC3");
272 axesList.addString("axisA2");
273 size_t nrOfRemappedAxes = 6;
274
275 REQUIRE(ddRemapper.open(pRemapper)); // controlboardremapper open reported successful
276
277 yarp::dev::IMultipleWrapper *imultwrap = nullptr;
278 REQUIRE(ddRemapper.view(imultwrap)); // interface for multiple wrapper correctly opened
279 REQUIRE(imultwrap);
280
281 CHECK(imultwrap->attachAll(fmcList)); // attachAll for controlboardremapper successful
282
283
284 // Test the controlboardremapper
285 checkRemapper(ddRemapper,200,nrOfRemappedAxes);
286
287 // Open the remotecontrolboardremapper
288 PolyDriver ddRemoteRemapper;
289 Property pRemoteRemapper;
290 pRemoteRemapper.put("device","remotecontrolboardremapper");
291 pRemoteRemapper.addGroup("axesNames");
292 Bottle & remoteAxesList = pRemoteRemapper.findGroup("axesNames").addList();
293 remoteAxesList.addString("axisA1");
294 remoteAxesList.addString("axisB1");
295 remoteAxesList.addString("axisC1");
296 remoteAxesList.addString("axisB3");
297 remoteAxesList.addString("axisC3");
298 remoteAxesList.addString("axisA2");
299
300 Bottle remoteControlBoards;
301 Bottle & remoteControlBoardsList = remoteControlBoards.addList();
302 remoteControlBoardsList.addString("/testRemapperRobot/a");
303 remoteControlBoardsList.addString("/testRemapperRobot/b");
304 remoteControlBoardsList.addString("/testRemapperRobot/c");
305 pRemoteRemapper.put("remoteControlBoards",remoteControlBoards.get(0));
306
307 pRemoteRemapper.put("localPortPrefix","/test/remoteControlBoardRemapper");
308
309 Property & opts = pRemoteRemapper.addGroup("REMOTE_CONTROLBOARD_OPTIONS");
310 opts.put("writeStrict","on");
311
312 REQUIRE(ddRemoteRemapper.open(pRemoteRemapper)); // remotecontrolboardremapper open reported successful, testing it
313
314 // Test the remotecontrolboardremapper
315 checkRemapper(ddRemoteRemapper,100,nrOfRemappedAxes);
316
317 // Close devices
318 imultwrap->detachAll();
319 ddRemapper.close();
320 ddRemoteRemapper.close();
321
322 for(int i=0; i < 3; i++)
323 {
324 wrappers[i]->close();
325 delete wrappers[i];
326 wrappers[i] = nullptr;
327 fmcbs[i]->close();
328 delete fmcbs[i];
329 fmcbs[i] = nullptr;
330 }
331 }
332
333 Network::setLocalMode(false);
