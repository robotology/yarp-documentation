30bool ImageRotation::create(const yarp::os::Property& options)
31{
32 //parse the user parameters
33 yarp::os::Property m_user_params;
34 yCDebug(IMAGEROTATION) << "user params:" << options.toString();
35 std::string str = options.find("carrier").asString();
36 getParamsFromCommandLine(str, m_user_params);
37 yCDebug(IMAGEROTATION) << "parsed params:" << m_user_params.toString();
38
39 if (m_user_params.check("options_rotate"))
40 {
41 m_options_rotate_str = m_user_params.find("options_rotate").asString();
42 }
43 if (m_options_rotate_str == std::string("rotate_cw"))
44 {
45 m_rot_flags = cv::ROTATE_90_CLOCKWISE;
46 }
47 else if (m_options_rotate_str == std::string("rotate_ccw"))
48 {
49 m_rot_flags = cv::ROTATE_90_COUNTERCLOCKWISE;
50 }
51 else
52 {
53 yCDebug(IMAGEROTATION) << "Invalid value of `options` parameter";
54 return false;
55 }
56 return true;
57}

72void ImageRotation::getParamsFromCommandLine(std::string carrierString, yarp::os::Property& prop)
73{
74 // Split command line string using '+' delimiter
75 std::vector<std::string> parameters;
76 split(carrierString, '+', parameters);
77
78 // Iterate over result strings
79 for (std::string param : parameters)
80 {
81 // If there is no '.', then the param is bad formatted, skip it.
82 auto pointPosition = param.find('.');
83 if (pointPosition == std::string::npos)
84 {
85 continue;
86 }
87
88 // Otherwise, separate key and value
89 std::string paramKey = param.substr(0, pointPosition);
90 yarp::os::Value paramValue;
91 std::string s = param.substr(pointPosition + 1, param.length());
92 paramValue.fromString(s.c_str());
93
94 //and append to the returned property
95 prop.put(paramKey, paramValue);
96 }
97 return;
98}
