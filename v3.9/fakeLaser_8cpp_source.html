252bool FakeLaser::LiangBarsky_clip(int edgeLeft, int edgeRight, int edgeTop, int edgeBottom,
253 XYCell_unbounded src, XYCell_unbounded dst,
254 XYCell& src_clipped, XYCell& dst_clipped)
255{
256 double t0 = 0.0; double t1 = 1.0;
257 double xdelta = double(dst.x - src.x);
258 double ydelta = double(dst.y - src.y);
259 double p, q, r;
260
261 for (int edge = 0; edge < 4; edge++)
262 {
263 if (edge == 0) { p = -xdelta; q = -(edgeLeft - src.x); }
264 if (edge == 1) { p = xdelta; q = (edgeRight - src.x); }
265 if (edge == 2) { p = -ydelta; q = -(edgeTop - src.y); }
266 if (edge == 3) { p = ydelta; q = (edgeBottom - src.y); }
267 r = q / p;
268 if (p == 0 && q < 0) {return false;} //line is outside (parallel)
269
270 if (p < 0)
271 {
272 if (r > t1) {return false;} //line is outside.
273 else if (r > t0) {t0 = r;} //line is clipped
274 }
275 else if (p > 0)
276 {
277 if (r < t0) {return false;} //line is outside.
278 else if (r < t1) {t1 = r;} //line is clipped
279 }
280 }
281
282 src_clipped.x = size_t(double(src.x) + t0 * xdelta);
283 src_clipped.y = size_t(double(src.y) + t0 * ydelta);
284 dst_clipped.x = size_t(double(src.x) + t1 * xdelta);
285 dst_clipped.y = size_t(double(src.y) + t1 * ydelta);
286
287 return true; //line is clipped
288}

290bool FakeLaser::acquireDataFromHW()
291{
292 m_laser_data.clear();
293 double t = yarp::os::Time::now();
294 static double t_orig = yarp::os::Time::now();
295 double size = (t - (t_orig));
296
297 static int test_count = 0;
298 static int test = 0;
299
300 if (m_test_mode == USE_PATTERN)
301 {
302 for (size_t i = 0; i < m_sensorsNum; i++)
303 {
304 double value = 0;
305 if (test == 0)
306 {
307 value = i / 100.0;
308 }
309 else if (test == 1)
310 {
311 value = size * 2;
312 }
313 else if (test == 2)
314 {
315 if (i <= 10) { value = 1.0 + i / 20.0; }
316 else if (i >= 90 && i <= 100) { value = 2.0 + (i - 90) / 20.0; }
317 else { value = m_min_distance; }
318 }
319 m_laser_data.push_back(value);
320 }
321
322 test_count++;
323 if (test_count == 60)
324 {
325 test_count = 0; test++; if (test > 2) { test = 0; }
326 t_orig = yarp::os::Time::now();
327 }
328 }
329 else if (m_test_mode == NO_OBSTACLES)
330 {
331 for (size_t i = 0; i < m_sensorsNum; i++)
332 {
333 m_laser_data.push_back(std::numeric_limits<double>::infinity());
334 }
335 }
336 else if (m_test_mode == USE_CONSTANT_VALUE)
337 {
338 for (size_t i = 0; i < m_sensorsNum; i++)
339 {
340 m_laser_data.push_back(m_const_value);
341 }
342 }
343 else if (m_test_mode == USE_MAPFILE)
344 {
345 if (m_loc_mode == LOC_NOT_SET)
346 {
347 //do nothing
348 }
349 else if (m_loc_mode == LOC_FROM_PORT)
350 {
351 Bottle* b = m_loc_port->read(false);
352 if (b)
353 {
354 m_robot_loc_x = b->get(0).asFloat64();
355 m_robot_loc_y = b->get(1).asFloat64();
356 m_robot_loc_t = b->get(2).asFloat64();
357 }
358 }
359 else if (m_loc_mode == LOC_FROM_CLIENT)
360 {
361 Map2DLocation loc;
362 if (m_iLoc->getCurrentPosition(loc))
363 {
364 m_robot_loc_x = loc.x;
365 m_robot_loc_y = loc.y;
366 m_robot_loc_t = loc.theta;
367 }
368 else
369 {
370 yCError(FAKE_LASER) << "Error occurred while getting current position from localization server";
371 }
372 }
373 else
374 {
375 yCDebug(FAKE_LASER) << "No localization mode selected. This branch should be not reachable.";
376 }
377
378 for (size_t i = 0; i < m_sensorsNum; i++)
379 {
380 //compute the ray in the robot reference frame
381 double ray_angle = i * m_resolution + m_min_angle;
382 double ray_target_x = m_max_distance * cos(ray_angle * DEG2RAD);
383 double ray_target_y = m_max_distance * sin(ray_angle * DEG2RAD);
384
385 //transforms the ray from the robot to the world reference frame
386 XYWorld ray_world;
387 ray_world.x = ray_target_x * cos(m_robot_loc_t * DEG2RAD) - ray_target_y * sin(m_robot_loc_t * DEG2RAD) + m_robot_loc_x;
388 ray_world.y = ray_target_x * sin(m_robot_loc_t * DEG2RAD) + ray_target_y * cos(m_robot_loc_t * DEG2RAD) + m_robot_loc_y;
389 XYCell src = m_map.world2Cell(XYWorld(m_robot_loc_x, m_robot_loc_y));
390
391 //beware! src is the robot position and it is always inside the image.
392 //dst is the end of the ray, and it can be out of the image!
393 //for this reason i am not going to call world2Cell() on dst, because that functions clips the point to the border without
394 //knowing the angular coefficient of the ray. I thus need the unclipped value, and run the LiangBarsky algorithm
395 //(which knows the angular coefficient of the ray) on it.
396 XYWorld ray_world_rot;
397 XYCell_unbounded dst;
398 ray_world_rot.x = ray_world.x * m_map.m_origin.get_ca() - ray_world.y * m_map.m_origin.get_sa();
399 ray_world_rot.y = ray_world.x * m_map.m_origin.get_sa() + ray_world.y * m_map.m_origin.get_ca();
400 dst.x = int((+ray_world_rot.x - this->m_map.m_origin.get_x()) / this->m_map.m_resolution) + 0;
401 dst.y = int((-ray_world_rot.y + this->m_map.m_origin.get_y()) / this->m_map.m_resolution) + (int)m_map.m_height - 1;
402
403 XYCell newsrc;
404 XYCell newdst;
405 double distance;
406 if (LiangBarsky_clip(0, (int)m_map.width(), 0, (int)m_map.height(),
407 XYCell_unbounded(src.x, src.y), dst, newsrc, newdst))
408 {
409 distance = checkStraightLine(src, newdst);
410 double simulated_noise = (*m_dis)(*m_gen);
411 m_laser_data.push_back(distance + simulated_noise);
412 }
413 else
414 {
415 //This should never happen, unless the robot is outside the map!
416 yDebug() << "Robot is outside the map?!";
417 m_laser_data.push_back(std::numeric_limits<double>::infinity());
418 }
419
420 }
421 }
422
423 //set the device status
424 m_device_status = yarp::dev::IRangefinder2D::DEVICE_OK_IN_USE;
425
426 return true;
427}

470void FakeLaser::trap_the_robot(double siz)
471{
472 double res;
473 m_map.getResolution(res);
474 size_t siz_cell = size_t(siz / res);
475 XYCell robot = m_map.world2Cell(XYWorld (m_robot_loc_x, m_robot_loc_y));
476 for (size_t x= robot.x- siz_cell; x< robot.x + siz_cell; x++)
477 {
478 size_t y=robot.y- siz_cell;
479 m_map.setMapFlag(XYCell(x, y), yarp::dev::Nav2D::MapGrid2D::map_flags::MAP_CELL_WALL);
480 }
481 for (size_t x = robot.x - siz_cell; x < robot.x + siz_cell; x++)
482 {
483 size_t y = robot.y + siz_cell;
484 m_map.setMapFlag(XYCell(x, y), yarp::dev::Nav2D::MapGrid2D::map_flags::MAP_CELL_WALL);
485 }
486 for (size_t y = robot.y - siz_cell; y < robot.y + siz_cell; y++)
487 {
488 size_t x = robot.x - siz_cell;
489 m_map.setMapFlag(XYCell(x, y), yarp::dev::Nav2D::MapGrid2D::map_flags::MAP_CELL_WALL);
490 }
491 for (size_t y = robot.y - siz_cell; y < robot.y + siz_cell; y++)
492 {
493 size_t x = robot.x + siz_cell;
494 m_map.setMapFlag(XYCell(x, y), yarp::dev::Nav2D::MapGrid2D::map_flags::MAP_CELL_WALL);
495 }

503bool FakeLaser::read(yarp::os::ConnectionReader& connection)
504{
505 yarp::os::Bottle command;
506 yarp::os::Bottle reply;
507 bool ok = command.read(connection);
508 if (!ok) {
509 return false;
510 }
511 reply.clear();
512
513 if (command.get(0).asString() == "trap")
514 {
515 if (command.size() == 1)
516 {
517 trap_the_robot();
518 reply.addVocab32(VOCAB_OK);
519 }
520 else if (command.size() == 2)
521 {
522 trap_the_robot(command.get(1).asFloat64());
523 reply.addVocab32(VOCAB_OK);
524 }
525 else
526 {
527 reply.addVocab32(VOCAB_ERR);
528 }
529 }
530 else if (command.get(0).asString() == "wall")
531 {
532 if (command.size() == 1)
533 {
534 wall_the_robot(1.0, 1.0);
535 wall_the_robot(1.0, 1.05);
536 reply.addVocab32(VOCAB_OK);
537 }
538 else if (command.size() == 2)
539 {
540 wall_the_robot(command.get(1).asFloat64(), 1.0);
541 wall_the_robot(command.get(1).asFloat64(), 1.05);
542 reply.addVocab32(VOCAB_OK);
543 }
544 else if (command.size() == 3)
545 {
546 wall_the_robot(command.get(1).asFloat64(), command.get(2).asFloat64());
547 wall_the_robot(command.get(1).asFloat64(), command.get(2).asFloat64()+0.05);
548 reply.addVocab32(VOCAB_OK);
549 }
550 else
551 {
552 reply.addVocab32(VOCAB_ERR);
553 }
554 }
555 else if (command.get(0).asString() == "free")
556 {
557 free_the_robot();
558 reply.addVocab32(VOCAB_OK);
559 }
560 else if (command.get(0).asString() == "help")
561 {
562 reply.addVocab32("many");
563 reply.addString("wall <size> <distance>: creates a wall in front of the robot");
564 reply.addString("trap <size>: traps the robot in a box obstacle");
565 reply.addString("free: removes all generated obstacles");
566 }
567 else
568 {
569 yCError(FAKE_LASER) << "Invalid command";
570 reply.addVocab32(VOCAB_ERR);
571 }
572
573 yarp::os::ConnectionWriter* returnToSender = connection.getWriter();
574 if (returnToSender != nullptr)
575 {
576 reply.write(*returnToSender);
577 }
578 return true;
579}

582void FakeLaser::drawStraightLine(XYCell src, XYCell dst)
583{
584 long int x, y;
585 long int dx, dy, dx1, dy1, px, py, xe, ye, i;
586 dx = (long int)dst.x - (long int)src.x;
587 dy = (long int)dst.y - (long int)src.y;
588 dx1 = abs(dx);
589 dy1 = abs(dy);

627 }

634 y = (long int)src.y;
635 ye = (long int)dst.y;
636 }
637 else

652 {

654 {

656 }

668double FakeLaser::checkStraightLine(XYCell src, XYCell dst)
669{
670 //BEWARE: src and dest must be already clipped and >0 in this function
671 XYCell test_point;
672 test_point.x = src.x;
673 test_point.y = src.y;
674
675 //here using the fast Bresenham algorithm
676 int dx = abs(int(dst.x - src.x));
677 int dy = abs(int(dst.y - src.y));
678 int err = dx - dy;
679
680 int sx;
681 int sy;
682 if (src.x < dst.x) { sx = 1; } else { sx = -1; }
683 if (src.y < dst.y) { sy = 1; } else { sy = -1; }
684
685 while (true)
686 {
687 //the test point is going to move from src until it reaches dst OR
688 //if it reaches the boundaries of the image
689 if (test_point.x==0 || test_point.y ==0 || test_point.x>=m_map.width() || test_point.y>=m_map.height())
690 {
691 break;
692 }
693
694 //if (m_map.isFree(src) == false)
695 if (m_map.isWall(XYCell(test_point.x,test_point.y)))
696 {
697 XYWorld world_start = m_map.cell2World(src);
698 XYWorld world_end = m_map.cell2World(XYCell(test_point.x, test_point.y));
699 double dist = sqrt(pow(world_start.x - world_end.x, 2) + pow(world_start.y - world_end.y, 2));
700 return dist;
701 }
702
703 if (test_point.x == dst.x && test_point.y == dst.y)
704 {
705 break;
706 }
707
708 int e2 = err * 2;
709 if (e2 > -dy)
710 {
711 err = err - dy;
712 test_point.x += sx;
713 }
714 if (e2 < dx)
715 {
716 err = err + dx;
717 test_point.y += sy;
718 }
719 }
720 return std::numeric_limits<double>::infinity();
721}
