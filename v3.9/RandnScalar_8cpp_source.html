71void RandnScalar::boxMuller()
72{
73 double x1 = 0.0;
74 double x2 = 0.0;
75 double w = 2.0;
76
77 while (w >= 1.0)
78 {
79 x1 = 2.0 * implementation(impl)->get() - 1.0;
80 x2 = 2.0 * implementation(impl)->get() - 1.0;
