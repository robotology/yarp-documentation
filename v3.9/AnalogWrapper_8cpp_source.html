82bool AnalogServerHandler::_handleIAnalog(yarp::os::Bottle &cmd, yarp::os::Bottle &reply)
83{
84 if (is == nullptr) {
85 return false;
86 }
87
88 const size_t msgsize=cmd.size();
89 int ret=IAnalogSensor::AS_ERROR;
90
91 int code=cmd.get(1).asVocab32();
92 switch (code)
93 {
94 case VOCAB_CALIBRATE:
95 if (msgsize == 2) {
96 ret = is->calibrateSensor();
97 } else if (msgsize > 2) {
98 size_t offset = 2;
99 Vector v(msgsize - offset);
100 for (unsigned int i = 0; i < v.size(); i++) {
101 v[i] = cmd.get(i + offset).asFloat64();
102 }
103 ret = is->calibrateSensor(v);
104 }
105 break;
106 case VOCAB_CALIBRATE_CHANNEL:
107 if (msgsize==3)
108 {
109 int ch=cmd.get(2).asInt32();
110 ret=is->calibrateChannel(ch);
111 }
112 else if (msgsize==4)
113 {
114 int ch=cmd.get(2).asInt32();
115 double v=cmd.get(3).asFloat64();
116 ret=is->calibrateChannel(ch, v);
117 }
118 break;
119 default:
120 return false;
121 }
122
123 reply.addInt32(ret);
124 return true;
125}

127bool AnalogServerHandler::read(yarp::os::ConnectionReader& connection)
128{
129 yarp::os::Bottle in;
130 yarp::os::Bottle out;
131 bool ok=in.read(connection);
132 if (!ok) {
133 return false;
134 }
135
136 // parse in, prepare out
137 int code = in.get(0).asVocab32();
138 bool ret=false;
139 if (code==VOCAB_IANALOG)
140 {
141 ret=_handleIAnalog(in, out);
142 }
143
144 if (!ret)
145 {
146 out.clear();
147 out.addVocab32(VOCAB_FAILED);
148 }
149
150 yarp::os::ConnectionWriter *returnToSender = connection.getWriter();
151 if (returnToSender!=nullptr) {
152 out.write(*returnToSender);
153 }
154 return true;
155}

250bool AnalogWrapper::openAndAttachSubDevice(Searchable &prop)
251{
252 Property p;
253 subDeviceOwned = new PolyDriver;
254 p.fromString(prop.toString());
255
256// p.setMonitor(prop.getMonitor(), "subdevice"); // pass on any monitoring
257 p.unput("device");
258 p.put("device", prop.find("subdevice").asString()); // subdevice was already checked before
259
260 // if errors occurred during open, quit here.
261 yCDebug(ANALOGWRAPPER, "opening AnalogWrapper subdevice...");
262 subDeviceOwned->open(p);
263
264 if (!subDeviceOwned->isValid())
265 {
266 yCError(ANALOGWRAPPER, "opening AnalogWrapper subdevice... FAILED\n");
267 return false;
268 }
269
270 subDeviceOwned->view(analogSensor_p);
271
272 if (analogSensor_p == nullptr)
273 {
274 yCError(ANALOGWRAPPER, "Opening IAnalogSensor interface of AnalogWrapper subdevice... FAILED\n");
275 return false;
276 }
277
278 int chNum = analogSensor_p->getChannels();
279
280 if (chNum <= 0)
281 {
282 yCError(ANALOGWRAPPER, "Calling analog sensor has invalid channels number %d.\n", chNum);
283 return false;
284 }
285
286 attach(analogSensor_p);
287 PeriodicThread::setPeriod(_rate / 1000.0);
288 return PeriodicThread::start();
289}

389bool AnalogWrapper::open(yarp::os::Searchable &config)
390{
391 yCWarning(ANALOGWRAPPER) << "The 'AnalogWrapper' device is deprecated.";
392 yCWarning(ANALOGWRAPPER) << "Possible alternatives, depending on the specific type sensor data, are:";
393 yCWarning(ANALOGWRAPPER) << "'MultipleAnalogSensorsRemapper`+`MultipleAnalogSensorsServer`, `PoseStampedRosPublisher`, `WrenchStampedRosPublisher`,`IMURosPublisher`,etc.";
394 yCWarning(ANALOGWRAPPER) << "The old device is no longer supported, and it will be deprecated in YARP 3.7 and removed in YARP 4.";
395 yCWarning(ANALOGWRAPPER) << "Please update your scripts.";
396
397 Property params;
398 params.fromString(config.toString());
399 yCTrace(ANALOGWRAPPER) << "AnalogWrapper params are: " << config.toString();
400
401 if (!config.check("period"))
402 {
403 yCError(ANALOGWRAPPER) << "AnalogWrapper: missing 'period' parameter. Check you configuration file\n";
404 return false;
405 }
406 else
407 {
408 _rate = config.find("period").asInt32();
409 }
410
411 if (config.check("deviceId"))
412 {
413 yCError(ANALOGWRAPPER) << "AnalogWrapper: the parameter 'deviceId' has been removed, please use parameter 'name' instead. \n"
414 << "e.g. In the FT wrapper configuration files of your robot, replace '<param name=""deviceId""> left_arm </param>' \n"
415 << "with '/icub/left_arm/analog:o' ";
416 return false;
417 }
418
419 if (!config.check("name"))
420 {
421 yCError(ANALOGWRAPPER) << "AnalogWrapper: missing 'name' parameter. Check you configuration file; it must be like:\n"
422 " name: full name of the port, like /robotName/deviceId/sensorType:o";
423 return false;
424 }
425 else
426 {
427 streamingPortName = config.find("name").asString();
428 }
429
430 if(!initialize_YARP(config) )
431 {
432 yCError(ANALOGWRAPPER) << sensorId << "Error initializing YARP ports";
433 return false;
434 }
435
436 // check if we need to create subdevice or if they are
437 // passed later on thorugh attachAll()
438 if(config.check("subdevice"))
439 {
440 ownDevices=true;
441 if(! openAndAttachSubDevice(config))
442 {
443 yCError(ANALOGWRAPPER, "AnalogWrapper: error while opening subdevice\n");
444 return false;
445 }
446 }
447 else
448 {
449 ownDevices=false;
450 if (!openDeferredAttach(config)) {
451 return false;
452 }
453 }
454
455 return true;

458bool AnalogWrapper::initialize_YARP(yarp::os::Searchable &params)
459{
460 // Create the list of ports
461 // port names are optional, do not check for correctness.
462 if(!params.check("ports"))
463 {
464 // if there is no "ports" section open only 1 port and use name as is.
465 if (Network::exists(streamingPortName + "/rpc:i") || Network::exists(streamingPortName))
466 {
467 yCError(ANALOGWRAPPER) << "AnalogWrapper: unable to open the analog server, address conflict";
468 return false;
469 }
470 createPort((streamingPortName ).c_str(), _rate );
471 // since createPort always return true, check the port is really been opened is done here
472 if (!Network::exists(streamingPortName + "/rpc:i")) {
473 return false;
474 }
475 }
476 else
477 {
478 Bottle *ports=params.find("ports").asList();
479
480 Value &deviceChannels = params.find("channels");
481 if (deviceChannels.isNull())
482 {
483 yCError(ANALOGWRAPPER, "AnalogWrapper: 'channels' parameters was not found in config file.");
484 return false;
485 }
486
487 int nports=ports->size();
488 int sumOfChannels = 0;
489 std::vector<AnalogPortEntry> tmpPorts;
490 tmpPorts.resize(nports);
491
492 for(size_t k=0; k<ports->size(); k++)
493 {
494 Bottle parameters=params.findGroup(ports->get(k).asString());
495
496 if (parameters.size()!=5)
497 {
498 yCError(ANALOGWRAPPER) << "AnalogWrapper: check skin port parameters in part description, I was expecting "
499 << ports->get(k).asString().c_str() << " followed by four integers";
500 yCError(ANALOGWRAPPER) << " your param is " << parameters.toString();
501 return false;
502 }
503
504 if (Network::exists(streamingPortName + "/" + std::string(ports->get(k).asString()) + "/rpc:i")
505 || Network::exists(streamingPortName + "/" + std::string(ports->get(k).asString())))
506 {
507 yCError(ANALOGWRAPPER) << "AnalogWrapper: unable to open the analog server, address conflict";
508 return false;
509 }
510 int wBase=parameters.get(1).asInt32();
511 int wTop=parameters.get(2).asInt32();
512 int base=parameters.get(3).asInt32();
513 int top=parameters.get(4).asInt32();
514
515 yCDebug(ANALOGWRAPPER) << "--> " << wBase << " " << wTop << " " << base << " " << top;
516
517 //check consistenty
518 if(wTop-wBase != top-base){
519 yCError(ANALOGWRAPPER) << "AnalogWrapper: numbers of mapped taxels do not match, check "
520 << ports->get(k).asString().c_str() << " port parameters in part description";
521 return false;
522 }
523 int portChannels = top-base+1;
524
525 tmpPorts[k].length = portChannels;
526 tmpPorts[k].offset = wBase;
527 yCDebug(ANALOGWRAPPER) << "opening port " << ports->get(k).asString().c_str();
528 tmpPorts[k].port_name = streamingPortName+ "/" + std::string(ports->get(k).asString());
529
530 sumOfChannels+=portChannels;
531 }
532 createPorts(tmpPorts, _rate);
533
534 if (sumOfChannels!=deviceChannels.asInt32())
535 {
536 yCError(ANALOGWRAPPER) << "AnalogWrapper: Total number of mapped taxels does not correspond to total taxels";
537 return false;
538 }
539 }
540 return true;

552void AnalogWrapper::run()
553{
554 int first, last, ret;
555
556 if (analogSensor_p!=nullptr)
557 {
558 ret=analogSensor_p->read(lastDataRead);
559
560 if (ret==yarp::dev::IAnalogSensor::AS_OK)
561 {
562 if (lastDataRead.size()>0)
563 {
564 if(1)
565 {
566 lastStateStamp.update();
567 // send the data on the port(s), splitting them as specified in the config file
568 for(auto& analogPort : analogPorts)
569 {
570 yarp::sig::Vector &pv = analogPort.port.prepare();
571 first = analogPort.offset;
572 if (analogPort.length == -1) { // read the max length available
573 last = lastDataRead.size()-1;
574 } else {
575 last = analogPort.offset + analogPort.length - 1;
576 }
577
578 // check vector limit
579 if(last>=(int)lastDataRead.size()){
580 yCError(ANALOGWRAPPER, )<<"AnalogWrapper: error while sending analog sensor output on port "<< analogPort.port_name
581 <<" Vector size expected to be at least "<<last<<" whereas it is "<< lastDataRead.size();
582 continue;
583 }
584 pv = lastDataRead.subVector(first, last);
585
586 analogPort.port.setEnvelope(lastStateStamp);
587 analogPort.port.write();
588 }
589 }
590 }
591 else
592 {
593 yCError(ANALOGWRAPPER, "AnalogWrapper: %s: vector size non valid: %lu", sensorId.c_str(), static_cast<unsigned long> (lastDataRead.size()));
594 }
595 }
596 else
597 {
598 switch(ret)
599 {
600 case IAnalogSensor::AS_OVF:
601 yCError(ANALOGWRAPPER, "AnalogWrapper: %s: Sensor returned overflow error (code %d).", sensorId.c_str(), ret);
602 break;
603 case IAnalogSensor::AS_TIMEOUT:
604 yCError(ANALOGWRAPPER, "AnalogWrapper: %s: Sensor returned timeout error (code %d).", sensorId.c_str(), ret);
605 break;
606 case IAnalogSensor::AS_ERROR:
607 default:
608 yCError(ANALOGWRAPPER, "AnalogWrapper: %s: Sensor returned error with code %d.", sensorId.c_str(), ret);
609 break;
610 }
611 }
612 }
