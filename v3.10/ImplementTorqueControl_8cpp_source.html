167bool ImplementTorqueControl::setRefTorques(const int n_joint, const int *joints, const double *t)
168{
169 if (!castToMapper(helper)->checkAxesIds(n_joint, joints)) {
170 return false;
171 }
172
173 yarp::dev::impl::Buffer<int> buffJoints = intBuffManager->getBuffer();
174 yarp::dev::impl::Buffer<double> buffValues = doubleBuffManager->getBuffer();
175
176 for(int idx=0; idx<n_joint; idx++)
177 {
178 buffValues[idx] = castToMapper(helper)->trqN2S(t[idx], joints[idx]);
179 buffJoints[idx] = castToMapper(helper)->toHw(joints[idx]);
180 }
181 bool ret = iTorqueRaw->setRefTorquesRaw(n_joint, buffJoints.getData(), buffValues.getData());
182
183 doubleBuffManager->releaseBuffer(buffValues);
184 intBuffManager->releaseBuffer(buffJoints);
185 return ret;
186}
