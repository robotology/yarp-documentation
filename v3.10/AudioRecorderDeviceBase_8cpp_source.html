29bool AudioRecorderDeviceBase::getSound(yarp::sig::Sound& sound, size_t min_number_of_samples, size_t max_number_of_samples, double max_samples_timeout_s)
30{
31 //check for something_to_record
32 {
33 #if AUTOMATIC_REC_START
34 if (m_isRecording == false)
35 {
36 this->startRecording();
37 }
38 #else
39 double debug_time = yarp::os::Time::now();
40 while (m_recording_enabled == false)
41 {
42 if (yarp::os::Time::now() - debug_time > 5.0)
43 {
44 yCInfo(AUDIORECORDER_BASE) << "getSound() is currently waiting. Use startRecording() to start the audio stream";
45 debug_time = yarp::os::Time::now();
46 }
47 yarp::os::SystemClock::delaySystem(c_sleep_time);
48 }
49 #endif
50 }
51
52 //prevents simultaneous start/stop/reset etc.
53 //std::lock_guard<std::mutex> lock(m_mutex); //This must be used carefully
54
55 //check on input parameters
56 if (max_number_of_samples < min_number_of_samples)
57 {
58 yCError(AUDIORECORDER_BASE) << "max_number_of_samples must be greater than min_number_of_samples!";
59 return false;
60 }
61 if (max_number_of_samples > this->m_audiorecorder_cfg.numSamples)
62 {
63 yCWarning(AUDIORECORDER_BASE) << "max_number_of_samples ("<< max_number_of_samples <<") is bigger than the internal audio buffer! It will be truncated to:" << this->m_audiorecorder_cfg.numSamples;
64 max_number_of_samples = this->m_audiorecorder_cfg.numSamples;
65 }
66
67 //wait until the desired number of samples are obtained
68 size_t buff_size = 0;
69 double start_time = yarp::os::Time::now();
70 double debug_time = yarp::os::Time::now();
71 do
72 {
73 buff_size = m_inputBuffer->size().getSamples();
74 if (buff_size >= max_number_of_samples) { break; }
75 if (buff_size >= min_number_of_samples && yarp::os::Time::now() - start_time > max_samples_timeout_s) { break; }
76 if (m_recording_enabled == false) { break; }
77
78 if (yarp::os::Time::now() - debug_time > 1.0)
79 {
80 debug_time = yarp::os::Time::now();
81 yCDebug(AUDIORECORDER_BASE) << "getSound() Buffer size is " << buff_size << "/" << max_number_of_samples << " after 1s";
82 }
83 yarp::os::SystemClock::delaySystem(c_sleep_time);
84 } while (true);
85
86 //prepare the sound data struct
87 size_t samples_to_be_copied = buff_size;
88 if (samples_to_be_copied > max_number_of_samples) {
89 samples_to_be_copied = max_number_of_samples;
90 }
91 if (sound.getChannels() != this->m_audiorecorder_cfg.numChannels || sound.getSamples() != samples_to_be_copied)
92 {
93 sound.resize(samples_to_be_copied, this->m_audiorecorder_cfg.numChannels);
94 }
95 sound.setFrequency(this->m_audiorecorder_cfg.frequency);
96
97 //fill the sound data struct, reading samples from the circular buffer
98 #if DEBUG_TIME_SPENT
99 double ct1 = yarp::os::Time::now();
100 #endif
101 for (size_t i = 0; i < samples_to_be_copied; i++)
102 {
103 for (size_t j = 0; j < this->m_audiorecorder_cfg.numChannels; j++)
104 {
105 int16_t s = (int16_t)(m_inputBuffer->read());
106 if (s > (std::numeric_limits<int16_t>::max() - m_cliptol) ||
107 s < (std::numeric_limits<int16_t>::min() + m_cliptol))
108 {
109 yCWarningThrottle(AUDIORECORDER_BASE, 0.1) << "Sound clipped!";
110 }
111 sound.set(s, i, j);
112 }
113 }
114
115 //amplify if required
116 if (m_sw_gain!=1.0) {sound.amplify(m_sw_gain);}
117
118 auto debug_p = sound.getInterleavedAudioRawData();
119 #if DEBUG_TIME_SPENT
120 double ct2 = yarp::os::Time::now();
121 yCDebug(AUDIORECORDER_BASE) << ct2 - ct1;
122 #endif
123 return true;
124}

216bool AudioRecorderDeviceBase::configureRecorderAudioDevice(yarp::os::Searchable& config, std::string device_name)
217{
218 std::string debug_cfg_string = config.toString();
219
220 m_audiorecorder_cfg.frequency = config.check("rate", Value(0), "audio sample rate (0=automatic)").asInt32();
221 m_audiorecorder_cfg.numSamples = config.check("samples", Value(0), "number of samples per network packet (0=automatic). For chunks of 1 second of recording set samples=rate. Channels number is handled internally.").asInt32();
222 m_audiorecorder_cfg.numChannels = config.check("channels", Value(0), "number of audio channels (0=automatic, max is 2)").asInt32();
223 m_hw_gain = config.check("hw_gain", Value(1.0), "HW gain").asFloat32();
224 m_sw_gain = config.check("sw_gain", Value(1.0), "SW gain").asFloat32();
225
226 if (m_audiorecorder_cfg.frequency == 0) {
227 m_audiorecorder_cfg.frequency = DEFAULT_SAMPLE_RATE;
228 }
229 if (m_audiorecorder_cfg.numChannels == 0) {
230 m_audiorecorder_cfg.numChannels = DEFAULT_NUM_CHANNELS;
231 }
232 if (m_audiorecorder_cfg.numSamples == 0) {
233 m_audiorecorder_cfg.numSamples = m_audiorecorder_cfg.frequency; // by default let's use chunks of 1 second
234 }
235
236 yCInfo(AUDIORECORDER_BASE) << "Device configured with the following options:";
237 yCInfo(AUDIORECORDER_BASE) << "Frequency:"<< m_audiorecorder_cfg.frequency;
238 yCInfo(AUDIORECORDER_BASE) << "Samples (buffer size):"<< m_audiorecorder_cfg.numSamples;
239 yCInfo(AUDIORECORDER_BASE) << "Channels:"<< m_audiorecorder_cfg.numChannels;
240 yCInfo(AUDIORECORDER_BASE) << "BytesForSample:"<< m_audiorecorder_cfg.bytesPerSample;
241 yCInfo(AUDIORECORDER_BASE) << "HW gain:" << m_hw_gain;
242 yCInfo(AUDIORECORDER_BASE) << "SW gain:" << m_sw_gain;
243
244 AudioBufferSize rec_buffer_size(m_audiorecorder_cfg.numSamples, m_audiorecorder_cfg.numChannels, m_audiorecorder_cfg.bytesPerSample);
245 if (m_inputBuffer == nullptr)
246 {
247 m_inputBuffer = new CircularAudioBuffer_16t(device_name, rec_buffer_size);
248 }
249
250 //additional options
251 m_enable_buffer_autoclear = config.check("buffer_autoclear", Value(true), "Automatically clear the buffer every time the devices is started/stopped").asBool();
252 m_audiobase_debug = config.check("debug", Value(false), "Enable debug mode").asBool();
253
254 return true;
