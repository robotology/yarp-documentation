26bool RobotDescriptionClient::open(yarp::os::Searchable &config)

29
30 if (!m_rpc_port.open(m_local))

32 yCError(ROBOTDESCRIPTIONCLIENT, "open(): Could not open rpc port %s, check network", m_local.c_str());

57bool RobotDescriptionClient::getAllDevicesByType(const std::string &type, std::vector<DeviceDescription>& dev_list)
58{
59 yarp::os::Bottle b;
60 yarp::os::Bottle resp;
61 dev_list.clear();
62
63 b.addVocab32(VOCAB_IROBOT_DESCRIPTION);
64 b.addVocab32(VOCAB_IROBOT_GET);
65 b.addVocab32(VOCAB_IROBOT_BY_TYPE);
66 b.addString(type);
67 bool ret = m_rpc_port.write(b, resp);
68 if (ret)
69 {
70 if (resp.get(0).asVocab32() != VOCAB_OK)
71 {
72 yCError(ROBOTDESCRIPTIONCLIENT) << "getAllDevices(): Received error from server";
73 return false;
74 }
75 else
76 {
77 Bottle *b = resp.get(1).asList();
78 for (size_t i = 0; i < b->size(); i += 2)
79 {
80 DeviceDescription desc;
81 desc.device_name = b->get(i).asString();
82 desc.device_type = b->get(i + 1).asString();
83 dev_list.push_back(desc);
84 }
85 return true;
86 }
87 }
88 else
89 {
90 yCError(ROBOTDESCRIPTIONCLIENT) << "Error on writing on rpc port";
91 return false;
92 }
93 return true;
94}
