61 yarp::robotinterface::experimental::ParamList yarp::robotinterface::experimental::mergeDuplicateGroups(const yarp::robotinterface::experimental::ParamList& list)
62 {
63  yarp::robotinterface::experimental::ParamList params = list;
64  for (auto it1 = params.begin(); it1 != params.end(); ++it1) {
65  yarp::robotinterface::experimental::Param& param1 = *it1;
66  for (auto it2 = it1 + 1; it2 != params.end();) {
67  yarp::robotinterface::experimental::Param& param2 = *it2;
68  if (param1.name() == param2.name()) {
69  if (!param1.isGroup() || !param2.isGroup()) {
70  yFatal() << "Duplicate parameter \"" << param1.name() << "\" found and at least one of them is not a group.";
71  }
72  param1.value() += std::string(" ");
73  param1.value() += param2.value();
74  it2 = params.erase(it2);
75  } else
76  it2++;
77  }
78  }
79  return params;
80 }
