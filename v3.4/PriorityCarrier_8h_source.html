127  void stimulate(double t) {
128 
129  temporalStimulation = getActualStimulation(t);
130  // if it is in active state, stimulates.
131  if(!isResting(temporalStimulation))
132  {
133  temporalStimulation += stimulation;
134  // respecting priority ceiling
135  if(temporalStimulation >= STIMUL_THRESHOLD)
136  {
137  temporalStimulation = STIMUL_THRESHOLD;
138  isActive = true;
139  }
140  // updating arrival time
141  timeArrival = t;
142  }
143  }
